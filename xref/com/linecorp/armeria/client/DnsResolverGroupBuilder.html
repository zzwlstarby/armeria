46  /**
47   * Builds an {@link AddressResolverGroup} which builds {@link AddressResolver}s that update DNS caches
48   * automatically. Standard {@link DnsNameResolver} will only expire a cache entry after TTL,
49   * meaning DNS queries after TTL will always take time to resolve. A refreshing {@link AddressResolver}
50   * on the other hand updates the DNS cache automatically when TTL elapses,
51   * meaning DNS queries after TTL will retrieve a refreshed result right away. If refreshing fails,
52   * the {@link AddressResolver} will retry with {@link #refreshBackoff(Backoff)}.
53   *
54   * <p>The refreshing {@link AddressResolver} will only start auto refresh for a given hostname
55   * on the second access before TTL to avoid auto-refreshing for queries that only happen once
56   * (e.g., requests during server startup).
57   */
58  public final class DnsResolverGroupBuilder {
59  
60      private Backoff refreshBackoff = Backoff.ofDefault();
61  
62      private int minTtl = 1;
63      private int maxTtl = Integer.MAX_VALUE;
64      private int negativeTtl;
65  
66      // DnsNameResolverBuilder properties:
67  
68      private boolean traceEnabled = true;
69      @Nullable
70      private Long queryTimeoutMillis;
71      @Nullable
72      private ResolvedAddressTypes resolvedAddressTypes;
73      @Nullable
74      private Boolean recursionDesired;
75      @Nullable
76      private Integer maxQueriesPerResolve;
77      @Nullable
78      private Integer maxPayloadSize;
79      @Nullable
80      private Boolean optResourceEnabled;
81      @Nullable
82      private HostsFileEntriesResolver hostsFileEntriesResolver;
83      @Nullable
84      private DnsServerAddressStreamProvider dnsServerAddressStreamProvider;
85      @Nullable
86      private DnsQueryLifecycleObserverFactory dnsQueryLifecycleObserverFactory;
87      @Nullable
88      private List<String> searchDomains;
89      @Nullable
90      private Integer ndots;
91      @Nullable
92      private Boolean decodeIdn;
93  
94      DnsResolverGroupBuilder() {}
95  
96      /**
97       * Sets {@link Backoff} which is used when the {@link DnsNameResolver} fails to update the cache.
98       */
99      public DnsResolverGroupBuilder refreshBackoff(Backoff refreshBackoff) {
100         this.refreshBackoff = requireNonNull(refreshBackoff, "refreshBackoff");
101         return this;
102     }
103 
104     /**
105      * Sets the minimum and maximum TTL of the cached DNS resource records in seconds. If the TTL of the DNS
106      * resource record returned by the DNS server is less than the minimum TTL or greater than the maximum TTL,
107      * this resolver will ignore the TTL from the DNS server and use the minimum TTL or the maximum TTL instead
108      * respectively.
109      * The default value is {@code 1} and {@link Integer#MAX_VALUE}, which practically tells this resolver to
110      * respect the TTL from the DNS server.
111      */
112     public DnsResolverGroupBuilder ttl(int minTtl, int maxTtl) {
113         checkArgument(minTtl > 0 && minTtl <= maxTtl,
114                       "minTtl: %s, maxTtl: %s (expected: 1 <= minTtl <= maxTtl)", minTtl, maxTtl);
115         this.minTtl = minTtl;
116         this.maxTtl = maxTtl;
117         return this;
118     }
119 
120     /**
121      * Sets the TTL of the cache for the failed DNS queries in seconds. The default value is {@code 0} which
122      * means that the {@link AddressResolver} does not cache when DNS queries are failed.
123      */
124     public DnsResolverGroupBuilder negativeTtl(int negativeTtl) {
125         checkArgument(negativeTtl >= 0, "negativeTtl: %s, (expected: >= 0)", negativeTtl);
126         this.negativeTtl = negativeTtl;
127         return this;
128     }
129 
130     /**
131      * Sets if this resolver should generate detailed trace information in exception messages so that
132      * it is easier to understand the cause of resolution failure. This flag is enabled by default.
133      */
134     public DnsResolverGroupBuilder traceEnabled(boolean traceEnabled) {
135         this.traceEnabled = traceEnabled;
136         return this;
137     }
138 
139     /**
140      * Sets the timeout of each DNS query performed by this resolver.
141      *
142      * @see DnsNameResolverBuilder#queryTimeoutMillis(long)
143      */
144     public DnsResolverGroupBuilder queryTimeout(Duration queryTimeout) {
145         requireNonNull(queryTimeout, "queryTimeout");
146         checkArgument(!queryTimeout.isNegative(), "queryTimeout: %s (expected: >= 0)",
147                       queryTimeout);
148         return queryTimeoutMillis(queryTimeout.toMillis());
149     }
150 
151     /**
152      * Sets the timeout of each DNS query performed by this resolver in milliseconds.
153      *
154      * @see DnsNameResolverBuilder#queryTimeoutMillis(long)
155      */
156     public DnsResolverGroupBuilder queryTimeoutMillis(long queryTimeoutMillis) {
157         checkArgument(queryTimeoutMillis >= 0, "queryTimeoutMillis: %s (expected: >= 0)", queryTimeoutMillis);
158         this.queryTimeoutMillis = queryTimeoutMillis;
159         return this;
160     }
161 
162     /**
163      * Sets {@link ResolvedAddressTypes} which is the list of the protocol families of the address resolved.
164      *
165      * @see DnsNameResolverBuilder#resolvedAddressTypes(ResolvedAddressTypes)
166      */
167     public DnsResolverGroupBuilder resolvedAddressTypes(
168             ResolvedAddressTypes resolvedAddressTypes) {
169         this.resolvedAddressTypes = requireNonNull(resolvedAddressTypes, "resolvedAddressTypes");
170         return this;
171     }

173     /**
174      * Sets if this resolver has to send a DNS query with the RD (recursion desired) flag set.
175      *
176      * @see DnsNameResolverBuilder#recursionDesired(boolean)
177      */
178     public DnsResolverGroupBuilder recursionDesired(boolean recursionDesired) {
179         this.recursionDesired = recursionDesired;
180         return this;
181     }

183     /**
184      * Returns the maximum allowed number of DNS queries to send when resolving a host name.
185      *
186      * @see DnsNameResolverBuilder#maxQueriesPerResolve(int)
187      */
188     public DnsResolverGroupBuilder maxQueriesPerResolve(int maxQueriesPerResolve) {
189         checkArgument(maxQueriesPerResolve > 0,
190                       "maxQueriesPerResolve: %s (expected: > 0)", maxQueriesPerResolve);
191         this.maxQueriesPerResolve = maxQueriesPerResolve;
192         return this;
193     }

196      * Sets the capacity of the datagram packet buffer in bytes.
197      *
198      * @see DnsNameResolverBuilder#maxPayloadSize(int)
199      */

201         this.maxPayloadSize = maxPayloadSize;
202         return this;
203     }

205     /**
206      * Enables the automatic inclusion of a optional records that tries to give the remote DNS server a hint
207      * about how much data the resolver can read per response. Some DNSServer may not support this and so
208      * fail to answer queries.
209      *
210      * @see DnsNameResolverBuilder#optResourceEnabled(boolean)

218      * Sets {@link HostsFileEntriesResolver} which is used to first check if the hostname is locally aliased.

255      * Sets the list of search domains of the resolver.

257      * @see DnsNameResolverBuilder#searchDomains(Iterable)

259     public DnsResolverGroupBuilder searchDomains(Iterable<String> searchDomains) {

265      * Sets the search domains of the resolver.
266      *
267      * @see DnsNameResolverBuilder#searchDomains(Iterable)

269     public DnsResolverGroupBuilder searchDomains(String... searchDomains) {
270         return searchDomains(ImmutableList.copyOf(requireNonNull(searchDomains, "searchDomains")));

272 
273     /**
274      * Sets the number of dots which must appear in a name before an initial absolute query is made.
275      *
276      * @see DnsNameResolverBuilder#ndots(int)
277      */
278     public DnsResolverGroupBuilder ndots(int ndots) {
279         checkArgument(ndots >= 0, "ndots: %s (expected: >= 0)", ndots);
280         this.ndots = ndots;
281         return this;
282     }
283 
284     /**
285      * Sets if the domain and host names should be decoded to unicode when received.
286      * See <a href="https://tools.ietf.org/html/rfc3492">rfc3492</a>. This flag is enabled by default.
287      *
288      * @see DnsNameResolverBuilder#decodeIdn(boolean)
289      */
290     public DnsResolverGroupBuilder decodeIdn(boolean decodeIdn) {
291         this.decodeIdn = decodeIdn;
292         return this;
293     }
294 
295     RefreshingAddressResolverGroup build(EventLoopGroup eventLoopGroup) {
296         final Consumer<DnsNameResolverBuilder> resolverConfigurator = builder -> {
297             builder.channelType(TransportType.datagramChannelType(eventLoopGroup))
298                    .socketChannelType(TransportType.socketChannelType(eventLoopGroup))
299                    .resolveCache(NoopDnsCache.INSTANCE)
300                    .authoritativeDnsServerCache(NoopAuthoritativeDnsServerCache.INSTANCE)
301                    .cnameCache(NoopDnsCnameCache.INSTANCE)
302                    .traceEnabled(traceEnabled)
303                    .completeOncePreferredResolved(true);
304 
305             if (queryTimeoutMillis != null) {
306                 builder.queryTimeoutMillis(queryTimeoutMillis);
307             }
308             if (resolvedAddressTypes != null) {
309                 builder.resolvedAddressTypes(resolvedAddressTypes);
310             }
311             if (recursionDesired != null) {
312                 builder.recursionDesired(recursionDesired);
313             }
314             if (maxQueriesPerResolve != null) {
315                 builder.maxQueriesPerResolve(maxQueriesPerResolve);
316             }
317             if (maxPayloadSize != null) {
318                 builder.maxPayloadSize(maxPayloadSize);
319             }
320             if (optResourceEnabled != null) {
321                 builder.optResourceEnabled(optResourceEnabled);
322             }
323             if (hostsFileEntriesResolver != null) {
324                 builder.hostsFileEntriesResolver(hostsFileEntriesResolver);
325             }
326             if (dnsServerAddressStreamProvider != null) {
327                 builder.nameServerProvider(dnsServerAddressStreamProvider);
328             }
329             if (dnsQueryLifecycleObserverFactory != null) {
330                 builder.dnsQueryLifecycleObserverFactory(dnsQueryLifecycleObserverFactory);
331             }
332             if (searchDomains != null) {
333                 builder.searchDomains(searchDomains);
334             }
335             if (ndots != null) {
336                 builder.ndots(ndots);
337             }
338             if (decodeIdn != null) {
339                 builder.decodeIdn(decodeIdn);
340             }
341         };
342         return new RefreshingAddressResolverGroup(resolverConfigurator, minTtl, maxTtl, negativeTtl,
343                                                   refreshBackoff, resolvedAddressTypes);
344     }
