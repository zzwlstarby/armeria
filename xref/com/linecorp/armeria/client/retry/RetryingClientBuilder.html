17  package com.linecorp.armeria.client.retry;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  import static com.google.common.base.Preconditions.checkState;
21  
22  import java.time.Duration;
23  import java.util.function.Function;
24  
25  import com.google.common.base.MoreObjects.ToStringHelper;
26  
27  import com.linecorp.armeria.client.HttpClient;
28  import com.linecorp.armeria.common.HttpResponse;
29  
30  /**
31   * Builds a new {@link RetryingClient} or its decorator function.
32   */
33  public class RetryingClientBuilder extends AbstractRetryingClientBuilder<HttpResponse> {
34  
35      private static final int DEFAULT_CONTENT_PREVIEW_LENGTH = Integer.MAX_VALUE;
36  
37      private boolean useRetryAfter;
38  
39      private int contentPreviewLength = DEFAULT_CONTENT_PREVIEW_LENGTH;
40  
41      private final boolean needsContentInStrategy;
42  
43      /**
44       * Creates a new builder with the specified {@link RetryStrategy}.
45       */
46      RetryingClientBuilder(RetryStrategy retryStrategy) {
47          super(retryStrategy);
48          needsContentInStrategy = false;
49      }
50  
51      /**
52       * Creates a new builder with the specified {@link RetryStrategyWithContent}.
53       */
54      RetryingClientBuilder(RetryStrategyWithContent<HttpResponse> retryStrategyWithContent) {
55          super(retryStrategyWithContent);
56          needsContentInStrategy = true;
57      }
58  
59      /**
60       * Whether retry should be attempted according to the {@code retryHeader} from the server or not.
61       * The web server may request a client to retry after specific time with {@code retryAfter} header.
62       * If you want to follow the direction from the server not by your {@link Backoff}, invoke this method
63       * with the {@code useRetryAfter} with {@code true} to request after the specified delay.
64       *
65       * @param useRetryAfter {@code true} if you want to retry after using the {@code retryAfter} header.
66       *                      {@code false} otherwise
67       * @return {@link RetryingClientBuilder} to support method chaining
68       */
69      public RetryingClientBuilder useRetryAfter(boolean useRetryAfter) {
70          this.useRetryAfter = useRetryAfter;
71          return this;
72      }
73  
74      /**
75       * Sets the length of content required to determine whether to retry or not. If the total length of content
76       * exceeds this length and there's no retry condition matched, it will hand over the stream to the client.
77       * Note that this property is useful only if you specified {@link RetryStrategyWithContent} when calling
78       * this builder's constructor. The default value of this property is
79       * {@value #DEFAULT_CONTENT_PREVIEW_LENGTH}.
80       *
81       * @param contentPreviewLength the content length to preview. {@code 0} does not disable the length limit.
82       *
83       * @return {@link RetryingClientBuilder} to support method chaining
84       *
85       * @throws IllegalStateException if this builder is created with a {@link RetryStrategy} rather than
86       *                               {@link RetryStrategyWithContent}
87       * @throws IllegalArgumentException if the specified {@code contentPreviewLength} is equal to or
88       *                                  less than {@code 0}
89       */
90      public RetryingClientBuilder contentPreviewLength(int contentPreviewLength) {
91          checkState(needsContentInStrategy, "cannot set contentPreviewLength when RetryStrategy is used; " +
92                                             "Use RetryStrategyWithContent to enable this feature.");
93          checkArgument(contentPreviewLength > 0,
94                        "contentPreviewLength: %s (expected: > 0)", contentPreviewLength);
95          this.contentPreviewLength = contentPreviewLength;
96          return this;
97      }
98  
99      /**
100      * Returns a newly-created {@link RetryingClient} based on the properties of this builder.
101      */
102     public RetryingClient build(HttpClient delegate) {
103         if (needsContentInStrategy) {
104             return new RetryingClient(delegate, retryStrategyWithContent(), maxTotalAttempts(),
105                                       responseTimeoutMillisForEachAttempt(), useRetryAfter,
106                                       contentPreviewLength);
107         }
108 
109         return new RetryingClient(delegate, retryStrategy(), maxTotalAttempts(),
110                                   responseTimeoutMillisForEachAttempt(), useRetryAfter);
111     }
112 
113     /**
114      * Returns a newly-created decorator that decorates an {@link HttpClient} with a new
115      * {@link RetryingClient} based on the properties of this builder.
116      */
117     public Function<? super HttpClient, RetryingClient> newDecorator() {
118         return this::build;
119     }
120 
121     @Override
122     public String toString() {
123         final ToStringHelper stringHelper = toStringHelper().add("useRetryAfter", this.useRetryAfter);
124         if (needsContentInStrategy) {
125             stringHelper.add("contentPreviewLength", contentPreviewLength);
126         }
127         return stringHelper.toString();
128     }
129 
130     // Methods that were overridden to change the return type.
131 
132     @Override
133     public RetryingClientBuilder maxTotalAttempts(int maxTotalAttempts) {
134         return (RetryingClientBuilder) super.maxTotalAttempts(maxTotalAttempts);
135     }
136 
137     @Override
138     public RetryingClientBuilder responseTimeoutMillisForEachAttempt(
139             long responseTimeoutMillisForEachAttempt) {
140         return (RetryingClientBuilder)
141                 super.responseTimeoutMillisForEachAttempt(responseTimeoutMillisForEachAttempt);
142     }
143 
144     @Override
145     public RetryingClientBuilder responseTimeoutForEachAttempt(Duration responseTimeoutForEachAttempt) {
146         return (RetryingClientBuilder) super.responseTimeoutForEachAttempt(responseTimeoutForEachAttempt);
147     }
148 }
