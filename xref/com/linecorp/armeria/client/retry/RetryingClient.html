124     /**
125      * Creates a new instance that decorates the specified {@link HttpClient}.
126      */
127     RetryingClient(HttpClient delegate,
128                    RetryStrategy retryStrategy, int totalMaxAttempts,
129                    long responseTimeoutMillisForEachAttempt, boolean useRetryAfter) {
130         super(delegate, retryStrategy, totalMaxAttempts, responseTimeoutMillisForEachAttempt);
131         needsContentInStrategy = false;
132         this.useRetryAfter = useRetryAfter;
133         contentPreviewLength = 0;
134     }

136     /**
137      * Creates a new instance that decorates the specified {@link HttpClient}.
138      */
139     RetryingClient(HttpClient delegate,
140                    RetryStrategyWithContent<HttpResponse> retryStrategyWithContent, int totalMaxAttempts,
141                    long responseTimeoutMillisForEachAttempt, boolean useRetryAfter,
142                    int contentPreviewLength) {
143         super(delegate, retryStrategyWithContent, totalMaxAttempts, responseTimeoutMillisForEachAttempt);
144         needsContentInStrategy = true;
145         this.useRetryAfter = useRetryAfter;
146         checkArgument(contentPreviewLength > 0,
147                       "contentPreviewLength: %s (expected: > 0)", contentPreviewLength);
148         this.contentPreviewLength = contentPreviewLength;
149     }

151     @Override
152     protected HttpResponse doExecute(ClientRequestContext ctx, HttpRequest req) throws Exception {
153         final CompletableFuture<HttpResponse> responseFuture = new CompletableFuture<>();
154         final HttpResponse res = HttpResponse.from(responseFuture, ctx.eventLoop());
155         final HttpRequestDuplicator reqDuplicator = req.toDuplicator(ctx.eventLoop(), 0);
156         doExecute0(ctx, reqDuplicator, req, res, responseFuture);
157         return res;
158     }
159 
160     private void doExecute0(ClientRequestContext ctx, HttpRequestDuplicator rootReqDuplicator,
161                             HttpRequest originalReq, HttpResponse returnedRes,
162                             CompletableFuture<HttpResponse> future) {
163         final int totalAttempts = getTotalAttempts(ctx);
164         final boolean initialAttempt = totalAttempts <= 1;
165         // The request or response has been aborted by the client before it receives a response,
166         // so stop retrying.
167         if (originalReq.whenComplete().isCompletedExceptionally()) {
168             originalReq.whenComplete().handle((unused, cause) -> {
169                 handleException(ctx, rootReqDuplicator, future, cause, initialAttempt);
170                 return null;
171             });
172             return;
173         }
174         if (returnedRes.isComplete()) {
175             returnedRes.whenComplete().handle((result, cause) -> {
176                 final Throwable abortCause = firstNonNull(cause, AbortedStreamException.get());
177                 handleException(ctx, rootReqDuplicator, future, abortCause, initialAttempt);
178                 return null;
179             });
180             return;
181         }
182 
183         if (!setResponseTimeout(ctx)) {
184             handleException(ctx, rootReqDuplicator, future, ResponseTimeoutException.get(), initialAttempt);
185             return;
186         }
187 
188         final HttpRequest duplicateReq;
189         if (initialAttempt) {
190             duplicateReq = rootReqDuplicator.duplicate();
191         } else {
192             final RequestHeadersBuilder newHeaders = originalReq.headers().toBuilder();
193             newHeaders.setInt(ARMERIA_RETRY_COUNT, totalAttempts - 1);
194             duplicateReq = rootReqDuplicator.duplicate(newHeaders.build());
195         }
196 
197         final ClientRequestContext derivedCtx = newDerivedContext(ctx, duplicateReq, ctx.rpcRequest(),
198                                                                   initialAttempt);
199         ctx.logBuilder().addChild(derivedCtx.log());
200 
201         final HttpResponse response = executeWithFallback(delegate(), derivedCtx,
202                                                           (context, cause) -> HttpResponse.ofFailure(cause));
203 
204         derivedCtx.log().whenAvailable(RequestLogProperty.RESPONSE_HEADERS).thenAccept(log -> {
205             try {
206                 if (needsContentInStrategy) {
207                     try (HttpResponseDuplicator duplicator =
208                                  response.toDuplicator(derivedCtx.eventLoop(),
209                                                        derivedCtx.maxResponseLength())) {
210                         final ContentPreviewResponse contentPreviewResponse = new ContentPreviewResponse(
211                                 duplicator.duplicate(), contentPreviewLength);
212                         final HttpResponse duplicated = duplicator.duplicate();
213                         retryStrategyWithContent().shouldRetry(derivedCtx, contentPreviewResponse)
214                                                   .handle(handleBackoff(ctx, derivedCtx, rootReqDuplicator,
215                                                                         originalReq, returnedRes, future,
216                                                                         duplicated, duplicator::abort));
217                     }
218                 } else {
219                     final Throwable responseCause =
220                             log.isAvailable(RequestLogProperty.RESPONSE_CAUSE) ? log.responseCause() : null;
221                     final Runnable originalResClosingTask =
222                             responseCause == null ? response::abort : () -> response.abort(responseCause);
223                     retryStrategy().shouldRetry(derivedCtx, responseCause)
224                                    .handle(handleBackoff(ctx, derivedCtx, rootReqDuplicator,
225                                                          originalReq, returnedRes, future, response,
226                                                          originalResClosingTask));
227                 }
228             } catch (Throwable t) {
229                 handleException(ctx, rootReqDuplicator, future, t, false);
230             }
231         });
232     }
233 
234     private static void handleException(ClientRequestContext ctx, HttpRequestDuplicator rootReqDuplicator,
235                                         CompletableFuture<HttpResponse> future, Throwable cause,
236                                         boolean endRequestLog) {
237         if (endRequestLog) {
238             ctx.logBuilder().endRequest(cause);
239         }
240         ctx.logBuilder().endResponse(cause);
241         future.completeExceptionally(cause);
242         rootReqDuplicator.abort(cause);
243     }
244 
245     private BiFunction<Backoff, Throwable, Void> handleBackoff(ClientRequestContext ctx,
246                                                                ClientRequestContext derivedCtx,
247                                                                HttpRequestDuplicator rootReqDuplicator,
248                                                                HttpRequest originalReq,
249                                                                HttpResponse returnedRes,
250                                                                CompletableFuture<HttpResponse> future,
251                                                                HttpResponse originalRes,
252                                                                Runnable originalResClosingTask) {
253         return (backoff, unused) -> {
254             if (backoff != null) {
255                 final long millisAfter = useRetryAfter ? getRetryAfterMillis(derivedCtx) : -1;
256                 final long nextDelay = getNextDelay(ctx, backoff, millisAfter);
257                 if (nextDelay >= 0) {
258                     originalResClosingTask.run();
259                     scheduleNextRetry(
260                             ctx, cause -> handleException(ctx, rootReqDuplicator, future, cause, false),
261                             () -> doExecute0(ctx, rootReqDuplicator, originalReq, returnedRes, future),
262                             nextDelay);
263                     return null;
264                 }
265             }
266             onRetryingComplete(ctx);
267             future.complete(originalRes);
268             rootReqDuplicator.close();
269             return null;
270         };
271     }
272 
273     private static long getRetryAfterMillis(ClientRequestContext ctx) {
274         final RequestLogAccess log = ctx.log();
275         final String value;
276         if (log.isAvailable(RequestLogProperty.RESPONSE_HEADERS)) {
277             value = log.partial().responseHeaders().get(HttpHeaderNames.RETRY_AFTER);
278         } else {
279             value = null;
280         }
281 
282         if (value != null) {
283             try {
284                 return Duration.ofSeconds(Integer.parseInt(value)).toMillis();
285             } catch (Exception ignored) {
286                 // Not a second value.
287             }
288 
289             try {
290                 @SuppressWarnings("UseOfObsoleteDateTimeApi")
291                 final Date date = DateFormatter.parseHttpDate(value);
292                 if (date != null) {
293                     return date.getTime() - System.currentTimeMillis();
294                 }
295             } catch (Exception ignored) {
296                 // `parseHttpDate()` can raise an exception rather than returning `null`
297                 // when the given value has more than 64 characters.
298             }
299 
300             logger.debug("The retryAfter: {}, from the server is neither an HTTP date nor a second.",
301                          value);
302         }
303 
304         return -1;
305     }

307     private static class ContentPreviewResponse extends FilteredHttpResponse {
308 
309         private final int contentPreviewLength;
310         private int contentLength;
311         @Nullable
312         private Subscription subscription;
313 
314         ContentPreviewResponse(HttpResponse delegate, int contentPreviewLength) {
315             super(delegate);
316             this.contentPreviewLength = contentPreviewLength;
317         }
318 
319         @Override
320         protected void beforeSubscribe(Subscriber<? super HttpObject> subscriber, Subscription subscription) {
321             this.subscription = subscription;
322         }
323 
324         @Override
325         protected HttpObject filter(HttpObject obj) {
326             if (obj instanceof HttpData) {
327                 final int dataLength = ((HttpData) obj).length();
328                 contentLength += dataLength;
329                 if (contentLength >= contentPreviewLength) {
330                     assert subscription != null;
331                     subscription.cancel();
332                 }
333             }
334             return obj;
335         }
336     }
