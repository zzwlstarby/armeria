56  final class HttpHealthChecker implements AsyncCloseable {
57  
58      private static final Logger logger = LoggerFactory.getLogger(HttpHealthChecker.class);
59  
60      private static final AsciiString ARMERIA_LPHC = HttpHeaderNames.of("armeria-lphc");
61  
62      private final HealthCheckerContext ctx;
63      private final WebClient webClient;
64      private final String authority;
65      private final String path;
66      private final boolean useGet;
67      private boolean wasHealthy;
68      private int maxLongPollingSeconds;
69      private int pingIntervalSeconds;
70      @Nullable
71      private HttpResponse lastResponse;
72      private final AsyncCloseableSupport closeable = AsyncCloseableSupport.of(this::closeAsync);
73  
74      HttpHealthChecker(HealthCheckerContext ctx, String path, boolean useGet) {
75          final Endpoint endpoint = ctx.endpoint();
76          this.ctx = ctx;
77          webClient = WebClient.builder(ctx.protocol(), endpoint)
78                               .options(ctx.clientOptions())
79                               .decorator(ResponseTimeoutUpdater::new)
80                               .build();
81          authority = endpoint.authority();
82          this.path = path;
83          this.useGet = useGet;
84      }
85  
86      void start() {
87          check();
88      }
89  
90      private synchronized void check() {
91          if (closeable.isClosing()) {
92              return;
93          }
94  
95          final RequestHeaders headers;
96          final RequestHeadersBuilder builder =
97                  RequestHeaders.builder(useGet ? HttpMethod.GET : HttpMethod.HEAD, path)
98                                .add(HttpHeaderNames.AUTHORITY, authority);
99          if (maxLongPollingSeconds > 0) {
100             headers = builder.add(HttpHeaderNames.IF_NONE_MATCH, wasHealthy ? "\"healthy\"" : "\"unhealthy\"")
101                              .add(HttpHeaderNames.PREFER, "wait=" + maxLongPollingSeconds)
102                              .build();
103         } else {
104             headers = builder.build();
105         }
106 
107         try (ClientRequestContextCaptor reqCtxCaptor = Clients.newContextCaptor()) {
108             lastResponse = webClient.execute(headers);
109             final ClientRequestContext reqCtx = reqCtxCaptor.get();
110             lastResponse.subscribe(new HealthCheckResponseSubscriber(reqCtx, lastResponse),
111                                    reqCtx.eventLoop(), SubscriptionOption.WITH_POOLED_OBJECTS);
112         }
113     }
114 
115     @Override
116     public CompletableFuture<?> closeAsync() {
117         return closeable.closeAsync();
118     }
119 
120     private synchronized void closeAsync(CompletableFuture<?> future) {
121         if (lastResponse == null) {
122             // Called even before the first request is sent.
123             future.complete(null);
124         } else {
125             lastResponse.abort();
126             lastResponse.whenComplete().handle((unused1, unused2) -> future.complete(null));
127         }
128     }
129 
130     @Override
131     public void close() {
132         closeable.close();
133     }
134 
135     private final class ResponseTimeoutUpdater extends SimpleDecoratingHttpClient {
136         ResponseTimeoutUpdater(HttpClient delegate) {
137             super(delegate);
138         }
139 
140         @Override
141         public HttpResponse execute(ClientRequestContext ctx, HttpRequest req) throws Exception {
142             if (maxLongPollingSeconds > 0) {
143                 ctx.extendResponseTimeoutMillis(TimeUnit.SECONDS.toMillis(maxLongPollingSeconds));
144             }
145             return delegate().execute(ctx, req);
146         }
147     }
148 
149     private class HealthCheckResponseSubscriber implements Subscriber<HttpObject> {
150 
151         private final ClientRequestContext reqCtx;
152         private final HttpResponse res;
153         @SuppressWarnings("NotNullFieldNotInitialized")
154         private Subscription subscription;
155         private boolean isHealthy;
156         private boolean receivedExpectedResponse;
157         private boolean updatedHealth;
158 
159         @Nullable
160         private ScheduledFuture<?> pingCheckFuture;
161         private long lastPingTimeNanos;
162 
163         HealthCheckResponseSubscriber(ClientRequestContext reqCtx, HttpResponse res) {
164             this.reqCtx = reqCtx;
165             this.res = res;
166         }
167 
168         @Override
169         public void onSubscribe(Subscription subscription) {
170             this.subscription = subscription;
171             subscription.request(1);
172             maybeSchedulePingCheck();
173         }
174 
175         @Override
176         public void onNext(HttpObject obj) {
177             // There's no chance that an exception is raised from the underlying logic, so we don't do
178             // try catch not to propagate the exception to the publisher.
179             // https://github.com/reactive-streams/reactive-streams-jvm#2.13
180 
181             if (closeable.isClosing()) {
182                 subscription.cancel();
183                 return;
184             }
185 
186             try {
187                 if (!(obj instanceof ResponseHeaders)) {
188                     ReferenceCountUtil.release(obj);
189                     return;
190                 }
191 
192                 final ResponseHeaders headers = (ResponseHeaders) obj;
193                 updateLongPollingSettings(headers);
194 
195                 final HttpStatus status = headers.status();
196                 final HttpStatusClass statusClass = status.codeClass();
197                 switch (statusClass) {
198                     case INFORMATIONAL:
199                         maybeSchedulePingCheck();
200                         break;
201                     case SERVER_ERROR:
202                         receivedExpectedResponse = true;
203                         break;
204                     case SUCCESS:
205                         isHealthy = true;
206                         receivedExpectedResponse = true;
207                         break;
208                     default:
209                         if (status == HttpStatus.NOT_MODIFIED) {
210                             isHealthy = wasHealthy;
211                             receivedExpectedResponse = true;
212                         } else {
213                             // Do not use long polling on an unexpected status for safety.
214                             maxLongPollingSeconds = 0;
215 
216                             if (statusClass == HttpStatusClass.CLIENT_ERROR) {
217                                 logger.warn("{} Unexpected 4xx health check response: {} A 4xx response " +
218                                             "generally indicates a misconfiguration of the client. " +
219                                             "Did you happen to forget to configure the {}'s client options?",
220                                             reqCtx, headers, HealthCheckedEndpointGroup.class.getSimpleName());
221                             } else {
222                                 logger.warn("{} Unexpected health check response: {}", reqCtx, headers);
223                             }
224                         }
225                 }
226             } finally {
227                 subscription.request(1);
228             }
229         }
230 
231         @Override
232         public void onError(Throwable t) {
233             updateHealth();
234         }
235 
236         @Override
237         public void onComplete() {
238             updateHealth();
239         }
240 
241         private void updateLongPollingSettings(ResponseHeaders headers) {
242             final String longPollingSettings = headers.get(ARMERIA_LPHC);
243             if (longPollingSettings == null) {
244                 maxLongPollingSeconds = 0;
245                 pingIntervalSeconds = 0;
246                 return;
247             }
248 
249             final int commaPos = longPollingSettings.indexOf(',');
250             int maxLongPollingSeconds = 0;
251             int pingIntervalSeconds = 0;
252             try {
253                 maxLongPollingSeconds = Integer.max(
254                         0, Integer.parseInt(longPollingSettings.substring(0, commaPos).trim()));
255                 pingIntervalSeconds = Integer.max(
256                         0, Integer.parseInt(longPollingSettings.substring(commaPos + 1).trim()));
257             } catch (Exception e) {
258                 // Ignore malformed settings.
259             }
260 
261             HttpHealthChecker.this.maxLongPollingSeconds = maxLongPollingSeconds;
262             if (maxLongPollingSeconds > 0 && pingIntervalSeconds < maxLongPollingSeconds) {
263                 HttpHealthChecker.this.pingIntervalSeconds = pingIntervalSeconds;
264             } else {
265                 HttpHealthChecker.this.pingIntervalSeconds = 0;
266             }
267         }
268 
269         // TODO(trustin): Remove once https://github.com/line/armeria/issues/1063 is fixed.
270         private void maybeSchedulePingCheck() {
271             lastPingTimeNanos = System.nanoTime();
272 
273             if (pingCheckFuture != null) {
274                 return;
275             }
276 
277             final int pingIntervalSeconds = HttpHealthChecker.this.pingIntervalSeconds;
278             if (pingIntervalSeconds <= 0) {
279                 return;
280             }
281 
282             final long pingTimeoutNanos = TimeUnit.SECONDS.toNanos(pingIntervalSeconds) * 2;
283             pingCheckFuture = reqCtx.eventLoop().scheduleWithFixedDelay(() -> {
284                 if (System.nanoTime() - lastPingTimeNanos >= pingTimeoutNanos) {
285                     // Did not receive a ping on time.
286                     res.abort(ResponseTimeoutException.get());
287                     isHealthy = false;
288                     receivedExpectedResponse = false;
289                     updateHealth();
290                 }
291             }, 1, 1, TimeUnit.SECONDS);
292         }
293 
294         private void updateHealth() {
295             if (pingCheckFuture != null) {
296                 pingCheckFuture.cancel(false);
297             }
298 
299             if (updatedHealth) {
300                 return;
301             }
302 
303             updatedHealth = true;
304 
305             ctx.updateHealth(isHealthy ? 1 : 0);
306             wasHealthy = isHealthy;
307 
308             final ScheduledExecutorService executor = ctx.executor();
309             try {
310                 // Send a long polling check immediately if:
311                 // - Server has long polling enabled.
312                 // - Server responded with 2xx or 5xx.
313                 if (maxLongPollingSeconds > 0 && receivedExpectedResponse) {
314                     executor.execute(HttpHealthChecker.this::check);
315                 } else {
316                     executor.schedule(HttpHealthChecker.this::check,
317                                       ctx.nextDelayMillis(), TimeUnit.MILLISECONDS);
318                 }
319             } catch (RejectedExecutionException ignored) {
320                 // Can happen if the Endpoint being checked has been disappeared from
321                 // the delegate EndpointGroup. See HealthCheckedEndpointGroupTest.disappearedEndpoint().
322             }
323         }
324     }
