17  package com.linecorp.armeria.client.endpoint;
18  
19  import static java.nio.file.StandardWatchEventKinds.ENTRY_DELETE;
20  import static java.nio.file.StandardWatchEventKinds.OVERFLOW;
21  
22  import java.nio.file.ClosedWatchServiceException;
23  import java.nio.file.Path;
24  import java.nio.file.WatchEvent;
25  import java.nio.file.WatchKey;
26  import java.nio.file.WatchService;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import com.google.common.base.MoreObjects;
32  
33  import com.linecorp.armeria.client.endpoint.FileWatcherRegistry.FileSystemWatchContext;
34  
35  /**
36   * A runnable which watches files.
37   */
38  class FileWatcherRunnable implements Runnable {
39      private static final Logger logger = LoggerFactory.getLogger(FileWatcherRunnable.class);
40  
41      private final WatchService watchService;
42      private final FileSystemWatchContext fileSystemWatchContext;
43  
44      /**
45       * Initializes a runnable which watches files.
46       * @param watchService the {@code WatchService} to receive events from
47       * @param fileSystemWatchContext the context which contains target files
48       */
49      FileWatcherRunnable(WatchService watchService, FileSystemWatchContext fileSystemWatchContext) {
50          this.watchService = watchService;
51          this.fileSystemWatchContext = fileSystemWatchContext;
52      }
53  
54      @Override
55      public void run() {
56          try {
57              WatchKey key;
58              while ((key = watchService.take()) != null) {
59                  final Path dirPath = (Path) key.watchable();
60                  for (WatchEvent<?> event : key.pollEvents()) {
61                      if (event.kind().type() == Path.class) {
62                          @SuppressWarnings("unchecked")
63                          final Path filePath = dirPath.resolve(((WatchEvent<Path>) event).context());
64                          if (event.kind().equals(ENTRY_DELETE)) {
65                              logger.warn("Ignoring a deleted file: {}", filePath);
66                              continue;
67                          }
68                          runCallback(filePath);
69                      } else if (event.kind().equals(OVERFLOW)) {
70                          logger.debug("Watch events may have been lost for path: {}", dirPath);
71                          runCallback(dirPath);
72                      } else {
73                          logger.debug("Ignoring unexpected event type: {}", event.kind().name());
74                      }
75                  }
76                  final boolean reset = key.reset();
77                  if (!reset) {
78                      logger.info("Path will no longer be watched: {}", dirPath);
79                  }
80              }
81          } catch (InterruptedException e) {
82              Thread.currentThread().interrupt();
83              logger.trace("File watching thread interrupted");
84          } catch (ClosedWatchServiceException e) {
85              // do nothing
86          }
87      }
88  
89      private void runCallback(Path path) {
90          fileSystemWatchContext.watchEvents().stream().filter(
91                  ctx -> path.startsWith(ctx.dirPath())).forEach(ctx -> {
92              try {
93                  ctx.runCallback();
94              } catch (Exception e) {
95                  logger.warn("Unexpected error from listener: {} ", path, e);
96              }
97          });
98      }
99  
100     /**
101      * Utility class which contains context info for each watched path.
102      */
103     static class FileWatchEvent {
104         private final WatchKey watchKey;
105         private final Runnable callback;
106         private final Path dirPath;
107 
108         /**
109          * Initializes the context for each watched path.
110          * @param watchKey the {@link WatchKey} registered for the current path
111          * @param callback the callback which is invoked on a watch event
112          * @param dirPath path which is watched
113          */
114         FileWatchEvent(WatchKey watchKey, Runnable callback, Path dirPath) {
115             this.watchKey = watchKey;
116             this.callback = callback;
117             this.dirPath = dirPath;
118         }
119 
120         private void runCallback() {
121             callback.run();
122         }
123 
124         Path dirPath() {
125             return dirPath;
126         }
127 
128         void cancel() {
129             watchKey.cancel();
130         }
131 
132         @Override
133         public String toString() {
134             return MoreObjects.toStringHelper(this)
135                               .add("dirPath", dirPath)
136                               .add("watchable", watchKey.watchable())
137                               .add("isValid", watchKey.isValid())
138                               .toString();
139         }
140     }
141 }
