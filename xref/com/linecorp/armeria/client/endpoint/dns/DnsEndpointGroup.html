54  /**
55   * {@link DynamicEndpointGroup} which resolves targets using DNS queries. This is useful for environments
56   * where service discovery is handled using DNS, e.g. Kubernetes uses SkyDNS for service discovery.
57   */
58  abstract class DnsEndpointGroup extends DynamicEndpointGroup {
59  
60      private final EventLoop eventLoop;
61      private final int minTtl;
62      private final int maxTtl;
63      private final Backoff backoff;
64      private final List<DnsQuestion> questions;
65      private final DefaultDnsNameResolver resolver;
66      private final Logger logger;
67      private final String logPrefix;
68  
69      private boolean started;
70      @Nullable
71      private volatile ScheduledFuture<?> scheduledFuture;
72      @VisibleForTesting
73      int attemptsSoFar;
74  
75      DnsEndpointGroup(EndpointSelectionStrategy selectionStrategy,
76                       EventLoop eventLoop, int minTtl, int maxTtl,
77                       DnsServerAddressStreamProvider serverAddressStreamProvider,
78                       Backoff backoff, Iterable<DnsQuestion> questions,
79                       Consumer<DnsNameResolverBuilder> resolverConfigurator) {
80  
81          super(selectionStrategy);
82  
83          this.eventLoop = eventLoop;
84          this.minTtl = minTtl;
85          this.maxTtl = maxTtl;
86          this.backoff = backoff;
87          this.questions = ImmutableList.copyOf(questions);
88          assert !this.questions.isEmpty();
89          logger = LoggerFactory.getLogger(getClass());
90          logPrefix = this.questions.stream()
91                                    .map(DnsQuestion::name)
92                                    .distinct()
93                                    .collect(Collectors.joining(", "));
94  
95          final DnsNameResolverBuilder resolverBuilder = new DnsNameResolverBuilder(eventLoop)
96                  .channelType(TransportType.datagramChannelType(eventLoop.parent()))
97                  .ttl(minTtl, maxTtl)
98                  .traceEnabled(true)
99                  .nameServerProvider(serverAddressStreamProvider);
100 
101         resolverConfigurator.accept(resolverBuilder);
102         resolver = new DefaultDnsNameResolver(resolverBuilder.build(), eventLoop);
103     }
104 
105     final Logger logger() {
106         return logger;
107     }
108 
109     final String logPrefix() {
110         return logPrefix;
111     }
112 
113     /**
114      * Invoke this method at the end of the subclass constructor to initiate the queries.
115      */
116     final void start() {
117         checkState(!started);
118         started = true;
119         eventLoop.execute(() -> sendQueries(questions));
120     }
121 
122     private void sendQueries(List<DnsQuestion> questions) {
123         if (isClosing()) {
124             return;
125         }
126 
127         final Future<List<DnsRecord>> future = resolver.sendQueries(questions, logPrefix);
128         attemptsSoFar++;
129         future.addListener(this::onDnsRecords);
130     }
131 
132     private void onDnsRecords(Future<? super List<DnsRecord>> future) {
133         if (isClosing()) {
134             if (future.isSuccess()) {
135                 @SuppressWarnings("unchecked")
136                 final List<DnsRecord> result = (List<DnsRecord>) future.getNow();
137                 result.forEach(ReferenceCountUtil::safeRelease);
138             }
139             return;
140         }
141 
142         if (!future.isSuccess()) {
143             // Failed. Try again with the delay given by Backoff.
144             final long delayMillis = backoff.nextDelayMillis(attemptsSoFar);
145             logger.warn("{} DNS query failed; retrying in {} ms (attempts so far: {}):",
146                         logPrefix, delayMillis, attemptsSoFar, future.cause());
147             scheduledFuture = eventLoop.schedule(() -> sendQueries(questions),
148                                                  delayMillis, TimeUnit.MILLISECONDS);
149             return;
150         }
151 
152         // Reset the counter so that Backoff is reset.
153         attemptsSoFar = 0;
154 
155         @SuppressWarnings("unchecked")
156         final List<DnsRecord> records = (List<DnsRecord>) future.getNow();
157         final long serverTtl = records.stream().mapToLong(DnsRecord::timeToLive).min().orElse(minTtl);
158         final int effectiveTtl = (int) Math.max(Math.min(serverTtl, maxTtl), minTtl);
159 
160         try {
161             setEndpoints(onDnsRecords(records, effectiveTtl));
162         } catch (Throwable t) {
163             logger.warn("{} Failed to process the DNS query result: {}", logPrefix, records, t);
164         } finally {
165             records.forEach(ReferenceCountUtil::safeRelease);
166             scheduledFuture = eventLoop.schedule(() -> sendQueries(questions), effectiveTtl, TimeUnit.SECONDS);
167         }
168     }
169 
170     /**
171      * Invoked when DNS records were retrieved from a DNS server. Implement this method to transform
172      * {@link DnsRecord}s into {@link Endpoint}s.
173      */
174     abstract ImmutableSortedSet<Endpoint> onDnsRecords(List<DnsRecord> records, int ttl) throws Exception;
175 
176     /**
177      * Stops polling DNS servers for service updates.
178      */
179     @Override
180     protected final void doCloseAsync(CompletableFuture<?> future) {
181         final ScheduledFuture<?> scheduledFuture = this.scheduledFuture;
182         if (scheduledFuture != null) {
183             scheduledFuture.cancel(true);
184         }
185         future.complete(null);
186     }
187 
188     /**
189      * Logs a warning message about an invalid record.
190      */
191     final void warnInvalidRecord(DnsRecordType type, ByteBuf content) {
192         DnsUtil.warnInvalidRecord(logger(), logPrefix(), type, content);
193     }
194 }
