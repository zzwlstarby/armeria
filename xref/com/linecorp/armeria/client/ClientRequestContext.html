59      /**
60       * Returns the client-side context of the {@link Request} that is being handled in the current thread.
61       *
62       * @throws IllegalStateException if the context is unavailable in the current thread or
63       *                               the current context is not a {@link ClientRequestContext}.
64       */
65      static ClientRequestContext current() {
66          final RequestContext ctx = RequestContext.current();
67          checkState(ctx instanceof ClientRequestContext,
68                     "The current context is not a client-side context: %s", ctx);
69          return (ClientRequestContext) ctx;
70      }

72      /**
73       * Returns the client-side context of the {@link Request} that is being handled in the current thread.
74       *
75       * @return the {@link ClientRequestContext} available in the current thread, or {@code null} if unavailable.
76       * @throws IllegalStateException if the current context is not a {@link ClientRequestContext}.
77       */
78      @Nullable
79      static ClientRequestContext currentOrNull() {
80          final RequestContext ctx = RequestContext.currentOrNull();
81          if (ctx == null) {
82              return null;
83          }
84          checkState(ctx instanceof ClientRequestContext,
85                     "The current context is not a client-side context: %s", ctx);
86          return (ClientRequestContext) ctx;
87      }

89      /**
90       * Maps the client-side context of the {@link Request} that is being handled in the current thread.
91       *
92       * @param mapper the {@link Function} that maps the {@link ClientRequestContext}
93       * @param defaultValueSupplier the {@link Supplier} that provides the value when the context is unavailable
94       *                             in the current thread. If {@code null}, the {@code null} will be returned
95       *                             when the context is unavailable in the current thread.
96       * @throws IllegalStateException if the current context is not a {@link ClientRequestContext}.
97       */
98      @Nullable
99      static <T> T mapCurrent(
100             Function<? super ClientRequestContext, T> mapper, @Nullable Supplier<T> defaultValueSupplier) {
101 
102         final ClientRequestContext ctx = currentOrNull();
103         if (ctx != null) {
104             return mapper.apply(ctx);
105         }
106 
107         if (defaultValueSupplier != null) {
108             return defaultValueSupplier.get();
109         }
110 
111         return null;
112     }

218     /**
219      * Returns the {@link Iterator} of all {@link Entry}s this context contains.
220      *
221      * <p>The {@link Iterator} returned by this method will also yield the {@link Entry}s from the
222      * {@link #root()} except those whose {@link AttributeKey} exist already in this context, e.g.
223      * <pre>{@code
224      * ClientRequestContext ctx = ...;
225      * assert ctx.ownAttr(KEY_A).equals("child_a");
226      * assert ctx.root().attr(KEY_A).equals("root_a");
227      * assert ctx.root().attr(KEY_B).equals("root_b");
228      *
229      * Iterator<Entry<AttributeKey<?>, Object>> attrs = ctx.attrs();
230      * assert attrs.next().getValue().equals("child_a"); // KEY_A
231      * // Skip KEY_A in the root.
232      * assert attrs.next().getValue().equals("root_b"); // KEY_B
233      * assert attrs.hasNext() == false;
234      * }</pre>
235      * Please note that any changes made to the {@link Entry} returned by {@link Iterator#next()} never
236      * affects the {@link Entry} owned by {@link #root()}. For example:
237      * <pre>{@code
238      * ClientRequestContext ctx = ...;
239      * assert ctx.root().attr(KEY).equals("root");
240      * assert ctx.ownAttr(KEY) == null;
241      *
242      * Iterator<Entry<AttributeKey<?>, Object>> attrs = ctx.attrs();
243      * Entry<AttributeKey<?>, Object> next = attrs.next();
244      * assert next.getKey() == KEY;
245      * // Overriding the root entry creates the client context's own entry.
246      * next.setValue("child");
247      * assert ctx.attr(KEY).equals("child");
248      * assert ctx.ownAttr(KEY).equals("child");
249      * // root attribute remains unaffected.
250      * assert ctx.root().attr(KEY).equals("root");
251      * }</pre>
252      * If you want to change the value from the root while iterating, please call
253      * {@link #attrs()} from {@link #root()}.
254      * <pre>{@code
255      * ClientRequestContext ctx = ...;
256      * assert ctx.root().attr(KEY).equals("root");
257      * assert ctx.ownAttr(KEY) == null;
258      *
259      * // Call attrs() from the root to set a value directly while iterating.
260      * Iterator<Entry<AttributeKey<?>, Object>> attrs = ctx.root().attrs();
261      * Entry<AttributeKey<?>, Object> next = attrs.next();
262      * assert next.getKey() == KEY;
263      * next.setValue("another_root");
264      * // The ctx does not have its own attribute.
265      * assert ctx.ownAttr(KEY) == null;
266      * assert ctx.attr(KEY).equals("another_root");
267      * }</pre>
268      *
269      * @see #ownAttrs()
270      */
271     @Override
272     Iterator<Entry<AttributeKey<?>, Object>> attrs();

298     /**
299      * Pushes this context to the thread-local stack. To pop the context from the stack, call
300      * {@link SafeCloseable#close()}, which can be done using a {@code try-with-resources} block:
301      * <pre>{@code
302      * try (SafeCloseable ignored = ctx.push()) {
303      *     ...
304      * }
305      * }</pre>
306      *
307      * <p>In order to call this method, the current thread-local state must meet one of the
308      * following conditions:
309      * <ul>
310      *   <li>the thread-local does not have any {@link RequestContext} in it</li>
311      *   <li>the thread-local has the same {@link ClientRequestContext} as this - reentrance</li>
312      *   <li>the thread-local has the {@link ServiceRequestContext} which is the same as {@link #root()}</li>
313      *   <li>the thread-local has the {@link ClientRequestContext} whose {@link #root()}
314      *       is the same {@link #root()}</li>
315      *   <li>the thread-local has the {@link ClientRequestContext} whose {@link #root()} is {@code null}
316      *       and this {@link #root()} is {@code null}</li>
317      * </ul>
318      * Otherwise, this method will throw an {@link IllegalStateException}.
319      */
320     @Override
321     default SafeCloseable push() {
322         final RequestContext oldCtx = RequestContextThreadLocal.getAndSet(this);
323         if (oldCtx == this) {
324             // Reentrance
325             return noopSafeCloseable();
326         }
327 
328         if (oldCtx == null) {
329             return RequestContextThreadLocal::remove;
330         }
331 
332         final ServiceRequestContext root = root();
333         if ((oldCtx instanceof ServiceRequestContext && oldCtx == root) ||
334             oldCtx instanceof ClientRequestContext && ((ClientRequestContext) oldCtx).root() == root) {
335             return () -> RequestContextThreadLocal.set(oldCtx);
336         }
337 
338         // Put the oldCtx back before throwing an exception.
339         RequestContextThreadLocal.set(oldCtx);
340         throw newIllegalContextPushingException(this, oldCtx);
341     }
