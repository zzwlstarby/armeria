61  /**
62   * Builds a new {@link ClientFactory}.
63   *
64   * <h2>Example</h2>
65   * <pre>{@code
66   * final ClientFactory factory =
67   *         ClientFactory.builder()
68   *                      // Set the connection timeout to 5 seconds.
69   *                      .connectTimeoutMillis(5000)
70   *                      // Set the socket send buffer to 1 MiB.
71   *                      .socketOption(ChannelOption.SO_SNDBUF, 1048576)
72   *                      // Disable certificate verification; never do this in production!
73   *                      .tlsNoVerify()
74   *                      .build();
75   * }</pre>
76   */
77  public final class ClientFactoryBuilder {
78  
79      private final Map<ClientFactoryOption<?>, ClientFactoryOptionValue<?>> options = new LinkedHashMap<>();

150     private void validateMaxNumEventLoopsPerEndpoint(int maxNumEventLoopsPerEndpoint) {
151         checkArgument(maxNumEventLoopsPerEndpoint > 0,
152                       "maxNumEventLoopsPerEndpoint: %s (expected: > 0)", maxNumEventLoopsPerEndpoint);
153         checkState(!options.containsKey(ClientFactoryOption.EVENT_LOOP_SCHEDULER_FACTORY),
154                    "maxNumEventLoopsPerEndpoint() and eventLoopSchedulerFactory() are mutually exclusive.");
155     }

157     /**
158      * Sets the {@link ToIntFunction} which takes an {@link Endpoint} and produces the maximum number of
159      * {@link EventLoop}s which will be used to handle connections to the specified {@link Endpoint}.
160      * The function should return {@code 0} or a negative value for the {@link Endpoint}s which it
161      * doesn't want to handle. For example: <pre>{@code
162      * ToIntFunction<Endpoint> function = endpoint -> {
163      *     if (endpoint.equals(Endpoint.of("foo.com"))) {
164      *         return 5;
165      *     }
166      *     if (endpoint.host().contains("bar.com")) {
167      *         return Integer.MAX_VALUE; // The value will be clamped at the number of event loops.
168      *     }
169      *     return -1; // Should return 0 or a negative value to use the default value.
170      * }
171      * }</pre>
172      */
173     public ClientFactoryBuilder maxNumEventLoopsFunction(ToIntFunction<Endpoint> maxNumEventLoopsFunction) {
174         checkState(!options.containsKey(ClientFactoryOption.EVENT_LOOP_SCHEDULER_FACTORY),
175                    "maxNumEventLoopsPerEndpoint() and eventLoopSchedulerFactory() are mutually exclusive.");
176         maxNumEventLoopsFunctions.add(requireNonNull(maxNumEventLoopsFunction, "maxNumEventLoopsFunction"));
177         return this;
178     }

195                       "connectTimeoutMillis: %s (expected: > 0)", connectTimeoutMillis);

210     /**
211      * Sets the options of sockets created by the {@link ClientFactory}.
212      */
213     public <T> ClientFactoryBuilder channelOption(ChannelOption<T> option, T value) {
214         requireNonNull(option, "option");
215         requireNonNull(value, "value");
216         channelOptions(ImmutableMap.of(option, value));
217         return this;
218     }

220     private void channelOptions(Map<ChannelOption<?>, Object> newChannelOptions) {
221         @SuppressWarnings("unchecked")
222         final ClientFactoryOptionValue<Map<ChannelOption<?>, Object>> castOptions =
223                 (ClientFactoryOptionValue<Map<ChannelOption<?>, Object>>) options.get(
224                         ClientFactoryOption.CHANNEL_OPTIONS);
225         if (castOptions == null) {
226             options.put(ClientFactoryOption.CHANNEL_OPTIONS,
227                         ClientFactoryOption.CHANNEL_OPTIONS.newValue(ImmutableMap.copyOf(newChannelOptions)));
228         } else {
229             final ImmutableMap.Builder<ChannelOption<?>, Object> builder =
230                     ImmutableMap.builderWithExpectedSize(newChannelOptions.size());
231             castOptions.value().forEach((channelOption, value) -> {
232                 if (!newChannelOptions.containsKey(channelOption)) {
233                     builder.put(channelOption, value);
234                 }
235             });
236             builder.putAll(newChannelOptions);
237 
238             options.put(ClientFactoryOption.CHANNEL_OPTIONS,
239                         ClientFactoryOption.CHANNEL_OPTIONS.newValue(builder.build()));
240         }
241     }

258     /**
259      * Disables the verification of server's key certificate chain. This method is a shortcut for:
260      * {@code tlsCustomizer(b -> b.trustManager(InsecureTrustManagerFactory.INSTANCE))}.
261      * <strong>Note:</strong> You should never use this in production but only for a testing purpose.
262      *
263      * @see InsecureTrustManagerFactory
264      * @see #tlsCustomizer(Consumer)

267         tlsCustomizer(b -> b.trustManager(InsecureTrustManagerFactory.INSTANCE));
268         return this;

271     /**
272      * Adds the {@link Consumer} which can arbitrarily configure the {@link SslContextBuilder} that will be
273      * applied to the SSL session. For example, use {@link SslContextBuilder#trustManager(TrustManagerFactory)}
274      * to configure a custom server CA or {@link SslContextBuilder#keyManager(KeyManagerFactory)} to configure
275      * a client certificate for SSL authorization.
276      */
277     public ClientFactoryBuilder tlsCustomizer(Consumer<? super SslContextBuilder> tlsCustomizer) {
278         requireNonNull(tlsCustomizer, "tlsCustomizer");
279         @SuppressWarnings("unchecked")
280         final ClientFactoryOptionValue<Consumer<? super SslContextBuilder>> oldTlsCustomizerValue =
281                 (ClientFactoryOptionValue<Consumer<? super SslContextBuilder>>)
282                         options.get(ClientFactoryOption.TLS_CUSTOMIZER);
283 
284         final Consumer<? super SslContextBuilder> oldTlsCustomizer =
285                 oldTlsCustomizerValue == null ? ClientFactoryOption.TLS_CUSTOMIZER.defaultValue()
286                                               : oldTlsCustomizerValue.value();
287         if (oldTlsCustomizer == ClientFactoryOption.TLS_CUSTOMIZER.defaultValue()) {
288             option(ClientFactoryOption.TLS_CUSTOMIZER, tlsCustomizer);
289         } else {
290             option(ClientFactoryOption.TLS_CUSTOMIZER, b -> {
291                 oldTlsCustomizer.accept(b);
292                 tlsCustomizer.accept(b);
293             });
294         }
295         return this;
296     }

491     /**
492      * Adds the specified {@link ClientFactoryOptionValue}.
493      */
494     public <T> ClientFactoryBuilder option(ClientFactoryOptionValue<T> optionValue) {
495         requireNonNull(optionValue, "optionValue");
496         if (ClientFactoryOption.CHANNEL_OPTIONS == optionValue.option()) {
497             @SuppressWarnings("unchecked")
498             final Map<ChannelOption<?>, Object> channelOptions =
499                     (Map<ChannelOption<?>, Object>) optionValue.value();
500             channelOptions(channelOptions);
501         } else {
502             options.put(optionValue.option(), optionValue);
503         }
504         return this;
505     }
506 
507     /**
508      * Adds the specified {@link ClientFactoryOptions}.
509      */
510     public ClientFactoryBuilder options(ClientFactoryOptions options) {
511         requireNonNull(options, "options");
512         options.forEach(this::option);
513         return this;
514     }

516     private ClientFactoryOptions buildOptions() {
517         options.computeIfAbsent(ClientFactoryOption.EVENT_LOOP_SCHEDULER_FACTORY, k -> {
518             final Function<? super EventLoopGroup, ? extends EventLoopScheduler> eventLoopSchedulerFactory =
519                     eventLoopGroup -> new DefaultEventLoopScheduler(
520                             eventLoopGroup, maxNumEventLoopsPerEndpoint, maxNumEventLoopsPerHttp1Endpoint,
521                             maxNumEventLoopsFunctions);
522             return ClientFactoryOption.EVENT_LOOP_SCHEDULER_FACTORY.newValue(eventLoopSchedulerFactory);
523         });
524 
525         options.computeIfAbsent(ClientFactoryOption.ADDRESS_RESOLVER_GROUP_FACTORY, k -> {
526             final Function<? super EventLoopGroup,
527                     ? extends AddressResolverGroup<? extends InetSocketAddress>> addressResolverGroupFactory =
528                     eventLoopGroup -> {
529                         // FIXME(ikhoon): Remove DefaultAddressResolverGroup registration after fixing Window
530                         //                domain name resolution failure.
531                         //                https://github.com/line/armeria/issues/2243
532                         if (Flags.useJdkDnsResolver() && dnsResolverGroupCustomizers == null) {
533                             return DefaultAddressResolverGroup.INSTANCE;
534                         }
535                         final DnsResolverGroupBuilder builder = new DnsResolverGroupBuilder();
536                         if (dnsResolverGroupCustomizers != null) {
537                             dnsResolverGroupCustomizers.forEach(consumer -> consumer.accept(builder));
538                         }
539                         return builder.build(eventLoopGroup);
540                     };
541             return ClientFactoryOption.ADDRESS_RESOLVER_GROUP_FACTORY.newValue(addressResolverGroupFactory);
542         });
543 
544         return ClientFactoryOptions.of(options.values());
545     }
546 
547     /**
548      * Returns a newly-created {@link ClientFactory} based on the properties of this builder.
549      */
550     public ClientFactory build() {
551         return new DefaultClientFactory(new HttpClientFactory(buildOptions()));
552     }
553 
554     @Override
555     public String toString() {
556         final ToStringHelper helper = MoreObjects.toStringHelper(this).omitNullValues();
557         helper.add("options", options);
558 
559         if (maxNumEventLoopsPerHttp1Endpoint > 0) {
560             helper.add("maxNumEventLoopsPerHttp1Endpoint", maxNumEventLoopsPerHttp1Endpoint);
561         }
562         if (maxNumEventLoopsPerEndpoint > 0) {
563             helper.add("maxNumEventLoopsPerEndpoint", maxNumEventLoopsPerEndpoint);
564         }
565         if (!maxNumEventLoopsFunctions.isEmpty()) {
566             helper.add("maxNumEventLoopsFunctions", maxNumEventLoopsFunctions);
567         }
568 
569         return helper.toString();
570     }
