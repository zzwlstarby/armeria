54  /**
55   * A {@link UnaryGrpcClient} can be used to make requests to a gRPC server without depending on gRPC stubs.
56   * This client takes care of deframing and framing with the gRPC wire format and handling appropriate headers.
57   *
58   * <p>This client does not support compression. If you need support for compression, please consider using
59   * normal gRPC stubs or file a feature request.
60   */
61  @UnstableApi
62  public final class UnaryGrpcClient {
63  
64      private final WebClient webClient;
65  
66      /**
67       * Constructs a {@link UnaryGrpcClient} for the given {@link WebClient}.
68       */
69      // TODO(anuraaga): We would ideally use our standard client building pattern, i.e.,
70      // Clients.builder(...).build(UnaryGrpcClient.class), but that requires mapping protocol schemes to media
71      // types, which cannot be duplicated. As this and normal gproto+ clients must use the same media type, we
72      // cannot currently implement this without rethinking / refactoring core and punt for now since this is an
73      // advanced API.
74      public UnaryGrpcClient(WebClient webClient) {
75          this.webClient = Clients.newDerivedClient(
76                  webClient,
77                  ClientOption.DECORATION.newValue(
78                          ClientDecoration.of(GrpcFramingDecorator::new)
79                  ));
80      }
81  
82      /**
83       * Executes a unary gRPC client request. The given {@code payload} will be framed and sent to the path at
84       * {@code uri}. {@code uri} should be the method's URI, which is always of the format
85       * {@code /:package-name.:service-name/:method}. For example, for the proto package
86       * {@code armeria.protocol}, the service name {@code CoolService} and the method name
87       * {@code RunWithoutStubs}, the {@code uri} would be {@code /armeria.protocol.CoolService/RunWithoutStubs}.
88       * If you aren't sure what the package, service name, and method name are for your method, you should
89       * probably use normal gRPC stubs instead of this class.
90       */
91      public CompletableFuture<byte[]> execute(String uri, byte[] payload) {
92          final HttpRequest request = HttpRequest.of(
93                  RequestHeaders.of(HttpMethod.POST, uri,
94                                    HttpHeaderNames.CONTENT_TYPE, "application/grpc+proto",
95                                    HttpHeaderNames.TE, HttpHeaderValues.TRAILERS),
96                  HttpData.wrap(payload));
97          return webClient.execute(request).aggregate()
98                          .thenApply(msg -> {
99                              if (!HttpStatus.OK.equals(msg.status())) {
100                                 throw new ArmeriaStatusException(
101                                         StatusCodes.INTERNAL,
102                                         "Non-successful HTTP response code: " + msg.status());
103                             }
104 
105                             // Status can either be in the headers or trailers depending on error
106                             String grpcStatus = msg.headers().get(GrpcHeaderNames.GRPC_STATUS);
107                             if (grpcStatus != null) {
108                                 checkGrpcStatus(grpcStatus, msg.headers());
109                             } else {
110                                 grpcStatus = msg.trailers().get(GrpcHeaderNames.GRPC_STATUS);
111                                 checkGrpcStatus(grpcStatus, msg.trailers());
112                             }
113 
114                             return msg.content().array();
115                         });
116     }
117 
118     private static void checkGrpcStatus(@Nullable String grpcStatus, HttpHeaders headers) {
119         if (grpcStatus != null && !"0".equals(grpcStatus)) {
120             String grpcMessage = headers.get(GrpcHeaderNames.GRPC_MESSAGE);
121             if (grpcMessage != null) {
122                 grpcMessage = StatusMessageEscaper.unescape(grpcMessage);
123             }
124             throw new ArmeriaStatusException(
125                     Integer.parseInt(grpcStatus),
126                     grpcMessage);
127         }
128     }
129 
130     private static final class GrpcFramingDecorator extends SimpleDecoratingHttpClient {
131 
132         private GrpcFramingDecorator(HttpClient delegate) {
133             super(delegate);
134         }
135 
136         @Override
137         public HttpResponse execute(ClientRequestContext ctx, HttpRequest req) {
138             return HttpResponse.from(
139                     req.aggregateWithPooledObjects(ctx.eventLoop(), ctx.alloc())
140                        .thenCompose(
141                                msg -> {
142                                    final ByteBuf buf;
143                                    if (msg.content() instanceof ByteBufHolder) {
144                                        buf = ((ByteBufHolder) msg.content()).content();
145                                    } else {
146                                        buf = Unpooled.wrappedBuffer(msg.content().array());
147                                    }
148                                    final HttpData framed;
149                                    try (ArmeriaMessageFramer framer = new ArmeriaMessageFramer(
150                                            ctx.alloc(), Integer.MAX_VALUE)) {
151                                        framed = framer.writePayload(buf);
152                                    }
153 
154                                    try {
155                                        return delegate().execute(ctx, HttpRequest.of(req.headers(), framed))
156                                                         .aggregateWithPooledObjects(ctx.eventLoop(),
157                                                                                     ctx.alloc());
158                                    } catch (Exception e) {
159                                        throw new ArmeriaStatusException(StatusCodes.INTERNAL,
160                                                                         "Error executing request.");
161                                    }
162                                })
163                        .thenCompose(msg -> {
164                            if (!msg.status().equals(HttpStatus.OK) || msg.content().isEmpty()) {
165                                // Nothing to deframe.
166                                return CompletableFuture.completedFuture(msg.toHttpResponse());
167                            }
168 
169                            final CompletableFuture<HttpResponse> responseFuture = new CompletableFuture<>();
170 
171                            try (ArmeriaMessageDeframer deframer = new ArmeriaMessageDeframer(new Listener() {
172                                @Override
173                                public void messageRead(DeframedMessage unframed) {
174                                    final ByteBuf buf = unframed.buf();
175                                    // Compression not supported.
176                                    assert buf != null;
177                                    responseFuture.complete(HttpResponse.of(
178                                            msg.headers(),
179                                            new ByteBufHttpData(buf, true),
180                                            msg.trailers()));
181                                }
182 
183                                @Override
184                                public void endOfStream() {
185                                    if (!responseFuture.isDone()) {
186                                        responseFuture.complete(HttpResponse.of(msg.headers(),
187                                                                                HttpData.empty(),
188                                                                                msg.trailers()));
189                                    }
190                                }
191                            }, Integer.MAX_VALUE, ctx.alloc())) {
192                                deframer.request(1);
193                                deframer.deframe(msg.content(), true);
194                            }
195                            return responseFuture;
196                        }), ctx.eventLoop());
197         }
198     }
