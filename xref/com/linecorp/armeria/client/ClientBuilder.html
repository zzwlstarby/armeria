35  /**
36   * Creates a new client that connects to the specified {@link URI} using the builder pattern. Use the factory
37   * methods in {@link Clients} if you do not have many options to override. If you are creating an
38   * {@link WebClient}, it is recommended to use the {@link WebClientBuilder} or
39   * factory methods in {@link WebClient}.
40   *
41   * <h3>How are decorators and HTTP headers configured?</h3>
42   *
43   * <p>Unlike other options, when a user calls {@link #option(ClientOption, Object)} or {@code options()} with
44   * a {@link ClientOption#DECORATION} or a {@link ClientOption#HTTP_HEADERS}, this builder will not simply
45   * replace the old option but <em>merge</em> the specified option into the previous option value. For example:
46   * <pre>{@code
47   * ClientOptionsBuilder b = ClientOptions.builder();
48   * b.option(ClientOption.HTTP_HEADERS, headersA);
49   * b.option(ClientOption.HTTP_HEADERS, headersB);
50   * b.option(ClientOption.DECORATION, decorationA);
51   * b.option(ClientOption.DECORATION, decorationB);
52   *
53   * ClientOptions opts = b.build();
54   * HttpHeaders httpHeaders = opts.httpHeaders();
55   * ClientDecoration decorations = opts.decoration();
56   * }</pre>
57   * {@code httpHeaders} will contain all HTTP headers of {@code headersA} and {@code headersB}.
58   * If {@code headersA} and {@code headersB} have the headers with the same name, the duplicate header in
59   * {@code headerB} will replace the one with the same name in {@code headerA}.
60   * Similarly, {@code decorations} will contain all decorators of {@code decorationA} and {@code decorationB},
61   * but there will be no replacement but only addition.
62   */

131     ClientBuilder(@Nullable URI uri, @Nullable Scheme scheme, @Nullable SessionProtocol protocol,
132                   @Nullable EndpointGroup endpointGroup) {
133         this.uri = uri;
134         this.scheme = scheme;
135         this.protocol = protocol;
136         this.endpointGroup = endpointGroup;
137     }
138 
139     /**
140      * Sets the {@code path} of the client.
141      */
142     public ClientBuilder path(String path) {
143         ensureEndpointGroup();
144         requireNonNull(path, "path");
145         checkArgument(path.startsWith("/"), "path: %s (expected: an absolute path starting with '/')", path);
146         this.path = path;
147         return this;
148     }
149 
150     /**
151      * Sets the {@link SerializationFormat} of the client. The default is {@link SerializationFormat#NONE}.
152      */
153     public ClientBuilder serializationFormat(SerializationFormat format) {
154         ensureEndpointGroup();
155         if (scheme != null) {
156             throw new IllegalStateException("scheme is already given");
157         }
158 
159         this.format = requireNonNull(format, "format");
160         return this;
161     }
162 
163     /**
164      * Returns a newly-created client which implements the specified {@code clientType}, based on the
165      * properties of this builder.
166      *
167      * @throws IllegalArgumentException if the scheme of the {@code uri} specified in
168      *                                  {@link Clients#builder(String)} or the specified {@code clientType} is
169      *                                  unsupported for the scheme
170      */
171     public <T> T build(Class<T> clientType) {
172         requireNonNull(clientType, "clientType");
173 
174         final Object client;
175         final ClientOptions options = buildOptions();
176         final ClientFactory factory = options.factory();
177         if (uri != null) {
178             client = factory.newClient(ClientBuilderParams.of(uri, clientType, options));
179         } else {
180             assert endpointGroup != null;
181             client = factory.newClient(ClientBuilderParams.of(scheme(), endpointGroup,
182                                                               path, clientType, options));
183         }
184 
185         @SuppressWarnings("unchecked")
186         final T cast = (T) client;
187         return cast;
188     }
189 
190     private Scheme scheme() {
191         return scheme == null ? Scheme.of(format, protocol) : scheme;
192     }
193 
194     private void ensureEndpointGroup() {
195         if (endpointGroup == null) {
196             throw new IllegalStateException(
197                     getClass().getSimpleName() + " must be created with an " +
198                     EndpointGroup.class.getSimpleName() + " to call this method.");
199         }
200     }
