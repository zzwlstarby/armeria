68      private final Retrofit.Builder retrofitBuilder;
69      private final String baseWebClientHost;
70      private final int baseWebClientPort;
71      private final WebClient webClient;
72      private boolean streaming;
73      private Executor callbackExecutor = CommonPools.blockingTaskExecutor();
74      @Nullable
75      private BiFunction<? super SessionProtocol, ? super Endpoint, ? extends WebClient> nonBaseClientFactory;
76  
77      ArmeriaRetrofitBuilder(WebClient webClient) {
78          this.webClient = webClient;
79          final URI uri = webClient.uri();
80          final SessionProtocol protocol = webClient.scheme().sessionProtocol();
81  
82          // Build a baseUrl that will pass Retrofit's validation.
83          final HttpUrl baseUrl = HttpUrl.get((protocol.isTls() ? "https" : "http") +
84                                              uri.toString().substring(protocol.uriText().length()));
85  
86          retrofitBuilder = new Retrofit.Builder().baseUrl(baseUrl);
87          baseWebClientHost = baseUrl.host();
88          baseWebClientPort = baseUrl.port();
89      }
90  
91      /**
92       * Specifies the {@link BiFunction} that creates a new non-base {@link WebClient}, which is used for
93       * sending requests to other authorities than that of base URL. If not specified, the non-base
94       * {@link WebClient} will have the same options with the base {@link WebClient}, which was specified
95       * with {@link ArmeriaRetrofit#of(WebClient)} or {@link ArmeriaRetrofit#builder(WebClient)}.
96       *
97       * <p>To avoid the overhead of repetitive instantiation of {@link WebClient}s, the {@link WebClient}s
98       * returned by the specified {@link BiFunction} will be cached for each combination of:
99       * <ul>
100      *   <li>Whether the connection is secured (HTTPS or HTTPS)</li>
101      *   <li>Host name</li>
102      *   <li>Port number</li>
103      * </ul></p>
104      *
105      * <p>You can use this method to create a customized non-base {@link WebClient}, for example to send
106      * an additional header, override the timeout or enforce HTTP/1:
107      * <pre>{@code
108      * ArmeriaRetrofit.builder("http://example.com/")
109      *                .nonBaseClientFactory((protocol, endpoint) -> {
110      *                    // Enforce HTTP/1.
111      *                    final SessionProtocol actualProtocol =
112      *                            protocol.isTls() ? SessionProtocol.H1 : SessionProtocol.H1C;
113      *
114      *                    return WebClient.builder(actualProtocol, endpoint)
115      *                                    // Derive most settings from 'defaultWebClient'.
116      *                                    .factory(defaultWebClient.factory())
117      *                                    .options(defaultWebClient.options())
118      *                                    // Set a custom header.
119      *                                    .setHttpHeader(HttpHeaderNames.AUTHORIZATION,
120      *                                                   "bearer my-access-token")
121      *                                    // Override the timeout.
122      *                                    .responseTimeout(Duration.ofSeconds(30))
123      *                                    .build();
124      *                })
125      *                .build();
126      * }</pre></p>
127      *
128      * <p>Note that the specified {@link BiFunction} is not used for sending requests to the base URL's
129      * authority. The default {@link WebClient} specified with {@link ArmeriaRetrofit#of(WebClient)} or
130      * {@link ArmeriaRetrofit#builder(WebClient)} will be used instead for such requests:
131      * <pre>{@code
132      * // No need to use 'nonBaseClientFactory()' method.
133      * ArmeriaRetrofit.of(WebClient.builder("http://example.com/")
134      *                             .setHttpHeader(HttpHeaderNames.AUTHORIZATION,
135      *                                            "bearer my-access-token")
136      *                             .build());
137      * }</pre></p>
138      */
139     public ArmeriaRetrofitBuilder nonBaseClientFactory(
140             BiFunction<? super SessionProtocol, ? super Endpoint, ? extends WebClient> nonBaseClientFactory) {
141         this.nonBaseClientFactory = requireNonNull(nonBaseClientFactory, "nonBaseClientFactory");
142         return this;
143     }

178     /**
179      * Sets the streaming flag to make Armeria client fully support {@link Streaming}.
180      * If this flag is {@code false}, Armeria client will buffer all data and call a callback after receiving
181      * the data completely, even if a service method was annotated with {@link Streaming}.
182      * By enabling this flag, Armeria client will use the {@link Executor} specified with
183      * {@link ArmeriaRetrofitBuilder#callbackExecutor(Executor)} to read the data in a blocking way.
184      *
185      * <p>Note: It is not recommended to have the service methods both with and without the
186      * {@link Streaming} annotation in the same interface. Consider separating them into different
187      * interfaces and using different builder to build the service.</p>
188      *
189      * @see Streaming
190      */
191     public ArmeriaRetrofitBuilder streaming(boolean streaming) {
192         this.streaming = streaming;
193         return this;
194     }

207     /**
208      * Returns a newly-created {@link Retrofit} based on the properties of this builder.
209      */
210     public Retrofit build() {
211         final SessionProtocol protocol = webClient.scheme().sessionProtocol();
212 
213         final ClientOptions retrofitOptions = buildOptions(webClient.options());
214         // Re-create the base client without a path, because Retrofit will always provide a full path.
215         final WebClient baseWebClient = WebClient.builder(protocol, webClient.endpointGroup())
216                                                  .options(retrofitOptions)
217                                                  .build();
218 
219         if (nonBaseClientFactory == null) {
220             nonBaseClientFactory = (p, url) -> WebClient.builder(p, Endpoint.of(url.host(), url.port()))
221                                                         .options(retrofitOptions)
222                                                         .build();
223         }
224 
225         retrofitBuilder.callFactory(new ArmeriaCallFactory(
226                 baseWebClientHost, baseWebClientPort, baseWebClient,
227                 streaming ? SubscriberFactory.streaming(callbackExecutor)
228                           : SubscriberFactory.blocking(), new CachedNonBaseClientFactory(nonBaseClientFactory)
229         ));
230 
231         return retrofitBuilder.build();
232     }
233 
234     private static class CachedNonBaseClientFactory
235             implements BiFunction<SessionProtocol, Endpoint, WebClient> {
236 
237         private final BiFunction<SessionProtocol, Endpoint, WebClient> nonBaseClientFactory;
238         private final Cache<Map.Entry<SessionProtocol, Endpoint>, WebClient> cache;
239 
240         @SuppressWarnings("unchecked")
241         CachedNonBaseClientFactory(
242                 BiFunction<? super SessionProtocol, ? super Endpoint,
243                         ? extends WebClient> nonBaseClientFactory) {
244 
245             this.nonBaseClientFactory = (BiFunction<SessionProtocol, Endpoint, WebClient>) nonBaseClientFactory;
246 
247             cache = Caffeine.newBuilder()
248                             .maximumSize(8192) // TODO(trustin): Add a flag if there's demand for it.
249                             .build();
250         }
251 
252         @Override
253         public WebClient apply(SessionProtocol protocol, Endpoint endpoint) {
254             final Map.Entry<SessionProtocol, Endpoint> key = Maps.immutableEntry(protocol, endpoint);
255             final WebClient webClient =
256                     cache.get(key, unused -> nonBaseClientFactory.apply(protocol, endpoint));
257             checkState(webClient != null, "nonBaseClientFactory returned null.");
258             return webClient;
259         }
260 
261         @Override
262         public String toString() {
263             return MoreObjects.toStringHelper(this)
264                               .add("nonBaseClientFactory", nonBaseClientFactory)
265                               .add("cache", cache)
266                               .toString();
267         }
268     }
