17  package com.linecorp.armeria.client;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.net.URI;
22  
23  import javax.annotation.Nullable;
24  
25  import com.linecorp.armeria.client.endpoint.EndpointGroup;
26  import com.linecorp.armeria.common.Scheme;
27  
28  /**
29   * Provides the construction parameters of a client.
30   */
31  public interface ClientBuilderParams {
32  
33      /**
34       * Returns a newly created {@link ClientBuilderParams} from the specified properties.
35       */
36      static ClientBuilderParams of(URI uri, Class<?> type, ClientOptions options) {
37          requireNonNull(uri, "uri");
38          requireNonNull(type, "type");
39          requireNonNull(options, "options");
40          return new DefaultClientBuilderParams(uri, type, options);
41      }
42  
43      /**
44       * Returns a newly created {@link ClientBuilderParams} from the specified properties.
45       */
46      static ClientBuilderParams of(Scheme scheme, EndpointGroup endpointGroup,
47                                    @Nullable String path, Class<?> type, ClientOptions options) {
48          requireNonNull(scheme, "scheme");
49          requireNonNull(endpointGroup, "endpointGroup");
50          requireNonNull(type, "type");
51          requireNonNull(options, "options");
52          return new DefaultClientBuilderParams(scheme, endpointGroup, path, type, options);
53      }
54  
55      /**
56       * Returns the {@link ClientFactory} who created the client.
57       *
58       * @deprecated Use {@link #options()} to get {@link ClientOption#FACTORY}.
59       */
60      @Deprecated
61      default ClientFactory factory() {
62          return options().get(ClientOption.FACTORY);
63      }
64  
65      /**
66       * Returns the {@link Scheme} of the client.
67       */
68      Scheme scheme();
69  
70      /**
71       * Returns the {@link EndpointGroup} of the client.
72       */
73      EndpointGroup endpointGroup();
74  
75      /**
76       * Returns the {@link String} that consists of path, query string and fragment.
77       */
78      String absolutePathRef(); // Name inspired by https://stackoverflow.com/a/47545070/55808
79  
80      /**
81       * Returns the endpoint URI of the client.
82       */
83      URI uri();
84  
85      /**
86       * Returns the type of the client.
87       */
88      Class<?> clientType();
89  
90      /**
91       * Returns the options of the client.
92       */
93      ClientOptions options();
94  }
