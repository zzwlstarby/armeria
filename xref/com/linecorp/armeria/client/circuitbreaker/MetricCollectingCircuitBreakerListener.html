17  package com.linecorp.armeria.client.circuitbreaker;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import com.linecorp.armeria.common.Flags;
22  import com.linecorp.armeria.common.metric.MeterIdPrefix;
23  import com.linecorp.armeria.internal.common.metric.MicrometerUtil;
24  
25  import io.micrometer.core.instrument.Meter;
26  import io.micrometer.core.instrument.MeterRegistry;
27  
28  /**
29   * A {@link CircuitBreakerListener} which exports the status of {@link CircuitBreaker}s to
30   * {@link MeterRegistry}.
31   *
32   * <p>This class will generate the following metrics:
33   *
34   * <table>
35   * <caption>metrics that will be generated by this class</caption>
36   * <tr><th>metric name</th><th>description</th></tr>
37   *
38   * <tr><td>{@code <name>.requests{name="<circuit breaker name>",result="success"}}</td>
39   * <td>The number of successful requests in the counter time window.</td></tr>
40   *
41   * <tr><td>{@code <name>.requests{name="<circuit breaker name>",result="failure"}}</td>
42   * <td>The number of failed requests in the counter time window.</td></tr>
43   *
44   * <tr><td>{@code <name>.transitions{name="<circuit breaker name>",state="CLOSED"}}</td>
45   * <td>The number of circuit breaker state transitions to {@link CircuitState#CLOSED}.</td></tr>
46   *
47   * <tr><td>{@code <name>.transitions{name="<circuit breaker name>",state="OPEN"}}</td>
48   * <td>The number of circuit breaker state transitions to {@link CircuitState#OPEN}.</td></tr>
49   *
50   * <tr><td>{@code <name>.transitions{name="<circuit breaker name>",state="HALF_OPEN"}}</td>
51   * <td>The number of circuit breaker state transitions to {@link CircuitState#HALF_OPEN}.</td></tr>
52   *
53   * <tr><td>{@code <name>.rejectedRequests{name="<circuit breaker name>"}}</td>
54   * <td>The number of requests rejected by the circuit breaker.</td></tr>
55   *
56   * </table>
57   */
58  public final class MetricCollectingCircuitBreakerListener implements CircuitBreakerListener {
59  
60      private static final String DEFAULT_METER_NAME = "armeria.client.circuit.breaker";
61      private static final String LEGACY_METER_NAME = "armeria.client.circuitBreaker";
62  
63      private final MeterRegistry registry;
64      private final String name;
65  
66      /**
67       * Creates a new instance with the default name {@value #DEFAULT_METER_NAME}.
68       */
69      public MetricCollectingCircuitBreakerListener(MeterRegistry registry) {
70          this(registry, Flags.useLegacyMeterNames() ? LEGACY_METER_NAME : DEFAULT_METER_NAME);
71      }
72  
73      /**
74       * Creates a new instance with the specified {@link Meter} name.
75       */
76      public MetricCollectingCircuitBreakerListener(MeterRegistry registry, String name) {
77          this.registry = requireNonNull(registry, "registry");
78          this.name = requireNonNull(name, "name");
79      }
80  
81      @Override
82      public void onStateChanged(String circuitBreakerName, CircuitState state) {
83          metricsOf(circuitBreakerName).onStateChanged(state);
84      }
85  
86      @Override
87      public void onEventCountUpdated(String circuitBreakerName, EventCount eventCount) {
88          metricsOf(circuitBreakerName).onCountUpdated(eventCount);
89      }
90  
91      @Override
92      public void onRequestRejected(String circuitBreakerName) {
93          metricsOf(circuitBreakerName).onRequestRejected();
94      }
95  
96      private CircuitBreakerMetrics metricsOf(String circuitBreakerName) {
97          final MeterIdPrefix idPrefix = new MeterIdPrefix(name, "name", circuitBreakerName);
98          return MicrometerUtil.register(registry, idPrefix,
99                                         CircuitBreakerMetrics.class,
100                                        CircuitBreakerMetrics::new);
101     }
102 }
