17  package com.linecorp.armeria.client;
18  
19  import com.linecorp.armeria.internal.common.AbstractHttp2ConnectionHandler;
20  
21  import io.netty.channel.Channel;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.handler.codec.http2.Http2ConnectionDecoder;
24  import io.netty.handler.codec.http2.Http2ConnectionEncoder;
25  import io.netty.handler.codec.http2.Http2Settings;
26  
27  final class Http2ClientConnectionHandler extends AbstractHttp2ConnectionHandler {
28  
29      private final HttpClientFactory clientFactory;
30      private final Http2ResponseDecoder responseDecoder;
31  
32      Http2ClientConnectionHandler(Http2ConnectionDecoder decoder, Http2ConnectionEncoder encoder,
33                                   Http2Settings initialSettings, Channel channel,
34                                   HttpClientFactory clientFactory) {
35  
36          super(decoder, encoder, initialSettings);
37  
38          this.clientFactory = clientFactory;
39          responseDecoder = new Http2ResponseDecoder(channel, encoder(), clientFactory);
40          connection().addListener(responseDecoder);
41          decoder().frameListener(responseDecoder);
42  
43          // Setup post build options
44          final long timeout = clientFactory.idleTimeoutMillis();
45          if (timeout > 0) {
46              gracefulShutdownTimeoutMillis(timeout);
47          } else {
48              // Timeout disabled
49              gracefulShutdownTimeoutMillis(-1);
50          }
51      }
52  
53      Http2ResponseDecoder responseDecoder() {
54          return responseDecoder;
55      }
56  
57      @Override
58      public void channelActive(ChannelHandlerContext ctx) throws Exception {
59          super.channelActive(ctx);
60  
61          // NB: Http2ConnectionHandler does not flush the preface string automatically.
62          ctx.flush();
63      }
64  
65      @Override
66      protected boolean needsImmediateDisconnection() {
67          return clientFactory.isClosing() || responseDecoder.goAwayHandler().receivedErrorGoAway();
68      }
69  }
