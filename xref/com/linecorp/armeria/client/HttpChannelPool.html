78      HttpChannelPool(HttpClientFactory clientFactory, EventLoop eventLoop,
79                      SslContext sslCtxHttp1Or2, SslContext sslCtxHttp1Only,
80                      ConnectionPoolListener listener) {
81          this.eventLoop = eventLoop;
82          pool = newEnumMap(
83                  Map.class,
84                  unused -> new HashMap<>(),
85                  SessionProtocol.H1, SessionProtocol.H1C,
86                  SessionProtocol.H2, SessionProtocol.H2C);
87          pendingAcquisitions = newEnumMap(
88                  Map.class,
89                  unused -> new HashMap<>(),
90                  SessionProtocol.HTTP, SessionProtocol.HTTPS,
91                  SessionProtocol.H1, SessionProtocol.H1C,
92                  SessionProtocol.H2, SessionProtocol.H2C);
93          allChannels = new IdentityHashMap<>();
94          this.listener = listener;
95  
96          final Bootstrap baseBootstrap = clientFactory.newBootstrap();
97          baseBootstrap.group(eventLoop);
98          bootstraps = newEnumMap(
99                  Bootstrap.class,
100                 desiredProtocol -> {
101                     final SslContext sslCtx = desiredProtocol == SessionProtocol.H1 ||
102                                               desiredProtocol == SessionProtocol.H1C ? sslCtxHttp1Only
103                                                                                      : sslCtxHttp1Or2;
104                     final Bootstrap bootstrap = baseBootstrap.clone();
105                     bootstrap.handler(new ChannelInitializer<Channel>() {
106                         @Override
107                         protected void initChannel(Channel ch) throws Exception {
108                             ch.pipeline().addLast(
109                                     new HttpClientPipelineConfigurator(clientFactory, desiredProtocol, sslCtx));
110                         }
111                     });
112                     return bootstrap;
113                 },
114                 SessionProtocol.HTTP, SessionProtocol.HTTPS,
115                 SessionProtocol.H1, SessionProtocol.H1C,
116                 SessionProtocol.H2, SessionProtocol.H2C);
117         connectTimeoutMillis = (Integer) baseBootstrap.config().options()
118                                                       .get(ChannelOption.CONNECT_TIMEOUT_MILLIS);
119     }

121     /**
122      * Returns an array whose index signifies {@link SessionProtocol#ordinal()}. Similar to {@link EnumMap}.
123      */
124     private static <T> T[] newEnumMap(Class<?> elementType,
125                                       Function<SessionProtocol, T> factory,
126                                       SessionProtocol... allowedProtocols) {
127         @SuppressWarnings("unchecked")
128         final T[] maps = (T[]) Array.newInstance(elementType, SessionProtocol.values().length);
129         // Attempting to access the array with an unallowed protocol will trigger NPE,
130         // which will help us find a bug.
131         for (SessionProtocol p : allowedProtocols) {
132             maps[p.ordinal()] = factory.apply(p);
133         }
134         return maps;
135     }

165     /**
166      * Attempts to acquire a {@link Channel} which is matched by the specified condition immediately.
167      *
168      * @return {@code null} is there's no match left in the pool and thus a new connection has to be
169      *         requested via {@link #acquireLater(SessionProtocol, PoolKey, ClientConnectionTimingsBuilder)}.
170      */
171     @Nullable
172     PooledChannel acquireNow(SessionProtocol desiredProtocol, PoolKey key) {
173         PooledChannel ch;
174         switch (desiredProtocol) {
175             case HTTP:
176                 ch = acquireNowExact(key, SessionProtocol.H2C);
177                 if (ch == null) {
178                     ch = acquireNowExact(key, SessionProtocol.H1C);
179                 }
180                 break;
181             case HTTPS:
182                 ch = acquireNowExact(key, SessionProtocol.H2);
183                 if (ch == null) {
184                     ch = acquireNowExact(key, SessionProtocol.H1);
185                 }
186                 break;
187             default:
188                 ch = acquireNowExact(key, desiredProtocol);
189         }
190         return ch;
191     }
192 
193     @Nullable
194     private PooledChannel acquireNowExact(PoolKey key, SessionProtocol protocol) {
195         final Deque<PooledChannel> queue = getPool(protocol, key);
196         if (queue == null) {
197             return null;
198         }
199 
200         // Find the most recently released channel while cleaning up the unhealthy channels.
201         for (int i = queue.size(); i > 0; i--) {
202             final PooledChannel pooledChannel = queue.peekLast();
203             assert pooledChannel != null;
204             if (!isHealthy(pooledChannel)) {
205                 queue.removeLast();
206                 continue;
207             }
208 
209             final HttpSession session = HttpSession.get(pooledChannel.get());
210             if (session.unfinishedResponses() >= session.maxUnfinishedResponses()) {
211                 // The channel is full of streams so we cannot create a new one.
212                 // Move the channel to the beginning of the queue so it has low priority.
213                 queue.removeLast();
214                 queue.addFirst(pooledChannel);
215                 continue;
216             }
217 
218             if (!protocol.isMultiplex()) {
219                 queue.removeLast();
220             }
221             return pooledChannel;
222         }
223 
224         return null;
225     }

227     private static boolean isHealthy(PooledChannel pooledChannel) {
228         final Channel ch = pooledChannel.get();
229         return ch.isActive() && HttpSession.get(ch).canSendRequest();
230     }
231 
232     @Nullable
233     private static SessionProtocol getProtocolIfHealthy(Channel ch) {
234         if (!ch.isActive()) {
235             return null;
236         }
237 
238         // Note that we do not need to check 'HttpSession.isActive()'
239         // because an inactive session always returns null.
240         return HttpSession.get(ch).protocol();
241     }

243     /**
244      * Acquires a new {@link Channel} which is matched by the specified condition by making a connection
245      * attempt or waiting for the current connection attempt in progress.
246      */
247     CompletableFuture<PooledChannel> acquireLater(SessionProtocol desiredProtocol, PoolKey key,
248                                                   ClientConnectionTimingsBuilder timingsBuilder) {
249         final CompletableFuture<PooledChannel> promise = new CompletableFuture<>();
250         if (!usePendingAcquisition(desiredProtocol, key, promise, timingsBuilder)) {
251             connect(desiredProtocol, key, promise, timingsBuilder);
252         }
253         return promise;
254     }

256     private CompletableFuture<PooledChannel> acquireLater(SessionProtocol desiredProtocol, PoolKey key,
257                                                           ClientConnectionTimingsBuilder timingsBuilder,
258                                                           CompletableFuture<PooledChannel> promise) {
259         if (!usePendingAcquisition(desiredProtocol, key, promise, timingsBuilder)) {
260             connect(desiredProtocol, key, promise, timingsBuilder);
261         }
262         return promise;
263     }

265     /**
266      * Tries to use the pending HTTP/2 connection to avoid creating an extra connection.
267      *
268      * @return {@code true} if succeeded to reuse the pending connection.
269      */
270     private boolean usePendingAcquisition(SessionProtocol desiredProtocol, PoolKey key,
271                                           CompletableFuture<PooledChannel> promise,
272                                           ClientConnectionTimingsBuilder timingsBuilder) {
273 
274         if (desiredProtocol == SessionProtocol.H1 || desiredProtocol == SessionProtocol.H1C) {
275             // Can't use HTTP/1 connections because they will not be available in the pool until
276             // the request is done.
277             return false;
278         }
279 
280         final CompletableFuture<PooledChannel> pendingAcquisition =
281                 getPendingAcquisition(desiredProtocol, key);
282 
283         if (pendingAcquisition == null) {
284             return false;
285         }
286 
287         timingsBuilder.pendingAcquisitionStart();
288         pendingAcquisition.handle((pch, cause) -> {
289             timingsBuilder.pendingAcquisitionEnd();
290 
291             if (cause == null) {
292                 final SessionProtocol actualProtocol = pch.protocol();
293                 if (actualProtocol.isMultiplex()) {
294                     final HttpSession session = HttpSession.get(pch.get());
295                     if (session.maxUnfinishedResponses() - session.unfinishedResponses() <= 1) {
296                         acquireLater(actualProtocol, key, timingsBuilder, promise);
297                     } else {
298                         promise.complete(pch);
299                     }
300                 } else {
301                     // Try to acquire again because the connection was not HTTP/2.
302                     // We use the exact protocol (H1 or H1C) instead of 'desiredProtocol' so that
303                     // we do not waste our time looking for pending acquisitions for the host
304                     // that does not support HTTP/2.
305                     final PooledChannel ch = acquireNow(actualProtocol, key);
306                     if (ch != null) {
307                         promise.complete(ch);
308                     } else {
309                         connect(actualProtocol, key, promise, timingsBuilder);
310                     }
311                 }
312             } else {
313                 // The pending connection attempt has failed.
314                 connect(desiredProtocol, key, promise, timingsBuilder);
315             }
316             return null;
317         });
318 
319         return true;
320     }

322     private void connect(SessionProtocol desiredProtocol, PoolKey key, CompletableFuture<PooledChannel> promise,
323                          ClientConnectionTimingsBuilder timingsBuilder) {
324 
325         setPendingAcquisition(desiredProtocol, key, promise);
326         timingsBuilder.socketConnectStart();
327 
328         final InetSocketAddress remoteAddress;
329         try {
330             remoteAddress = toRemoteAddress(key);
331         } catch (UnknownHostException e) {
332             notifyConnect(desiredProtocol, key, eventLoop.newFailedFuture(e), promise, timingsBuilder);
333             return;
334         }
335 
336         // Fail immediately if it is sure that the remote address doesn't support the desired protocol.
337         if (SessionProtocolNegotiationCache.isUnsupported(remoteAddress, desiredProtocol)) {
338             notifyConnect(desiredProtocol, key,
339                           eventLoop.newFailedFuture(
340                                   new SessionProtocolNegotiationException(
341                                           desiredProtocol, "previously failed negotiation")),
342                           promise, timingsBuilder);
343             return;
344         }
345 
346         // Create a new connection.
347         final Promise<Channel> sessionPromise = eventLoop.newPromise();
348         connect(remoteAddress, desiredProtocol, sessionPromise);
349 
350         if (sessionPromise.isDone()) {
351             notifyConnect(desiredProtocol, key, sessionPromise, promise, timingsBuilder);
352         } else {
353             sessionPromise.addListener((Future<Channel> future) -> {
354                 notifyConnect(desiredProtocol, key, future, promise, timingsBuilder);
355             });
356         }
357     }

359     /**
360      * A low-level operation that triggers a new connection attempt. Used only by:
361      * <ul>
362      *   <li>{@link #connect(SessionProtocol, PoolKey, CompletableFuture, ClientConnectionTimingsBuilder)} -
363      *       The pool has been exhausted.</li>
364      *   <li>{@link HttpSessionHandler} - HTTP/2 upgrade has failed.</li>
365      * </ul>
366      */
367     void connect(SocketAddress remoteAddress, SessionProtocol desiredProtocol,
368                  Promise<Channel> sessionPromise) {
369         final Bootstrap bootstrap = getBootstrap(desiredProtocol);
370         final ChannelFuture connectFuture = bootstrap.connect(remoteAddress);
371 
372         connectFuture.addListener((ChannelFuture future) -> {
373             if (future.isSuccess()) {
374                 initSession(desiredProtocol, future, sessionPromise);
375             } else {
376                 sessionPromise.setFailure(future.cause());
377             }
378         });
379     }
380 
381     private static InetSocketAddress toRemoteAddress(PoolKey key) throws UnknownHostException {
382         final InetAddress inetAddr = InetAddress.getByAddress(
383                 key.host, NetUtil.createByteArrayFromIpAddressString(key.ipAddr));
384         return new InetSocketAddress(inetAddr, key.port);
385     }
386 
387     private void initSession(SessionProtocol desiredProtocol, ChannelFuture connectFuture,
388                              Promise<Channel> sessionPromise) {
389         assert connectFuture.isSuccess();
390 
391         final Channel ch = connectFuture.channel();
392         final EventLoop eventLoop = ch.eventLoop();
393         assert eventLoop.inEventLoop();
394 
395         final ScheduledFuture<?> timeoutFuture = eventLoop.schedule(() -> {
396             if (sessionPromise.tryFailure(new SessionProtocolNegotiationException(
397                     desiredProtocol, "connection established, but session creation timed out: " + ch))) {
398                 ch.close();
399             }
400         }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
401 
402         ch.pipeline().addLast(new HttpSessionHandler(this, ch, sessionPromise, timeoutFuture));
403     }

405     private void notifyConnect(SessionProtocol desiredProtocol, PoolKey key, Future<Channel> future,
406                                CompletableFuture<PooledChannel> promise,
407                                ClientConnectionTimingsBuilder timingsBuilder) {
408         assert future.isDone();
409         removePendingAcquisition(desiredProtocol, key);
410 
411         timingsBuilder.socketConnectEnd();
412         try {
413             if (future.isSuccess()) {
414                 final Channel channel = future.getNow();
415                 final SessionProtocol protocol = getProtocolIfHealthy(channel);
416                 if (protocol == null || closeable.isClosing()) {
417                     channel.close();
418                     promise.completeExceptionally(
419                             new UnprocessedRequestException(
420                                     new ClosedSessionException("acquired an unhealthy connection")));
421                     return;
422                 }
423 
424                 allChannels.put(channel, Boolean.TRUE);
425 
426                 try {
427                     listener.connectionOpen(protocol,
428                                             (InetSocketAddress) channel.remoteAddress(),
429                                             (InetSocketAddress) channel.localAddress(),
430                                             channel);
431                 } catch (Exception e) {
432                     if (logger.isWarnEnabled()) {
433                         logger.warn("{} Exception handling {}.connectionOpen()",
434                                     channel, listener.getClass().getName(), e);
435                     }
436                 }
437 
438                 final HttpSession session = HttpSession.get(channel);
439                 if (session.unfinishedResponses() < session.maxUnfinishedResponses()) {
440                     if (protocol.isMultiplex()) {
441                         final Http2PooledChannel pooledChannel = new Http2PooledChannel(channel, protocol);
442                         addToPool(protocol, key, pooledChannel);
443                         promise.complete(pooledChannel);
444                     } else {
445                         promise.complete(new Http1PooledChannel(channel, protocol, key));
446                     }
447                 } else {
448                     // Server set MAX_CONCURRENT_STREAMS to 0, which means we can't send anything.
449                     channel.close();
450                     promise.completeExceptionally(
451                             new UnprocessedRequestException(RefusedStreamException.get()));
452                 }
453 
454                 channel.closeFuture().addListener(f -> {
455                     allChannels.remove(channel);
456 
457                     // Clean up old unhealthy channels by iterating from the beginning of the queue.
458                     final Deque<PooledChannel> queue = getPool(protocol, key);
459                     if (queue != null) {
460                         for (;;) {
461                             final PooledChannel pooledChannel = queue.peekFirst();
462                             if (pooledChannel == null || isHealthy(pooledChannel)) {
463                                 break;
464                             }
465                             queue.removeFirst();
466                         }
467                     }
468 
469                     try {
470                         listener.connectionClosed(protocol,
471                                                   (InetSocketAddress) channel.remoteAddress(),
472                                                   (InetSocketAddress) channel.localAddress(),
473                                                   channel);
474                     } catch (Exception e) {
475                         if (logger.isWarnEnabled()) {
476                             logger.warn("{} Exception handling {}.connectionClosed()",
477                                         channel, listener.getClass().getName(), e);
478                         }
479                     }
480                 });
481             } else {
482                 promise.completeExceptionally(new UnprocessedRequestException(future.cause()));
483             }
484         } catch (Exception e) {
485             promise.completeExceptionally(new UnprocessedRequestException(e));
486         }
487     }

489     /**
490      * Adds a {@link Channel} to this pool.
491      */
492     private void addToPool(SessionProtocol actualProtocol, PoolKey key, PooledChannel pooledChannel) {
493         assert eventLoop.inEventLoop() : Thread.currentThread().getName();
494         getOrCreatePool(actualProtocol, key).addLast(pooledChannel);
495     }

497     @Override
498     public CompletableFuture<?> closeAsync() {
499         return closeable.closeAsync();
500     }
501 
502     private void closeAsync(CompletableFuture<?> future) {
503         if (!eventLoop.inEventLoop()) {
504             eventLoop.execute(() -> closeAsync(future));
505             return;
506         }
507 
508         // NB: Make a copy first, because close() will trigger the closeFuture listener
509         //     which mutates allChannels back, causing ConcurrentModificationException.
510         final Channel[] allChannels = this.allChannels.keySet().toArray(EMPTY_CHANNELS);
511         final int numAllChannels = allChannels.length;
512         if (numAllChannels == 0) {
513             future.complete(null);
514             return;
515         }
516 
517         // Complete the given future when all channels are closed.
518         final ChannelFutureListener listener = new ChannelFutureListener() {
519             private int numRemainingChannels = numAllChannels;
520 
521             @Override
522             public void operationComplete(ChannelFuture unused) throws Exception {
523                 if (--numRemainingChannels <= 0) {
524                     future.complete(null);
525                 }
526             }
527         };
528 
529         for (Channel ch : allChannels) {
530             ch.close().addListener(listener);
531         }
532     }
533 
534     /**
535      * Closes all {@link Channel}s managed by this pool.
536      */
537     @Override
538     public void close() {
539         if (Thread.currentThread() instanceof NonBlocking) {
540             // Avoid blocking operation if we're in an event loop, because otherwise we might see a dead lock
541             // while waiting for the channels to be closed.
542             closeable.closeAsync();
543         } else {
544             closeable.close();
545         }
546     }

548     static final class PoolKey {
549         final String host;
550         final String ipAddr;
551         final int port;
552         final int hashCode;
553 
554         PoolKey(String host, String ipAddr, int port) {
555             this.host = host;
556             this.ipAddr = ipAddr;
557             this.port = port;
558             hashCode = (host.hashCode() * 31 + ipAddr.hashCode()) * 31 + port;
559         }
560 
561         @Override
562         public boolean equals(@Nullable Object o) {
563             if (this == o) {
564                 return true;
565             }
566 
567             if (!(o instanceof PoolKey)) {
568                 return false;
569             }
570 
571             final PoolKey that = (PoolKey) o;
572             // Compare IP address first, which is most likely to differ.
573             return ipAddr.equals(that.ipAddr) &&
574                    port == that.port &&
575                    host.equals(that.host);
576         }
577 
578         @Override
579         public int hashCode() {
580             return hashCode;
581         }
582 
583         @Override
584         public String toString() {
585             return MoreObjects.toStringHelper(this)
586                               .add("host", host)
587                               .add("ipAddr", ipAddr)
588                               .add("port", port)
589                               .toString();
590         }
591     }

593     static final class Http2PooledChannel extends PooledChannel {
594         Http2PooledChannel(Channel channel, SessionProtocol protocol) {
595             super(channel, protocol);
596         }

598         @Override
599         public void release() {
600             // There's nothing to do here because we keep the connection in the pool after acquisition.
601         }

604     final class Http1PooledChannel extends PooledChannel {
605         private final PoolKey key;
606 
607         Http1PooledChannel(Channel channel, SessionProtocol protocol, PoolKey key) {
608             super(channel, protocol);
609             this.key = key;
610         }
611 
612         @Override
613         public void release() {
614             if (!eventLoop.inEventLoop()) {
615                 eventLoop.execute(this::doRelease);
616             } else {
617                 doRelease();
618             }
619         }
620 
621         private void doRelease() {
622             if (isHealthy(this)) {
623                 // Channel turns out to be healthy. Add it back to the pool.
624                 addToPool(protocol(), key, this);
625             } else {
626                 // Channel not healthy. Do not add it back to the pool.
627             }
628         }
629     }
