53      private static final Logger logger = LoggerFactory.getLogger(Http2ResponseDecoder.class);

55      private final Http2Connection conn;
56      private final Http2ConnectionEncoder encoder;
57      private final Http2GoAwayHandler goAwayHandler;
58  
59      Http2ResponseDecoder(Channel channel, Http2ConnectionEncoder encoder, HttpClientFactory clientFactory) {
60          super(channel,
61                InboundTrafficController.ofHttp2(channel, clientFactory.http2InitialConnectionWindowSize()));
62          conn = encoder.connection();
63          this.encoder = encoder;
64          goAwayHandler = new Http2GoAwayHandler();
65      }
66  
67      @Override
68      HttpResponseWrapper addResponse(
69              int id, DecodedHttpResponse res, @Nullable ClientRequestContext ctx,
70              EventLoop eventLoop, long responseTimeoutMillis, long maxContentLength) {
71  
72          final HttpResponseWrapper resWrapper =
73                  super.addResponse(id, res, ctx, eventLoop, responseTimeoutMillis, maxContentLength);
74  
75          resWrapper.whenComplete().handle((unused, cause) -> {
76              if (eventLoop.inEventLoop()) {
77                  onWrapperCompleted(resWrapper, id, cause);
78              } else {
79                  eventLoop.execute(() -> onWrapperCompleted(resWrapper, id, cause));
80              }
81              return null;
82          });
83          return resWrapper;
84      }
85  
86      private void onWrapperCompleted(HttpResponseWrapper resWrapper, int id, @Nullable Throwable cause) {
87          // Cancel timeout future and abort the request if it exists.
88          resWrapper.onSubscriptionCancelled(cause);
89  
90          if (cause != null) {
91              // We are not closing the connection but just send a RST_STREAM,
92              // so we have to remove the response manually.
93              removeResponse(id);
94  
95              // Reset the stream.
96              final int streamId = idToStreamId(id);
97              final int lastStreamId = conn.local().lastStreamKnownByPeer();
98              if (lastStreamId < 0 || // Did not receive a GOAWAY yet or
99                  streamId <= lastStreamId) { // received a GOAWAY and the request's streamId <= lastStreamId
100                 final ChannelHandlerContext ctx = channel().pipeline().lastContext();
101                 if (ctx != null) {
102                     encoder.writeRstStream(ctx, streamId, Http2Error.CANCEL.code(), ctx.newPromise());
103                     ctx.flush();
104                 } else {
105                     // The pipeline has been cleaned up due to disconnection.
106                 }
107             }
108         }
109     }

124     @Override
125     public void onStreamClosed(Http2Stream stream) {
126         goAwayHandler.onStreamClosed(channel(), stream);
127 
128         final HttpResponseWrapper res = getResponse(streamIdToId(stream.id()), true);
129         if (res == null) {
130             return;
131         }
132 
133         if (!goAwayHandler.receivedGoAway()) {
134             res.close(ClosedStreamException.get());
135             return;
136         }
137 
138         final int lastStreamId = conn.local().lastStreamKnownByPeer();
139         if (stream.id() > lastStreamId) {
140             res.close(new UnprocessedRequestException(GoAwayReceivedException.get()));
141         } else {
142             res.close(ClosedStreamException.get());
143         }
144 
145         // Send a GOAWAY frame if the connection has been scheduled for disconnection and
146         // it did not receive or send a GOAWAY frame.
147         if (needsToDisconnectNow() && !goAwayHandler.sentGoAway() && !goAwayHandler.receivedGoAway()) {
148             channel().close();
149         }
150     }

175     @Override
176     public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
177                               boolean endOfStream) throws Http2Exception {
178         final HttpResponseWrapper res = getResponse(streamIdToId(streamId), endOfStream);
179         if (res == null) {
180             if (conn.streamMayHaveExisted(streamId)) {
181                 if (logger.isDebugEnabled()) {
182                     logger.debug("{} Received a late HEADERS frame for a closed stream: {}",
183                                  ctx.channel(), streamId);
184                 }
185                 return;
186             }
187 
188             throw connectionError(PROTOCOL_ERROR, "received a HEADERS frame for an unknown stream: %d",
189                                   streamId);
190         }
191 
192         res.logResponseFirstBytesTransferred();
193 
194         final HttpHeaders converted = ArmeriaHttpUtil.toArmeria(headers, false, endOfStream);
195         try {
196             res.initTimeout();
197             res.tryWrite(converted);
198         } catch (Throwable t) {
199             res.close(t);
200             throw connectionError(INTERNAL_ERROR, t, "failed to consume a HEADERS frame");
201         }
202 
203         if (endOfStream) {
204             res.close();
205         }
206     }

216     @Override
217     public int onDataRead(
218             ChannelHandlerContext ctx, int streamId, ByteBuf data,
219             int padding, boolean endOfStream) throws Http2Exception {
220 
221         final int dataLength = data.readableBytes();
222         final HttpResponseWrapper res = getResponse(streamIdToId(streamId), endOfStream);
223         if (res == null) {
224             if (conn.streamMayHaveExisted(streamId)) {
225                 if (logger.isDebugEnabled()) {
226                     logger.debug("{} Received a late DATA frame for a closed stream: {}",
227                                  ctx.channel(), streamId);
228                 }
229                 return dataLength + padding;
230             }
231 
232             throw connectionError(PROTOCOL_ERROR, "received a DATA frame for an unknown stream: %d",
233                                   streamId);
234         }
235 
236         final long maxContentLength = res.maxContentLength();
237         if (maxContentLength > 0 && res.writtenBytes() > maxContentLength - dataLength) {
238             res.close(ContentTooLargeException.get());
239             throw connectionError(INTERNAL_ERROR,
240                                   "content length too large: %d + %d > %d (stream: %d)",
241                                   res.writtenBytes(), dataLength, maxContentLength, streamId);
242         }
243 
244         try {
245             res.tryWrite(new ByteBufHttpData(data.retain(), endOfStream));
246         } catch (Throwable t) {
247             res.close(t);
248             throw connectionError(INTERNAL_ERROR, t, "failed to consume a DATA frame");
249         }
250 
251         if (endOfStream) {
252             res.close();
253         }
254 
255         // All bytes have been processed.
256         return dataLength + padding;
257     }
258 
259     @Override
260     public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception {
261         final HttpResponseWrapper res = removeResponse(streamIdToId(streamId));
262         if (res == null) {
263             if (conn.streamMayHaveExisted(streamId)) {
264                 if (logger.isDebugEnabled()) {
265                     logger.debug("{} Received a late RST_STREAM frame for a closed stream: {}",
266                                  ctx.channel(), streamId);
267                 }
268             } else {
269                 throw connectionError(PROTOCOL_ERROR,
270                                       "received a RST_STREAM frame for an unknown stream: %d", streamId);
271             }
272             return;
273         }
274 
275         res.close(new ClosedStreamException("received a RST_STREAM frame: " + Http2Error.valueOf(errorCode)));
276     }

302     private static int streamIdToId(int streamId) {
303         return streamId - 1 >>> 1;
304     }
305 
306     private static int idToStreamId(int id) {
307         return (id << 1) + 1;
308     }
