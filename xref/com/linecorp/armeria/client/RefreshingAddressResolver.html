53  final class RefreshingAddressResolver extends AbstractAddressResolver<InetSocketAddress> {
54  
55      private static final Logger logger = LoggerFactory.getLogger(RefreshingAddressResolver.class);
56  
57      private final ConcurrentMap<String, CompletableFuture<CacheEntry>> cache;
58      private final DefaultDnsNameResolver resolver;
59      private final List<DnsRecordType> dnsRecordTypes;
60      private final int minTtl;
61      private final int maxTtl;
62      private final int negativeTtl;
63      private final Backoff refreshBackoff;
64  
65      private volatile boolean resolverClosed;
66  
67      RefreshingAddressResolver(EventLoop eventLoop,
68                                ConcurrentMap<String, CompletableFuture<CacheEntry>> cache,
69                                DefaultDnsNameResolver resolver, List<DnsRecordType> dnsRecordTypes,
70                                int minTtl, int maxTtl, int negativeTtl, Backoff refreshBackoff) {
71          super(eventLoop);
72          this.cache = cache;
73          this.resolver = resolver;
74          this.dnsRecordTypes = dnsRecordTypes;
75          this.minTtl = minTtl;
76          this.maxTtl = maxTtl;
77          this.negativeTtl = negativeTtl;
78          this.refreshBackoff = refreshBackoff;
79      }
80  
81      @Override
82      protected boolean doIsResolved(InetSocketAddress address) {
83          return !requireNonNull(address, "address").isUnresolved();
84      }
85  
86      @Override
87      protected void doResolve(InetSocketAddress unresolvedAddress, Promise<InetSocketAddress> promise)
88              throws Exception {
89          requireNonNull(unresolvedAddress, "unresolvedAddress");
90          requireNonNull(promise, "promise");
91          if (resolverClosed) {
92              promise.tryFailure(new IllegalStateException("resolver is closed already."));
93              return;
94          }
95          final String hostname = unresolvedAddress.getHostString();
96          final int port = unresolvedAddress.getPort();
97          final CompletableFuture<CacheEntry> entryFuture = cache.get(hostname);
98          if (entryFuture != null) {
99              handleFromCache(entryFuture, promise, port);
100             return;
101         }
102 
103         final CompletableFuture<CacheEntry> result = new CompletableFuture<>();
104         final CompletableFuture<CacheEntry> previous = cache.putIfAbsent(hostname, result);
105         if (previous != null) {
106             handleFromCache(previous, promise, port);
107             return;
108         }
109 
110         final List<DnsQuestion> questions =
111                 dnsRecordTypes.stream()
112                               .map(type -> DnsQuestionWithoutTrailingDot.of(hostname, type))
113                               .collect(toImmutableList());
114         sendQueries(questions, hostname, result);
115         result.handle((entry, unused) -> {
116             final Throwable cause = entry.cause();
117             if (cause != null) {
118                 if (entry.hasCacheableCause() && negativeTtl > 0) {
119                     executor().schedule(() -> cache.remove(hostname), negativeTtl, TimeUnit.SECONDS);
120                 } else {
121                     cache.remove(hostname);
122                 }
123                 promise.tryFailure(cause);
124                 return null;
125             }
126 
127             entry.scheduleRefresh(entry.ttlMillis());
128             promise.trySuccess(new InetSocketAddress(entry.address(), port));
129             return null;
130         });
131     }
132 
133     private void handleFromCache(CompletableFuture<CacheEntry> future, Promise<InetSocketAddress> promise,
134                                  int port) {
135         future.handle((entry, unused) -> {
136             final Throwable cause = entry.cause();
137             if (cause != null) {
138                 promise.tryFailure(cause);
139                 return null;
140             }
141             entry.servedFromCache();
142             promise.trySuccess(new InetSocketAddress(entry.address(), port));
143             return null;
144         });
145     }
146 
147     private void sendQueries(List<DnsQuestion> questions, String hostname,
148                              CompletableFuture<CacheEntry> result) {
149         final Future<List<DnsRecord>> recordsFuture = resolver.sendQueries(questions, hostname);
150         recordsFuture.addListener(f -> {
151             if (!f.isSuccess()) {
152                 final Throwable cause = f.cause();
153 
154                 // TODO(minwoox): In Netty, DnsNameResolver only caches if the failure was not because of an
155                 //                IO error / timeout that was caused by the query itself.
156                 //                To figure that out, we need to check the cause of the UnknownHostException.
157                 //                If it's null, then we can cache the cause. However, this is very fragile
158                 //                because Netty can change the behavior while we are not noticing that.
159                 //                So sending a PR to upstream would be the best solution.
160                 final boolean hasCacheableCause;
161                 if (cause instanceof UnknownHostException) {
162                     final UnknownHostException unknownHostException = (UnknownHostException) cause;
163                     hasCacheableCause = unknownHostException.getCause() == null;
164                 } else {
165                     hasCacheableCause = false;
166                 }
167                 result.complete(new CacheEntry(null, -1, questions, cause, hasCacheableCause));
168                 return;
169             }
170 
171             @SuppressWarnings("unchecked")
172             final List<DnsRecord> records = (List<DnsRecord>) f.getNow();
173             InetAddress inetAddress = null;
174             long ttlMillis = -1;
175             try {
176                 for (DnsRecord r : records) {
177                     final byte[] addrBytes = extractAddressBytes(r, logger, hostname);
178                     if (addrBytes == null) {
179                         continue;
180                     }
181                     try {
182                         inetAddress = InetAddress.getByAddress(hostname, addrBytes);
183                         ttlMillis = TimeUnit.SECONDS.toMillis(
184                                 Math.max(Math.min(r.timeToLive(), maxTtl), minTtl));
185                         break;
186                     } catch (UnknownHostException e) {
187                         // Should never reach here because we already validated it in extractAddressBytes.
188                         result.complete(new CacheEntry(null, -1, questions, new IllegalArgumentException(
189                                 "Invalid address: " + hostname, e), false));
190                         return;
191                     }
192                 }
193             } finally {
194                 records.forEach(ReferenceCountUtil::safeRelease);
195             }
196 
197             final CacheEntry cacheEntry;
198             if (inetAddress == null) {
199                 cacheEntry = new CacheEntry(null, -1, questions, new UnknownHostException(
200                         "failed to receive DNS records for " + hostname), true);
201             } else {
202                 cacheEntry = new CacheEntry(inetAddress, ttlMillis, questions, null, false);
203             }
204             result.complete(cacheEntry);
205         });
206     }
207 
208     @Override
209     protected void doResolveAll(InetSocketAddress unresolvedAddress, Promise<List<InetSocketAddress>> promise)
210             throws Exception {
211         throw new UnsupportedOperationException();
212     }
213 
214     /**
215      * Closes all the resources allocated and used by this resolver.
216      *
217      * <p>Please note that this method does not clear the {@link CacheEntry} because the {@link #cache} is
218      * created by {@link RefreshingAddressResolverGroup} and shared across from all
219      * {@link RefreshingAddressResolver}s. {@link CacheEntry} is cleared when
220      * {@link RefreshingAddressResolverGroup#close()} is called.
221      */
222     @Override
223     public void close() {
224         resolverClosed = true;
225         resolver.close();
226     }
227 
228     final class CacheEntry implements Runnable {
229 
230         @Nullable
231         private final InetAddress address;
232         private final long ttlMillis;
233         private final List<DnsQuestion> questions;
234 
235         @Nullable
236         private final Throwable cause;
237         private final boolean hasCacheableCause;
238 
239         /**
240          * No need to be volatile because updated only by the {@link #executor()}.
241          */
242         private int numAttemptsSoFar = 1;
243 
244         private volatile boolean servedFromCache;
245 
246         @VisibleForTesting
247         @Nullable
248         ScheduledFuture<?> refreshFuture;
249 
250         CacheEntry(@Nullable InetAddress address, long ttlMillis, List<DnsQuestion> questions,
251                    @Nullable Throwable cause, boolean hasCacheableCause) {
252             this.address = address;
253             this.ttlMillis = ttlMillis;
254             this.questions = questions;
255             this.cause = cause;
256             this.hasCacheableCause = hasCacheableCause;
257         }
258 
259         void servedFromCache() {
260             servedFromCache = true;
261         }
262 
263         @Nullable
264         InetAddress address() {
265             return address;
266         }
267 
268         long ttlMillis() {
269             return ttlMillis;
270         }
271 
272         @Nullable
273         Throwable cause() {
274             return cause;
275         }
276 
277         boolean hasCacheableCause() {
278             return hasCacheableCause;
279         }
280 
281         void scheduleRefresh(long nextDelayMillis) {
282             if (resolverClosed) {
283                 return;
284             }
285             refreshFuture = executor().schedule(this, nextDelayMillis, TimeUnit.MILLISECONDS);
286         }
287 
288         void clear() {
289             assert resolverClosed;
290             if (refreshFuture != null) {
291                 refreshFuture.cancel(false);
292             }
293         }
294 
295         @Override
296         public void run() {
297             if (resolverClosed) {
298                 return;
299             }
300 
301             assert address != null;
302             final String hostName = address.getHostName();
303             if (!servedFromCache) {
304                 cache.remove(hostName);
305                 return;
306             }
307 
308             final CompletableFuture<CacheEntry> result = new CompletableFuture<>();
309             sendQueries(questions, hostName, result);
310             result.handle((entry, unused) -> {
311                 if (resolverClosed) {
312                     return null;
313                 }
314 
315                 final Throwable cause = entry.cause();
316                 if (cause != null) {
317                     final long nextDelayMillis = refreshBackoff.nextDelayMillis(numAttemptsSoFar++);
318                     if (nextDelayMillis < 0) {
319                         cache.remove(hostName);
320                         return null;
321                     }
322                     scheduleRefresh(nextDelayMillis);
323                     return null;
324                 }
325 
326                 // Got the response successfully so reset the state.
327                 servedFromCache = false;
328                 numAttemptsSoFar = 1;
329 
330                 if (entry.address().equals(address) && entry.ttlMillis() == ttlMillis) {
331                     scheduleRefresh(ttlMillis);
332                 } else {
333                     // Replace the old entry with the new one.
334                     cache.put(hostName, result);
335                     entry.scheduleRefresh(entry.ttlMillis());
336                 }
337                 return null;
338             });
339         }
340 
341         @Override
342         public String toString() {
343             return MoreObjects.toStringHelper(this).omitNullValues()
344                               .add("address", address)
345                               .add("ttlMillis", ttlMillis)
346                               .add("questions", questions)
347                               .add("cause", cause)
348                               .add("hasCacheableCause", hasCacheableCause)
349                               .add("servedFromCache", servedFromCache)
350                               .toString();
351         }
352     }
