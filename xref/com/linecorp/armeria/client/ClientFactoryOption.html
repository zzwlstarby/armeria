75      // Do not accept 1) the options that may break Armeria and 2) the deprecated options.
76      @SuppressWarnings("deprecation")
77      private static final Set<ChannelOption<?>> PROHIBITED_SOCKET_OPTIONS = ImmutableSet.of(
78              ChannelOption.ALLOW_HALF_CLOSURE, ChannelOption.AUTO_READ,
79              ChannelOption.AUTO_CLOSE, ChannelOption.MAX_MESSAGES_PER_READ,
80              ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, ChannelOption.WRITE_BUFFER_LOW_WATER_MARK,
81              EpollChannelOption.EPOLL_MODE);
82  
83      /**
84       * The {@link ChannelOption}s of the sockets created by the {@link ClientFactory}.
85       */
86      public static final ClientFactoryOption<Map<ChannelOption<?>, Object>> CHANNEL_OPTIONS =
87              define("CHANNEL_OPTIONS", ImmutableMap.of(), newOptions -> {
88                  for (ChannelOption<?> channelOption : PROHIBITED_SOCKET_OPTIONS) {
89                      checkArgument(!newOptions.containsKey(channelOption),
90                                    "prohibited channel option: %s", channelOption);
91                  }
92                  return newOptions;
93              }, (oldValue, newValue) -> {
94                  final Map<ChannelOption<?>, Object> newOptions = newValue.value();
95                  if (newOptions.isEmpty()) {
96                      return oldValue;
97                  }
98                  final Map<ChannelOption<?>, Object> oldOptions = oldValue.value();
99                  if (oldOptions.isEmpty()) {
100                     return newValue;
101                 }
102                 final ImmutableMap.Builder<ChannelOption<?>, Object> builder =
103                         ImmutableMap.builderWithExpectedSize(oldOptions.size() + newOptions.size());
104                 oldOptions.forEach((key, value) -> {
105                     if (!newOptions.containsKey(key)) {
106                         builder.put(key, value);
107                     }
108                 });
109                 builder.putAll(newOptions);
110                 return newValue.option().newValue(builder.build());
111             });

217     /**
218      * Returns the all available {@link ClientFactoryOption}s.
219      */
220     public static Set<ClientFactoryOption<?>> allOptions() {
221         return allOptions(ClientFactoryOption.class);
222     }

224     /**
225      * Returns the {@link ClientFactoryOption} with the specified {@code name}.
226      *
227      * @throws NoSuchElementException if there's no such option defined.
228      */
229     public static ClientFactoryOption<?> of(String name) {
230         return of(ClientFactoryOption.class, name);
231     }
232 
233     /**
234      * Defines a new {@link ClientFactoryOption} of the specified name and default value.
235      *
236      * @param name the name of the option.
237      * @param defaultValue the default value of the option, which will be used when unspecified.
238      *
239      * @throws IllegalStateException if an option with the specified name exists already.
240      */
241     public static <T> ClientFactoryOption<T> define(String name, T defaultValue) {
242         return define(name, defaultValue, Function.identity(), (oldValue, newValue) -> newValue);
243     }
244 
245     /**
246      * Defines a new {@link ClientFactoryOption} of the specified name, default value and merge function.
247      *
248      * @param name the name of the option.
249      * @param defaultValue the default value of the option, which will be used when unspecified.
250      * @param validator the {@link Function} which is used for validating and normalizing an option value.
251      * @param mergeFunction the {@link BiFunction} which is used for merging old and new option values.
252      *
253      * @throws IllegalStateException if an option with the specified name exists already.
254      */
255     public static <T> ClientFactoryOption<T> define(
256             String name,
257             T defaultValue,
258             Function<T, T> validator,
259             BiFunction<ClientFactoryOptionValue<T>,
260                     ClientFactoryOptionValue<T>,
261                     ClientFactoryOptionValue<T>> mergeFunction) {
262         return define(ClientFactoryOption.class, name, defaultValue,
263                       ClientFactoryOption::new, validator, mergeFunction);
264     }
265 
266     private ClientFactoryOption(
267             String name,
268             T defaultValue,
269             Function<T, T> validator,
270             BiFunction<ClientFactoryOptionValue<T>,
271                     ClientFactoryOptionValue<T>,
272                     ClientFactoryOptionValue<T>> mergeFunction) {
273         super(name, defaultValue, validator, mergeFunction);
274     }
275 
276     @Override
277     protected ClientFactoryOptionValue<T> doNewValue(T value) {
278         return new ClientFactoryOptionValue<>(this, value);
279     }
