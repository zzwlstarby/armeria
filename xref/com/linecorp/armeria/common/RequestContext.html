67  /**
68   * Provides information about a {@link Request}, its {@link Response} and related utilities.
69   * A server-side {@link Request} has a {@link ServiceRequestContext} and
70   * a client-side {@link Request} has a {@link ClientRequestContext}.
71   */
72  public interface RequestContext {
73  
74      /**
75       * Returns the context of the {@link Request} that is being handled in the current thread.
76       *
77       * @throws IllegalStateException if the context is unavailable in the current thread
78       */
79      static <T extends RequestContext> T current() {
80          final T ctx = currentOrNull();
81          if (ctx == null) {
82              throw new IllegalStateException(RequestContext.class.getSimpleName() + " unavailable");
83          }
84          return ctx;
85      }
86  
87      /**
88       * Returns the context of the {@link Request} that is being handled in the current thread.
89       *
90       * @return the {@link RequestContext} available in the current thread, or {@code null} if unavailable.
91       */
92      @Nullable
93      static <T extends RequestContext> T currentOrNull() {
94          return RequestContextThreadLocal.get();
95      }

97      /**
98       * Maps the context of the {@link Request} that is being handled in the current thread.
99       *
100      * @param mapper the {@link Function} that maps the {@link RequestContext}
101      * @param defaultValueSupplier the {@link Supplier} that provides the value when the context is unavailable
102      *                             in the current thread. If {@code null}, the {@code null} will be returned
103      *                             when the context is unavailable in the current thread.
104      */
105     @Nullable
106     static <T> T mapCurrent(
107             Function<? super RequestContext, T> mapper, @Nullable Supplier<T> defaultValueSupplier) {
108 
109         final RequestContext ctx = currentOrNull();
110         if (ctx != null) {
111             return mapper.apply(ctx);
112         }
113 
114         if (defaultValueSupplier != null) {
115             return defaultValueSupplier.get();
116         }
117 
118         return null;
119     }

392     /**
393      * Replaces the current {@link RequestContext} in the thread-local with this context without any validation.
394      * This method also does not run any callbacks.
395      *
396      * <p><strong>Note:</strong> Do not use this if you don't know what you are doing. This method does not
397      * prevent the situation where a wrong {@link RequestContext} is pushed into the thread-local.
398      * Use {@link #push()} instead.
399      *
400      * @see ClientRequestContext#push()
401      * @see ServiceRequestContext#push()
402      */
403     default SafeCloseable replace() {
404         final RequestContext oldCtx = RequestContextThreadLocal.getAndSet(this);
405         if (oldCtx == null) {
406             return RequestContextThreadLocal::remove;
407         }
408         return () -> RequestContextThreadLocal.set(oldCtx);
409     }

557     default <T> CompletionStage<T> makeContextAware(CompletionStage<T> stage) {
558         final CompletableFuture<T> future = JavaVersionSpecific.get().newRequestContextAwareFuture(this);

561                 if (cause != null) {
562                     future.completeExceptionally(cause);
563                 } else {
564                     future.complete(result);
565                 }
566             } catch (Throwable t) {
567                 future.completeExceptionally(t);
568             }
569             return null;
570         });
571         return future;
572     }

591     /**
592      * Resolves the specified {@code promise} with the specified {@code result} so that the {@code promise} is
593      * marked as 'done'. If {@code promise} is done already, this method does the following:
594      * <ul>
595      *   <li>Log a warning about the failure, and</li>
596      *   <li>Release {@code result} if it is {@linkplain ReferenceCounted a reference-counted object},
597      *       such as {@link ByteBuf} and {@link FullHttpResponse}.</li>
598      * </ul>
599      * Note that a {@link Promise} can be done already even if you did not call this method in the following
600      * cases:
601      * <ul>
602      *   <li>Invocation timeout - The invocation associated with the {@link Promise} has been timed out.</li>
603      *   <li>User error - A service implementation called any of the following methods more than once:
604      *     <ul>
605      *       <li>{@link #resolvePromise(Promise, Object)}</li>
606      *       <li>{@link #rejectPromise(Promise, Throwable)}</li>
607      *       <li>{@link Promise#setSuccess(Object)}</li>
608      *       <li>{@link Promise#setFailure(Throwable)}</li>
609      *       <li>{@link Promise#cancel(boolean)}</li>
610      *     </ul>
611      *   </li>
612      * </ul>
613      *
614      * @deprecated Use {@link CompletableFuture} instead.
615      */
616     @Deprecated
617     default void resolvePromise(Promise<?> promise, Object result) {
618         @SuppressWarnings("unchecked")
619         final Promise<Object> castPromise = (Promise<Object>) promise;
620 
621         if (castPromise.trySuccess(result)) {
622             // Resolved successfully.
623             return;
624         }
625 
626         try {
627             if (!(promise.cause() instanceof TimeoutException)) {
628                 // Log resolve failure unless it is due to a timeout.
629                 LoggerFactory.getLogger(RequestContext.class).warn(
630                         "Failed to resolve a completed promise ({}) with {}", promise, result);
631             }
632         } finally {
633             ReferenceCountUtil.safeRelease(result);
634         }
635     }

637     /**
638      * Rejects the specified {@code promise} with the specified {@code cause}. If {@code promise} is done
639      * already, this method logs a warning about the failure. Note that a {@link Promise} can be done already
640      * even if you did not call this method in the following cases:
641      * <ul>
642      *   <li>Invocation timeout - The invocation associated with the {@link Promise} has been timed out.</li>
643      *   <li>User error - A service implementation called any of the following methods more than once:
644      *     <ul>
645      *       <li>{@link #resolvePromise(Promise, Object)}</li>
646      *       <li>{@link #rejectPromise(Promise, Throwable)}</li>
647      *       <li>{@link Promise#setSuccess(Object)}</li>
648      *       <li>{@link Promise#setFailure(Throwable)}</li>
649      *       <li>{@link Promise#cancel(boolean)}</li>
650      *     </ul>
651      *   </li>
652      * </ul>
653      *
654      * @deprecated Use {@link CompletableFuture} instead.
655      */
656     @Deprecated
657     default void rejectPromise(Promise<?> promise, Throwable cause) {
658         if (promise.tryFailure(cause)) {
659             // Fulfilled successfully.
660             return;
661         }
662 
663         final Throwable firstCause = promise.cause();
664         if (firstCause instanceof TimeoutException) {
665             // Timed out already.
666             return;
667         }
668 
669         if (Exceptions.isExpected(cause)) {
670             // The exception that was thrown after firstCause (often a transport-layer exception)
671             // was a usual expected exception, not an error.
672             return;
673         }
674 
675         LoggerFactory.getLogger(RequestContext.class).warn(
676                 "Failed to reject a completed promise ({}) with {}", promise, cause, cause);
677     }
