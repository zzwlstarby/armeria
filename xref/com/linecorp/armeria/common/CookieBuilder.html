16  package com.linecorp.armeria.common;
17  
18  import static com.google.common.base.Preconditions.checkArgument;
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.BitSet;
22  
23  import javax.annotation.Nullable;
24  
25  /**
26   * Builds a {@link Cookie}.
27   */
28  public final class CookieBuilder {
29  
30      private static final BitSet VALID_COOKIE_ATTRIBUTE_VALUE_OCTETS = validCookieAttributeValueOctets();
31  
32      // path-value        = <any CHAR except CTLs or ";">
33      private static BitSet validCookieAttributeValueOctets() {
34          final BitSet bits = new BitSet();
35          for (int i = 32; i < 127; i++) {
36              bits.set(i);
37          }
38          bits.set(';', false);
39          return bits;
40      }
41  
42      private static String validateAttributeValue(String value, String valueName) {
43          value = requireNonNull(value, valueName).trim();
44          checkArgument(!value.isEmpty(), "%s is empty.", valueName);
45          final int i = CookieUtil.firstInvalidOctet(value, VALID_COOKIE_ATTRIBUTE_VALUE_OCTETS);
46          if (i >= 0) {
47              throw new IllegalArgumentException(
48                      valueName + " contains a prohibited character: " + value.charAt(i));
49          }
50          return value;
51      }
52  
53      private String name;
54      private String value;
55      private boolean valueQuoted;
56      @Nullable
57      private String domain;
58      @Nullable
59      private String path;
60      long maxAge = Cookie.UNDEFINED_MAX_AGE;
61      private boolean secure;
62      private boolean httpOnly;
63      @Nullable
64      private String sameSite;
65  
66      // These two fields are only used by ClientCookieDecoder.
67      int expiresStart;
68      int expiresEnd;
69  
70      CookieBuilder(String name, String value) {
71          this.name = requireNonNull(name, "name");
72          this.value = requireNonNull(value, "value");
73      }
74  
75      CookieBuilder(Cookie cookie) {
76          name = cookie.name();
77          value = cookie.value();
78          valueQuoted = cookie.isValueQuoted();
79          domain = cookie.domain();
80          path = cookie.path();
81          maxAge = cookie.maxAge();
82          secure = cookie.isSecure();
83          httpOnly = cookie.isHttpOnly();
84          sameSite = cookie.sameSite();
85      }
86  
87      /**
88       * Sets the name of the {@link Cookie}.
89       */
90      public CookieBuilder name(String name) {
91          this.name = requireNonNull(name, "name");
92          return this;
93      }
94  
95      /**
96       * Sets the value of the {@link Cookie}.
97       */
98      public CookieBuilder value(String value) {
99          this.value = requireNonNull(value, "value");
100         return this;
101     }
102 
103     /**
104      * Sets whether the value of the {@link Cookie} needs to be wrapped with double quotes when encoding.
105      * If unspecified, the {@link Cookie} will not be wrapped with double quotes.
106      */
107     public CookieBuilder valueQuoted(boolean valueQuoted) {
108         this.valueQuoted = valueQuoted;
109         return this;
110     }
111 
112     /**
113      * Sets the domain of the {@link Cookie}.
114      */
115     public CookieBuilder domain(String domain) {
116         this.domain = validateAttributeValue(domain, "domain");
117         return this;
118     }
119 
120     /**
121      * Sets the path of the {@link Cookie}.
122      */
123     public CookieBuilder path(String path) {
124         this.path = validateAttributeValue(path, "path");
125         return this;
126     }
127 
128     /**
129      * Sets the maximum age of the {@link Cookie} in seconds. If an age of {@code 0} is specified,
130      * the {@link Cookie} will be automatically removed by browser because it will expire immediately.
131      * If {@link Cookie#UNDEFINED_MAX_AGE} is specified, this {@link Cookie} will be removed when the
132      * browser is closed. If unspecified, {@link Cookie#UNDEFINED_MAX_AGE} will be used.
133      *
134      * @param maxAge The maximum age of this {@link Cookie} in seconds
135      */
136     public CookieBuilder maxAge(long maxAge) {
137         this.maxAge = maxAge;
138         return this;
139     }
140 
141     /**
142      * Sets the security status of the {@link Cookie}. If unspecified, {@code false} will be used.
143      */
144     public CookieBuilder secure(boolean secure) {
145         this.secure = secure;
146         return this;
147     }
148 
149     /**
150      * Sets whether the {@link Cookie} is HTTP only. If {@code true}, the {@link Cookie} cannot be accessed
151      * by a client side script. However, this works only if the browser supports it. For more information,
152      * please look <a href="http://www.owasp.org/index.php/HTTPOnly">here</a>. If unspecified, {@code false}
153      * will be used.
154      */
155     public CookieBuilder httpOnly(boolean httpOnly) {
156         this.httpOnly = httpOnly;
157         return this;
158     }
159 
160     /**
161      * Sets the <a href="https://tools.ietf.org/html/draft-ietf-httpbis-rfc6265bis-03#section-4.1.2.7"
162      * >{@code SameSite}</a> attribute of the {@link Cookie}. The value is supposed to be one of {@code "Lax"},
163      * {@code "Strict"} or {@code "None"}. Note that this attribute is server-side only.
164      */
165     public CookieBuilder sameSite(String sameSite) {
166         this.sameSite = validateAttributeValue(sameSite, "sameSite");
167         return this;
168     }
169 
170     /**
171      * Returns a newly created {@link Cookie} with the properties set so far.
172      */
173     public Cookie build() {
174         return new DefaultCookie(name, value, valueQuoted, domain, path, maxAge, secure, httpOnly, sameSite);
175     }
176 }
