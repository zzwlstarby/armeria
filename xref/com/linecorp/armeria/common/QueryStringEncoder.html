44  package com.linecorp.armeria.common;
45  
46  import java.util.Map.Entry;
47  
48  import io.netty.util.internal.StringUtil;
49  
50  final class QueryStringEncoder {
51  
52      // Forked from netty-4.1.43 and guava-28.1
53      // https://github.com/netty/netty/blob/bd8cea644a07890f5bada18ddff0a849b58cd861/codec-http/src/main/java/io/netty/handler/codec/http/QueryStringEncoder.java
54      // https://github.com/google/guava/blob/13e39cd167a49aad525be462e61d9e5f2b1781ec/guava/src/com/google/common/net/PercentEscaper.java
55  
56      private static final char[] UTF_UNKNOWN = { '%', '3', 'F' }; // Percent encoded question mark
57      private static final char[] UPPER_HEX_DIGITS = "0123456789ABCDEF".toCharArray();
58      private static final byte[] SAFE_OCTETS = new byte[Character.MAX_VALUE + 1];
59  
60      static {
61          final String safeOctetStr = "-_.*abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
62          for (int i = 0; i < safeOctetStr.length(); i++) {
63              SAFE_OCTETS[safeOctetStr.charAt(i)] = -1;
64          }
65      }
66  
67      static void encodeParams(StringBuilder buf, QueryParamGetters params) {
68          for (Entry<String, String> e : params) {
69              encodeComponent(buf, e.getKey()).append('=');
70              encodeComponent(buf, e.getValue()).append('&');
71          }
72  
73          buf.setLength(buf.length() - 1);
74      }
75  
76      private static StringBuilder encodeComponent(StringBuilder buf, String s) {
77          final int len = s.length();
78          for (int i = 0; i < len; i++) {
79              final char c = s.charAt(i);
80              if (SAFE_OCTETS[c] == 0) {
81                  if (i != 0) {
82                      buf.append(s, 0, i);
83                  }
84                  encodeUtf8Component(buf, s, i);
85                  return buf;
86              }
87          }
88  
89          return buf.append(s);
90      }
91  
92      /**
93       * Encodes a query component (name or value). The octet at {@code start} must be an unsafe octet.
94       */
95      private static void encodeUtf8Component(StringBuilder buf, String s, int start) {
96          final int end = s.length();
97          if (start == end) {
98              return;
99          }
100 
101         final char[] tmp = new char[12];
102         tmp[0] = tmp[3] = tmp[6] = tmp[9] = '%'; // Pre-set '%' so we don't have to set anymore.
103 
104         int i = start;
105         for (;;) {
106             char c = s.charAt(i++);
107             if (c < 0x80) {
108                 if (c == ' ') {
109                     buf.append('+');
110                 } else {
111                     tmp[2] = UPPER_HEX_DIGITS[c & 0xF];
112                     tmp[1] = UPPER_HEX_DIGITS[c >>> 4];
113                     buf.append(tmp, 0, 3);
114                 }
115             } else if (c < 0x800) {
116                 tmp[5] = UPPER_HEX_DIGITS[c & 0xF];
117                 c >>>= 4;
118                 tmp[4] = UPPER_HEX_DIGITS[0x8 | (c & 0x3)];
119                 c >>>= 2;
120                 tmp[2] = UPPER_HEX_DIGITS[c & 0xF];
121                 tmp[1] = UPPER_HEX_DIGITS[0xC | (c >>> 4)];
122                 buf.append(tmp, 0, 6);
123             } else if (!StringUtil.isSurrogate(c)) {
124                 tmp[8] = UPPER_HEX_DIGITS[c & 0xF];
125                 c >>>= 4;
126                 tmp[7] = UPPER_HEX_DIGITS[0x8 | (c & 0x3)];
127                 c >>>= 2;
128                 tmp[5] = UPPER_HEX_DIGITS[c & 0xF];
129                 c >>>= 4;
130                 tmp[4] = UPPER_HEX_DIGITS[0x8 | (c & 0x3)];
131                 tmp[2] = UPPER_HEX_DIGITS[(c >>> 2) & 0xF];
132                 tmp[1] = 'E';
133                 buf.append(tmp, 0, 9);
134             } else if (!Character.isHighSurrogate(c) || i == end) {
135                 buf.append(UTF_UNKNOWN);
136             } else {
137                 final char c2 = s.charAt(i++);
138                 if (!Character.isLowSurrogate(c2)) {
139                     buf.append(UTF_UNKNOWN);
140                     if (Character.isHighSurrogate(c2)) {
141                         buf.append(UTF_UNKNOWN);
142                     } else {
143                         tmp[2] = UPPER_HEX_DIGITS[c2 & 0xF];
144                         tmp[1] = UPPER_HEX_DIGITS[c2 >>> 4];
145                         buf.append(tmp, 0, 3);
146                     }
147                 } else {
148                     int codePoint = Character.toCodePoint(c, c2);
149                     // See http://www.unicode.org/versions/Unicode7.0.0/ch03.pdf#G2630.
150                     tmp[11] = UPPER_HEX_DIGITS[codePoint & 0xF];
151                     codePoint >>>= 4;
152                     tmp[10] = UPPER_HEX_DIGITS[0x8 | (codePoint & 0x3)];
153                     codePoint >>>= 2;
154                     tmp[8] = UPPER_HEX_DIGITS[codePoint & 0xF];
155                     codePoint >>>= 4;
156                     tmp[7] = UPPER_HEX_DIGITS[0x8 | (codePoint & 0x3)];
157                     codePoint >>>= 2;
158                     tmp[5] = UPPER_HEX_DIGITS[codePoint & 0xF];
159                     codePoint >>>= 4;
160                     tmp[4] = UPPER_HEX_DIGITS[0x8 | (codePoint & 0x3)];
161                     tmp[2] = UPPER_HEX_DIGITS[(codePoint >>> 2) & 0xF];
162                     tmp[1] = 'F';
163                     buf.append(tmp, 0, 12);
164                 }
165             }
166 
167             // Find and append the safe region as-is.
168             final int nextUnsafeOctetIndex = indexOfUnsafeOctet(s, i);
169             if (nextUnsafeOctetIndex < 0) {
170                 if (i != end) {
171                     buf.append(s, i, end);
172                 }
173                 break;
174             }
175 
176             if (nextUnsafeOctetIndex != i) {
177                 buf.append(s, i, nextUnsafeOctetIndex);
178                 i = nextUnsafeOctetIndex;
179             }
180         }
181     }
182 
183     private static int indexOfUnsafeOctet(String s, int start) {
184         final int len = s.length();
185         for (int i = start; i < len; i++) {
186             final char c = s.charAt(i);
187             if (SAFE_OCTETS[c] == 0) {
188                 return i;
189             }
190         }
191 
192         return -1;
193     }
194 
195     private QueryStringEncoder() {}
