16  package com.linecorp.armeria.common.util;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.util.Collections;
21  import java.util.Iterator;
22  import java.util.Map;
23  
24  import com.google.common.collect.Iterators;
25  
26  import it.unimi.dsi.fastutil.objects.Reference2ReferenceOpenHashMap;
27  
28  /**
29   * A set of configuration options and their respective values.
30   *
31   * @param <T> the type of the option.
32   * @param <U> the type of the option value holder.
33   *
34   * @see AbstractOption
35   * @see AbstractOptionValue
36   */
37  public abstract class AbstractOptions<
38          T extends AbstractOption<T, U, Object>,
39          U extends AbstractOptionValue<U, T, Object>> implements Iterable<U> {
40  
41      private final Reference2ReferenceOpenHashMap<T, U> valueMap;
42  
43      /**
44       * Creates a new instance.
45       *
46       * @param values the option values
47       */
48      protected AbstractOptions(Iterable<? extends AbstractOptionValue<?, ?, ?>> values) {
49          requireNonNull(values, "values");
50          valueMap = init(values);
51      }
52  
53      /**
54       * Creates a new instance.
55       *
56       * @param baseOptions the base options to merge
57       * @param additionalValues the option values
58       */
59      protected AbstractOptions(AbstractOptions<T, U> baseOptions,
60                                Iterable<? extends AbstractOptionValue<?, ?, ?>> additionalValues) {
61          requireNonNull(baseOptions, "baseOptions");
62          requireNonNull(additionalValues, "additionalValues");
63  
64          if (baseOptions.valueMap.isEmpty()) {
65              valueMap = init(additionalValues);
66          } else {
67              valueMap = init(baseOptions, additionalValues);
68          }
69      }
70  
71      private Reference2ReferenceOpenHashMap<T, U> init(
72              Iterable<? extends AbstractOptionValue<?, ?, ?>> additionalValues) {
73  
74          if (additionalValues instanceof AbstractOptions) {
75              @SuppressWarnings("unchecked")
76              final Reference2ReferenceOpenHashMap<T, U> clone =
77                      ((AbstractOptions<T, U>) additionalValues).valueMap.clone();
78              return clone;
79          }
80  
81          final Reference2ReferenceOpenHashMap<T, U> map = new Reference2ReferenceOpenHashMap<>();
82          for (final AbstractOptionValue<?, ?, ?> value : additionalValues) {
83              @SuppressWarnings("unchecked")
84              final U newValue = (U) value;
85              map.put(newValue.option(), newValue);
86          }
87          return map;
88      }
89  
90      private Reference2ReferenceOpenHashMap<T, U> init(
91              AbstractOptions<T, U> baseOptions,
92              Iterable<? extends AbstractOptionValue<?, ?, ?>> additionalValues) {
93  
94          // Use cheaper Iterable if possible.
95          if (additionalValues instanceof AbstractOptions) {
96              @SuppressWarnings("unchecked")
97              final Reference2ReferenceOpenHashMap<T, U> additionalValueMap =
98                      ((AbstractOptions<T, U>) additionalValues).valueMap;
99              additionalValues = additionalValueMap.values();
100         }
101 
102         // Merge all options.
103         final Reference2ReferenceOpenHashMap<T, U> map = baseOptions.valueMap.clone();
104         for (final AbstractOptionValue<?, ?, ?> value : additionalValues) {
105             @SuppressWarnings("unchecked")
106             final U newValue = (U) value;
107             final T option = newValue.option();
108             final U oldValue = map.putIfAbsent(option, newValue);
109             if (oldValue != null) {
110                 map.put(option, option.merge(oldValue, newValue));
111             }
112         }
113         return map;
114     }
115 
116     /**
117      * Returns the value of the specified {@code option}.
118      *
119      * @param <V> the type of the value
120      */
121     public final <V> V get(AbstractOption<?, ?, V> option) {
122         requireNonNull(option, "option");
123         final U optionValue = valueMap.get(option);
124         if (optionValue == null) {
125             return option.defaultValue();
126         }
127 
128         @SuppressWarnings("unchecked")
129         final V cast = (V) optionValue.value();
130         return cast;
131     }
132 
133     /**
134      * Returns an immutable {@link Iterator} of user-specified options.
135      */
136     @Override
137     public Iterator<U> iterator() {
138         return Iterators.unmodifiableIterator(valueMap.values().iterator());
139     }
140 
141     /**
142      * Returns an immutable {@link Map} of user-specified options.
143      */
144     public final Map<T, U> asMap() {
145         return Collections.unmodifiableMap(valueMap);
146     }
147 
148     @Override
149     public String toString() {
150         return getClass().getSimpleName() + Iterators.toString(valueMap.values().iterator());
151     }
152 }
