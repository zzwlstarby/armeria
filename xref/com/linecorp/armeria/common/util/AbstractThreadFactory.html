17  package com.linecorp.armeria.common.util;
18  
19  import static com.google.common.base.Preconditions.checkState;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.concurrent.ThreadFactory;
23  import java.util.function.Function;
24  
25  import javax.annotation.Nullable;
26  
27  import io.netty.util.concurrent.DefaultThreadFactory;
28  
29  /**
30   * A skeletal {@link ThreadFactory} implementation.
31   */
32  abstract class AbstractThreadFactory implements ThreadFactory {
33      // Note that we did not extend DefaultThreadFactory directly to hide it from the class hierarchy.
34      private final ThreadFactoryImpl delegate;
35      private final Function<? super Runnable, ? extends Runnable> taskFunction;
36  
37      AbstractThreadFactory(String threadNamePrefix, boolean daemon, int priority,
38                            @Nullable ThreadGroup threadGroup,
39                            Function<? super Runnable, ? extends Runnable> taskFunction) {
40          delegate = new ThreadFactoryImpl(requireNonNull(threadNamePrefix, "threadNamePrefix"),
41                                           daemon, priority, threadGroup);
42          this.taskFunction = requireNonNull(taskFunction, "taskFunction");
43      }
44  
45      @Override
46      public final Thread newThread(Runnable r) {
47          final Runnable newRunnable = taskFunction.apply(r);
48          checkState(newRunnable != null, "taskFunction.apply() returned null.");
49          return delegate.newThread(newRunnable);
50      }
51  
52      abstract Thread newThread(@Nullable ThreadGroup threadGroup, Runnable r, String name);
53  
54      private final class ThreadFactoryImpl extends DefaultThreadFactory {
55          ThreadFactoryImpl(String threadNamePrefix) {
56              super(threadNamePrefix);
57          }
58  
59          ThreadFactoryImpl(String threadNamePrefix, boolean daemon) {
60              super(threadNamePrefix, daemon);
61          }
62  
63          ThreadFactoryImpl(String threadNamePrefix, int priority) {
64              super(threadNamePrefix, priority);
65          }
66  
67          ThreadFactoryImpl(String threadNamePrefix, boolean daemon, int priority) {
68              super(threadNamePrefix, daemon, priority);
69          }
70  
71          ThreadFactoryImpl(String threadNamePrefix, boolean daemon, int priority,
72                            @Nullable ThreadGroup threadGroup) {
73              super(threadNamePrefix, daemon, priority, threadGroup);
74          }
75  
76          @Override
77          protected Thread newThread(Runnable r, String name) {
78              return AbstractThreadFactory.this.newThread(threadGroup, r, name);
79          }
80      }
81  }
