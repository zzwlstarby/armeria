43  /**
44   * Provides utilities for accessing the information about the current system and process.
45   */
46  public final class SystemInfo {
47  
48      private static final Logger logger = LoggerFactory.getLogger(SystemInfo.class);
49  
50      private static final int JAVA_VERSION;
51  
52      private static boolean JETTY_ALPN_OPTIONAL_OR_AVAILABLE;
53  
54      private static final OsType osType;
55  
56      static {
57          int javaVersion = -1;
58          try {
59              final String spec = System.getProperty("java.specification.version");
60              if (spec != null) {
61                  final String[] strValues = spec.split("\\.");
62                  final int major;
63                  final int minor;
64  
65                  switch (strValues.length) {
66                      case 0:
67                          major = 0;
68                          minor = 0;
69                          break;
70                      case 1:
71                          major = Integer.parseInt(strValues[0]);
72                          minor = 0;
73                          break;
74                      default:
75                          major = Integer.parseInt(strValues[0]);
76                          minor = Integer.parseInt(strValues[1]);
77                  }
78  
79                  if (major > 1) {
80                      javaVersion = major;
81                  } else if (major == 1) {
82                      if (minor == 0) {
83                          javaVersion = 1;
84                      } else if (minor > 0) {
85                          javaVersion = minor;
86                      }
87                  }
88              }
89  
90              if (javaVersion > 0) {
91                  logger.debug("Java version: {}", javaVersion);
92              } else {
93                  logger.warn("'java.specification.version' contains an unexpected value: {}", spec);
94              }
95          } catch (Throwable t) {
96              logger.warn("Failed to determine Java version", t);
97          }
98  
99          JAVA_VERSION = javaVersion > 0 ? javaVersion : 8;
100 
101         // ALPN check from https://github.com/netty/netty/blob/1065e0f26e0d47a67c479b0fad81efab5d9438d9/handler/src/main/java/io/netty/handler/ssl/JettyAlpnSslEngine.java
102         if (JAVA_VERSION >= 9) {
103             JETTY_ALPN_OPTIONAL_OR_AVAILABLE = true;
104         } else {
105             try {
106                 // Always use bootstrap class loader.
107                 Class.forName("sun.security.ssl.ALPNExtension", true, null);
108                 JETTY_ALPN_OPTIONAL_OR_AVAILABLE = true;
109             } catch (Throwable ignore) {
110                 // alpn-boot was not loaded.
111                 JETTY_ALPN_OPTIONAL_OR_AVAILABLE = false;
112             }
113         }
114 
115         final String osName = Ascii.toUpperCase(System.getProperty("os.name", ""));
116         if (osName.startsWith("WINDOWS")) {
117             osType = OsType.WINDOWS;
118         } else if (osName.startsWith("LINUX")) {
119             osType = OsType.LINUX;
120         } else if (osName.startsWith("MAC")) {
121             osType = OsType.MAC;
122         } else {
123             osType = OsType.OTHERS;
124         }
125     }

148     /**
149      * Returns the current process ID.
150      *
151      * @throws IllegalStateException if failed to retrieve the current process ID.
152      */
153     public static int pid() {
154         if (Pid.PID <= 0) {
155             throw new IllegalStateException("Failed to retrieve the current PID.");
156         }
157         return Pid.PID;
158     }

161      * Returns the number of microseconds since the epoch (00:00:00, 01-Jan-1970, GMT). The precision of the
162      * returned value may vary depending on {@linkplain #javaVersion() Java version}. Currently, Java 9 or
163      * above is required for microsecond precision. {@code System.currentTimeMillis() * 1000} is returned on
164      * Java 8.

186     private static final class Hostname {
187 
188         private static final Pattern HOSTNAME_PATTERN = Pattern.compile(
189                 "^(?:[-_a-zA-Z0-9]|[-_a-zA-Z0-9][-_.a-zA-Z0-9]*[-_a-zA-Z0-9])$");
190 
191         static final String HOSTNAME;
192 
193         static {
194             // Try /proc/sys/kernel/hostname on Linux.
195             String hostname = null;
196             if (isLinux()) {
197                 try {
198                     final List<String> lines = Files.readAllLines(Paths.get("/proc/sys/kernel/hostname"));
199                     if (!lines.isEmpty()) {
200                         hostname = normalizeHostname(lines.get(0));
201                     }
202                     if (hostname != null) {
203                         logger.info("Hostname: {} (from /proc/sys/kernel/hostname)", hostname);
204                     } else {
205                         logger.debug("/proc/sys/kernel/hostname does not contain a valid hostname: {}", lines);
206                     }
207                 } catch (Throwable t) {
208                     logger.debug("Failed to get the hostname from /proc/sys/kernel/hostname; " +
209                                  "using the 'hostname' command instead", t);
210                 }
211             }
212 
213             // Try /usr/bin/hostname.
214             if (hostname == null) {
215                 Process process = null;
216                 try {
217                     process = Runtime.getRuntime().exec("hostname");
218                     final BufferedReader in = new BufferedReader(
219                             new InputStreamReader(process.getInputStream()));
220                     final String line = in.readLine();
221                     if (line != null) {
222                         hostname = normalizeHostname(line);
223                     }
224 
225                     if (hostname != null) {
226                         logger.info("Hostname: {} (from 'hostname' command)", hostname);
227                     } else {
228                         logger.debug("The 'hostname' command returned a non-hostname ({}); " +
229                                      "using InetAddress.getLocalHost() instead", line);
230                     }
231                 } catch (Throwable t) {
232                     logger.debug("Failed to get the hostname using the 'hostname' command; " +
233                                  "using InetAddress.getLocalHost() instead", t);
234                 } finally {
235                     if (process != null) {
236                         process.destroy();
237                     }
238                 }
239             }
240 
241             if (hostname == null) {
242                 try {
243                     final String jdkHostname = InetAddress.getLocalHost().getHostName();
244                     hostname = normalizeHostname(jdkHostname);
245                     if (hostname == null) {
246                         logger.warn("InetAddress.getLocalHost() returned an invalid hostname ({}); " +
247                                     "using 'localhost' instead", jdkHostname);
248                     } else {
249                         logger.info("Hostname: {} (from InetAddress.getLocalHost())", hostname);
250                     }
251                 } catch (Throwable t) {
252                     logger.warn("Failed to get the hostname using InetAddress.getLocalHost(); " +
253                                 "using 'localhost' instead", t);
254                 }
255             }
256 
257             HOSTNAME = firstNonNull(hostname, "localhost");
258         }
259 
260         @Nullable
261         private static String normalizeHostname(String line) {
262             final String hostname = IDN.toASCII(line.trim(), IDN.ALLOW_UNASSIGNED);
263             if (!HOSTNAME_PATTERN.matcher(hostname).matches()) {
264                 return null;
265             }
266             return Ascii.toLowerCase(hostname);
267         }
268     }

270     private static final class Pid {
271 
272         static final int PID;
273 
274         static {
275             int pid = -1;
276             // Try ProcessHandle.pid() if Java 9+.
277             if (javaVersion() >= 9) {
278                 try {
279                     final Class<?> handleClass = Class.forName("java.lang.ProcessHandle", true,
280                                                                Process.class.getClassLoader());
281                     final Method currentMethod = handleClass.getDeclaredMethod("current");
282                     final Method pidMethod = handleClass.getDeclaredMethod("pid");
283                     final Object currentHandle = currentMethod.invoke(null);
284                     final Object result = pidMethod.invoke(currentHandle);
285                     pid = validatePid(result);
286                     if (pid <= 0) {
287                         logger.warn("ProcessHandle.pid() returned an invalid PID: {}", result);
288                     } else {
289                         logger.info("PID: {} (from ProcessHandle.pid())", pid);
290                     }
291                 } catch (Throwable t) {
292                     logFailure("ProcessHandle.current()", true, t);
293                 }
294             }
295 
296             // Try sun.management.VMManagement.getProcessId().
297             if (pid <= 0) {
298                 try {
299                     final RuntimeMXBean runtime = ManagementFactory.getRuntimeMXBean();
300                     final Field jvmField = runtime.getClass().getDeclaredField("jvm");
301                     jvmField.setAccessible(true);
302                     final Object jvm = jvmField.get(runtime);
303                     final Method getProcessIdMethod = jvm.getClass().getDeclaredMethod("getProcessId");
304                     getProcessIdMethod.setAccessible(true);
305                     final Object result = getProcessIdMethod.invoke(jvm);
306                     pid = validatePid(result);
307                     if (pid <= 0) {
308                         logger.warn("VMManagement.getProcessId() returned an invalid PID: {}", result);
309                     } else {
310                         logger.info("PID: {} (from VMManagement.getProcessId())", pid);
311                     }
312                 } catch (Throwable t) {
313                     logFailure("VMManagement.getProcessId()", false, t);
314                 }
315             }
316 
317             // Try /proc/self (Linux only)
318             if (pid <= 0 && isLinux()) {
319                 try {
320                     final Path path = Paths.get("/proc/self");
321                     if (Files.isSymbolicLink(path)) {
322                         final Path realPath = path.toRealPath();
323                         pid = validatePid(Integer.parseInt(realPath.getFileName().toString()));
324                         if (pid <= 0) {
325                             logger.warn("/proc/self does not refer to a PID-named file: {}", realPath);
326                         } else {
327                             logger.info("PID: {} (from /proc/self)", pid);
328                         }
329                     }
330                 } catch (Throwable t) {
331                     logFailure("/proc/self", false, t);
332                 }
333             }
334 
335             // Try RuntimeMXBean.getName() as the last resort.
336             if (pid <= 0) {
337                 try {
338                     final String result = ManagementFactory.getRuntimeMXBean().getName();
339                     final String[] values = result.split("@");
340                     pid = validatePid(values.length > 0 ? Integer.parseInt(values[0]) : -1);
341                     if (pid <= 0) {
342                         logger.warn("RuntimeMXBean.getName() returned an unexpected value: {}", result);
343                     } else {
344                         logger.info("PID: {} (from RuntimeMXBean.getName())", pid);
345                     }
346                 } catch (Throwable t) {
347                     logFailure("RuntimeMXBean.getName()", true, t);
348                 }
349             }
350 
351             PID = pid;
352         }
353 
354         private static int validatePid(@Nullable Object value) {
355             if (!(value instanceof Number)) {
356                 return -1;
357             } else {
358                  final int pid = ((Number) value).intValue();
359                  return pid > 0 ? pid : -1;
360             }
361         }
362 
363         private static void logFailure(String method, boolean warn, Throwable cause) {
364             cause = Exceptions.peel(cause);
365             if (cause instanceof UnsupportedOperationException ||
366                 cause instanceof SecurityException ||
367                 cause instanceof IllegalAccessException) {
368                 logger.debug("An access to {} not possible due to platform restriction:", method, cause);
369                 return;
370             }
371 
372             final String msg = "Failed to retrieve the current PID from {}:";
373             if (warn) {
374                 logger.warn(msg, method, cause);
375             } else {
376                 logger.debug(msg, method, cause);
377             }
378         }
