30  package com.linecorp.armeria.common.util;
31  
32  import static com.google.common.base.Preconditions.checkArgument;
33  
34  import java.util.concurrent.TimeUnit;
35  import java.util.concurrent.atomic.AtomicInteger;
36  import java.util.concurrent.atomic.AtomicLong;
37  
38  import com.google.common.annotations.VisibleForTesting;
39  
40  /**
41   * The rate-limited sampler allows you to choose an amount of traces to accept on a per-second
42   * interval. The minimum number is 0 and the max is 2,147,483,647 (max int).
43   *
44   * <p>For example, to allow 10 traces per second, you'd initialize the following:
45   * <pre>{@code
46   * tracingBuilder.sampler(RateLimitingSampler.create(10));
47   * }</pre>
48   *
49   * <h3>Appropriate Usage</h3>
50   *
51   * <p>If the rate is 10 or more traces per second, an attempt is made to distribute the accept
52   * decisions equally across the second. For example, if the rate is 100, 10 will pass every
53   * decisecond as opposed to bunching all pass decisions at the beginning of the second.
54   *
55   * <p>However, this sampler is insensitive to the trace ID and will operate correctly even if they are
56   * not perfectly random.
57   *
58   * <h3>Implementation</h3>
59   *
60   * <p>The implementation uses {@link System#nanoTime} and tracks how many yes decisions occur
61   * across a second window. When the rate is at least 10/s, the yes decisions are equally split over
62   * 10 deciseconds, allowing a roll-over of unused yes decisions up until the end of the second.
63   *
64   * <p>Forked from brave-core 5.6.9 at b8c00c594cbf75a33788d3dc990f94b9c6f41c01
65   */
66  final class RateLimitingSampler<T> implements Sampler<T> {
67  
68      static <T> Sampler<T> create(int samplesPerSecond) {
69          checkArgument(samplesPerSecond >= 0,
70                        "samplesPerSecond: %s (expected: >= 0)", samplesPerSecond);
71          if (samplesPerSecond == 0) {
72              return Sampler.never();
73          }
74          return new RateLimitingSampler<>(samplesPerSecond);
75      }
76  
77      static final long NANOS_PER_SECOND = TimeUnit.SECONDS.toNanos(1);
78      static final long NANOS_PER_DECISECOND = NANOS_PER_SECOND / 10;
79  
80      @VisibleForTesting
81      final MaxFunction maxFunction;
82      private final AtomicInteger usage = new AtomicInteger();
83      private final AtomicLong nextReset;
84  
85      RateLimitingSampler(int samplesPerSecond) {
86          maxFunction =
87                  samplesPerSecond < 10 ? new LessThan10(samplesPerSecond) : new AtLeast10(samplesPerSecond);
88          final long now = System.nanoTime();
89          nextReset = new AtomicLong(now + NANOS_PER_SECOND);
90      }
91  
92      @Override
93      public boolean isSampled(Object ignored) {
94          final long now = System.nanoTime();
95          final long updateAt = nextReset.get();
96  
97          // First task is to determine if this request is later than the one second sampling window
98          final long nanosUntilReset = -(now - updateAt); // because nanoTime can be negative
99          if (nanosUntilReset <= 0) {
100             // Attempt to move into the next sampling interval.
101             // nanosUntilReset is now invalid regardless of race winner, so we can't sample based on it.
102             if (nextReset.compareAndSet(updateAt, now + NANOS_PER_SECOND)) {
103                 usage.set(0);
104             }
105 
106             // recurse as it is simpler than resetting all the locals.
107             // reset happens once per second, this code doesn't take a second, so no infinite recursion.
108             return isSampled(ignored);
109         }
110 
111         // Now, we determine the amount of samples allowed for this interval, and sample accordingly
112         final int max = maxFunction.max(nanosUntilReset);
113         int prev;
114         int next;
115         do { // same form as java 8 AtomicLong.getAndUpdate
116             prev = usage.get();
117             next = prev + 1;
118             if (next > max) {
119                 return false;
120             }
121         } while (!usage.compareAndSet(prev, next));
122         return true;
123     }
124 
125     @Override
126     public String toString() {
127         return "RateLimitingSampler()";
128     }
129 
130     private abstract static class MaxFunction {
131         abstract int max(long nanosUntilReset);
132     }
133 
134     /**
135      * For a reservoir of less than 10, we permit draining it completely at any time in the second.
136      */
137     @VisibleForTesting
138     static final class LessThan10 extends MaxFunction {
139         final int samplesPerSecond;
140 
141         LessThan10(int samplesPerSecond) {
142             this.samplesPerSecond = samplesPerSecond;
143         }
144 
145         @Override
146         int max(long nanosUntilResetIgnored) {
147             return samplesPerSecond;
148         }
149     }
150 
151     /**
152      * For a reservoir of at least 10, we permit draining up to a decisecond watermark. Because the
153      * rate could be odd, we may have a remainder, which is arbitrarily available. We allow any
154      * remainders in the 1st decisecond or any time thereafter.
155      *
156      * <p>Ex. If the rate is 10/s then you can use 1 in the first decisecond, another 1 in the 2nd,
157      * or up to 10 by the last.
158      *
159      * <p>Ex. If the rate is 103/s then you can use 13 in the first decisecond, another 10 in the
160      * 2nd, or up to 103 by the last.
161      */
162     private static final class AtLeast10 extends MaxFunction {
163         final int[] max;
164 
165         AtLeast10(int samplesPerSecond) {
166             final int samplesPerDecisecond = samplesPerSecond / 10;
167             final int remainder = samplesPerSecond % 10;
168             max = new int[10];
169             max[0] = samplesPerDecisecond + remainder;
170             for (int i = 1; i < 10; i++) {
171                 max[i] = max[i - 1] + samplesPerDecisecond;
172             }
173         }
174 
175         @Override
176         int max(long nanosUntilReset) {
177             // Check to see if we are in the first or last interval
178             if (nanosUntilReset > NANOS_PER_SECOND - NANOS_PER_DECISECOND) {
179                 return max[0];
180             }
181             if (nanosUntilReset < NANOS_PER_DECISECOND) {
182                 return max[9];
183             }
184 
185             // Choose a slot based on the remaining deciseconds
186             final int decisecondsUntilReset = (int) (nanosUntilReset / NANOS_PER_DECISECOND);
187             return max[10 - decisecondsUntilReset];
188         }
189     }
