35  /**
36   * Provides asynchronous start-stop life cycle support.
37   *
38   * @param <T> the type of the startup argument. Use {@link Void} if unused.
39   * @param <U> the type of the shutdown argument. Use {@link Void} if unused.
40   * @param <V> the type of the startup result. Use {@link Void} if unused.
41   * @param <L> the type of the life cycle event listener. Use {@link Void} if unused.
42   */
43  public abstract class StartStopSupport<T, U, V, L> implements ListenableAsyncCloseable {
44  
45      private static final Logger logger = LoggerFactory.getLogger(StartStopSupport.class);
46  
47      enum State {
48          STARTING,
49          STARTED,
50          STOPPING,
51          STOPPED
52      }
53  
54      private final Executor executor;
55      private final List<L> listeners = new CopyOnWriteArrayList<>();
56      private final AsyncCloseableSupport closeable = AsyncCloseableSupport.of(this::closeAsync);
57  
58      private volatile State state = State.STOPPED;
59  
60      /**
61       * This future is {@code V}-typed when STARTING/STARTED and {@link Void}-typed when STOPPING/STOPPED.
62       */
63      private UnmodifiableFuture<?> future = completedFuture(null);
64  
65      /**
66       * Creates a new instance.
67       *
68       * @param executor the {@link Executor} which will be used for invoking the extension points of this class:
69       *                 <ul>
70       *                   <li>{@link #doStart(Object)}</li>
71       *                   <li>{@link #doStop(Object)}</li>
72       *                   <li>{@link #rollbackFailed(Throwable)}</li>
73       *                   <li>{@link #notificationFailed(Object, Throwable)}</li>
74       *                   <li>All listener notifications</li>
75       *                 </ul>
76       *                 .. except {@link #closeFailed(Throwable)} which is invoked at the caller thread.
77       */
78      protected StartStopSupport(Executor executor) {
79          this.executor = requireNonNull(executor, "executor");
80      }
81  
82      /**
83       * Adds the specified {@code listener}, so that it is notified when the state of this
84       * {@link StartStopSupport} changes.
85       */
86      public final void addListener(L listener) {
87          listeners.add(requireNonNull(listener, "listener"));
88      }
89  
90      /**
91       * Removes the specified {@code listener}, so that it is not notified anymore.
92       */
93      public final boolean removeListener(L listener) {
94          return listeners.remove(requireNonNull(listener, "listener"));
95      }

97      /**
98       * Begins the startup procedure without an argument by calling {@link #doStart(Object)}, ensuring that
99       * neither {@link #doStart(Object)} nor {@link #doStop(Object)} is invoked concurrently. When the startup
100      * fails, {@link #stop()} will be invoked automatically to roll back the side effect caused by this method
101      * and any exceptions that occurred during the rollback will be reported to
102      * {@link #rollbackFailed(Throwable)}. This method is a shortcut for
103      * {@code start(null, null, failIfStarted)}.
104      *
105      * @param failIfStarted whether to fail the returned {@link CompletableFuture} with
106      *                      an {@link IllegalStateException} when the startup procedure is already
107      *                      in progress or done
108      */
109     public final CompletableFuture<V> start(boolean failIfStarted) {
110         return start(null, null, failIfStarted);
111     }

113     /**
114      * Begins the startup procedure without an argument by calling {@link #doStart(Object)}, ensuring that
115      * neither {@link #doStart(Object)} nor {@link #doStop(Object)} is invoked concurrently. When the startup
116      * fails, {@link #stop()} will be invoked automatically to roll back the side effect caused by this method
117      * and any exceptions that occurred during the rollback will be reported to
118      * {@link #rollbackFailed(Throwable)}. This method is a shortcut for
119      * {@code start(arg, null, failIfStarted)}.
120      *
121      * @param arg           the argument to pass to {@link #doStart(Object)},
122      *                      or {@code null} to pass no argument.
123      * @param failIfStarted whether to fail the returned {@link CompletableFuture} with
124      *                      an {@link IllegalStateException} when the startup procedure is already
125      *                      in progress or done
126      */
127     public final CompletableFuture<V> start(@Nullable T arg, boolean failIfStarted) {
128         return start(arg, null, failIfStarted);
129     }
130 
131     /**
132      * Begins the startup procedure by calling {@link #doStart(Object)}, ensuring that neither
133      * {@link #doStart(Object)} nor {@link #doStop(Object)} is invoked concurrently. When the startup fails,
134      * {@link #stop(Object)} will be invoked with the specified {@code rollbackArg} automatically to roll back
135      * the side effect caused by this method and any exceptions that occurred during the rollback will be
136      * reported to {@link #rollbackFailed(Throwable)}.
137      *
138      * @param arg           the argument to pass to {@link #doStart(Object)},
139      *                      or {@code null} to pass no argument.
140      * @param rollbackArg   the argument to pass to {@link #doStop(Object)} when rolling back.
141      * @param failIfStarted whether to fail the returned {@link CompletableFuture} with
142      *                      an {@link IllegalStateException} when the startup procedure is already
143      *                      in progress or done
144      */
145     public final CompletableFuture<V> start(@Nullable T arg, @Nullable U rollbackArg, boolean failIfStarted) {
146         return start0(arg, rollbackArg, failIfStarted);
147     }
148 
149     private synchronized UnmodifiableFuture<V> start0(@Nullable T arg,
150                                                       @Nullable U rollbackArg,
151                                                       boolean failIfStarted) {
152         if (closeable.isClosing()) {
153             return exceptionallyCompletedFuture(new IllegalStateException("closed already"));
154         }
155 
156         switch (state) {
157             case STARTING:
158             case STARTED:
159                 if (failIfStarted) {
160                     return exceptionallyCompletedFuture(
161                             new IllegalStateException("must be stopped to start; currently " + state));
162                 } else {
163                     @SuppressWarnings("unchecked")
164                     final UnmodifiableFuture<V> castFuture = (UnmodifiableFuture<V>) future;
165                     return castFuture;
166                 }
167             case STOPPING:
168                 // A user called start() to restart, but not stopped completely yet.
169                 // Try again once stopped.
170                 return UnmodifiableFuture.wrap(
171                         future.exceptionally(unused -> null)
172                               .thenComposeAsync(unused -> start(arg, failIfStarted), executor));
173         }
174 
175         assert state == State.STOPPED : "state: " + state;
176         state = State.STARTING;
177 
178         // Attempt to start.
179         final CompletableFuture<V> startFuture = new CompletableFuture<>();
180         boolean submitted = false;
181         try {
182             executor.execute(() -> {
183                 try {
184                     notifyListeners(State.STARTING, arg, null, null);
185                     final CompletionStage<V> f = doStart(arg);
186                     checkState(f != null, "doStart() returned null.");
187 
188                     f.handle((result, cause) -> {
189                         if (cause != null) {
190                             startFuture.completeExceptionally(cause);
191                         } else {
192                             startFuture.complete(result);
193                         }
194                         return null;
195                     });
196                 } catch (Exception e) {
197                     startFuture.completeExceptionally(e);
198                 }
199             });
200             submitted = true;
201         } catch (Exception e) {
202             return exceptionallyCompletedFuture(e);
203         } finally {
204             if (!submitted) {
205                 state = State.STOPPED;
206             }
207         }
208 
209         final UnmodifiableFuture<V> future = UnmodifiableFuture.wrap(
210                 startFuture.handleAsync((result, cause) -> {
211                     if (cause != null) {
212                         // Failed to start. Stop and complete with the start failure cause.
213                         final CompletableFuture<Void> rollbackFuture =
214                                 stop(rollbackArg, true).exceptionally(stopCause -> {
215                                     rollbackFailed(Exceptions.peel(stopCause));
216                                     return null;
217                                 });
218 
219                         return rollbackFuture.<V>thenCompose(unused -> exceptionallyCompletedFuture(cause));
220                     } else {
221                         enter(State.STARTED, arg, null, result);
222                         return completedFuture(result);
223                     }
224                 }, executor).thenCompose(Function.identity()));
225 
226         this.future = future;
227         return future;
228     }

230     /**
231      * Begins the shutdown procedure without an argument by calling {@link #doStop(Object)}, ensuring that
232      * neither {@link #doStart(Object)} nor {@link #doStop(Object)} is invoked concurrently. This method is
233      * a shortcut of {@code stop(null)}.
234      */
235     public final CompletableFuture<Void> stop() {
236         return stop(null);
237     }

239     /**
240      * Begins the shutdown procedure by calling {@link #doStop(Object)}, ensuring that neither
241      * {@link #doStart(Object)} nor {@link #doStop(Object)} is invoked concurrently.
242      *
243      * @param arg the argument to pass to {@link #doStop(Object)}, or {@code null} to pass no argument.
244      */
245     public final CompletableFuture<Void> stop(@Nullable U arg) {
246         return stop(arg, false);
247     }

249     private CompletableFuture<Void> stop(@Nullable U arg, boolean rollback) {
250         return stop0(arg, rollback);
251     }
252 
253     private synchronized UnmodifiableFuture<Void> stop0(@Nullable U arg, boolean rollback) {
254         switch (state) {
255             case STARTING:
256                 if (!rollback) {
257                     // Try again once started.
258                     return UnmodifiableFuture.wrap(
259                             future.exceptionally(unused -> null) // Ignore the exception.
260                                   .thenComposeAsync(unused -> stop(arg), executor));
261                 } else {
262                     break;
263                 }
264             case STOPPING:
265             case STOPPED:
266                 @SuppressWarnings("unchecked")
267                 final UnmodifiableFuture<Void> castFuture =
268                         (UnmodifiableFuture<Void>) future;
269                 return castFuture;
270         }
271 
272         assert state == State.STARTED || rollback : "state: " + state + ", rollback: " + rollback;
273         final State oldState = state;
274         state = State.STOPPING;
275 
276         final CompletableFuture<Void> stopFuture = new CompletableFuture<>();
277         boolean submitted = false;
278         try {
279             executor.execute(() -> {
280                 try {
281                     notifyListeners(State.STOPPING, null, arg, null);
282                     final CompletionStage<Void> f = doStop(arg);
283                     checkState(f != null, "doStop() returned null.");
284 
285                     f.handle((unused, cause) -> {
286                         if (cause != null) {
287                             stopFuture.completeExceptionally(cause);
288                         } else {
289                             stopFuture.complete(null);
290                         }
291                         return null;
292                     });
293                 } catch (Exception e) {
294                     stopFuture.completeExceptionally(e);
295                 }
296             });
297             submitted = true;
298         } catch (Exception e) {
299             return exceptionallyCompletedFuture(e);
300         } finally {
301             if (!submitted) {
302                 state = oldState;
303             }
304         }
305 
306         final UnmodifiableFuture<Void> future = UnmodifiableFuture.wrap(
307                 stopFuture.whenCompleteAsync((unused1, cause) -> enter(State.STOPPED, null, arg, null),
308                                              executor));
309         this.future = future;
310         return future;
311     }

333     private void closeAsync(CompletableFuture<?> future) {
334         stop(null).handle((result, cause) -> {
335             if (cause != null) {
336                 future.completeExceptionally(cause);
337             } else {
338                 future.complete(null);
339             }
340             return null;
341         });
342     }

344     /**
345      * A synchronous version of {@link #stop(Object)}. Exceptions occurred during shutdown are reported to
346      * {@link #closeFailed(Throwable)}. No argument (i.e. {@code null}) is passed.
347      */
348     @Override
349     public final void close() {
350         try {
351             closeable.close();
352         } catch (Throwable e) {
353             closeFailed(Exceptions.peel(e));
354         }
355     }

357     private void enter(State state, @Nullable T startArg, @Nullable U stopArg, @Nullable V startResult) {
358         synchronized (this) {
359             assert this.state != state : "transition to the same state: " + state;
360             this.state = state;
361         }
362         notifyListeners(state, startArg, stopArg, startResult);
363     }
364 
365     private void notifyListeners(State state, @Nullable T startArg, @Nullable U stopArg,
366                                  @Nullable V startResult) {
367         for (L l : listeners) {
368             try {
369                 switch (state) {
370                     case STARTING:
371                         notifyStarting(l, startArg);
372                         break;
373                     case STARTED:
374                         notifyStarted(l, startArg, startResult);
375                         break;
376                     case STOPPING:
377                         notifyStopping(l, stopArg);
378                         break;
379                     case STOPPED:
380                         notifyStopped(l, stopArg);
381                         break;
382                     default:
383                         throw new Error("unknown state: " + state);
384                 }
385             } catch (Exception cause) {
386                 notificationFailed(l, cause);
387             }
388         }
389     }

394      * @param arg the argument passed from {@link #start(Object, boolean)},

399     /**
400      * Invoked by {@link #stop(Object)} to perform the actual startup, or indirectly by
401      * {@link #start(Object, boolean)} when startup failed.

404      *            or {@code null} if no argument was specified.
405      */
406     protected abstract CompletionStage<Void> doStop(@Nullable U arg) throws Exception;

408     /**
409      * Invoked when the startup procedure begins.

417     /**
418      * Invoked when the startup procedure is finished.

443     protected void notifyStopped(L listener, @Nullable U arg) throws Exception {}

445     /**
446      * Invoked when failed to stop during the rollback after startup failure.
447      */
448     protected void rollbackFailed(Throwable cause) {
449         logStopFailure(cause);
450     }
451 
452     /**
453      * Invoked when an event listener raises an exception.
454      */
455     protected void notificationFailed(L listener, Throwable cause) {
456         logger.warn("Failed to notify a listener: {}", listener, cause);
457     }
458 
459     /**
460      * Invoked when failed to stop in {@link #close()}.
461      */
462     protected void closeFailed(Throwable cause) {
463         logStopFailure(cause);
464     }
465 
466     private static void logStopFailure(Throwable cause) {
467         logger.warn("Failed to stop: {}", cause.getMessage(), cause);
468     }
