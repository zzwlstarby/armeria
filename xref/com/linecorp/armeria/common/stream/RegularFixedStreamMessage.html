17  package com.linecorp.armeria.common.stream;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import org.reactivestreams.Subscriber;
22  
23  import com.linecorp.armeria.common.util.UnstableApi;
24  
25  import io.netty.util.ReferenceCountUtil;
26  
27  /**
28   * A {@link FixedStreamMessage} that publishes an arbitrary number of objects. It is recommended to use
29   * {@link EmptyFixedStreamMessage}, {@link OneElementFixedStreamMessage}, or
30   * {@link TwoElementFixedStreamMessage} when publishing less than three objects.
31   */
32  @UnstableApi
33  public class RegularFixedStreamMessage<T> extends FixedStreamMessage<T> {
34  
35      private final T[] objs;
36  
37      private int fulfilled;
38  
39      private boolean inOnNext;
40  
41      /**
42       * Creates a new instance with the specified elements.
43       */
44      protected RegularFixedStreamMessage(T[] objs) {
45          requireNonNull(objs, "objs");
46          for (int i = 0; i < objs.length; i++) {
47              if (objs[i] == null) {
48                  throw new NullPointerException("objs[" + i + "] is null");
49              }
50          }
51  
52          this.objs = objs.clone();
53      }
54  
55      @Override
56      final void cleanupObjects() {
57          while (fulfilled < objs.length) {
58              final T obj = objs[fulfilled];
59              objs[fulfilled++] = null;
60              try {
61                  onRemoval(obj);
62              } finally {
63                  ReferenceCountUtil.safeRelease(obj);
64              }
65          }
66      }
67  
68      @Override
69      final void doRequest(SubscriptionImpl subscription, long n) {
70          final int oldDemand = requested();
71          if (oldDemand >= objs.length) {
72              // Already enough demand to finish the stream so don't need to do anything.
73              return;
74          }
75          // As objs.length is fixed, we can safely cap the demand to it here.
76          if (n >= objs.length) {
77              setRequested(objs.length);
78          } else {
79              // As objs.length is an int, large demand will always fall into the above branch and there is no
80              // chance of overflow, so just simply add the demand.
81              setRequested((int) Math.min(oldDemand + n, objs.length));
82          }
83          if (requested() > oldDemand) {
84              doNotify(subscription);
85          }
86      }
87  
88      private void doNotify(SubscriptionImpl subscription) {
89          if (inOnNext) {
90              // Do not let Subscriber.onNext() reenter, because it can lead to weird-looking event ordering
91              // for a Subscriber implemented like the following:
92              //
93              //   public void onNext(Object e) {
94              //       subscription.request(1);
95              //       ... Handle 'e' ...
96              //   }
97              //
98              // Note that we do not call this method again, because we are already in the notification loop
99              // and it will consume the element we've just added in addObjectOrEvent() from the queue as
100             // expected.
101             //
102             // We do not need to worry about synchronizing the access to 'inOnNext' because the subscriber
103             // methods must be on the same thread, or synchronized, according to Reactive Streams spec.
104             return;
105         }
106 
107         final Subscriber<Object> subscriber = subscription.subscriber();
108         for (;;) {
109             if (closeEvent() != null) {
110                 cleanup(subscription);
111                 return;
112             }
113 
114             if (fulfilled == objs.length) {
115                 notifySubscriberOfCloseEvent(subscription, SUCCESSFUL_CLOSE);
116                 return;
117             }
118 
119             final int requested = requested();
120 
121             if (fulfilled == requested) {
122                 break;
123             }
124 
125             while (fulfilled < requested) {
126                 if (closeEvent() != null) {
127                     cleanup(subscription);
128                     return;
129                 }
130 
131                 T o = objs[fulfilled];
132                 objs[fulfilled++] = null;
133                 o = prepareObjectForNotification(subscription, o);
134                 inOnNext = true;
135                 try {
136                     subscriber.onNext(o);
137                 } finally {
138                     inOnNext = false;
139                 }
140             }
141         }
142     }
143 
144     @Override
145     public boolean isEmpty() {
146         return false;
147     }
148 }
