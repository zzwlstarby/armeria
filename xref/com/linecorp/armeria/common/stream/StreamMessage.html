37  /**
38   * A variant of <a href="http://www.reactive-streams.org/">Reactive Streams</a> {@link Publisher}, which allows
39   * only one {@link Subscriber}. Unlike a usual {@link Publisher}, a {@link StreamMessage} can stream itself
40   * only once. It has the following additional operations on top of what the Reactive Streams API provides:
41   * <ul>
42   *   <li>{@link #isOpen()}</li>
43   *   <li>{@link #isEmpty()}</li>
44   *   <li>{@link #whenComplete()}</li>
45   *   <li>{@link #abort()}</li>
46   * </ul>
47   *
48   * <h3>When is a {@link StreamMessage} fully consumed?</h3>
49   *
50   * <p>A {@link StreamMessage} is <em>complete</em> (or 'fully consumed') when:
51   * <ul>
52   *   <li>the {@link Subscriber} consumes all elements and {@link Subscriber#onComplete()} is invoked,</li>
53   *   <li>an error occurred and {@link Subscriber#onError(Throwable)} is invoked,</li>
54   *   <li>the {@link Subscription} has been cancelled or</li>
55   *   <li>{@link #abort()} has been requested.</li>
56   * </ul>
57   *
58   * <p>When fully consumed, the {@link CompletableFuture} returned by {@link StreamMessage#whenComplete()}
59   * will complete, which you may find useful because {@link Subscriber} does not notify you when a stream is
60   * {@linkplain Subscription#cancel() cancelled}.
61   *
62   * <h3>Publication and Consumption of {@link ReferenceCounted} objects</h3>
63   *
64   * <p>{@link StreamMessage} will reject the publication request of a {@link ReferenceCounted} object except
65   * {@link ByteBuf} and {@link ByteBufHolder}.
66   *
67   * <p>{@link StreamMessage} will discard the publication request of a {@link ByteBuf} or a {@link ByteBufHolder}
68   * silently and release it automatically when the publication is attempted after the stream is closed.
69   *
70   * <p>For {@link ByteBuf} and {@link ByteBufHolder}, {@link StreamMessage} will convert them into their
71   * respective unpooled versions that never leak, so that the {@link Subscriber} does not need to worry about
72   * leaks.
73   *
74   * <p>If a {@link Subscriber} does not want a {@link StreamMessage} to make a copy of a {@link ByteBufHolder},
75   * specify {@link SubscriptionOption#WITH_POOLED_OBJECTS} when you subscribe. Note that the {@link Subscriber}
76   * is responsible for releasing the objects given with {@link Subscriber#onNext(Object)}.
77   *
78   * <p>{@link Subscriber#onError(Throwable)} is invoked when any exception is raised except the
79   * {@link CancelledSubscriptionException} which is caused by {@link Subscription#cancel()}. If you want your
80   * {@link Subscriber} get notified by {@link Subscriber#onError(Throwable)} when {@link Subscription#cancel()}
81   * is called, specify {@link SubscriptionOption#NOTIFY_CANCELLATION} when you subscribe.
82   *
83   * @param <T> the type of element signaled
84   */
85  public interface StreamMessage<T> extends Publisher<T> {
86      /**
87       * Creates a new {@link StreamMessage} that will publish no objects, just a close event.
88       */
89      static <T> StreamMessage<T> of() {
90          return new EmptyFixedStreamMessage<>();
91      }
92  
93      /**
94       * Creates a new {@link StreamMessage} that will publish the single {@code obj}.
95       */
96      static <T> StreamMessage<T> of(T obj) {
97          requireNonNull(obj, "obj");
98          return new OneElementFixedStreamMessage<>(obj);
99      }
100 
101     /**
102      * Creates a new {@link StreamMessage} that will publish the two {@code obj1} and {@code obj2}.
103      */
104     static <T> StreamMessage<T> of(T obj1, T obj2) {
105         requireNonNull(obj1, "obj1");
106         requireNonNull(obj2, "obj2");
107         return new TwoElementFixedStreamMessage<>(obj1, obj2);
108     }
109 
110     /**
111      * Creates a new {@link StreamMessage} that will publish the given {@code objs}.
112      */
113     @SafeVarargs
114     static <T> StreamMessage<T> of(T... objs) {
115         requireNonNull(objs, "objs");
116         switch (objs.length) {
117             case 0:
118                 return of();
119             case 1:
120                 return of(objs[0]);
121             case 2:
122                 return of(objs[0], objs[1]);
123             default:
124                 for (int i = 0; i < objs.length; i++) {
125                     if (objs[i] == null) {
126                         throw new NullPointerException("objs[" + i + "] is null");
127                     }
128                 }
129                 return new RegularFixedStreamMessage<>(objs);
130         }
131     }
132 
133     /**
134      * Returns {@code true} if this stream is not closed yet. Note that a stream may not be
135      * {@linkplain #whenComplete() complete} even if it's closed; a stream is complete when it's fully
136      * consumed by a {@link Subscriber}.
137      */
138     boolean isOpen();
139 
140     /**
141      * Returns {@code true} if this stream has been closed and did not publish any elements.
142      * Note that this method will not return {@code true} when the stream is open even if it has not
143      * published anything so far, because it may publish something later.
144      */
145     boolean isEmpty();
146 
147     /**
148      * Returns {@code true} if this stream is complete, either successfully or exceptionally,
149      * including cancellation and abortion.
150      *
151      * <p>A {@link StreamMessage} is <em>complete</em> (or 'fully consumed') when:
152      * <ul>
153      *   <li>the {@link Subscriber} consumes all elements and {@link Subscriber#onComplete()} is invoked,</li>
154      *   <li>an error occurred and {@link Subscriber#onError(Throwable)} is invoked,</li>
155      *   <li>the {@link Subscription} has been cancelled or</li>
156      *   <li>{@link #abort()} has been requested.</li>
157      * </ul>
158      */
159     default boolean isComplete() {
160         return whenComplete().isDone();
161     }
162 
163     /**
164      * Returns a {@link CompletableFuture} that completes when this stream is complete,
165      * either successfully or exceptionally, including cancellation and abortion.
166      *
167      * <p>A {@link StreamMessage} is <em>complete</em>
168      * (or 'fully consumed') when:
169      * <ul>
170      *   <li>the {@link Subscriber} consumes all elements and {@link Subscriber#onComplete()} is invoked,</li>
171      *   <li>an error occurred and {@link Subscriber#onError(Throwable)} is invoked,</li>
172      *   <li>the {@link Subscription} has been cancelled or</li>
173      *   <li>{@link #abort()} has been requested.</li>
174      * </ul>
175      */
176     CompletableFuture<Void> whenComplete();
177 
178     /**
179      * Returns a {@link CompletableFuture} that completes when this stream is complete,
180      * either successfully or exceptionally, including cancellation and abortion.
181      *
182      * <p>A {@link StreamMessage} is <em>complete</em>
183      * (or 'fully consumed') when:
184      * <ul>
185      *   <li>the {@link Subscriber} consumes all elements and {@link Subscriber#onComplete()} is invoked,</li>
186      *   <li>an error occurred and {@link Subscriber#onError(Throwable)} is invoked,</li>
187      *   <li>the {@link Subscription} has been cancelled or</li>
188      *   <li>{@link #abort()} has been requested.</li>
189      * </ul>
190      *
191      * @deprecated Use {@link #whenComplete()}.
192      */
193     @Deprecated
194     default CompletableFuture<Void> completionFuture() {
195         return whenComplete();
196     }
197 
198     /**
199      * Returns a {@link CompletableFuture} that completes when this stream is complete,
200      * either successfully or exceptionally, including cancellation and abortion.
201      *
202      * @deprecated Use {@link #whenComplete()} instead.
203      */
204     @Deprecated
205     default CompletableFuture<Void> closeFuture() {
206         return whenComplete();
207     }
208 
209     /**
210      * Requests to start streaming data to the specified {@link Subscriber}. If there is a problem subscribing,
211      * {@link Subscriber#onError(Throwable)} will be invoked with one of the following exceptions:
212      * <ul>
213      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
214      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
215      *   <li>{@link CancelledSubscriptionException} if this stream has been
216      *       {@linkplain Subscription#cancel() cancelled} and {@link SubscriptionOption#NOTIFY_CANCELLATION} is
217      *       specified when subscribed.</li>
218      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
219      * </ul>
220      */
221     @Override
222     default void subscribe(Subscriber<? super T> subscriber) {
223         subscribe(subscriber, defaultSubscriberExecutor());
224     }
225 
226     /**
227      * Requests to start streaming data to the specified {@link Subscriber}. If there is a problem subscribing,
228      * {@link Subscriber#onError(Throwable)} will be invoked with one of the following exceptions:
229      * <ul>
230      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
231      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
232      *   <li>{@link CancelledSubscriptionException} if this stream has been
233      *       {@linkplain Subscription#cancel() cancelled} and {@link SubscriptionOption#NOTIFY_CANCELLATION} is
234      *       specified when subscribed.</li>
235      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
236      * </ul>
237      *
238      * @param options {@link SubscriptionOption}s to subscribe with
239      */
240     default void subscribe(Subscriber<? super T> subscriber, SubscriptionOption... options) {
241         subscribe(subscriber, defaultSubscriberExecutor(), options);
242     }
243 
244     /**
245      * Requests to start streaming data to the specified {@link Subscriber}. If there is a problem subscribing,
246      * {@link Subscriber#onError(Throwable)} will be invoked with one of the following exceptions:
247      * <ul>
248      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
249      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
250      *   <li>{@link CancelledSubscriptionException} if this stream has been
251      *       {@linkplain Subscription#cancel() cancelled} and {@link SubscriptionOption#NOTIFY_CANCELLATION} is
252      *       specified when subscribed.</li>
253      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
254      * </ul>
255      *
256      * @param executor the executor to subscribe
257      */
258     void subscribe(Subscriber<? super T> subscriber, EventExecutor executor);
259 
260     /**
261      * Requests to start streaming data to the specified {@link Subscriber}. If there is a problem subscribing,
262      * {@link Subscriber#onError(Throwable)} will be invoked with one of the following exceptions:
263      * <ul>
264      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
265      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
266      *   <li>{@link CancelledSubscriptionException} if this stream has been
267      *       {@linkplain Subscription#cancel() cancelled} and {@link SubscriptionOption#NOTIFY_CANCELLATION} is
268      *       specified when subscribed.</li>
269      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
270      * </ul>
271      *
272      * @param executor the executor to subscribe
273      * @param options {@link SubscriptionOption}s to subscribe with
274      */
275     void subscribe(Subscriber<? super T> subscriber, EventExecutor executor, SubscriptionOption... options);
276 
277     /**
278      * Subscribes to this {@link StreamMessage} and retrieves all elements from it.
279      * The returned {@link CompletableFuture} may be completed exceptionally with the following exceptions:
280      * <ul>
281      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
282      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
283      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
284      * </ul>
285      *
286      * @return the {@link CompletableFuture} which will be completed with the list of the elements retrieved.
287      */
288     default CompletableFuture<List<T>> drainAll() {
289         return drainAll(defaultSubscriberExecutor());
290     }
291 
292     /**
293      * Subscribes to this {@link StreamMessage} and retrieves all elements from it.
294      * The returned {@link CompletableFuture} may be completed exceptionally with the following exceptions:
295      * <ul>
296      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
297      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
298      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
299      * </ul>
300      *
301      * @param options {@link SubscriptionOption}s to subscribe with. Note that
302      *                {@link SubscriptionOption#NOTIFY_CANCELLATION} is ineffective because there's no
303      *                cancelling while draining all elements.
304      * @return the {@link CompletableFuture} which will be completed with the list of the elements retrieved.
305      */
306     default CompletableFuture<List<T>> drainAll(SubscriptionOption... options) {
307         return drainAll(defaultSubscriberExecutor(), options);
308     }
309 
310     /**
311      * Subscribes to this {@link StreamMessage} and retrieves all elements from it.
312      * The returned {@link CompletableFuture} may be completed exceptionally with the following exceptions:
313      * <ul>
314      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
315      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
316      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
317      * </ul>
318      *
319      * @param executor the executor to retrieve all elements
320      * @return the {@link CompletableFuture} which will be completed with the list of the elements retrieved.
321      */
322     CompletableFuture<List<T>> drainAll(EventExecutor executor);
323 
324     /**
325      * Subscribes to this {@link StreamMessage} and retrieves all elements from it.
326      * The returned {@link CompletableFuture} may be completed exceptionally with the following exceptions:
327      * <ul>
328      *   <li>{@link IllegalStateException} if other {@link Subscriber} subscribed to this stream already.</li>
329      *   <li>{@link AbortedStreamException} if this stream has been {@linkplain #abort() aborted}.</li>
330      *   <li>Other exceptions that occurred due to an error while retrieving the elements.</li>
331      * </ul>
332      *
333      * @param executor the executor to retrieve all elements
334      * @param options {@link SubscriptionOption}s to subscribe with. Note that
335      *                {@link SubscriptionOption#NOTIFY_CANCELLATION} is ineffective because there's no
336      *                cancelling while draining all elements.
337      * @return the {@link CompletableFuture} which will be completed with the list of the elements retrieved.
338      */
339     CompletableFuture<List<T>> drainAll(EventExecutor executor, SubscriptionOption... options);
340 
341     /**
342      * Returns a new {@link StreamMessageDuplicator} that duplicates this {@link StreamMessage} into one or
343      * more {@link StreamMessage}s, which publish the same elements.
344      * Note that you cannot subscribe to this {@link StreamMessage} anymore after you call this method.
345      * To subscribe, call {@link StreamMessageDuplicator#duplicate()} from the returned
346      * {@link StreamMessageDuplicator}.
347      */
348     default StreamMessageDuplicator<T> toDuplicator() {
349         return toDuplicator(defaultSubscriberExecutor());
350     }
351 
352     /**
353      * Returns a new {@link StreamMessageDuplicator} that duplicates this {@link StreamMessage} into one or
354      * more {@link StreamMessage}s, which publish the same elements.
355      * Note that you cannot subscribe to this {@link StreamMessage} anymore after you call this method.
356      * To subscribe, call {@link StreamMessageDuplicator#duplicate()} from the returned
357      * {@link StreamMessageDuplicator}.
358      *
359      * @param executor the executor to duplicate
360      */
361     default StreamMessageDuplicator<T> toDuplicator(EventExecutor executor) {
362         requireNonNull(executor, "executor");
363         return new DefaultStreamMessageDuplicator<>(this, unused -> 0, executor, 0 /* no limit for length */);
364     }
365 
366     /**
367      * Returns the default {@link EventExecutor} which will be used when a user subscribes using
368      * {@link #subscribe(Subscriber)}, {@link #subscribe(Subscriber, SubscriptionOption...)},
369      * {@link #drainAll()} and {@link #drainAll(SubscriptionOption...)}.
370      *
371      * <p>Please note that if this method is called multiple times, the returned {@link EventExecutor}s can be
372      * different depending on this {@link StreamMessage} implementation.
373      */
374     default EventExecutor defaultSubscriberExecutor() {
375         final EventLoop eventExecutor = RequestContext.mapCurrent(RequestContext::eventLoop,
376                                                                   CommonPools.workerGroup()::next);
377         assert eventExecutor != null;
378         return eventExecutor;
379     }
380 
381     /**
382      * Closes this stream with {@link AbortedStreamException} and prevents further subscription.
383      * A {@link Subscriber} that attempts to subscribe to an aborted stream will be notified with
384      * an {@link AbortedStreamException} via {@link Subscriber#onError(Throwable)}. Calling this method
385      * on a closed or aborted stream has no effect.
386      */
387     void abort();
388 
389     /**
390      * Closes this stream with the specified {@link Throwable} and prevents further subscription.
391      * A {@link Subscriber} that attempts to subscribe to an aborted stream will be notified with
392      * the specified {@link Throwable} via {@link Subscriber#onError(Throwable)}. Calling this method
393      * on a closed or aborted stream has no effect.
394      */
395     void abort(Throwable cause);
