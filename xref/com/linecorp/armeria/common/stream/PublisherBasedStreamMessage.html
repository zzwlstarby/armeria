25  import java.util.concurrent.atomic.AtomicReferenceFieldUpdater;

27  import javax.annotation.Nullable;

29  import org.reactivestreams.Publisher;
30  import org.reactivestreams.Subscriber;
31  import org.reactivestreams.Subscription;
32  
33  import com.google.common.annotations.VisibleForTesting;
34  import com.spotify.futures.CompletableFutures;

36  import com.linecorp.armeria.common.util.EventLoopCheckingFuture;

39  import io.netty.util.concurrent.EventExecutor;
40  import io.netty.util.concurrent.ImmediateEventExecutor;

42  /**
43   * Adapts a {@link Publisher} into a {@link StreamMessage}.
44   *
45   * @param <T> the type of element signaled
46   */
47  @UnstableApi
48  public class PublisherBasedStreamMessage<T> implements StreamMessage<T> {
49  
50      @SuppressWarnings("rawtypes")
51      private static final AtomicReferenceFieldUpdater<PublisherBasedStreamMessage, AbortableSubscriber>
52              subscriberUpdater = AtomicReferenceFieldUpdater.newUpdater(
53              PublisherBasedStreamMessage.class, AbortableSubscriber.class, "subscriber");
54  
55      private final Publisher<? extends T> publisher;
56      private final CompletableFuture<Void> completionFuture = new EventLoopCheckingFuture<>();
57      @Nullable
58      @SuppressWarnings("unused") // Updated only via subscriberUpdater.
59      private volatile AbortableSubscriber subscriber;
60      private volatile boolean publishedAny;
61  
62      /**
63       * Creates a new instance with the specified delegate {@link Publisher}.
64       */
65      public PublisherBasedStreamMessage(Publisher<? extends T> publisher) {
66          this.publisher = publisher;
67      }
68  
69      /**
70       * Returns the delegate {@link Publisher}.
71       */
72      protected final Publisher<? extends T> delegate() {
73          return publisher;
74      }
75  
76      @Override
77      public boolean isOpen() {
78          return !completionFuture.isDone();
79      }
80  
81      @Override
82      public boolean isEmpty() {
83          return !isOpen() && !publishedAny;
84      }
85  
86      @Override
87      public void subscribe(Subscriber<? super T> subscriber, EventExecutor executor) {
88          subscribe0(subscriber, executor, false);
89      }
90  
91      @Override
92      public void subscribe(Subscriber<? super T> subscriber, EventExecutor executor,
93                            SubscriptionOption... options) {
94          requireNonNull(options, "options");
95  
96          final boolean notifyCancellation = containsNotifyCancellation(options);
97          subscribe0(subscriber, executor, notifyCancellation);
98      }
99  
100     private void subscribe0(Subscriber<? super T> subscriber, EventExecutor executor,
101                             boolean notifyCancellation) {
102         requireNonNull(subscriber, "subscriber");
103         requireNonNull(executor, "executor");
104 
105         if (!subscribe1(subscriber, executor, notifyCancellation)) {
106             final AbortableSubscriber oldSubscriber = this.subscriber;
107             assert oldSubscriber != null;
108             failLateSubscriber(executor, subscriber, oldSubscriber.subscriber);
109         }
110     }
111 
112     private boolean subscribe1(Subscriber<? super T> subscriber, EventExecutor executor,
113                                boolean notifyCancellation) {
114         final AbortableSubscriber s = new AbortableSubscriber(this, subscriber, executor, notifyCancellation);
115         if (!subscriberUpdater.compareAndSet(this, null, s)) {
116             return false;
117         }
118 
119         publisher.subscribe(s);
120 
121         return true;
122     }
123 
124     private static void failLateSubscriber(EventExecutor executor,
125                                            Subscriber<?> lateSubscriber, Subscriber<?> oldSubscriber) {
126         final Throwable cause = abortedOrLate(oldSubscriber);
127 
128         executor.execute(() -> {
129             lateSubscriber.onSubscribe(NoopSubscription.INSTANCE);
130             lateSubscriber.onError(cause);
131         });
132     }
133 
134     @Override
135     public CompletableFuture<List<T>> drainAll(EventExecutor executor) {
136         requireNonNull(executor, "executor");
137 
138         final StreamMessageDrainer<T> drainer = new StreamMessageDrainer<>(false);
139         if (!subscribe1(drainer, executor, false)) {
140             final AbortableSubscriber subscriber = this.subscriber;
141             assert subscriber != null;
142             return CompletableFutures.exceptionallyCompletedFuture(abortedOrLate(subscriber.subscriber));
143         }
144 
145         return drainer.future();
146     }
147 
148     @Override
149     public CompletableFuture<List<T>> drainAll(EventExecutor executor, SubscriptionOption... options) {
150         requireNonNull(options, "options");
151 
152         return drainAll(executor);
153     }
154 
155     @Override
156     public void abort() {
157         abort0(AbortedStreamException.get());
158     }
159 
160     @Override
161     public void abort(Throwable cause) {
162         requireNonNull(cause, "cause");
163         abort0(cause);
164     }
165 
166     private void abort0(Throwable cause) {
167         final AbortableSubscriber subscriber = this.subscriber;
168         if (subscriber != null) {
169             subscriber.abort(cause);
170             return;
171         }
172 
173         final AbortableSubscriber abortable = new AbortableSubscriber(this, AbortingSubscriber.get(cause),
174                                                                       ImmediateEventExecutor.INSTANCE,
175                                                                       false);
176         if (!subscriberUpdater.compareAndSet(this, null, abortable)) {
177             this.subscriber.abort(cause);
178             return;
179         }
180 
181         abortable.abort(cause);
182         abortable.onSubscribe(NoopSubscription.INSTANCE);
183     }
184 
185     @Override
186     public CompletableFuture<Void> whenComplete() {
187         return completionFuture;
188     }
189 
190     @VisibleForTesting
191     static final class AbortableSubscriber implements Subscriber<Object>, Subscription {
192         private final PublisherBasedStreamMessage<?> parent;
193         private final EventExecutor executor;
194         private final boolean notifyCancellation;
195         private Subscriber<Object> subscriber;
196         @Nullable
197         private volatile Subscription subscription;
198         @Nullable
199         private volatile Throwable abortCause;
200 
201         @SuppressWarnings("unchecked")
202         AbortableSubscriber(PublisherBasedStreamMessage<?> parent, Subscriber<?> subscriber,
203                             EventExecutor executor, boolean notifyCancellation) {
204             this.parent = parent;
205             this.subscriber = (Subscriber<Object>) subscriber;
206             this.executor = executor;
207             this.notifyCancellation = notifyCancellation;
208         }
209 
210         @Override
211         public void request(long n) {
212             final Subscription subscription = this.subscription;
213             assert subscription != null;
214             subscription.request(n);
215         }
216 
217         @Override
218         public void cancel() {
219             // 'subscription' can never be null here because 'subscriber.onSubscriber()' is invoked
220             // only after 'subscription' is set. See onSubscribe0().
221             assert subscription != null;
222 
223             // Don't cancel but just abort if abort is pending.
224             cancelOrAbort(abortCause == null);
225         }
226 
227         void abort(Throwable cause) {
228             if (abortCause == null) {
229                 abortCause = cause;
230             }
231             if (subscription != null) {
232                 cancelOrAbort(false);
233             }
234         }
235 
236         private void cancelOrAbort(boolean cancel) {
237             if (executor.inEventLoop()) {
238                 cancelOrAbort0(cancel);
239             } else {
240                 executor.execute(() -> cancelOrAbort0(cancel));
241             }
242         }
243 
244         private void cancelOrAbort0(boolean cancel) {
245             final CompletableFuture<Void> completionFuture = parent.whenComplete();
246             if (completionFuture.isDone()) {
247                 return;
248             }
249 
250             final Subscriber<Object> subscriber = this.subscriber;
251             // Replace the subscriber with a placeholder so that it can be garbage-collected and
252             // we conform to the Reactive Streams specification rule 3.13.
253             if (!(subscriber instanceof AbortingSubscriber)) {
254                 this.subscriber = NoopSubscriber.get();
255             }
256 
257             final Throwable cause = cancel ? CancelledSubscriptionException.get() : abortCause;
258             try {
259                 if (!cancel || notifyCancellation) {
260                     subscriber.onError(cause);
261                 }
262             } finally {
263                 try {
264                     subscription.cancel();
265                 } finally {
266                     completionFuture.completeExceptionally(cause);
267                 }
268             }
269         }
270 
271         @Override
272         public void onSubscribe(Subscription subscription) {
273             if (executor.inEventLoop()) {
274                 onSubscribe0(subscription);
275             } else {
276                 executor.execute(() -> onSubscribe0(subscription));
277             }
278         }
279 
280         private void onSubscribe0(Subscription subscription) {
281             try {
282                 this.subscription = subscription;
283                 subscriber.onSubscribe(this);
284             } finally {
285                 if (abortCause != null) {
286                     cancelOrAbort0(false);
287                 }
288             }
289         }
290 
291         @Override
292         public void onNext(Object obj) {
293             parent.publishedAny = true;
294             if (executor.inEventLoop()) {
295                 subscriber.onNext(obj);
296             } else {
297                 executor.execute(() -> subscriber.onNext(obj));
298             }
299         }
300 
301         @Override
302         public void onError(Throwable cause) {
303             if (executor.inEventLoop()) {
304                 onError0(cause);
305             } else {
306                 executor.execute(() -> onError0(cause));
307             }
308         }
309 
310         private void onError0(Throwable cause) {
311             try {
312                 subscriber.onError(cause);
313             } finally {
314                 parent.whenComplete().completeExceptionally(cause);
315             }
316         }
317 
318         @Override
319         public void onComplete() {
320             if (executor.inEventLoop()) {
321                 onComplete0();
322             } else {
323                 executor.execute(this::onComplete0);
324             }
325         }
326 
327         private void onComplete0() {
328             try {
329                 subscriber.onComplete();
330             } finally {
331                 parent.whenComplete().complete(null);
332             }
333         }
334     }
