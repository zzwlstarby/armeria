17  package com.linecorp.armeria.common.logging;
18  
19  import java.nio.charset.Charset;
20  import java.util.List;
21  import java.util.function.BiFunction;
22  
23  import com.google.common.collect.ImmutableList;
24  
25  import com.linecorp.armeria.common.HttpHeaders;
26  import com.linecorp.armeria.common.MediaType;
27  import com.linecorp.armeria.common.RequestContext;
28  import com.linecorp.armeria.common.RequestHeaders;
29  import com.linecorp.armeria.common.ResponseHeaders;
30  
31  import io.netty.buffer.ByteBuf;
32  
33  final class DefaultContentPreviewFactory implements ContentPreviewerFactory {
34  
35      private static final List<String> textSubTypes = ImmutableList.of("json", "xml");
36      private static final List<String> textSubTypeSuffixes = ImmutableList.of("+json", "+xml");
37  
38      private final int maxLength;
39      private final List<PreviewSpec> previewSpecs;
40      private final Charset defaultCharset;
41  
42      DefaultContentPreviewFactory(List<PreviewSpec> previewSpecs, int maxLength, Charset defaultCharset) {
43          this.maxLength = maxLength;
44          this.previewSpecs = previewSpecs;
45          this.defaultCharset = defaultCharset;
46      }
47  
48      @Override
49      public ContentPreviewer requestContentPreviewer(RequestContext ctx, RequestHeaders headers) {
50          return contentPreviewer(ctx, headers);
51      }
52  
53      @Override
54      public ContentPreviewer responseContentPreviewer(RequestContext ctx, ResponseHeaders resHeaders) {
55          return contentPreviewer(ctx, resHeaders);
56      }
57  
58      private ContentPreviewer contentPreviewer(RequestContext ctx, HttpHeaders headers) {
59          for (PreviewSpec previewSpec : previewSpecs) {
60              if (previewSpec.predicate().test(ctx, headers)) {
61                  switch (previewSpec.mode()) {
62                      case TEXT:
63                          final Charset charset = charset(headers);
64                          return new TextContentPreviewer(maxLength, charset);
65                      case BINARY:
66                          final BiFunction<? super HttpHeaders, ? super ByteBuf, String> producer =
67                                  previewSpec.producer();
68                          assert producer != null;
69                          return new ProducerBasedContentPreviewer(maxLength, headers, producer);
70                      case DISABLED:
71                          return ContentPreviewer.disabled();
72                  }
73              }
74          }
75  
76          final MediaType contentType = headers.contentType();
77          if (contentType != null) {
78              final Charset charset = contentType.charset();
79              if (charset != null) {
80                  return new TextContentPreviewer(maxLength, charset);
81              }
82  
83              if ("text".equals(contentType.type()) ||
84                  textSubTypes.contains(contentType.subtype()) ||
85                  textSubTypeSuffixes.stream().anyMatch(contentType.subtype()::endsWith) ||
86                  contentType.is(MediaType.FORM_DATA)) {
87                  return new TextContentPreviewer(maxLength, defaultCharset);
88              }
89          }
90  
91          return ContentPreviewer.disabled();
92      }
93  
94      private Charset charset(HttpHeaders headers) {
95          final MediaType contentType = headers.contentType();
96          if (contentType != null) {
97              return contentType.charset(defaultCharset);
98          } else {
99              return defaultCharset;
100         }
101     }
102 }
