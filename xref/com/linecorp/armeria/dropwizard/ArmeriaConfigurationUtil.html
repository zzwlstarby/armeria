97      /**
98       * Configures the {@link ServerBuilder} with the specified {@code settings}.
99       */
100     static void configureServer(ServerBuilder serverBuilder, ArmeriaSettings settings) {
101         requireNonNull(serverBuilder, "serverBuilder");
102         requireNonNull(settings, "settings");
103 
104         if (settings.getGracefulShutdownQuietPeriodMillis() >= 0 &&
105             settings.getGracefulShutdownTimeoutMillis() >= 0) {
106             serverBuilder.gracefulShutdownTimeoutMillis(settings.getGracefulShutdownQuietPeriodMillis(),
107                                                         settings.getGracefulShutdownTimeoutMillis());
108             logger.debug("Set graceful shutdown timeout: quiet period {} ms, timeout {} ms",
109                          settings.getGracefulShutdownQuietPeriodMillis(),
110                          settings.getGracefulShutdownTimeoutMillis());
111         }
112         if (settings.getMaxRequestLength() != null) {
113             serverBuilder.maxRequestLength(settings.getMaxRequestLength());
114         }
115         if (settings.getMaxNumConnections() != null) {
116             serverBuilder.maxNumConnections(settings.getMaxNumConnections());
117         }
118         if (!settings.isDateHeaderEnabled()) {
119             serverBuilder.disableDateHeader();
120         }
121         if (!settings.isServerHeaderEnabled()) {
122             serverBuilder.disableServerHeader();
123         }
124         if (settings.getDefaultHostname() != null) {
125             serverBuilder.defaultHostname(settings.getDefaultHostname());
126         }
127         if (settings.isVerboseResponses()) {
128             serverBuilder.verboseResponses(true);
129         }
130 
131         if (settings.getPorts().isEmpty()) {
132             serverBuilder.port(new ServerPort(DEFAULT_PORT.getPort(), DEFAULT_PORT.getProtocols()));
133         } else {
134             configurePorts(serverBuilder, settings.getPorts());
135         }
136 
137         if (settings.getSsl() != null) {
138             configureTls(serverBuilder, settings.getSsl());
139         }
140         if (settings.getCompression() != null) {
141             configureCompression(serverBuilder, settings.getCompression());
142         }
143         if (settings.getHttp1() != null) {
144             configureHttp1(serverBuilder, settings.getHttp1());
145         }
146         if (settings.getHttp2() != null) {
147             configureHttp2(serverBuilder, settings.getHttp2());
148         }
149         if (settings.getProxy() != null) {
150             configureProxy(serverBuilder, settings.getProxy());
151         }
152         if (settings.getAccessLog() != null) {
153             configureAccessLog(serverBuilder, settings.getAccessLog());
154         }
155     }

157     /**
158      * Adds {@link Port}s to the specified {@link ServerBuilder}.
159      */
160     private static void configurePorts(ServerBuilder server, List<Port> ports) {
161         requireNonNull(server, "server");
162         requireNonNull(ports, "ports");
163         ports.forEach(p -> {
164             final String ip = p.getIp();
165             final String iface = p.getIface();
166             final int port = p.getPort();
167             final List<SessionProtocol> protocols = firstNonNull(p.getProtocols(),
168                                                                  ImmutableList.of(SessionProtocol.HTTP));
169 
170             if (ip == null) {
171                 if (iface == null) {
172                     server.port(new ServerPort(port, protocols));
173                 } else {
174                     try {
175                         final Enumeration<InetAddress> e = NetworkInterface.getByName(iface).getInetAddresses();
176                         while (e.hasMoreElements()) {
177                             server.port(new ServerPort(new InetSocketAddress(e.nextElement(), port),
178                                                        protocols));
179                         }
180                     } catch (SocketException e) {
181                         throw new IllegalStateException("Failed to find an iface: " + iface, e);
182                     }
183                 }
184             } else if (iface == null) {
185                 if (NetUtil.isValidIpV4Address(ip) || NetUtil.isValidIpV6Address(ip)) {
186                     final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(ip);
187                     try {
188                         server.port(new ServerPort(new InetSocketAddress(
189                                 InetAddress.getByAddress(bytes), port), protocols));
190                     } catch (UnknownHostException e) {
191                         // Should never happen.
192                         throw new Error(e);
193                     }
194                 } else {
195                     throw new IllegalStateException("invalid IP address: " + ip);
196                 }
197             } else {
198                 throw new IllegalStateException("A port cannot have both IP and iface: " + p);
199             }
200         });
201     }

210     /**
211      * Adds SSL/TLS context to the specified {@link ServerBuilder}.
212      */
213     private static void configureTls(ServerBuilder sb, ArmeriaSettings.Ssl ssl,
214                                      @Nullable Supplier<KeyStore> keyStoreSupplier,
215                                      @Nullable Supplier<KeyStore> trustStoreSupplier) {
216         if (!ssl.isEnabled()) {
217             return;
218         }
219         try {
220             if (keyStoreSupplier == null && trustStoreSupplier == null &&
221                 ssl.getKeyStore() == null && ssl.getTrustStore() == null) {
222                 logger.warn("Configuring TLS with a self-signed certificate " +
223                             "because no key or trust store was specified");
224                 sb.tlsSelfSigned();
225                 return;
226             }
227 
228             final KeyManagerFactory keyManagerFactory = getKeyManagerFactory(ssl, keyStoreSupplier);
229             final TrustManagerFactory trustManagerFactory = getTrustManagerFactory(ssl, trustStoreSupplier);
230 
231             sb.tls(keyManagerFactory);
232             sb.tlsCustomizer(sslContextBuilder -> {
233                 sslContextBuilder.trustManager(trustManagerFactory);
234 
235                 final SslProvider sslProvider = ssl.getProvider();
236                 if (sslProvider != null) {
237                     sslContextBuilder.sslProvider(sslProvider);
238                 }
239                 final List<String> enabledProtocols = ssl.getEnabledProtocols();
240                 if (enabledProtocols != null) {
241                     sslContextBuilder.protocols(enabledProtocols.toArray(EMPTY_PROTOCOL_NAMES));
242                 }
243                 final List<String> ciphers = ssl.getCiphers();
244                 if (ciphers != null) {
245                     sslContextBuilder.ciphers(ImmutableList.copyOf(ciphers),
246                                               SupportedCipherSuiteFilter.INSTANCE);
247                 }
248                 final ClientAuth clientAuth = ssl.getClientAuth();
249                 if (clientAuth != null) {
250                     sslContextBuilder.clientAuth(clientAuth);
251                 }
252             });
253         } catch (Exception e) {
254             throw new IllegalStateException("Failed to configure TLS: " + e, e);
255         }
256     }

258     private static KeyManagerFactory getKeyManagerFactory(
259             ArmeriaSettings.Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
260         final KeyStore store;
261         if (sslStoreProvider != null) {
262             store = sslStoreProvider.get();
263         } else {
264             store = loadKeyStore(ssl.getKeyStoreType(), ssl.getKeyStore(), ssl.getKeyStorePassword());
265         }
266 
267         KeyManagerFactory keyManagerFactory =
268                 KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
269         if (ssl.getKeyAlias() != null) {
270             keyManagerFactory = new CustomAliasKeyManagerFactory(keyManagerFactory, ssl.getKeyAlias());
271         }
272 
273         String keyPassword = ssl.getKeyPassword();
274         if (keyPassword == null) {
275             keyPassword = ssl.getKeyStorePassword();
276         }
277 
278         keyManagerFactory.init(store, keyPassword != null ? keyPassword.toCharArray()
279                                                           : null);
280         return keyManagerFactory;
281     }
282 
283     private static TrustManagerFactory getTrustManagerFactory(
284             ArmeriaSettings.Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
285         final KeyStore store;
286         if (sslStoreProvider != null) {
287             store = sslStoreProvider.get();
288         } else {
289             store = loadKeyStore(ssl.getTrustStoreType(), ssl.getTrustStore(), ssl.getTrustStorePassword());
290         }
291 
292         final TrustManagerFactory trustManagerFactory =
293                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
294         trustManagerFactory.init(store);
295         return trustManagerFactory;
296     }

298     @Nullable
299     private static KeyStore loadKeyStore(
300             @Nullable String type,
301             @Nullable String resource,
302             @Nullable String password) throws IOException, GeneralSecurityException {
303         if (resource == null) {
304             return null;
305         }
306         final KeyStore store = KeyStore.getInstance(firstNonNull(type, "JKS"));
307         final URL url = getURL(resource);
308         store.load(url.openStream(), password != null ? password.toCharArray()
309                                                       : null);
310         return store;
311     }

313     /**
314      * Configures a decorator for encoding the content of the HTTP responses sent from the server.
315      */
316     private static Function<? super HttpService, EncodingService> contentEncodingDecorator(
317             @Nullable String[] mimeTypes, @Nullable String[] excludedUserAgents,
318             int minBytesToForceChunkedAndEncoding) {
319         final Predicate<MediaType> encodableContentTypePredicate;
320         if (mimeTypes == null || mimeTypes.length == 0) {
321             encodableContentTypePredicate = contentType -> true;
322         } else {
323             final List<MediaType> encodableContentTypes =
324                     Arrays.stream(mimeTypes).map(MediaType::parse).collect(toImmutableList());
325             encodableContentTypePredicate = contentType ->
326                     encodableContentTypes.stream().anyMatch(contentType::is);
327         }
328 
329         final Predicate<? super RequestHeaders> encodableRequestHeadersPredicate;
330         if (excludedUserAgents == null || excludedUserAgents.length == 0) {
331             encodableRequestHeadersPredicate = headers -> true;
332         } else {
333             final List<Pattern> patterns =
334                     Arrays.stream(excludedUserAgents).map(Pattern::compile).collect(toImmutableList());
335             encodableRequestHeadersPredicate = headers -> {
336                 // No User-Agent header will be converted to an empty string.
337                 final String userAgent = headers.get(HttpHeaderNames.USER_AGENT, "");
338                 return patterns.stream().noneMatch(pattern -> pattern.matcher(userAgent).matches());
339             };
340         }
341 
342         return EncodingService.builder()
343                               .encodableContentTypePredicate(encodableContentTypePredicate)
344                               .encodableRequestHeadersPredicate(encodableRequestHeadersPredicate)
345                               .minBytesToForceChunkedEncoding(minBytesToForceChunkedAndEncoding)
346                               .newDecorator();
347     }

349     /**
350      * Parses the data size text as a decimal {@code long}.
351      *
352      * @param dataSizeText the data size text, i.e. {@code 1}, {@code 1B}, {@code 1KB}, {@code 1MB},
353      *                     {@code 1GB} or {@code 1TB}
354      */
355     private static long parseDataSize(String dataSizeText) {
356         requireNonNull(dataSizeText, "text");
357         final Matcher matcher = DATA_SIZE_PATTERN.matcher(dataSizeText);
358         checkArgument(matcher.matches(),
359                       "Invalid data size text: %s (expected: %s)",
360                       dataSizeText, DATA_SIZE_PATTERN);
361 
362         final long unit;
363         final String unitText = matcher.group(2);
364         if (Strings.isNullOrEmpty(unitText)) {
365             unit = 1L;
366         } else {
367             switch (Ascii.toLowerCase(unitText)) {
368                 case "b":
369                     unit = 1L;
370                     break;
371                 case "kb":
372                     unit = 1024L;
373                     break;
374                 case "mb":
375                     unit = 1024L * 1024L;
376                     break;
377                 case "gb":
378                     unit = 1024L * 1024L * 1024L;
379                     break;
380                 case "tb":
381                     // TODO(ikhoon): Simplify with Math.pow?
382                     unit = 1024L * 1024L * 1024L * 1024L;
383                     break;
384                 default:
385                     throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
386                                                        " (expected: " + DATA_SIZE_PATTERN + ')');
387             }
388         }
389         try {
390             final long amount = Long.parseLong(matcher.group(1));
391             return LongMath.checkedMultiply(amount, unit);
392         } catch (Exception e) {
393             throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
394                                                " (expected: " + DATA_SIZE_PATTERN + ')', e);
395         }
396     }

398     /**
399      * Resolves the given resource location to a {@link URL}.
400      *
401      * <p>Does not check whether the URL actually exists; simply returns
402      * the URL that the given location would correspond to.
403      *
404      * @param resourceLocation the resource location to resolve: either a
405      *                         "classpath:" pseudo URL, a "file:" URL, or a plain file path
406      * @return a corresponding URL object
407      * @throws FileNotFoundException if the resource cannot be resolved to a URL
408      */
409     private static URL getURL(String resourceLocation) throws FileNotFoundException {
410         // Forked from Spring 5.2.2
411         // https://github.com/spring-projects/spring-framework/blob/22a888b53df620b0905ce8beb6b0cf71981086d8/spring-core/src/main/java/org/springframework/util/ResourceUtils.java#L129
412         if (resourceLocation.startsWith("classpath:")) {
413             final String path = resourceLocation.substring("classpath:".length());
414             final URL resource = ArmeriaConfigurationUtil.class.getClassLoader().getResource(path);
415             if (resource == null) {
416                 throw new FileNotFoundException("class path resource [" + path +
417                                                 "] cannot be resolved to URL because it does not exist");
418             }
419             return resource;
420         }
421         try {
422             // try URL
423             return new URL(resourceLocation);
424         } catch (MalformedURLException ex) {
425             // no URL -> treat as file path
426             try {
427                 return new File(resourceLocation).toURI().toURL();
428             } catch (MalformedURLException ex2) {
429                 throw new FileNotFoundException("Resource location [" + resourceLocation +
430                                                 "] is neither a URL nor a well-formed file path");
431             }
432         }
433     }

435     private static void configureCompression(ServerBuilder serverBuilder, Compression compression) {
436         if (compression.isEnabled()) {
437             final int minBytesToForceChunkedAndEncoding;
438             final String minResponseSize = compression.getMinResponseSize();
439             if (minResponseSize == null) {
440                 minBytesToForceChunkedAndEncoding = DEFAULT_MIN_BYTES_TO_FORCE_CHUNKED_AND_ENCODING;
441             } else {
442                 minBytesToForceChunkedAndEncoding = Ints.saturatedCast(parseDataSize(minResponseSize));
443             }
444             serverBuilder.decorator(contentEncodingDecorator(compression.getMimeTypes(),
445                                                              compression.getExcludedUserAgents(),
446                                                              minBytesToForceChunkedAndEncoding));
447         }
448     }

450     private static void configureHttp1(ServerBuilder serverBuilder, Http1 http1) {
451         if (http1.getMaxInitialLineLength() != null) {
452             serverBuilder.http1MaxInitialLineLength(http1.getMaxInitialLineLength());
453         }
454         if (http1.getMaxChunkSize() != null) {
455             serverBuilder.http1MaxChunkSize((int) parseDataSize(http1.getMaxChunkSize()));
456         }
457     }

459     private static void configureHttp2(ServerBuilder serverBuilder, Http2 http2) {
460         if (http2.getInitialConnectionWindowSize() != null) {
461             serverBuilder.http2InitialConnectionWindowSize(
462                     (int) parseDataSize(http2.getInitialConnectionWindowSize()));
463         }
464         if (http2.getInitialStreamWindowSize() != null) {
465             serverBuilder.http2InitialStreamWindowSize((int) parseDataSize(http2.getInitialStreamWindowSize()));
466         }
467         if (http2.getMaxFrameSize() != null) {
468             serverBuilder.http2MaxFrameSize((int) parseDataSize(http2.getMaxFrameSize()));
469         }
470         if (http2.getMaxHeaderListSize() != null) {
471             serverBuilder.http2MaxHeaderListSize((int) parseDataSize(http2.getMaxHeaderListSize()));
472         }
473     }

475     private static void configureProxy(ServerBuilder serverBuilder, Proxy proxy) {
476         if (proxy.getMaxTlvSize() != null) {
477             serverBuilder.proxyProtocolMaxTlvSize((int) parseDataSize(proxy.getMaxTlvSize()));
478         }

481     private static void configureAccessLog(ServerBuilder serverBuilder, AccessLog accessLog) {
482         if ("common".equals(accessLog.getType())) {
483             serverBuilder.accessLogWriter(AccessLogWriter.common(), true);
484         } else if ("combined".equals(accessLog.getType())) {
485             serverBuilder.accessLogWriter(AccessLogWriter.combined(), true);
486         } else if ("custom".equals(accessLog.getType())) {
487             serverBuilder
488                     .accessLogWriter(AccessLogWriter.custom(accessLog.getFormat()), true);
489         }
490     }
