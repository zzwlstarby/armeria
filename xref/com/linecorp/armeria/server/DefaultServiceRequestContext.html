125     /**
126      * Creates a new instance.
127      *
128      * @param cfg the {@link ServiceConfig}
129      * @param ch the {@link Channel} that handles the invocation
130      * @param meterRegistry the {@link MeterRegistry} that collects various stats
131      * @param sessionProtocol the {@link SessionProtocol} of the invocation
132      * @param id the {@link RequestId} that represents the identifier of the current {@link Request}
133      *           and {@link Response} pair.
134      * @param routingContext the parameters which are used when finding a matched {@link Route}
135      * @param routingResult the result of finding a matched {@link Route}
136      * @param req the request associated with this context
137      * @param sslSession the {@link SSLSession} for this invocation if it is over TLS
138      * @param proxiedAddresses source and destination addresses delivered through proxy servers
139      * @param clientAddress the address of a client who initiated the request
140      * @param requestStartTimeNanos {@link System#nanoTime()} value when the request started.
141      * @param requestStartTimeMicros the number of microseconds since the epoch,
142      *                               e.g. {@code System.currentTimeMillis() * 1000}.
143      */
144     public DefaultServiceRequestContext(
145             ServiceConfig cfg, Channel ch, MeterRegistry meterRegistry, SessionProtocol sessionProtocol,
146             RequestId id, RoutingContext routingContext, RoutingResult routingResult, HttpRequest req,
147             @Nullable SSLSession sslSession, ProxiedAddresses proxiedAddresses, InetAddress clientAddress,
148             long requestStartTimeNanos, long requestStartTimeMicros) {
149 
150         super(meterRegistry, sessionProtocol, id,
151               requireNonNull(routingContext, "routingContext").method(), routingContext.path(),
152               requireNonNull(routingResult, "routingResult").query(),
153               requireNonNull(req, "req"), null, null);
154 
155         this.ch = requireNonNull(ch, "ch");
156         this.cfg = requireNonNull(cfg, "cfg");
157         this.routingContext = routingContext;
158         this.routingResult = routingResult;
159         this.sslSession = sslSession;
160         this.proxiedAddresses = requireNonNull(proxiedAddresses, "proxiedAddresses");
161         this.clientAddress = requireNonNull(clientAddress, "clientAddress");
162 
163         log = RequestLog.builder(this);
164         log.startRequest(requestStartTimeNanos, requestStartTimeMicros);
165         log.session(ch, sessionProtocol, sslSession, null);
166         log.requestHeaders(req.headers());
167 
168         // For the server, request headers are processed well before ServiceRequestContext is created. It means
169         // there is some delay between the actual channel read and this logging, but it's the best we can do for
170         // now.
171         log.requestFirstBytesTransferred();
172 
173         requestTimeoutMillis = cfg.requestTimeoutMillis();
174         maxRequestLength = cfg.maxRequestLength();
175         additionalResponseHeaders = HttpHeaders.of();
176         additionalResponseTrailers = HttpHeaders.of();
177     }

200     @Override
201     public ServiceRequestContext newDerivedContext(RequestId id,
202                                                    @Nullable HttpRequest req,
203                                                    @Nullable RpcRequest rpcReq) {
204         requireNonNull(req, "req");
205         if (rpcRequest() != null) {
206             requireNonNull(rpcReq, "rpcReq");
207         }
208 
209         final DefaultServiceRequestContext ctx = new DefaultServiceRequestContext(
210                 cfg, ch, meterRegistry(), sessionProtocol(), id, routingContext,
211                 routingResult, req, sslSession(), proxiedAddresses(), clientAddress(),
212                 System.nanoTime(), SystemInfo.currentTimeMicros());
213 
214         if (rpcReq != null) {
215             ctx.updateRpcRequest(rpcReq);
216         }
217 
218         final HttpHeaders additionalHeaders = additionalResponseHeaders();
219         if (!additionalHeaders.isEmpty()) {
220             ctx.setAdditionalResponseHeaders(additionalHeaders);
221         }
222 
223         final HttpHeaders additionalTrailers = additionalResponseTrailers();
224         if (!additionalTrailers.isEmpty()) {
225             ctx.setAdditionalResponseTrailers(additionalTrailers);
226         }
227 
228         for (final Iterator<Entry<AttributeKey<?>, Object>> i = attrs(); i.hasNext();/* noop */) {
229             ctx.addAttr(i.next());
230         }
231         return ctx;
232     }

322     @Override
323     public void clearRequestTimeout() {
324         if (requestTimeoutMillis == 0) {
325             return;
326         }
327 
328         final TimeoutController requestTimeoutController = this.requestTimeoutController;
329         requestTimeoutMillis = 0;
330         if (requestTimeoutController != null) {
331             if (eventLoop().inEventLoop()) {
332                 requestTimeoutController.cancelTimeout();
333             } else {
334                 eventLoop().execute(requestTimeoutController::cancelTimeout);
335             }
336         } else {
337             addPendingTimeoutTask(TimeoutController::cancelTimeout);
338         }
339     }

341     @Override
342     public void setRequestTimeoutMillis(long requestTimeoutMillis) {
343         checkArgument(requestTimeoutMillis >= 0,
344                       "requestTimeoutMillis: %s (expected: >= 0)", requestTimeoutMillis);
345         if (requestTimeoutMillis == 0) {
346             clearRequestTimeout();
347             return;
348         }
349 
350         if (this.requestTimeoutMillis == 0) {
351             setRequestTimeoutAfterMillis(requestTimeoutMillis);
352             return;
353         }
354 
355         final long adjustmentMillis =
356                 LongMath.saturatedSubtract(requestTimeoutMillis, this.requestTimeoutMillis);
357         extendRequestTimeoutMillis(adjustmentMillis);
358     }

365     @Override
366     public void extendRequestTimeoutMillis(long adjustmentMillis) {
367         if (adjustmentMillis == 0 || requestTimeoutMillis == 0) {
368             return;
369         }
370 
371         final long oldRequestTimeoutMillis = requestTimeoutMillis;
372         requestTimeoutMillis = LongMath.saturatedAdd(oldRequestTimeoutMillis, adjustmentMillis);
373         final TimeoutController requestTimeoutController = this.requestTimeoutController;
374         if (requestTimeoutController != null) {
375             if (eventLoop().inEventLoop()) {
376                 requestTimeoutController.extendTimeout(adjustmentMillis);
377             } else {
378                 eventLoop().execute(() -> requestTimeoutController.extendTimeout(adjustmentMillis));
379             }
380         } else {
381             addPendingTimeoutTask(timeoutController -> timeoutController.extendTimeout(adjustmentMillis));
382         }
383     }

390     @Override
391     public void setRequestTimeoutAfterMillis(long requestTimeoutMillis) {
392         checkArgument(requestTimeoutMillis > 0,
393                       "requestTimeoutMillis: %s (expected: > 0)", requestTimeoutMillis);
394 
395         long passedTimeMillis = 0;
396         final TimeoutController requestTimeoutController = this.requestTimeoutController;
397         if (requestTimeoutController != null) {
398             final Long startTimeNanos = requestTimeoutController.startTimeNanos();
399             if (startTimeNanos != null) {
400                 passedTimeMillis = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTimeNanos);
401             }
402             if (eventLoop().inEventLoop()) {
403                 requestTimeoutController.resetTimeout(requestTimeoutMillis);
404             } else {
405                 eventLoop().execute(() -> requestTimeoutController
406                         .resetTimeout(requestTimeoutMillis));
407             }
408         } else {
409             final long startTimeNanos = System.nanoTime();
410             addPendingTimeoutTask(timeoutController -> {
411                 final long passedTimeMillis0 =
412                         TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTimeNanos);
413                 final long timeoutMillis = Math.max(1, requestTimeoutMillis - passedTimeMillis0);
414                 timeoutController.resetTimeout(timeoutMillis);
415             });
416         }
417 
418         this.requestTimeoutMillis = LongMath.saturatedAdd(passedTimeMillis, requestTimeoutMillis);
419     }

426     @Override
427     public void setRequestTimeoutAtMillis(long requestTimeoutAtMillis) {
428         checkArgument(requestTimeoutAtMillis >= 0,
429                       "requestTimeoutAtMillis: %s (expected: >= 0)", requestTimeoutAtMillis);
430         final long requestTimeoutAfter = requestTimeoutAtMillis - System.currentTimeMillis();
431 
432         if (requestTimeoutAfter <= 0) {
433             final TimeoutController requestTimeoutController = this.requestTimeoutController;
434             if (requestTimeoutController != null) {
435                 if (eventLoop().inEventLoop()) {
436                     requestTimeoutController.timeoutNow();
437                 } else {
438                     eventLoop().execute(requestTimeoutController::timeoutNow);
439                 }
440             } else {
441                 addPendingTimeoutTask(TimeoutController::timeoutNow);
442             }
443         } else {
444             setRequestTimeoutAfterMillis(requestTimeoutAfter);
445         }
446     }

532     private void updateAdditionalResponseHeaders(
533             AtomicReferenceFieldUpdater<DefaultServiceRequestContext, HttpHeaders> atomicUpdater,
534             Function<HttpHeadersBuilder, HttpHeadersBuilder> valueUpdater) {
535         for (;;) {
536             final HttpHeaders oldValue = atomicUpdater.get(this);
537             final HttpHeaders newValue = valueUpdater.apply(oldValue.toBuilder()).build();
538             if (atomicUpdater.compareAndSet(this, oldValue, newValue)) {
539                 return;
540             }
541         }
542     }

549     private boolean removeAdditionalResponseHeader(
550             AtomicReferenceFieldUpdater<DefaultServiceRequestContext, HttpHeaders> atomicUpdater,
551             CharSequence name) {
552         requireNonNull(name, "name");
553         for (;;) {
554             final HttpHeaders oldValue = atomicUpdater.get(this);
555             if (oldValue.isEmpty() || !oldValue.contains(name)) {
556                 return false;
557             }
558 
559             final HttpHeaders newValue = oldValue.toBuilder().removeAndThen(name).build();
560             if (atomicUpdater.compareAndSet(this, oldValue, newValue)) {
561                 return true;
562             }
563         }
564     }

621     /**
622      * Sets the {@code requestTimeoutController} that is set to a new timeout when
623      * the {@linkplain #requestTimeoutMillis()} request timeout} of the request is changed.
624      *
625      * <p>Note: This method is meant for internal use by server-side protocol implementation to reschedule
626      * a timeout task when a user updates the request timeout configuration.
627      */
628     void setRequestTimeoutController(TimeoutController requestTimeoutController) {
629         requireNonNull(requestTimeoutController, "requestTimeoutController");
630         checkState(this.requestTimeoutController == null, "requestTimeoutController is set already.");
631         this.requestTimeoutController = requestTimeoutController;
632 
633         final Consumer<TimeoutController> pendingTimeoutTask = this.pendingTimeoutTask;
634         if (pendingTimeoutTask != null) {
635             if (eventLoop().inEventLoop()) {
636                 pendingTimeoutTask.accept(requestTimeoutController);
637             } else {
638                 eventLoop().execute(() -> pendingTimeoutTask.accept(requestTimeoutController));
639             }
640         }
641     }
642 
643     private void addPendingTimeoutTask(Consumer<TimeoutController> pendingTimeoutTask) {
644         if (this.pendingTimeoutTask == null) {
645             this.pendingTimeoutTask = pendingTimeoutTask;
646         } else {
647             this.pendingTimeoutTask = this.pendingTimeoutTask.andThen(pendingTimeoutTask);
648         }
649     }

660     private String toStringSlow() {
661         // Prepare all properties required for building a String, so that we don't have a chance of
662         // building one String with a thread-local StringBuilder while building another String with
663         // the same StringBuilder. See TemporaryThreadLocals for more information.
664         final String sreqId = id().shortText();
665         final String chanId = ch.id().asShortText();
666         final InetSocketAddress raddr = remoteAddress();
667         final InetSocketAddress laddr = localAddress();
668         final InetAddress caddr = clientAddress();
669         final String proto = sessionProtocol().uriText();
670         final String authority = virtualHost().defaultHostname();
671         final String path = path();
672         final String method = method().name();
673 
674         // Build the string representation.
675         final StringBuilder buf = TemporaryThreadLocals.get().stringBuilder();
676         buf.append("[sreqId=").append(sreqId)
677            .append(", chanId=").append(chanId);
678 
679         if (!Objects.equals(caddr, raddr.getAddress())) {
680             buf.append(", caddr=");
681             TextFormatter.appendInetAddress(buf, caddr);
682         }
683 
684         buf.append(", raddr=");
685         TextFormatter.appendSocketAddress(buf, raddr);
686         buf.append(", laddr=");
687         TextFormatter.appendSocketAddress(buf, laddr);
688         buf.append("][")
689            .append(proto).append("://").append(authority).append(path).append('#').append(method)
690            .append(']');
691 
692         return strVal = buf.toString();
693     }
