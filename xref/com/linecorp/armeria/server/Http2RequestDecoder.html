58      private static final ByteBuf DATA_MISSING_METHOD =
59              Unpooled.copiedBuffer(HttpResponseStatus.BAD_REQUEST + "\nMissing method",
60                                    StandardCharsets.UTF_8).asReadOnly();
61      private static final ByteBuf DATA_UNSUPPORTED_METHOD =
62              Unpooled.copiedBuffer(HttpResponseStatus.METHOD_NOT_ALLOWED + "\nUnsupported method",
63                                    StandardCharsets.UTF_8).asReadOnly();
64      private static final ByteBuf DATA_INVALID_CONTENT_LENGTH =
65              Unpooled.copiedBuffer(HttpResponseStatus.BAD_REQUEST + "\nInvalid content length",
66                                    StandardCharsets.UTF_8).asReadOnly();

75      private final IntObjectMap<DecodedHttpRequest> requests = new IntObjectHashMap<>();
76      private int nextId;

78      Http2RequestDecoder(ServerConfig cfg, Channel channel, Http2ConnectionEncoder writer, String scheme) {
79          this.cfg = cfg;
80          this.channel = channel;
81          this.writer = writer;
82          this.scheme = scheme;
83          inboundTrafficController =
84                  InboundTrafficController.ofHttp2(channel, cfg.http2InitialConnectionWindowSize());
85          goAwayHandler = new Http2GoAwayHandler();
86      }

88      Http2GoAwayHandler goAwayHandler() {
89          return goAwayHandler;
90      }

97      @Override
98      public void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
99                                boolean endOfStream) throws Http2Exception {
100         DecodedHttpRequest req = requests.get(streamId);
101         if (req == null) {
102             // Validate the method.
103             final CharSequence method = headers.method();
104             if (method == null) {
105                 writeErrorResponse(ctx, streamId, HttpResponseStatus.BAD_REQUEST, DATA_MISSING_METHOD);
106                 return;
107             }
108             if (!HttpMethod.isSupported(method.toString())) {
109                 writeErrorResponse(ctx, streamId, HttpResponseStatus.METHOD_NOT_ALLOWED,
110                                    DATA_UNSUPPORTED_METHOD);
111                 return;
112             }
113 
114             // Validate the 'content-length' header if exists.
115             final boolean contentEmpty;
116             if (headers.contains(HttpHeaderNames.CONTENT_LENGTH)) {
117                 final long contentLength = headers.getLong(HttpHeaderNames.CONTENT_LENGTH, -1L);
118                 if (contentLength < 0) {
119                     writeErrorResponse(ctx, streamId, HttpResponseStatus.BAD_REQUEST,
120                                        DATA_INVALID_CONTENT_LENGTH);
121                     return;
122                 }
123                 contentEmpty = contentLength == 0;
124             } else {
125                 contentEmpty = true;
126             }
127 
128             if (!handle100Continue(ctx, streamId, headers)) {
129                 writeErrorResponse(ctx, streamId, HttpResponseStatus.EXPECTATION_FAILED, null);
130                 return;
131             }
132 
133             req = new DecodedHttpRequest(ctx.channel().eventLoop(), ++nextId, streamId,
134                                          ArmeriaHttpUtil.toArmeriaRequestHeaders(ctx, headers, endOfStream,
135                                                                                  scheme, cfg),
136                                          true, inboundTrafficController,
137                                          // FIXME(trustin): Use a different maxRequestLength
138                                          //                 for a different host.
139                                          cfg.defaultVirtualHost().maxRequestLength());
140 
141             // Close the request early when it is sure that there will be
142             // neither content nor trailers.
143             if (contentEmpty && endOfStream) {
144                 req.close();
145             }
146 
147             requests.put(streamId, req);
148             ctx.fireChannelRead(req);
149         } else {
150             try {
151                 req.write(ArmeriaHttpUtil.toArmeria(headers, true, endOfStream));
152             } catch (Throwable t) {
153                 req.close(t);
154                 throw connectionError(INTERNAL_ERROR, t, "failed to consume a HEADERS frame");
155             }
156         }
157 
158         if (endOfStream) {
159             req.close();
160         }
161     }

163     @Override
164     public void onHeadersRead(
165             ChannelHandlerContext ctx, int streamId, Http2Headers headers,
166             int streamDependency, short weight, boolean exclusive, int padding,
167             boolean endOfStream) throws Http2Exception {
168 
169         onHeadersRead(ctx, streamId, headers, padding, endOfStream);
170     }
171 
172     private boolean handle100Continue(ChannelHandlerContext ctx, int streamId, Http2Headers headers) {
173         final CharSequence expectValue = headers.get(HttpHeaderNames.EXPECT);
174         if (expectValue == null) {
175             // No 'expect' header.
176             return true;
177         }
178 
179         // '100-continue' is the only allowed expectation.
180         if (!AsciiString.contentEqualsIgnoreCase(HttpHeaderValues.CONTINUE, expectValue)) {
181             return false;
182         }
183 
184         // Send a '100 Continue' response.
185         writer.writeHeaders(
186                 ctx, streamId,
187                 new DefaultHttp2Headers(false).status(HttpStatus.CONTINUE.codeAsText()),
188                 0, false, ctx.voidPromise());
189 
190         // Remove the 'expect' header so that it's handled in a way invisible to a Service.
191         headers.remove(HttpHeaderNames.EXPECT);
192         return true;
193     }

195     @Override
196     public void onStreamClosed(Http2Stream stream) {
197         goAwayHandler.onStreamClosed(channel, stream);
198 
199         final DecodedHttpRequest req = requests.remove(stream.id());
200         if (req != null) {
201             // Ignored if the stream has already been closed.
202             req.close(ClosedStreamException.get());
203         }
204     }
205 
206     @Override
207     public int onDataRead(
208             ChannelHandlerContext ctx, int streamId, ByteBuf data,
209             int padding, boolean endOfStream) throws Http2Exception {
210 
211         final DecodedHttpRequest req = requests.get(streamId);
212         if (req == null) {
213             throw connectionError(PROTOCOL_ERROR, "received a DATA Frame for an unknown stream: %d",
214                                   streamId);
215         }
216 
217         final int dataLength = data.readableBytes();
218         if (dataLength == 0) {
219             // Received an empty DATA frame
220             if (endOfStream) {
221                 req.close();
222             }
223             return padding;
224         }
225 
226         req.increaseTransferredBytes(dataLength);
227 
228         final long maxContentLength = req.maxRequestLength();
229         if (maxContentLength > 0 && req.transferredBytes() > maxContentLength) {
230             final Http2Stream stream = writer.connection().stream(streamId);
231             if (isWritable(stream)) {
232                 writeErrorResponse(ctx, streamId, HttpResponseStatus.REQUEST_ENTITY_TOO_LARGE, null);
233                 writer.writeRstStream(ctx, streamId, Http2Error.CANCEL.code(), ctx.voidPromise());
234                 if (req.isOpen()) {
235                     req.close(ContentTooLargeException.get());
236                 }
237             } else {
238                 // The response has been started already. Abort the request and let the response continue.
239                 req.abort();
240             }
241         } else if (req.isOpen()) {
242             try {
243                 req.write(new ByteBufHttpData(data.retain(), endOfStream));
244             } catch (Throwable t) {
245                 req.close(t);
246                 throw connectionError(INTERNAL_ERROR, t, "failed to consume a DATA frame");
247             }
248 
249             if (endOfStream) {
250                 req.close();
251             }
252         }
253 
254         // All bytes have been processed.
255         return dataLength + padding;
256     }
257 
258     private static boolean isWritable(Http2Stream stream) {
259         switch (stream.state()) {
260             case OPEN:
261             case HALF_CLOSED_REMOTE:
262                 return !stream.isHeadersSent();
263             default:
264                 return false;
265         }
266     }
267 
268     private void writeErrorResponse(ChannelHandlerContext ctx, int streamId, HttpResponseStatus status,
269                                     @Nullable ByteBuf content) throws Http2Exception {
270         final ByteBuf data =
271                 content != null ? content
272                                 : Unpooled.wrappedBuffer(status.toString().getBytes(StandardCharsets.UTF_8));
273 
274         writer.writeHeaders(
275                 ctx, streamId,
276                 new DefaultHttp2Headers(false)
277                         .status(status.codeAsText())
278                         .set(HttpHeaderNames.CONTENT_TYPE, MediaType.PLAIN_TEXT_UTF_8.toString())
279                         .setInt(HttpHeaderNames.CONTENT_LENGTH, data.readableBytes()),
280                 0, false, ctx.voidPromise());
281 
282         writer.writeData(ctx, streamId, data, 0, true, ctx.voidPromise());
283 
284         final Http2Stream stream = writer.connection().stream(streamId);
285         if (stream != null && writer.flowController().hasFlowControlled(stream)) {
286             // Ensure to flush the error response if it's flow-controlled so that it is sent
287             // before an RST_STREAM frame.
288             writer.flowController().writePendingBytes();
289         }
290     }

292     @Override
293     public void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception {
294         final DecodedHttpRequest req = requests.get(streamId);
295         if (req == null) {
296             throw connectionError(PROTOCOL_ERROR,
297                                   "received a RST_STREAM frame for an unknown stream: %d", streamId);
298         }
299 
300         req.abortResponse(new ClosedStreamException(
301                 "received a RST_STREAM frame: " + Http2Error.valueOf(errorCode)));
302     }

304     @Override
305     public void onPushPromiseRead(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
306                                   Http2Headers headers, int padding) throws Http2Exception {
307         throw connectionError(PROTOCOL_ERROR, "received a PUSH_PROMISE frame which only a server can send");
308     }
