57      /**
58       * Returns the default implementation of the {@link Router} to find a {@link ServiceConfig}.
59       * It consists of several router implementations which use one of Trie and List. It also includes
60       * cache mechanism to improve its performance.
61       */
62      public static Router<ServiceConfig> ofVirtualHost(VirtualHost virtualHost, Iterable<ServiceConfig> configs,
63                                                        RejectedRouteHandler rejectionHandler) {
64          requireNonNull(virtualHost, "virtualHost");
65          requireNonNull(configs, "configs");
66          requireNonNull(rejectionHandler, "rejectionHandler");
67  
68          final BiConsumer<Route, Route> rejectionConsumer = (route, existingRoute) -> {
69              try {
70                  rejectionHandler.handleDuplicateRoute(virtualHost, route, existingRoute);
71              } catch (Exception e) {
72                  logger.warn("Unexpected exception from a {}:",
73                              RejectedRouteHandler.class.getSimpleName(), e);
74              }
75          };
76          final Set<Route> ambiguousRoutes =
77                  resolveAmbiguousRoutes(StreamSupport.stream(configs.spliterator(), false)
78                                                      .map(ServiceConfig::route)
79                                                      .collect(toImmutableList()));
80          return wrapVirtualHostRouter(defaultRouter(configs, virtualHost.fallbackServiceConfig(),
81                                                     ServiceConfig::route, rejectionConsumer),
82                                       ambiguousRoutes);
83      }

85      /**
86       * Returns the default implementation of the {@link Router} to find a {@link CompositeServiceEntry}.
87       */
88      public static <T extends Service<?, ?>> Router<T> ofCompositeService(
89              List<CompositeServiceEntry<T>> entries) {
90          requireNonNull(entries, "entries");
91  
92          final Router<CompositeServiceEntry<T>> delegate = wrapCompositeServiceRouter(defaultRouter(
93                  entries, /* fallbackValue */ null, CompositeServiceEntry::route,
94                  (mapping, existingMapping) -> {
95                      final String a = mapping.toString();
96                      final String b = existingMapping.toString();
97                      if (a.equals(b)) {
98                          throw new IllegalStateException(
99                                  "Your composite service has a duplicate path mapping: " + a);
100                     }
101 
102                     throw new IllegalStateException(
103                             "Your composite service has path mappings with a conflict: " +
104                             a + " vs. " + b);
105                 }), resolveAmbiguousRoutes(entries.stream()
106                                                   .map(CompositeServiceEntry::route)
107                                                   .collect(toImmutableList())));
108 
109         return new CompositeRouter<>(delegate, result ->
110                 result.isPresent() ? Routed.of(result.route(), result.routingResult(),
111                                                result.value().service())
112                                    : Routed.empty());
113     }

115     /**
116      * Returns the default implementation of the {@link Router} to find a {@link RouteDecoratingService}.
117      */
118     public static Router<RouteDecoratingService> ofRouteDecoratingService(
119             List<RouteDecoratingService> routeDecoratingServices) {
120         return wrapRouteDecoratingServiceRouter(
121                 defaultRouter(routeDecoratingServices, null,
122                               RouteDecoratingService::route,
123                               (route1, route2) -> {/* noop */}),
124                 resolveAmbiguousRoutes(routeDecoratingServices.stream()
125                                                               .map(RouteDecoratingService::route)
126                                                               .collect(toImmutableList())));
127     }

129     /**
130      * Finds the {@link Route}s that are not unique based on the following properties.
131      * <ul>
132      *     <li>{@link Route#pathType()}</li>
133      *     <li>{@link Route#paths()}</li>
134      *     <li>{@link Route#methods()}</li>
135      *     <li>{@link Route#consumes()}</li>
136      *     <li>{@link Route#produces()}</li>
137      * </ul>
138      */
139     private static Set<Route> resolveAmbiguousRoutes(List<Route> allRoutes) {
140         final Map<List<Object>, List<Route>> dup = new HashMap<>();
141         allRoutes.forEach(route -> {
142             final List<Object> key = ImmutableList.builder()
143                                                   .add(route.pathType())
144                                                   .addAll(route.paths())
145                                                   .addAll(route.methods())
146                                                   .addAll(route.consumes())
147                                                   .addAll(route.produces())
148                                                   .build();
149             dup.computeIfAbsent(key, unused -> new ArrayList<>())
150                .add(route);
151         });
152         return dup.values().stream()
153                   .filter(routes -> routes.size() > 1)  // ambiguous routes
154                   .flatMap(Collection::stream).collect(toImmutableSet());
155     }

157     /**
158      * Returns the default implementation of {@link Router}. It consists of several router implementations
159      * which use one of Trie and List. Consecutive {@link ServiceConfig}s would be grouped according to whether
160      * it is able to produce trie path string or not while traversing the list, then each group would be
161      * transformed to a {@link Router}.
162      */
163     private static <V> Router<V> defaultRouter(Iterable<V> values, @Nullable V fallbackValue,
164                                                Function<V, Route> routeResolver,
165                                                BiConsumer<Route, Route> rejectionHandler) {
166         return new CompositeRouter<>(routers(values, fallbackValue, routeResolver, rejectionHandler),
167                                      Function.identity());
168     }

170     /**
171      * Returns a list of {@link Router}s.
172      */
173     @VisibleForTesting
174     static <V> List<Router<V>> routers(Iterable<V> values, @Nullable V fallbackValue,
175                                        Function<V, Route> routeResolver,
176                                        BiConsumer<Route, Route> rejectionHandler) {
177         rejectDuplicateMapping(values, routeResolver, rejectionHandler);
178 
179         final ImmutableList.Builder<Router<V>> builder = ImmutableList.builder();
180         final List<V> group = new ArrayList<>();
181 
182         boolean addingTrie = true;
183 
184         for (V value : values) {
185             final Route route = routeResolver.apply(value);
186             final boolean hasTriePath = route.pathType().hasTriePath();
187             if (addingTrie && hasTriePath || !addingTrie && !hasTriePath) {
188                 // We are adding the same type of Route to 'group'.
189                 group.add(value);
190                 continue;
191             }
192 
193             // Changed the router type.
194             if (!group.isEmpty()) {
195                 builder.add(router(addingTrie, group, fallbackValue, routeResolver));
196             }
197             addingTrie = !addingTrie;
198             group.add(value);
199         }
200         if (!group.isEmpty()) {
201             builder.add(router(addingTrie, group, fallbackValue, routeResolver));
202         }
203         return builder.build();
204     }

206     private static <V> void rejectDuplicateMapping(
207             Iterable<V> values, Function<V, Route> routeResolver,
208             BiConsumer<Route, Route> rejectionHandler) {
209 
210         final Map<String, List<Route>> triePath2Routes = new HashMap<>();
211         for (V v : values) {
212             final Route route = routeResolver.apply(v);
213             final boolean hasTriePath = route.pathType().hasTriePath();
214             if (!hasTriePath) {
215                 continue;
216             }
217             final String triePath = route.paths().get(1);
218             final List<Route> existingRoutes =
219                     triePath2Routes.computeIfAbsent(triePath, unused -> new ArrayList<>());
220             for (Route existingRoute : existingRoutes) {
221                 if (route.complexity() != existingRoute.complexity()) {
222                     continue;
223                 }
224 
225                 if (route.getClass() != existingRoute.getClass()) {
226                     continue;
227                 }
228 
229                 if (route.complexity() == 0) {
230                     rejectionHandler.accept(route, existingRoute);
231                     return;
232                 }
233 
234                 if (route.methods().stream().noneMatch(
235                         method -> existingRoute.methods().contains(method))) {
236                     // No overlap in supported methods.
237                     continue;
238                 }
239                 if (!route.consumes().isEmpty() &&
240                     route.consumes().stream().noneMatch(
241                             mediaType -> existingRoute.consumes().contains(mediaType))) {
242                     // No overlap in consume types.
243                     continue;
244                 }
245                 if (!route.produces().isEmpty() &&
246                     route.produces().stream().noneMatch(
247                             mediaType -> existingRoute.produces().contains(mediaType))) {
248                     // No overlap in produce types.
249                     continue;
250                 }
251 
252                 rejectionHandler.accept(route, existingRoute);
253                 return;
254             }
255 
256             existingRoutes.add(route);
257         }
258     }

260     /**
261      * Returns a {@link Router} implementation which is using one of {@link RoutingTrie} and {@link List}.
262      */
263     private static <V> Router<V> router(boolean isTrie, List<V> values, @Nullable V fallbackValue,
264                                         Function<V, Route> routeResolver) {
265         final Comparator<V> valueComparator =
266                 Comparator.comparingInt(e -> -1 * routeResolver.apply(e).complexity());
267 
268         final Router<V> router;
269         if (isTrie) {
270             final RoutingTrieBuilder<V> builder = new RoutingTrieBuilder<>();
271             // Set a comparator to sort services by the number of conditions to check in a descending order.
272             builder.comparator(valueComparator);
273             for (V v : values) {
274                 final Route route = routeResolver.apply(v);
275                 builder.add(route.paths().get(1), v);
276 
277                 if (fallbackValue != null) {
278                     // Add an extra route without a trailing slash for a redirect.
279                     // Note that `path.length()` must be greater than 1 because path is `/` when 1.
280                     final String path = route.paths().get(0);
281                     final int pathLen = path.length();
282                     if (pathLen > 1 && path.charAt(pathLen - 1) == '/') {
283                         builder.add(path.substring(0, pathLen - 1),
284                                     fallbackValue, /* hasHighPrecedence */ false);
285                     }
286                 }
287             }
288             router = new TrieRouter<>(builder.build(), routeResolver);
289         } else {
290             values.sort(valueComparator);
291             router = new SequentialRouter<>(values, routeResolver);
292         }
293 
294         if (logger.isDebugEnabled()) {
295             logger.debug("Router created for {} service(s): {}",
296                          values.size(), router.getClass().getSimpleName());
297             for (V v : values) {
298                 final Route route = routeResolver.apply(v);
299                 logger.debug("meterTag: {}, complexity: {}", route.meterTag(), route.complexity());
300             }
301         }
302         values.clear();
303         return router;
304     }

306     /**
307      * Finds the most suitable service from the given {@link ServiceConfig} list.
308      */
309     private static <V> Routed<V> findBest(RoutingContext routingCtx, @Nullable List<V> values,
310                                           Function<V, Route> routeResolver) {
311         Routed<V> result = Routed.empty();
312         if (values != null) {
313             for (V value : values) {
314                 final Route route = routeResolver.apply(value);
315                 final RoutingResult routingResult = route.apply(routingCtx);
316                 if (routingResult.isPresent()) {
317                     //
318                     // The services are sorted as follows:
319                     //
320                     // 1) the service with method and media type negotiation
321                     //    (consumable and producible)
322                     // 2) the service with method and producible media type negotiation
323                     // 3) the service with method and consumable media type negotiation
324                     // 4) the service with method negotiation
325                     // 5) the other services (in a registered order)
326                     //
327                     // 1) and 2) may produce a score between the lowest and the highest because they should
328                     // negotiate the produce type with the value of 'Accept' header.
329                     // 3), 4) and 5) always produces the lowest score.
330                     //
331 
332                     // Found the best matching.
333                     if (routingResult.hasHighestScore()) {
334                         result = Routed.of(route, routingResult, value);
335                         break;
336                     }
337 
338                     // We have still a chance to find a better matching.
339                     if (result.isPresent()) {
340                         if (routingResult.score() > result.routingResult().score()) {
341                             // Replace the candidate with the new one only if the score is better.
342                             // If the score is same, we respect the order of service registration.
343                             result = Routed.of(route, routingResult, value);
344                         }
345                     } else {
346                         // Keep the result as a candidate.
347                         result = Routed.of(route, routingResult, value);
348                     }
349                 }
350             }
351         }
352         return result;
353     }

355     private static <V> List<Routed<V>> findAll(RoutingContext routingCtx, List<V> values,
356                                                Function<V, Route> routeResolver) {
357         final ImmutableList.Builder<Routed<V>> builder = ImmutableList.builderWithExpectedSize(values.size());
358 
359         for (V value : values) {
360             final Route route = routeResolver.apply(value);
361             final RoutingResult routingResult = route.apply(routingCtx);
362             if (routingResult.isPresent()) {
363                 builder.add(Routed.of(route, routingResult, value));
364             }
365         }
366         return builder.build();
367     }

369     private static final class TrieRouter<V> implements Router<V> {
370 
371         private final RoutingTrie<V> trie;
372         private final Function<V, Route> routeResolver;
373 
374         TrieRouter(RoutingTrie<V> trie, Function<V, Route> routeResolver) {
375             this.trie = requireNonNull(trie, "trie");
376             this.routeResolver = requireNonNull(routeResolver, "routeResolver");
377         }
378 
379         @Override
380         public Routed<V> find(RoutingContext routingCtx) {
381             return findBest(routingCtx, trie.find(routingCtx.path()), routeResolver);
382         }
383 
384         @Override
385         public List<Routed<V>> findAll(RoutingContext routingCtx) {
386             return Routers.findAll(routingCtx, trie.findAll(routingCtx.path()), routeResolver);
387         }
388 
389         @Override
390         public void dump(OutputStream output) {
391             trie.dump(output);
392         }
393     }
394 
395     private static final class SequentialRouter<V> implements Router<V> {
396 
397         private final List<V> values;
398         private final Function<V, Route> routeResolver;
399 
400         SequentialRouter(List<V> values, Function<V, Route> routeResolver) {
401             this.values = ImmutableList.copyOf(requireNonNull(values, "values"));
402             this.routeResolver = requireNonNull(routeResolver, "routeResolver");
403         }
404 
405         @Override
406         public Routed<V> find(RoutingContext routingCtx) {
407             return findBest(routingCtx, values, routeResolver);
408         }
409 
410         @Override
411         public List<Routed<V>> findAll(RoutingContext routingCtx) {
412             return Routers.findAll(routingCtx, values, routeResolver);
413         }
414 
415         @Override
416         public void dump(OutputStream output) {
417             // Do not close this writer in order to keep output stream open.
418             final PrintWriter p = new PrintWriter(new OutputStreamWriter(output, StandardCharsets.UTF_8));
419             p.printf("Dump of %s:%n", this);
420             for (int i = 0; i < values.size(); i++) {
421                 p.printf("<%d> %s%n", i, values.get(i));
422             }
423             p.flush();
424         }
425     }
