36  /**
37   * An {@link HttpService} that sends a redirect response such as {@code "307 Temporary Redirect"}.
38   * You have to specify a template or a {@link Function} that generates the value of the {@code "Location"}
39   * header.
40   *
41   * <h3>Using a location template</h3>
42   *
43   * <p>You can choose one of the following template styles where the path parameters are substituted with
44   * the values retrieved from {@link ServiceRequestContext#pathParam(String)}:</p>
45   *
46   * <ul>
47   *   <li>{@code /new} (no path parameters)
48   *     <pre>{@code
49   *     ServerBuilder sb = Server.builder();
50   *     // e.g. /old -> /new
51   *     sb.service("/old", new RedirectService("/new");
52   *     }</pre>
53   *   </li>
54   *   <li>{@code /new/{var}} (curly-brace style parameters)
55   *     <pre>{@code
56   *     // e.g. /old/foo -> /new/foo
57   *     sb.service("/old/{var}", new RedirectService("/new/{var}");
58   *     }</pre>
59   *   </li>
60   *   <li>{@code /new/:var1/:var2} (colon style parameters)
61   *     <pre>{@code
62   *     // e.g. /old/foo/bar -> /new/foo/bar
63   *     sb.service("/old/:var1/:var2", new RedirectService("/new/:var1/:var2"));
64   *     }</pre>
65   *   </li>
66   *   <li>{@code http://host/new} (full URL without path parameters)
67   *     <pre>{@code
68   *     // e.g. /old -> http://host/new
69   *     sb.service("/old", new RedirectService("http://host/new"));
70   *     }</pre>
71   *   </li>
72   *   <li>{@code http://host/new/{var}} (full URL with curly-brace style parameters)
73   *     <pre>{@code
74   *     // e.g. /old/foo -> http://host/new/foo
75   *     sb.service("/old/{var}", new RedirectService("http://host/new/{var}"));
76   *     }</pre>
77   *   </li>
78   *   <li>{@code http://host/new/:var1/:var2} (full URL with colon style parameters)
79   *     <pre>{@code
80   *     // e.g. /old/foo/bar -> http://host/new/foo/bar
81   *     sb.service("/old/:var1/:var2", new RedirectService("http://host/new/:var1/:var2"));
82   *     }</pre>
83   *   </li>
84   * </ul>
85   *
86   * <h3>Using a location function</h3>
87   *
88   * <p>You can also specify a custom function to generate a location which cannot be generated with a location
89   * template:</p>
90   *
91   * <pre>{@code
92   * ServerBuilder sb = Server.builder();
93   * // e.g. /foo/bar -> /NNNNNN/foo_bar
94   * sb.service("/:var1/:var2", new RedirectService(ctx -> {
95   *     String name = ctx.pathParam("var1") + "_" + ctx.pathParam("var2");
96   *     return String.format("/%d/%s", name.hashCode(), name);
97   * });
98   * }</pre>
99   *
100  * <h3>Specifying an alternative status code</h3>
101  *
102  * <p>By default, {@link RedirectService} responds with {@link HttpStatus#TEMPORARY_REDIRECT 307 Temporary
103  * Redirect} status. You can specify alternative status such as {@link HttpStatus#MOVED_PERMANENTLY 301 Moved
104  * Permanently} when calling the constructor.</p>
105  *
106  * <h3>Preserving a query string (or not)</h3>
107  *
108  * <p>By default, {@link RedirectService} preserves the query string in the request URI when generating
109  * a new location. For example, if you bound {@code new RedirectService("/new")} at {@code "/old"},
110  * a request to {@code "/old?foo=bar"} will be redirected to {@code "/new?foo=bar"}. You can disable
111  * this behavior by specifying {@code false} for the {@code preserveQueryString} parameter
112  * when constructing the service.</p>
113  *
114  * <p>Note that {@link RedirectService} will not append the query string if the generated location already
115  * contains a query string, regardless of the {@code preserveQueryString} parameter value. For example,
116  * the following location function never preserves the original query string:</p>
117  *
118  * <pre>{@code
119  * ServiceBuilder sb = Server.builder();
120  * // /old?foo=bar -> /new?redirected=1 (?foo=bar is ignored.)
121  * sb.service("/old", new RedirectService("/new?redirected=1"));
122  * }</pre>
123  */
124 public final class RedirectService extends AbstractHttpService {
125 
126     private static final Pattern VALID_DEFAULT_URI_PATTERN = Pattern
127             .compile("(?:(?:^https?:/{2}(?:([^:]+:)?[^:@]+@)?[^:]+)(?::[0-9]{1,5})?)?" +
128                     "(?:/[^/{}:]+|/:[^/{}]+|/\\{[^/{}]+})+/?");
129     private static final Pattern PATTERN_PARAMS_START = Pattern.compile("/:|/\\{");
130 
131     private final HttpStatus httpStatus;
132     private final Function<? super ServiceRequestContext, String> locationFunction;
133     private final boolean preserveQueryString;
134 
135     @Nullable
136     private Set<String> paramNames;
137 
138     /**
139      * Creates a new instance that redirects to the location constructed with the specified
140      * {@code locationPattern}, preserving the query string in the request URI.
141      *
142      * @param locationPattern the location pattern that is used to generate a redirect location.
143      *
144      * @throws IllegalArgumentException if the specified {@code locationPattern} is unsupported or invalid
145      */
146     public RedirectService(String locationPattern) {
147         this(locationPattern, true);
148     }
149 
150     /**
151      * Creates a new instance that redirects to the location constructed with the specified
152      * {@code locationPattern}.
153      *
154      * @param locationPattern the location pattern that is used to generate a redirect location.
155      * @param preserveQueryString whether to preserve the query string in the generated redirect location.
156      *
157      * @throws IllegalArgumentException if the specified {@code locationPattern} is unsupported or invalid
158      */
159     public RedirectService(String locationPattern, boolean preserveQueryString) {
160         this(HttpStatus.TEMPORARY_REDIRECT, locationPattern, preserveQueryString);
161     }
162 
163     /**
164      * Creates a new instance that redirects to the location returned by {@code locationFunction},
165      * preserving the query string in the request URI.
166      *
167      * @param locationFunction a {@link Function} that takes a {@link ServiceRequestContext}
168      *                         and returns a new location.
169      */
170     public RedirectService(Function<? super ServiceRequestContext, String> locationFunction) {
171         this(locationFunction, true);
172     }
173 
174     /**
175      * Creates a new instance that redirects to the location returned by {@code locationFunction}.
176      *
177      * @param locationFunction a {@link Function} that takes a {@link ServiceRequestContext}
178      *                         and returns a new location.
179      * @param preserveQueryString whether to preserve the query string in the generated redirect location.
180      */
181     public RedirectService(Function<? super ServiceRequestContext, String> locationFunction,
182                            boolean preserveQueryString) {
183         this(HttpStatus.TEMPORARY_REDIRECT, locationFunction, preserveQueryString);
184     }
185 
186     /**
187      * Creates a new instance that redirects to the location constructed with the specified
188      * {@code locationPattern}, preserving the query string in the request URI.
189      *
190      * @param redirectStatus the {@link HttpStatus} that the {@link HttpService} will return.
191      * @param locationPattern the location pattern that is used to generate a redirect location.
192      *
193      * @throws IllegalArgumentException if the specified {@code locationPattern} is unsupported or invalid
194      */
195     public RedirectService(HttpStatus redirectStatus, String locationPattern) {
196         this(redirectStatus, locationPattern, true);
197     }
198 
199     /**
200      * Creates a new instance that redirects to the location constructed with the specified
201      * {@code locationPattern}.
202      *
203      * @param redirectStatus the {@link HttpStatus} that the {@link HttpService} will return.
204      * @param locationPattern the location pattern that is used to generate a redirect location.
205      * @param preserveQueryString whether to preserve the query string in the generated redirect location.
206      *
207      * @throws IllegalArgumentException if the specified {@code locationPattern} is unsupported or invalid
208      */
209     public RedirectService(HttpStatus redirectStatus, String locationPattern, boolean preserveQueryString) {
210         this(redirectStatus, toLocationFunction(locationPattern), preserveQueryString);
211 
212         final Matcher m = PATTERN_PARAMS_START.matcher(locationPattern);
213         if (m.find()) {
214             paramNames = Route.builder().path(locationPattern.substring(m.start())).build().paramNames();
215         }
216     }
217 
218     /**
219      * Creates a new instance that redirects to the location returned by {@code locationFunction},
220      * preserving the query string in the request URI.
221      *
222      * @param redirectStatus the {@link HttpStatus} that the {@link HttpService} will return.
223      * @param locationFunction a {@link Function} that takes a {@link ServiceRequestContext}
224      *                         and returns a new location.
225      */
226     public RedirectService(HttpStatus redirectStatus,
227                            Function<? super ServiceRequestContext, String> locationFunction) {
228         this(redirectStatus, locationFunction, true);
229     }
230 
231     /**
232      * Creates a new instance that redirects to the location returned by {@code locationFunction}.
233      *
234      * @param redirectStatus the {@link HttpStatus} that the {@link HttpService} will return.
235      * @param locationFunction a {@link Function} that takes a {@link ServiceRequestContext}
236      *                         and returns a new location.
237      * @param preserveQueryString whether to preserve the query string in the generated redirect location.
238      */
239     public RedirectService(HttpStatus redirectStatus,
240                            Function<? super ServiceRequestContext, String> locationFunction,
241                            boolean preserveQueryString) {
242         requireNonNull(redirectStatus, "redirectStatus");
243         requireNonNull(locationFunction, "locationFunction");
244         if (redirectStatus.compareTo(HttpStatus.MULTIPLE_CHOICES) < 0 ||
245             redirectStatus.compareTo(HttpStatus.TEMPORARY_REDIRECT) > 0) {
246             throw new IllegalArgumentException("redirectStatus: " + redirectStatus + " (expected: 300 .. 307)");
247         }
248         httpStatus = redirectStatus;
249         this.locationFunction = locationFunction;
250         this.preserveQueryString = preserveQueryString;
251     }
252 
253     /**
254      * NB: For now we redirect all methods.
255      */
256     @Override
257     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
258         String location = locationFunction.apply(ctx);
259         requireNonNull(location, "locationFunction returned null.");
260 
261         if (preserveQueryString) {
262             location = appendQueryString(ctx, location);
263         }
264 
265         return HttpResponse.of(ResponseHeaders.of(httpStatus,
266                                                   HttpHeaderNames.LOCATION, location));
267     }
268 
269     private static String appendQueryString(ServiceRequestContext ctx, String location) {
270         final String query = ctx.query();
271         if (query == null || location.lastIndexOf('?') >= 0) {
272             // The request URI does not have a query string or
273             // The new location includes a query string already.
274             return location;
275         }
276 
277         return location + '?' + query;
278     }
279 
280     @Override
281     public void serviceAdded(ServiceConfig cfg) throws Exception {
282         if (paramNames != null) {
283             final Set<String> params = cfg.route().paramNames();
284 
285             // Find out if old path and new path are compatible
286             for (String param : paramNames) {
287                 if (!params.contains(param)) {
288                     throw new IllegalArgumentException("pathParams: " + param + " (no matching param in " +
289                                                        params + ')');
290                 }
291             }
292             // We don't need the paramNames anymore
293             paramNames = null;
294         }
295 
296         super.serviceAdded(cfg);
297     }
298 
299     private static Function<? super ServiceRequestContext, String> toLocationFunction(String locationPattern) {
300         requireNonNull(locationPattern, "locationPattern");
301         if (!isDefaultUriPattern(locationPattern)) {
302             throw new IllegalArgumentException("locationPattern: " + locationPattern);
303         }
304         return ctx -> populatePatternParams(locationPattern, ctx.pathParams());
305     }
306 
307     private static String populatePatternParams(String pathPattern, Map<String, String> pathParams) {
308         for (Entry<String, String> e : pathParams.entrySet()) {
309             @SuppressWarnings("RegExpRedundantEscape")
310             final String tokenPattern = "\\{" + e.getKey() + "\\}|:" + e.getKey();
311             pathPattern = pathPattern.replaceAll(tokenPattern, e.getValue());
312         }
313         return pathPattern;
314     }
315 
316     private static boolean isDefaultUriPattern(String pathPattern) {
317         return VALID_DEFAULT_URI_PATTERN.matcher(pathPattern).matches();
318     }
