17  package com.linecorp.armeria.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.List;
22  
23  import javax.annotation.Nullable;
24  
25  import com.linecorp.armeria.common.HttpHeaderNames;
26  import com.linecorp.armeria.common.HttpMethod;
27  import com.linecorp.armeria.common.HttpRequest;
28  import com.linecorp.armeria.common.MediaType;
29  import com.linecorp.armeria.common.QueryParams;
30  import com.linecorp.armeria.common.RequestHeaders;
31  import com.linecorp.armeria.internal.common.ArmeriaHttpUtil;
32  
33  /**
34   * Holds the parameters which are required to find a service available to handle the request.
35   */
36  public interface RoutingContext {
37  
38      /**
39       * Returns the {@link VirtualHost} instance which belongs to this {@link RoutingContext}.
40       */
41      VirtualHost virtualHost();
42  
43      /**
44       * Returns the virtual host name of the request.
45       */
46      String hostname();
47  
48      /**
49       * Returns {@link HttpMethod} of the request.
50       */
51      HttpMethod method();
52  
53      /**
54       * Returns the absolute path retrieved from the request,
55       * as defined in <a href="https://tools.ietf.org/html/rfc3986">RFC3986</a>.
56       */
57      String path();
58  
59      /**
60       * Returns the query retrieved from the request,
61       * as defined in <a href="https://tools.ietf.org/html/rfc3986">RFC3986</a>.
62       */
63      @Nullable
64      String query();
65  
66      /**
67       * Returns the query parameters retrieved from the request path.
68       */
69      QueryParams params();
70  
71      /**
72       * Returns {@link MediaType} specified by 'Content-Type' header of the request.
73       */
74      @Nullable
75      MediaType contentType();
76  
77      /**
78       * Returns a list of {@link MediaType}s that are specified in {@link HttpHeaderNames#ACCEPT} in the order
79       * of client-side preferences. If the client does not send the header, this will contain only
80       * {@link MediaType#ANY_TYPE}.
81       */
82      List<MediaType> acceptTypes();
83  
84      /**
85       * Returns the {@link RequestHeaders} retrieved from the request.
86       */
87      RequestHeaders headers();
88  
89      /**
90       * Defers throwing an {@link HttpStatusException} until reaching the end of the service list.
91       */
92      void deferStatusException(HttpStatusException cause);
93  
94      /**
95       * Returns a deferred {@link HttpStatusException} which was previously set via
96       * {@link #deferStatusException(HttpStatusException)}.
97       */
98      @Nullable
99      HttpStatusException deferredStatusException();
100 
101     /**
102      * Returns a wrapped {@link RoutingContext} which holds the specified {@code path}.
103      * It is usually used to find an {@link HttpService} with a prefix-stripped path.
104      */
105     default RoutingContext overridePath(String path) {
106         requireNonNull(path, "path");
107         return new RoutingContextWrapper(this) {
108             @Override
109             public String path() {
110                 return path;
111             }
112         };
113     }
114 
115     /**
116      * Returns {@code true} if this context is for a CORS preflight request.
117      *
118      * @see ArmeriaHttpUtil#isCorsPreflightRequest(HttpRequest)
119      */
120     boolean isCorsPreflight();
121 
122     /**
123      * Returns {@code true} if this context requires matching the predicates for query parameters.
124      *
125      * @see RouteBuilder#matchesParams(Iterable)
126      */
127     default boolean requiresMatchingParamsPredicates() {
128         return true;
129     }
130 
131     /**
132      * Returns {@code true} if this context requires matching the predicates for HTTP headers.
133      *
134      * @see RouteBuilder#matchesHeaders(Iterable)
135      */
136     default boolean requiresMatchingHeadersPredicates() {
137         return true;
138     }
139 }
