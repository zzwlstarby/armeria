56      DefaultRoute(PathMapping pathMapping, Set<HttpMethod> methods,
57                   Set<MediaType> consumes, Set<MediaType> produces,
58                   List<RoutingPredicate<QueryParams>> paramPredicates,
59                   List<RoutingPredicate<HttpHeaders>> headerPredicates) {
60          this.pathMapping = requireNonNull(pathMapping, "pathMapping");
61          this.methods = Sets.immutableEnumSet(requireNonNull(methods, "methods"));
62          this.consumes = ImmutableSet.copyOf(requireNonNull(consumes, "consumes"));
63          this.produces = ImmutableSet.copyOf(requireNonNull(produces, "produces"));
64          this.paramPredicates = ImmutableList.copyOf(requireNonNull(paramPredicates, "paramPredicates"));
65          this.headerPredicates = ImmutableList.copyOf(requireNonNull(headerPredicates, "headerPredicates"));
66  
67          loggerName = generateLoggerName(pathMapping.loggerName(), methods, consumes, produces,
68                                          paramPredicates, headerPredicates);
69  
70          meterTag = generateMeterTag(pathMapping.meterTag(), methods, consumes, produces,
71                                      paramPredicates, headerPredicates);
72  
73          int complexity = 0;
74          if (!methods.isEmpty()) {
75              complexity += 1;
76          }
77          if (!consumes.isEmpty()) {
78              complexity += 1 << 1;
79          }
80          if (!produces.isEmpty()) {
81              complexity += 1 << 2;
82          }
83          if (!paramPredicates.isEmpty()) {
84              complexity += 1 << 3;
85          }
86          if (!headerPredicates.isEmpty()) {
87              complexity += 1 << 4;
88          }
89          this.complexity = complexity;
90      }

92      @Override
93      public RoutingResult apply(RoutingContext routingCtx) {
94          final RoutingResultBuilder builder = pathMapping.apply(requireNonNull(routingCtx, "routingCtx"));
95          if (builder == null) {
96              return RoutingResult.empty();
97          }
98  
99          if (!methods.isEmpty() && !methods.contains(routingCtx.method())) {
100             // '415 Unsupported Media Type' and '406 Not Acceptable' is more specific than
101             // '405 Method Not Allowed'. So 405 would be set if there is no status code set before.
102             if (routingCtx.deferredStatusException() == null) {
103                 routingCtx.deferStatusException(HttpStatusException.of(HttpStatus.METHOD_NOT_ALLOWED));
104             }
105 
106             return emptyOrCorsPreflightResult(routingCtx, builder);
107         }
108 
109         final MediaType contentType = routingCtx.contentType();
110         boolean contentTypeMatched = false;
111         if (contentType == null) {
112             if (consumes.isEmpty()) {
113                 contentTypeMatched = true;
114             }
115         } else if (!consumes.isEmpty()) {
116             for (MediaType consumeType : consumes) {
117                 contentTypeMatched = contentType.belongsTo(consumeType);
118                 if (contentTypeMatched) {
119                     break;
120                 }
121             }
122             if (!contentTypeMatched) {
123                 routingCtx.deferStatusException(HttpStatusException.of(HttpStatus.UNSUPPORTED_MEDIA_TYPE));
124                 return emptyOrCorsPreflightResult(routingCtx, builder);
125             }
126         }
127 
128         final List<MediaType> acceptTypes = routingCtx.acceptTypes();
129         if (acceptTypes.isEmpty()) {
130             if (contentTypeMatched && produces.isEmpty()) {
131                 builder.score(HIGHEST_SCORE);
132             }
133             for (MediaType produceType : produces) {
134                 if (!isAnyType(produceType)) {
135                     builder.negotiatedResponseMediaType(produceType);
136                     break;
137                 }
138             }
139         } else if (!produces.isEmpty()) {
140             boolean found = false;
141             for (MediaType produceType : produces) {
142                 for (int i = 0; i < acceptTypes.size(); i++) {
143                     final MediaType acceptType = acceptTypes.get(i);
144                     if (produceType.belongsTo(acceptType)) {
145                         // To early stop path mapping traversal,
146                         // we set the score as the best score when the index is 0.
147 
148                         final int score = i == 0 ? HIGHEST_SCORE : -1 * i;
149                         builder.score(score);
150                         if (!isAnyType(produceType)) {
151                             builder.negotiatedResponseMediaType(produceType);
152                         }
153                         found = true;
154                         break;
155                     }
156                 }
157                 if (found) {
158                     break;
159                 }
160             }
161             if (!found) {
162                 routingCtx.deferStatusException(HttpStatusException.of(HttpStatus.NOT_ACCEPTABLE));
163                 return emptyOrCorsPreflightResult(routingCtx, builder);
164             }
165         }
166 
167         if (routingCtx.requiresMatchingParamsPredicates()) {
168             if (!paramPredicates.isEmpty() &&
169                 !paramPredicates.stream().allMatch(p -> p.test(routingCtx.params()))) {
170                 return RoutingResult.empty();
171             }
172         }
173         if (routingCtx.requiresMatchingHeadersPredicates()) {
174             if (!headerPredicates.isEmpty() &&
175                 !headerPredicates.stream().allMatch(p -> p.test(routingCtx.headers()))) {
176                 return RoutingResult.empty();
177             }
178         }
179         return builder.build();
180     }
181 
182     private static RoutingResult emptyOrCorsPreflightResult(RoutingContext routingCtx,
183                                                             RoutingResultBuilder builder) {
184         if (routingCtx.isCorsPreflight()) {
185             return builder.type(RoutingResultType.CORS_PREFLIGHT).build();
186         }
187 
188         return RoutingResult.empty();
189     }

191     private static boolean isAnyType(MediaType contentType) {
192         // Ignores all parameters including the quality factor.
193         return "*".equals(contentType.type()) || "*".equals(contentType.subtype());
194     }

241     private static String generateLoggerName(String prefix, Set<HttpMethod> methods,
242                                              Set<MediaType> consumes, Set<MediaType> produces,
243                                              List<RoutingPredicate<QueryParams>> paramPredicates,
244                                              List<RoutingPredicate<HttpHeaders>> headerPredicates) {
245         final StringJoiner name = new StringJoiner(".");
246         name.add(prefix);
247         if (!methods.isEmpty()) {
248             name.add(loggerNameJoiner.join(methods.stream().sorted().iterator()));
249         }
250 
251         // The following three cases should be different to each other.
252         // Each name would be produced as follows:
253         //
254         // consumes: text/plain, text/html               -> consumes.text_plain.text_html
255         // consumes: text/plain, produces: text/html -> consumes.text_plain.produces.text_html
256         // produces: text/plain, text/html               -> produces.text_plain.text_html
257 
258         if (!consumes.isEmpty()) {
259             name.add("consumes");
260             consumes.forEach(e -> name.add(e.type() + '_' + e.subtype()));
261         }
262         if (!produces.isEmpty()) {
263             name.add("produces");
264             produces.forEach(e -> name.add(e.type() + '_' + e.subtype()));
265         }
266         if (!paramPredicates.isEmpty()) {
267             name.add("params");
268             name.add(loggerNameJoiner.join(
269                     paramPredicates.stream().map(RoutingPredicate::name).sorted().distinct().iterator()));
270         }
271         if (!headerPredicates.isEmpty()) {
272             name.add("headers");
273             name.add(loggerNameJoiner.join(
274                     headerPredicates.stream().map(RoutingPredicate::name).sorted().distinct().iterator()));
275         }
276         return name.toString();
277     }

279     private static String generateMeterTag(String parentTag, Set<HttpMethod> methods,
280                                            Set<MediaType> consumes, Set<MediaType> produces,
281                                            List<RoutingPredicate<QueryParams>> paramPredicates,
282                                            List<RoutingPredicate<HttpHeaders>> headerPredicates) {
283 
284         final StringJoiner name = new StringJoiner(",");
285         name.add(parentTag);
286         if (!methods.isEmpty()) {
287             name.add("methods:" + meterTagJoiner.join(methods.stream().sorted().iterator()));
288         }
289 
290         // The following three cases should be different to each other.
291         // Each name would be produced as follows:
292         //
293         // consumes: text/plain, text/html               -> "consumes:text/plain,text/html"
294         // consumes: text/plain, produces: text/html -> "consumes:text/plain,produces:text/html"
295         // produces: text/plain, text/html               -> "produces:text/plain,text/html"
296 
297         addMediaTypes(name, "consumes", consumes);
298         addMediaTypes(name, "produces", produces);
299 
300         if (!paramPredicates.isEmpty()) {
301             name.add("params");
302             name.add(meterTagJoiner.join(
303                     paramPredicates.stream().map(RoutingPredicate::name).sorted().distinct().iterator()));
304         }
305         if (!headerPredicates.isEmpty()) {
306             name.add("headers");
307             name.add(meterTagJoiner.join(
308                     headerPredicates.stream().map(RoutingPredicate::name).sorted().distinct().iterator()));
309         }
310         return name.toString();
311     }
312 
313     private static void addMediaTypes(StringJoiner builder, String prefix, Set<MediaType> mediaTypes) {
314         if (!mediaTypes.isEmpty()) {
315             final StringBuilder buf = new StringBuilder();
316             buf.append(prefix).append(':');
317             for (MediaType t : mediaTypes) {
318                 buf.append(t.type());
319                 buf.append('/');
320                 buf.append(t.subtype());
321                 buf.append(',');
322             }
323             buf.setLength(buf.length() - 1);
324             builder.add(buf.toString());
325         }
326     }

333     @Override
334     public boolean equals(@Nullable Object o) {
335         if (this == o) {
336             return true;
337         }
338 
339         if (!(o instanceof DefaultRoute)) {
340             return false;
341         }
342 
343         final DefaultRoute that = (DefaultRoute) o;
344         return Objects.equals(pathMapping, that.pathMapping) &&
345                methods.equals(that.methods) &&
346                consumes.equals(that.consumes) &&
347                produces.equals(that.produces) &&
348                headerPredicates.equals(that.headerPredicates) &&
349                paramPredicates.equals(that.paramPredicates);
350     }
