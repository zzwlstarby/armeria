16  package com.linecorp.armeria.server.healthcheck;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.io.IOException;
21  import java.nio.charset.Charset;
22  import java.nio.charset.StandardCharsets;
23  import java.util.concurrent.CompletionStage;
24  
25  import com.fasterxml.jackson.databind.JsonNode;
26  import com.fasterxml.jackson.databind.ObjectMapper;
27  import com.fasterxml.jackson.databind.node.JsonNodeType;
28  
29  import com.linecorp.armeria.common.AggregatedHttpRequest;
30  import com.linecorp.armeria.common.HttpRequest;
31  import com.linecorp.armeria.common.HttpStatus;
32  import com.linecorp.armeria.common.MediaType;
33  import com.linecorp.armeria.server.HttpStatusException;
34  import com.linecorp.armeria.server.ServiceRequestContext;
35  
36  final class DefaultHealthCheckUpdateHandler implements HealthCheckUpdateHandler {
37  
38      private static final ObjectMapper mapper = new ObjectMapper();
39  
40      static final DefaultHealthCheckUpdateHandler INSTANCE = new DefaultHealthCheckUpdateHandler();
41  
42      private DefaultHealthCheckUpdateHandler() {}
43  
44      @Override
45      public CompletionStage<HealthCheckUpdateResult> handle(ServiceRequestContext ctx,
46                                                             HttpRequest req) throws Exception {
47          requireNonNull(req, "req");
48          switch (req.method()) {
49              case PUT:
50              case POST:
51                  return req.aggregate().thenApply(DefaultHealthCheckUpdateHandler::handlePut);
52              case PATCH:
53                  return req.aggregate().thenApply(DefaultHealthCheckUpdateHandler::handlePatch);
54              default:
55                  throw HttpStatusException.of(HttpStatus.METHOD_NOT_ALLOWED);
56          }
57      }
58  
59      private static HealthCheckUpdateResult handlePut(AggregatedHttpRequest req) {
60          final JsonNode json = toJsonNode(req);
61          if (json.getNodeType() != JsonNodeType.OBJECT) {
62              throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
63          }
64  
65          final JsonNode healthy = json.get("healthy");
66          if (healthy == null) {
67              throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
68          }
69          if (healthy.getNodeType() != JsonNodeType.BOOLEAN) {
70              throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
71          }
72  
73          return healthy.booleanValue() ? HealthCheckUpdateResult.HEALTHY
74                                        : HealthCheckUpdateResult.UNHEALTHY;
75      }
76  
77      private static HealthCheckUpdateResult handlePatch(AggregatedHttpRequest req) {
78          final JsonNode json = toJsonNode(req);
79          if (json.getNodeType() != JsonNodeType.ARRAY ||
80              json.size() != 1) {
81              throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
82          }
83  
84          final JsonNode patchCommand = json.get(0);
85          final JsonNode op = patchCommand.get("op");
86          final JsonNode path = patchCommand.get("path");
87          final JsonNode value = patchCommand.get("value");
88          if (op == null || path == null || value == null ||
89              !"replace".equals(op.textValue()) ||
90              !"/healthy".equals(path.textValue()) ||
91              !value.isBoolean()) {
92              throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
93          }
94  
95          return value.booleanValue() ? HealthCheckUpdateResult.HEALTHY
96                                      : HealthCheckUpdateResult.UNHEALTHY;
97      }
98  
99      private static JsonNode toJsonNode(AggregatedHttpRequest req) {
100         final MediaType contentType = req.contentType();
101         if (contentType != null && !contentType.is(MediaType.JSON)) {
102             throw HttpStatusException.of(HttpStatus.UNSUPPORTED_MEDIA_TYPE);
103         }
104 
105         final Charset charset = contentType == null ? StandardCharsets.UTF_8
106                                                     : contentType.charset(StandardCharsets.UTF_8);
107         try {
108             return StandardCharsets.UTF_8.equals(charset) ? mapper.readTree(req.content().array())
109                                                           : mapper.readTree(req.content(charset));
110         } catch (IOException e) {
111             throw HttpStatusException.of(HttpStatus.BAD_REQUEST);
112         }
113     }
114 }
