54  /**
55   * An access log component to generate a log message.
56   */
57  @FunctionalInterface
58  interface AccessLogComponent {
59  
60      /**
61       * Returns a part of a log message.
62       */
63      @Nullable
64      Object getMessage(RequestLog log);
65  
66      /**
67       * Returns whether adding quotes between a log message.
68       */
69      default boolean addQuote() {
70          return false;
71      }
72  
73      static AccessLogComponent ofText(String text) {
74          return new TextComponent(text);
75      }
76  
77      static AccessLogComponent ofDefaultRequestTimestamp() {
78          return new TimestampComponent(false, null);
79      }
80  
81      static AccessLogComponent ofPredefinedCommon(AccessLogType type) {
82          return ofPredefinedCommon(type, null);
83      }
84  
85      static AccessLogComponent ofPredefinedCommon(AccessLogType type, @Nullable String variable) {
86          return new CommonComponent(type, type == AccessLogType.REQUEST_LINE, null, variable);
87      }
88  
89      static AccessLogComponent ofQuotedRequestHeader(CharSequence headerName) {
90          return new HttpHeaderComponent(AccessLogType.REQUEST_HEADER,
91                                         headerName, true, null);
92      }
93  
94      /**
95       * A text component of a log message.
96       */
97      class TextComponent implements AccessLogComponent {
98  
99          static boolean isSupported(AccessLogType type) {
100             return type == AccessLogType.TEXT;
101         }
102 
103         private final String text;
104 
105         TextComponent(String text) {
106             this.text = requireNonNull(text, "text");
107         }
108 
109         @Override
110         public Object getMessage(RequestLog log) {
111             return text;
112         }
113     }

115     /**
116      * A timestamp component of a log message.
117      */
118     class TimestampComponent implements AccessLogComponent {
119 
120         @VisibleForTesting
121         static final DateTimeFormatter defaultDateTimeFormatter =
122                 DateTimeFormatter.ofPattern("dd/MMM/yyyy:HH:mm:ss Z", Locale.ENGLISH);
123         @VisibleForTesting
124         static final ZoneId defaultZoneId = ZoneId.systemDefault();
125 
126         private static final Map<String, DateTimeFormatter> predefinedFormatters =
127                 getFields(DateTimeFormatter.class, field -> {
128                     if (field == null) {
129                         return false;
130                     } else {
131                         final int m = field.getModifiers();
132                         // public static final DateTimeFormatter ...
133                         return Modifier.isPublic(m) && Modifier.isStatic(m) && Modifier.isFinal(m) &&
134                                (field.getType() == DateTimeFormatter.class);
135                     }
136                 }).stream().collect(Collectors.toMap(Field::getName, f -> {
137                     try {
138                         return (DateTimeFormatter) f.get(null);
139                     } catch (Throwable cause) {
140                         // Should not reach here.
141                         throw new Error(cause);
142                     }
143                 }));
144 
145         static boolean isSupported(AccessLogType type) {
146             return type == AccessLogType.REQUEST_TIMESTAMP;
147         }
148 
149         private final boolean addQuote;
150         private final DateTimeFormatter formatter;
151 
152         TimestampComponent(boolean addQuote, @Nullable String variable) {
153             this.addQuote = addQuote;
154             formatter = findFormatter(variable);
155         }
156 
157         @Nullable
158         @Override
159         public Object getMessage(RequestLog log) {
160             return formatter.format(ZonedDateTime.ofInstant(
161                     Instant.ofEpochMilli(log.requestStartTimeMillis()), defaultZoneId));
162         }
163 
164         @Override
165         public boolean addQuote() {
166             return addQuote;
167         }
168 
169         static DateTimeFormatter findFormatter(@Nullable String variable) {
170             if (variable == null) {
171                 return defaultDateTimeFormatter;
172             }
173 
174             final DateTimeFormatter predefined = predefinedFormatters.get(variable);
175             if (predefined != null) {
176                 return predefined;
177             }
178 
179             // User-defined pattern.
180             try {
181                 return DateTimeFormatter.ofPattern(variable, Locale.ENGLISH);
182             } catch (Throwable cause) {
183                 throw new IllegalArgumentException("unexpected date/time format variable: " +
184                                                    variable, cause);
185             }
186         }
187     }

189     /**
190      * An abstract class to support a condition whether a log component is applied or not.
191      */
192     abstract class ResponseHeaderConditional implements AccessLogComponent {
193 
194         @Nullable
195         private final Function<ResponseHeaders, Boolean> condition;
196         private final boolean addQuote;
197 
198         protected ResponseHeaderConditional(@Nullable Function<ResponseHeaders, Boolean> condition,
199                                             boolean addQuote) {
200             this.condition = condition;
201             this.addQuote = addQuote;
202         }
203 
204         @Nullable
205         @VisibleForTesting
206         Function<ResponseHeaders, Boolean> condition() {
207             return condition;
208         }
209 
210         @Nullable
211         @Override
212         public final Object getMessage(RequestLog log) {
213             if (condition != null &&
214                 !condition.apply(log.responseHeaders())) {
215                 return null;
216             }
217             return getMessage0(log);
218         }
219 
220         @Nullable
221         abstract Object getMessage0(RequestLog log);
222 
223         @Override
224         public boolean addQuote() {
225             return addQuote;
226         }
227     }

229     /**
230      * Common log components of a log message.
231      */
232     class CommonComponent extends ResponseHeaderConditional {
233 
234         static boolean isSupported(AccessLogType type) {
235             switch (type) {
236                 case LOCAL_IP_ADDRESS:
237                 case REMOTE_IP_ADDRESS:
238                 case REMOTE_HOST:
239                 case RFC931:
240                 case AUTHENTICATED_USER:
241                 case REQUEST_LINE:
242                 case RESPONSE_STATUS_CODE:
243                 case RESPONSE_LENGTH:
244                 case REQUEST_ID:
245                     return true;
246                 default:
247                     return false;
248             }
249         }
250 
251         private final AccessLogType type;
252         @Nullable
253         private final String variable;
254 
255         CommonComponent(AccessLogType type, boolean addQuote,
256                         @Nullable Function<ResponseHeaders, Boolean> condition,
257                         @Nullable String variable) {
258             super(condition, addQuote);
259             checkArgument(isSupported(requireNonNull(type, "type")),
260                           "Type '%s' is not acceptable by %s", type, CommonComponent.class.getName());
261             this.type = type;
262             this.variable = variable;
263         }
264 
265         @Nullable
266         @Override
267         public Object getMessage0(RequestLog log) {
268             switch (type) {
269                 case LOCAL_IP_ADDRESS:
270                     final InetSocketAddress local = log.context().localAddress();
271                     return local == null || local.isUnresolved() ? null : local.getAddress().getHostAddress();
272 
273                 case REMOTE_IP_ADDRESS:
274                     if ("c".equals(variable)) {
275                         // %{c}a means the remote address of the underlying channel.
276                         final InetSocketAddress remote = log.context().remoteAddress();
277                         return remote == null || remote.isUnresolved() ? null
278                                                                        : remote.getAddress().getHostAddress();
279                     } else {
280                         // %a means the client address who initiated a request.
281                         final ServiceRequestContext ctx = (ServiceRequestContext) log.context();
282                         return ctx.clientAddress().getHostAddress();
283                     }
284                 case REMOTE_HOST:
285                     final SocketAddress ra = log.context().remoteAddress();
286                     return ra instanceof InetSocketAddress ? ((InetSocketAddress) ra).getHostString() : null;
287 
288                 case RFC931:
289                 case AUTHENTICATED_USER:
290                     // We do not support these kinds of log types now.
291                     return null;
292 
293                 case REQUEST_LINE:
294                     final String httpMethodName = log.requestHeaders().method().name();
295                     final String path = log.requestHeaders().path();
296                     final String logName = log.name();
297                     final String protocol = firstNonNull(log.sessionProtocol(),
298                                                          log.context().sessionProtocol()).uriText();
299 
300                     final StringBuilder requestLine = TemporaryThreadLocals.get().stringBuilder();
301                     requestLine.append(httpMethodName).append(' ').append(path);
302                     if (logName != null) {
303                         requestLine.append('#')
304                                    .append(UrlEscapers.urlFragmentEscaper().escape(logName));
305                     }
306                     requestLine.append(' ').append(protocol);
307                     return requestLine.toString();
308 
309                 case RESPONSE_STATUS_CODE:
310                     return log.responseHeaders().status().code();
311                 case RESPONSE_LENGTH:
312                     return log.responseLength();
313                 case REQUEST_ID:
314                     final RequestId id = log.context().id();
315                     if ("short".equals(variable)) {
316                         return id.shortText();
317                     } else {
318                         return id.text();
319                     }
320             }
321             return null;
322         }
323     }

325     /**
326      * An HTTP header component of a log message.
327      */
328     class HttpHeaderComponent extends ResponseHeaderConditional {
329 
330         static boolean isSupported(AccessLogType type) {
331             return type == AccessLogType.REQUEST_HEADER ||
332                    type == AccessLogType.RESPONSE_HEADER;
333         }
334 
335         private final AsciiString headerName;
336         private final Function<RequestLog, HttpHeaders> httpHeaders;
337 
338         HttpHeaderComponent(AccessLogType logType, CharSequence headerName, boolean addQuote,
339                             @Nullable Function<ResponseHeaders, Boolean> condition) {
340             super(condition, addQuote);
341             this.headerName = HttpHeaderNames.of(requireNonNull(headerName, "headerName"));
342             if (logType == AccessLogType.REQUEST_HEADER) {
343                 httpHeaders = RequestLog::requestHeaders;
344             } else {
345                 assert logType == AccessLogType.RESPONSE_HEADER : logType.name();
346                 httpHeaders = RequestLog::responseHeaders;
347             }
348         }
349 
350         @VisibleForTesting
351         AsciiString headerName() {
352             return headerName;
353         }
354 
355         @Override
356         public Object getMessage0(RequestLog log) {
357             return httpHeaders.apply(log).get(headerName);
358         }
359     }

361     /**
362      * An attribute component of a log message.
363      */
364     class AttributeComponent extends ResponseHeaderConditional {
365 
366         static boolean isSupported(AccessLogType type) {
367             return type == AccessLogType.ATTRIBUTE;
368         }
369 
370         private final AttributeKey<?> key;
371         private final Function<Object, String> stringifer;
372 
373         AttributeComponent(String attributeName, Function<Object, String> stringifer, boolean addQuote,
374                            @Nullable Function<ResponseHeaders, Boolean> condition) {
375             super(condition, addQuote);
376             key = AttributeKey.valueOf(requireNonNull(attributeName, "attributeName"));
377             this.stringifer = requireNonNull(stringifer, "stringifer");
378         }
379 
380         @VisibleForTesting
381         AttributeKey<?> key() {
382             return key;
383         }
384 
385         @Nullable
386         @Override
387         Object getMessage0(RequestLog log) {
388             final Object value = log.context().attr(key);
389             return value != null ? stringifer.apply(value) : null;
390         }
391     }

393     /**
394      * A {@link RequestLog} component of a log message.
395      */
396     class RequestLogComponent extends ResponseHeaderConditional {
397 
398         static boolean isSupported(AccessLogType type) {
399             return type == AccessLogType.REQUEST_LOG;
400         }
401 
402         private final Function<RequestLog, Object> resolver;
403 
404         RequestLogComponent(String variable, boolean addQuote,
405                             @Nullable Function<ResponseHeaders, Boolean> condition) {
406             super(condition, addQuote);
407             resolver = findResolver(requireNonNull(variable, "variable"));
408         }
409 
410         @Nullable
411         @Override
412         Object getMessage0(RequestLog log) {
413             return resolver.apply(log);
414         }
415 
416         @Nullable
417         private static String handleThrowable(@Nullable Throwable cause) {
418             if (cause == null) {
419                 return null;
420             }
421             cause = Exceptions.peel(cause);
422             final String message = cause.getMessage();
423             return message != null ? cause.getClass().getSimpleName() + ": " + message
424                                    : cause.getClass().getSimpleName();
425         }
426 
427         private static Function<RequestLog, Object> findResolver(String variable) {
428             // The same order as methods in the RequestLog interface.
429             switch (variable) {
430                 case "method":
431                     return log -> log.requestHeaders().method();
432                 case "path":
433                     return log -> log.context().path();
434                 case "query":
435                     return log -> log.context().query();
436 
437                 case "requestStartTimeMillis":
438                     return RequestLog::requestStartTimeMillis;
439                 case "requestEndTimeMillis":
440                     return log -> Instant.ofEpochMilli(log.requestStartTimeMillis())
441                                          .plusNanos(log.requestDurationNanos()).toEpochMilli();
442                 case "requestDurationMillis":
443                     return log -> Duration.ofNanos(log.requestDurationNanos()).toMillis();
444                 case "requestDurationNanos":
445                     return RequestLog::requestDurationNanos;
446                 case "requestLength":
447                     return RequestLog::requestLength;
448                 case "requestCause":
449                     return log -> handleThrowable(log.requestCause());
450                 case "requestContentPreview":
451                     return RequestLog::requestContentPreview;
452 
453                 case "responseStartTimeMillis":
454                     return RequestLog::responseStartTimeMillis;
455                 case "responseEndTimeMillis":
456                     return log -> Instant.ofEpochMilli(log.responseStartTimeMillis())
457                                          .plusNanos(log.responseDurationNanos()).toEpochMilli();
458                 case "responseDurationMillis":
459                     return log -> Duration.ofNanos(log.responseDurationNanos()).toMillis();
460                 case "responseDurationNanos":
461                     return RequestLog::responseDurationNanos;
462                 case "responseLength":
463                     return RequestLog::responseLength;
464                 case "responseCause":
465                     return log -> handleThrowable(log.responseCause());
466                 case "responseContentPreview":
467                     return RequestLog::responseContentPreview;
468 
469                 case "totalDurationMillis":
470                     return log -> Duration.ofNanos(log.totalDurationNanos()).toMillis();
471                 case "totalDurationNanos":
472                     return RequestLog::totalDurationNanos;
473 
474                 case "sessionProtocol":
475                     return RequestLog::sessionProtocol;
476                 case "serializationFormat":
477                     return log -> log.scheme().serializationFormat();
478                 case "scheme":
479                     return RequestLog::scheme;
480                 case "host":
481                     return log -> {
482                         final String authority = log.requestHeaders().authority();
483                         if ("?".equals(authority)) {
484                             final InetSocketAddress remoteAddr = log.context().remoteAddress();
485                             assert remoteAddr != null;
486                             return remoteAddr.getHostString();
487                         }
488                         return authority;
489                     };
490                 case "status":
491                     return log -> log.responseHeaders().status();
492                 case "statusCode":
493                     return log -> log.responseHeaders().status().code();
494 
495                 default:
496                     throw new IllegalArgumentException("unexpected request log variable: " + variable);
497             }
498         }
