88  /**
89   * Encapsulates the state of a single server call, reading messages from the client, passing to business logic
90   * via {@link ServerCall.Listener}, and writing messages passed back to the response.
91   */
92  final class ArmeriaServerCall<I, O> extends ServerCall<I, O>
93          implements ArmeriaMessageDeframer.Listener, TransportStatusListener {

147     ArmeriaServerCall(HttpHeaders clientHeaders,
148                       MethodDescriptor<I, O> method,
149                       CompressorRegistry compressorRegistry,
150                       DecompressorRegistry decompressorRegistry,
151                       HttpResponseWriter res,
152                       int maxInboundMessageSizeBytes,
153                       int maxOutboundMessageSizeBytes,
154                       ServiceRequestContext ctx,
155                       SerializationFormat serializationFormat,
156                       @Nullable MessageMarshaller jsonMarshaller,
157                       boolean unsafeWrapRequestBuffers,
158                       boolean useBlockingTaskExecutor,
159                       ResponseHeaders defaultHeaders) {
160         requireNonNull(clientHeaders, "clientHeaders");
161         this.method = requireNonNull(method, "method");
162         this.ctx = requireNonNull(ctx, "ctx");
163         this.serializationFormat = requireNonNull(serializationFormat, "serializationFormat");
164         this.defaultHeaders = requireNonNull(defaultHeaders, "defaultHeaders");
165         messageReader = new HttpStreamReader(
166                 requireNonNull(decompressorRegistry, "decompressorRegistry"),
167                 new ArmeriaMessageDeframer(
168                         this,
169                         maxInboundMessageSizeBytes,
170                         ctx.alloc())
171                         .decompressor(clientDecompressor(clientHeaders, decompressorRegistry)),
172                 this);
173         messageFramer = new ArmeriaMessageFramer(ctx.alloc(), maxOutboundMessageSizeBytes);
174         this.res = requireNonNull(res, "res");
175         this.compressorRegistry = requireNonNull(compressorRegistry, "compressorRegistry");
176         clientAcceptEncoding =
177                 Strings.emptyToNull(clientHeaders.get(GrpcHeaderNames.GRPC_ACCEPT_ENCODING));
178         marshaller = new GrpcMessageMarshaller<>(ctx.alloc(), serializationFormat, method, jsonMarshaller,
179                                                  unsafeWrapRequestBuffers);
180         this.unsafeWrapRequestBuffers = unsafeWrapRequestBuffers;
181         blockingExecutor = useBlockingTaskExecutor ?
182                            MoreExecutors.newSequentialExecutor(ctx.blockingTaskExecutor()) : null;
183 
184         res.whenComplete().handleAsync((unused, t) -> {
185             if (!closeCalled) {
186                 // Closed by client, not by server.
187                 cancelled = true;
188                 try (SafeCloseable ignore = ctx.push()) {
189                     close(Status.CANCELLED, new Metadata());
190                 }
191             }
192             return null;
193         }, ctx.eventLoop());
194     }

196     @Override
197     public void request(int numMessages) {
198         if (ctx.eventLoop().inEventLoop()) {
199             messageReader.request(numMessages);
200         } else {
201             ctx.eventLoop().submit(() -> messageReader.request(numMessages));
202         }
203     }

205     @Override
206     public void sendHeaders(Metadata metadata) {
207         if (ctx.eventLoop().inEventLoop()) {
208             doSendHeaders(metadata);
209         } else {
210             ctx.eventLoop().submit(() -> doSendHeaders(metadata));
211         }
212     }
213 
214     private void doSendHeaders(Metadata metadata) {
215         checkState(!sendHeadersCalled, "sendHeaders already called");
216         checkState(!closeCalled, "call is closed");
217 
218         if (compressor == null || !messageCompression || clientAcceptEncoding == null) {
219             compressor = Codec.Identity.NONE;
220         } else {
221             final List<String> acceptedEncodingsList =
222                     ACCEPT_ENCODING_SPLITTER.splitToList(clientAcceptEncoding);
223             if (!acceptedEncodingsList.contains(compressor.getMessageEncoding())) {
224                 // resort to using no compression.
225                 compressor = Codec.Identity.NONE;
226             }
227         }
228         messageFramer.setCompressor(ForwardingCompressor.forGrpc(compressor));
229 
230         ResponseHeaders headers = defaultHeaders;
231 
232         if (compressor != Codec.Identity.NONE || InternalMetadata.headerCount(metadata) > 0) {
233             headers = headers.withMutations(builder -> {
234                 if (compressor != Codec.Identity.NONE) {
235                     builder.set(GrpcHeaderNames.GRPC_ENCODING, compressor.getMessageEncoding());
236                 }
237                 MetadataUtil.fillHeaders(metadata, builder);
238             });
239         }
240 
241         sendHeadersCalled = true;
242         res.write(headers);
243     }
244 
245     @Override
246     public void sendMessage(O message) {
247         pendingMessagesUpdater.incrementAndGet(this);
248         if (ctx.eventLoop().inEventLoop()) {
249             doSendMessage(message);
250         } else {
251             ctx.eventLoop().submit(() -> doSendMessage(message));
252         }
253     }
254 
255     private void doSendMessage(O message) {
256         checkState(sendHeadersCalled, "sendHeaders has not been called");
257         checkState(!closeCalled, "call is closed");
258 
259         if (firstResponse == null) {
260             firstResponse = message;
261         }
262 
263         try {
264             res.write(messageFramer.writePayload(marshaller.serializeResponse(message)));
265             res.whenConsumed().thenRun(() -> {
266                 if (pendingMessagesUpdater.decrementAndGet(this) == 0) {
267                     if (blockingExecutor != null) {
268                         blockingExecutor.execute(this::invokeOnReady);
269                     } else {
270                         invokeOnReady();
271                     }
272                 }
273             });
274         } catch (RuntimeException e) {
275             close(GrpcStatus.fromThrowable(e), new Metadata());
276             throw e;
277         } catch (Throwable t) {
278             close(GrpcStatus.fromThrowable(t), new Metadata());
279             throw new RuntimeException(t);
280         }
281     }
282 
283     private void invokeOnReady() {
284         try {
285             listener.onReady();
286         } catch (Throwable t) {
287             close(GrpcStatus.fromThrowable(t), new Metadata());
288         }
289     }

291     @Override
292     public boolean isReady() {
293         return !closeCalled && pendingMessages == 0;
294     }

296     @Override
297     public void close(Status status, Metadata metadata) {
298         if (ctx.eventLoop().inEventLoop()) {
299             doClose(status, metadata);
300         } else {
301             ctx.eventLoop().submit(() -> doClose(status, metadata));
302         }
303     }
304 
305     private void doClose(Status status, Metadata metadata) {
306         checkState(!closeCalled, "call already closed");
307 
308         closeCalled = true;
309         if (cancelled) {
310             // No need to write anything to client if cancelled already.
311             closeListener(status);
312             return;
313         }
314 
315         final HttpHeaders trailers = statusToTrailers(status, metadata, sendHeadersCalled);
316         final HttpObject trailersObj;
317         if (sendHeadersCalled && GrpcSerializationFormats.isGrpcWeb(serializationFormat)) {
318             // Normal trailers are not supported in grpc-web and must be encoded as a message.
319             // Message compression is not supported in grpc-web, so we don't bother using the normal
320             // ArmeriaMessageFramer.
321             trailersObj = serializeTrailersAsMessage(trailers);
322         } else {
323             trailersObj = trailers;
324         }
325         try {
326             if (res.tryWrite(trailersObj)) {
327                 res.close();
328             }
329         } finally {
330             closeListener(status);
331         }
332     }

363     @Override
364     public void messageRead(DeframedMessage message) {
365 
366         final I request;
367         final ByteBuf buf = message.buf();
368 
369         boolean success = false;
370         try {
371             // Special case for unary calls.
372             if (messageReceived && method.getType() == MethodType.UNARY) {
373                 closeListener(Status.INTERNAL.withDescription(
374                         "More than one request messages for unary call or server streaming call"));
375                 return;
376             }
377             messageReceived = true;
378 
379             if (isCancelled()) {
380                 return;
381             }
382             success = true;
383         } finally {
384             if (buf != null && !success) {
385                 buf.release();
386             }
387         }
388 
389         try {
390             request = marshaller.deserializeRequest(message);
391         } catch (IOException e) {
392             throw new UncheckedIOException(e);
393         }
394 
395         if (!ctx.log().isAvailable(RequestLogProperty.REQUEST_CONTENT)) {
396             ctx.logBuilder().requestContent(GrpcLogUtil.rpcRequest(method, request), null);
397         }
398 
399         if (unsafeWrapRequestBuffers && buf != null) {
400             GrpcUnsafeBufferUtil.storeBuffer(buf, request, ctx);
401         }
402 
403         if (blockingExecutor != null) {
404             blockingExecutor.execute(() -> invokeOnMessage(request));
405         } else {
406             invokeOnMessage(request);
407         }
408     }
409 
410     private void invokeOnMessage(I request) {
411         try (SafeCloseable ignored = ctx.push()) {
412             listener.onMessage(request);
413         } catch (Throwable t) {
414             close(GrpcStatus.fromThrowable(t), new Metadata());
415         }
416     }

418     @Override
419     public void endOfStream() {
420         setClientStreamClosed();
421         if (!closeCalled) {
422             if (!ctx.log().isAvailable(RequestLogProperty.REQUEST_CONTENT)) {
423                 ctx.logBuilder().requestContent(GrpcLogUtil.rpcRequest(method), null);
424             }
425 
426             if (blockingExecutor != null) {
427                 blockingExecutor.execute(this::invokeHalfClose);
428             } else {
429                 invokeHalfClose();
430             }
431         }
432     }
433 
434     private void invokeHalfClose() {
435         try (SafeCloseable ignored = ctx.push()) {
436             listener.onHalfClose();
437         } catch (Throwable t) {
438             close(GrpcStatus.fromThrowable(t), new Metadata());
439         }
440     }

442     @Override
443     public void transportReportStatus(Status status, Metadata unused) {
444         // A server doesn't see trailers from the client so will never have Metadata here.
445 
446         if (closeCalled) {
447             // We've already called close on the server-side and will close the listener with the server-side
448             // status, so we ignore client transport status's at this point (it's usually the RST_STREAM
449             // corresponding to a successful stream ending in practice, but even if it was an actual transport
450             // failure there's no need to notify the server listener of it).
451             return;
452         }
453         closeListener(status);
454     }
455 
456     private void closeListener(Status newStatus) {
457         if (!listenerClosed) {
458             listenerClosed = true;
459             setClientStreamClosed();
460             messageFramer.close();
461             ctx.logBuilder().responseContent(GrpcLogUtil.rpcResponse(newStatus, firstResponse), null);
462             if (newStatus.isOk()) {
463                 if (blockingExecutor != null) {
464                     blockingExecutor.execute(this::invokeOnComplete);
465                 } else {
466                     invokeOnComplete();
467                 }
468             } else {
469                 cancelled = true;
470                 if (blockingExecutor != null) {
471                     blockingExecutor.execute(this::invokeOnCancel);
472                 } else {
473                     invokeOnCancel();
474                 }
475                 // Transport error, not business logic error, so reset the stream.
476                 if (!closeCalled) {
477                     final StatusException statusException = newStatus.asException();
478                     final Throwable cause = statusException.getCause();
479                     if (cause != null) {
480                         res.close(cause);
481                     } else {
482                         res.abort();
483                     }
484                 }
485             }
486         }
487     }
488 
489     private void invokeOnComplete() {
490         try (SafeCloseable ignored = ctx.push()) {
491             listener.onComplete();
492         } catch (Throwable t) {
493             // This should not be possible with normal generated stubs which do not implement
494             // onComplete, but is conceivable for a completely manually constructed stub.
495             logger.warn("Error in gRPC onComplete handler.", t);
496         }
497     }
498 
499     private void invokeOnCancel() {
500         try (SafeCloseable ignored = ctx.push()) {
501             listener.onCancel();
502         } catch (Throwable t) {
503             if (!closeCalled) {
504                 // A custom error when dealing with client cancel or transport issues should be
505                 // returned. We have already closed the listener, so it will not receive any more
506                 // callbacks as designed.
507                 close(GrpcStatus.fromThrowable(t), new Metadata());
508             }
509         }
510     }
511 
512     private void setClientStreamClosed() {
513         if (!clientStreamClosed) {
514             messageReader().cancel();
515             clientStreamClosed = true;
516         }
517     }

519     private HttpHeaders statusToTrailers(Status status, Metadata metadata, boolean headersSent) {
520         return statusToTrailers(ctx, status, metadata, headersSent);

523     // Returns ResponseHeaders if headersSent == false or HttpHeaders otherwise.
524     static HttpHeaders statusToTrailers(
525             ServiceRequestContext ctx, Status status, Metadata metadata, boolean headersSent) {
526         final HttpHeadersBuilder trailers = GrpcTrailersUtil.statusToTrailers(
527                 status.getCode().value(), status.getDescription(), headersSent);
528 
529         MetadataUtil.fillHeaders(metadata, trailers);
530 
531         if (ctx.verboseResponses() && status.getCause() != null) {
532             final ThrowableProto proto = GrpcStatus.serializeThrowable(status.getCause());
533             trailers.add(GrpcHeaderNames.ARMERIA_GRPC_THROWABLEPROTO_BIN,
534                          Base64.getEncoder().encodeToString(proto.toByteArray()));
535         }
536 
537         return trailers.build();
538     }

549     private HttpData serializeTrailersAsMessage(HttpHeaders trailers) {
550         final ByteBuf serialized = ctx.alloc().buffer();
551         boolean success = false;
552         try {
553             serialized.writeByte(TRAILERS_FRAME_HEADER);
554             // Skip, we'll set this after serializing the headers.
555             serialized.writeInt(0);
556             for (Map.Entry<AsciiString, String> trailer : trailers) {
557                 encodeHeader(trailer.getKey(), trailer.getValue(), serialized);
558             }
559             final int messageSize = serialized.readableBytes() - 5;
560             serialized.setInt(1, messageSize);
561             success = true;
562         } finally {
563             if (!success) {
564                 serialized.release();
565             }
566         }
567         return new ByteBufHttpData(serialized, true);
568     }

570     @Nullable
571     private static Decompressor clientDecompressor(HttpHeaders headers, DecompressorRegistry registry) {
572         final String encoding = headers.get(GrpcHeaderNames.GRPC_ENCODING);
573         if (encoding == null) {
574             return ForwardingDecompressor.forGrpc(Identity.NONE);
575         }
576         final io.grpc.Decompressor decompressor = registry.lookupDecompressor(encoding);
577         if (decompressor != null) {
578             return ForwardingDecompressor.forGrpc(decompressor);
579         }
580         return ForwardingDecompressor.forGrpc(Identity.NONE);
581     }

583     // Copied from io.netty.handler.codec.http.HttpHeadersEncoder
584     private static void encodeHeader(CharSequence name, CharSequence value, ByteBuf buf) {
585         final int nameLen = name.length();
586         final int valueLen = value.length();
587         final int entryLen = nameLen + valueLen + 4;
588         buf.ensureWritable(entryLen);
589         int offset = buf.writerIndex();
590         writeAscii(buf, offset, name, nameLen);
591         offset += nameLen;
592         buf.setByte(offset++, ':');
593         buf.setByte(offset++, ' ');
594         writeAscii(buf, offset, value, valueLen);
595         offset += valueLen;
596         buf.setByte(offset++, '\r');
597         buf.setByte(offset++, '\n');
598         buf.writerIndex(offset);
599     }
600 
601     private static void writeAscii(ByteBuf buf, int offset, CharSequence value, int valueLen) {
602         if (value instanceof AsciiString) {
603             ByteBufUtil.copy((AsciiString) value, 0, buf, offset, valueLen);
604         } else {
605             writeCharSequence(buf, offset, value, valueLen);
606         }
607     }
608 
609     private static void writeCharSequence(ByteBuf buf, int offset, CharSequence value, int valueLen) {
610         for (int i = 0; i < valueLen; ++i) {
611             buf.setByte(offset++, c2b(value.charAt(i)));
612         }
613     }
