17  package com.linecorp.armeria.server.grpc.protocol;
18  
19  import java.util.concurrent.CompletableFuture;
20  
21  import com.linecorp.armeria.common.util.UnstableApi;
22  
23  import io.netty.buffer.ByteBuf;
24  import io.netty.buffer.ByteBufUtil;
25  import io.netty.buffer.Unpooled;
26  
27  /**
28   * An {@link AbstractUnaryGrpcService} can be used to implement a gRPC service without depending on gRPC stubs.
29   * This service takes care of deframing and framing with the gRPC wire format and handling appropriate headers.
30   *
31   * <p>This service does not support compression. If you need support for compression, please consider using
32   * normal gRPC stubs or file a feature request.
33   */
34  @UnstableApi
35  public abstract class AbstractUnaryGrpcService extends AbstractUnsafeUnaryGrpcService {
36  
37      /**
38       * Returns an unframed response message to return to the client, given an unframed request message. It is
39       * expected that the implementation has the logic to know how to parse the request and serialize a response
40       * into {@code byte[]}. The returned {@code byte[]} will be framed and returned to the client.
41       */
42      protected abstract CompletableFuture<byte[]> handleMessage(byte[] message);
43  
44      @Override
45      protected final CompletableFuture<ByteBuf> handleMessage(ByteBuf message) {
46          final byte[] bytes;
47          try {
48              bytes = ByteBufUtil.getBytes(message);
49          } finally {
50              message.release();
51          }
52          return handleMessage(bytes).thenApply(Unpooled::wrappedBuffer);
53      }
54  }
