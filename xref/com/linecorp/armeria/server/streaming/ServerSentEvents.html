39  /**
40   * A utility class which helps to create a <a href="https://www.w3.org/TR/eventsource/">Server-Sent Events</a>
41   * stream from a content {@link Publisher} or {@link Stream}.
42   *
43   * <p>A user simply creates a streaming {@link HttpResponse} which emits Server-Sent Events, e.g.
44   * <pre>{@code
45   * Server server =
46   *     Server.builder()
47   *           // Emit Server-Sent Events with the SeverSentEvent instances published by a publisher.
48   *           .service("/sse1",
49   *                    (ctx, req) -> ServerSentEvents.fromPublisher(
50   *                            Flux.just(ServerSentEvent.ofData("foo"), ServerSentEvent.ofData("bar"))))
51   *           // Emit Server-Sent Events with converting instances published by a publisher into
52   *           // ServerSentEvent instances.
53   *           .service("/sse2",
54   *                    (ctx, req) -> ServerSentEvents.fromPublisher(
55   *                            Flux.just("foo", "bar"), ServerSentEvent::ofData))
56   *           .build();
57   * }</pre>
58   */
59  public final class ServerSentEvents {

107      * Creates a new Server-Sent Events stream from the specified {@link Publisher}.
108      *
109      * @param headers the HTTP headers supposed to send
110      * @param contentPublisher the {@link Publisher} which publishes the objects supposed to send as contents
111      * @param trailers the HTTP trailers
112      */
113     public static HttpResponse fromPublisher(ResponseHeaders headers,
114                                              Publisher<? extends ServerSentEvent> contentPublisher,
115                                              HttpHeaders trailers) {
116         requireNonNull(headers, "headers");
117         requireNonNull(contentPublisher, "contentPublisher");
118         requireNonNull(trailers, "trailers");
119         return streamingFrom(contentPublisher, sanitizeHeaders(headers), trailers,
120                              ServerSentEvents::toHttpData);
121     }

287         return HttpResponse.of(sanitizeHeaders(headers), toHttpData(sse), trailers);

290     private static ResponseHeaders sanitizeHeaders(ResponseHeaders headers) {
291         if (headers == defaultHttpHeaders) {
292             return headers;
293         }
294         return ensureContentType(ensureHttpStatus(headers));
295     }
296 
297     static ResponseHeaders ensureHttpStatus(ResponseHeaders headers) {
298         final HttpStatus status = headers.status();
299         if (status.equals(HttpStatus.OK)) {
300             return headers;
301         }
302 
303         if (!warnedStatusCode) {
304             logger.warn(
305                     "Overwriting the HTTP status code from '{}' to '{}' for Server-Sent Events. " +
306                     "Do not set an HTTP status code on the HttpHeaders when calling factory methods in '{}', " +
307                     "or set '{}' if you want to specify its status code. " +
308                     "Please refer to https://www.w3.org/TR/eventsource/ for more information.",
309                     status, HttpStatus.OK, ServerSentEvents.class.getSimpleName(), HttpStatus.OK);
310             warnedStatusCode = true;
311         }
312         return headers.toBuilder().status(HttpStatus.OK).build();
313     }
314 
315     static ResponseHeaders ensureContentType(ResponseHeaders headers) {
316         final MediaType contentType = headers.contentType();
317         if (contentType == null) {
318             return headers.toBuilder()
319                           .add(HttpHeaderNames.CONTENT_TYPE, MediaType.EVENT_STREAM.toString())
320                           .build();
321         }
322 
323         if (contentType.is(MediaType.EVENT_STREAM)) {
324             return headers;
325         }
326 
327         if (!warnedContentType) {
328             logger.warn("Overwriting content-type from '{}' to '{}' for Server-Sent Events. " +
329                         "Do not set a content-type on the HttpHeaders when calling factory methods in '{}', " +
330                         "or set '{}' if you want to specify its content-type. " +
331                         "Please refer to https://www.w3.org/TR/eventsource/ for more information.",
332                         contentType, MediaType.EVENT_STREAM,
333                         ServerSentEvents.class.getSimpleName(), MediaType.EVENT_STREAM);
334             warnedContentType = true;
335         }
336         return headers.toBuilder()
337                       .contentType(MediaType.EVENT_STREAM)
338                       .build();
339     }

341     private static HttpData toHttpData(ServerSentEvent sse) {
342         final StringBuilder sb = new StringBuilder();
343 
344         // Write a comment first because a user might want to explain his or her event at first line.
345         final String comment = sse.comment();
346         if (comment != null) {
347             appendField(sb, "", comment, false);
348         }
349 
350         final String id = sse.id();
351         if (id != null) {
352             appendField(sb, "id", id, true);
353         }
354 
355         final String event = sse.event();
356         if (event != null) {
357             appendField(sb, "event", event, true);
358         }
359 
360         final String data = sse.data();
361         if (data != null) {
362             appendField(sb, "data", data, true);
363         }
364 
365         final Duration retry = sse.retry();
366         if (retry != null) {
367             // Reconnection time, in milliseconds.
368             sb.append("retry:").append(retry.toMillis()).append(LINE_FEED);
369         }
370 
371         return sb.length() == 0 ? HttpData.empty()
372                                 : HttpData.ofUtf8(sb.append(LINE_FEED).toString());
373     }
374 
375     private static <T> HttpData toHttpData(
376             Function<? super T, ? extends ServerSentEvent> converter, T content) {
377         final ServerSentEvent sse = converter.apply(content);
378         return sse == null ? HttpData.empty() : toHttpData(sse);
379     }
380 
381     private static void appendField(StringBuilder sb, String name, String value,
382                                     boolean emitFieldForEmptyValue) {
383         if (value.isEmpty()) {
384             if (emitFieldForEmptyValue) {
385                 // Emit name only if the value is an empty string.
386                 sb.append(name).append(LINE_FEED);
387             }
388         } else {
389             sb.append(name).append(':');
390 
391             final String[] values = value.split("\n");
392             assert values.length > 0;
393             if (values.length == 1) {
394                 sb.append(value);
395             } else {
396                 final int len = values.length - 1;
397                 for (int i = 0; i < len; i++) {
398                     sb.append(values[i]).append(LINE_FEED).append(name).append(':');
399                 }
400                 sb.append(values[len]);
401             }
402             sb.append(LINE_FEED);
403         }
404     }
