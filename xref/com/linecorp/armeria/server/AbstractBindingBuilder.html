219     private void addRouteBuilder(String pathPattern, HttpMethod method) {
220         addRouteBuilder(Route.builder().path(requireNonNull(pathPattern, "pathPattern")), EnumSet.of(method));
221     }
222 
223     private void addRouteBuilder(RouteBuilder routeBuilder, Set<HttpMethod> methods) {
224         final Set<HttpMethod> methodSet = routeBuilders.computeIfAbsent(
225                 routeBuilder, key -> EnumSet.noneOf(HttpMethod.class));
226 
227         for (HttpMethod method : methods) {
228             if (!methodSet.add(method)) {
229                 throw new IllegalArgumentException("duplicate HTTP method: " + method +
230                                                    ", for: " + routeBuilder);
231             }
232         }
233     }

300     /**
301      * Sets the {@link Route} to accept a request if it matches all the specified predicates for
302      * HTTP parameters. The predicate can be one of the following forms:
303      * <ul>
304      *     <li>{@code some-param=some-value} which means that the request must have a
305      *     {@code some-param=some-value} parameter</li>
306      *     <li>{@code some-param!=some-value} which means that the request must not have a
307      *     {@code some-param=some-value} parameter</li>
308      *     <li>{@code some-param} which means that the request must contain a {@code some-param} parameter</li>
309      *     <li>{@code !some-param} which means that the request must not contain a {@code some-param}
310      *     parameter</li>
311      * </ul>
312      *
313      * <p>Note that these predicates can be evaluated only with the query string of the request URI.
314      * Also note that each predicate will be evaluated with the decoded value of HTTP parameters,
315      * so do not use percent-encoded value in the predicate.
316      *
317      * @see MatchesParam
318      */
319     public AbstractBindingBuilder matchesParams(String... paramPredicates) {
320         return matchesParams(ImmutableList.copyOf(requireNonNull(paramPredicates, "paramPredicates")));
321     }

411     /**
412      * Returns a newly-created {@link Route}s based on the properties of this builder.
413      */
414     final List<Route> buildRouteList() {
415         final Builder<Route> builder = ImmutableList.builder();
416 
417         if (pathBuilders.isEmpty() && routeBuilders.isEmpty()) {
418             throw new IllegalStateException(
419                     "Should set at least one path that the service is bound to before calling this.");
420         }
421         if (pathBuilders.isEmpty() && !methods.isEmpty()) {
422             throw new IllegalStateException("Should set a path when the methods are set: " + methods);
423         }
424 
425         if (!pathBuilders.isEmpty()) {
426             final Set<HttpMethod> pathMethods = methods.isEmpty() ? HttpMethod.knownMethods() : methods;
427             pathBuilders.forEach(pathBuilder -> addRouteBuilder(pathBuilder, pathMethods));
428         }
429         routeBuilders.forEach((routeBuilder, routeMethods) -> {
430             builder.add(routeBuilder.methods(routeMethods)
431                                     .consumes(consumeTypes)
432                                     .produces(produceTypes)
433                                     .matchesParams(paramPredicates)
434                                     .matchesHeaders(headerPredicates)
435                                     .build());
436         });
437 
438         return builder.build();
439     }
