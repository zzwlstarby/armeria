267     /**
268      * Adds the {@link DocServiceFilter} that checks whether a method will be <b>included</b> while building
269      * {@link DocService}. The {@link DocServiceFilter} will be invoked with the plugin, service and
270      * method name. The rule is as follows:
271      * <ul>
272      *   <li>No {@link #include(DocServiceFilter)} and {@link #exclude(DocServiceFilter)} is called -
273      *       include all methods.</li>
274      *   <li>Only {@link #exclude(DocServiceFilter)} is called -
275      *       include all methods except the methods which the exclusion filter returns {@code true}.</li>
276      *   <li>Only {@link #include(DocServiceFilter)} is called -
277      *       include the methods which the inclusion filter returns {@code true}.</li>
278      *   <li>{@link #include(DocServiceFilter)} and {@link #exclude(DocServiceFilter)} is called -
279      *       include the methods which the inclusion filter returns {@code true} and the exclusion filter
280      *       returns {@code false}.</li>
281      * </ul>
282      *
283      * <P>Note that this can be called multiple times and the {@link DocServiceFilter}s are composed using
284      * {@link DocServiceFilter#or(DocServiceFilter)} and {@link DocServiceFilter#and(DocServiceFilter)}.
285      *
286      * @see #exclude(DocServiceFilter)
287      * @see DocService to find out how DocService generates documentaion
288      */
289     public DocServiceBuilder include(DocServiceFilter filter) {
290         requireNonNull(filter, "filter");
291         if (includeFilter == ALL_SERVICES) {
292             includeFilter = filter;
293         } else {
294             includeFilter = includeFilter.or(filter);
295         }
296         return this;
297     }

308      *   <li>Only {@link #include(DocServiceFilter)} is called -

310      *   <li>{@link #include(DocServiceFilter)} and {@link #exclude(DocServiceFilter)} is called -
311      *       include the methods which the inclusion filter returns {@code true} and the exclusion filter
312      *       returns {@code false}.</li>
313      * </ul>
314      *
315      * <P>Note that this can be called multiple times and the {@link DocServiceFilter}s are composed using
316      * {@link DocServiceFilter#or(DocServiceFilter)} and {@link DocServiceFilter#and(DocServiceFilter)}.

388     private static String serializeExampleRequest(
389             String serviceName, String methodName, Object exampleRequest) {
390 
391         if (exampleRequest instanceof CharSequence) {
392             return exampleRequest.toString();
393         }
394 
395         for (DocServicePlugin plugin : DocService.plugins) {
396             // Skip if the plugin does not support it.
397             if (plugin.supportedExampleRequestTypes().stream()
398                       .noneMatch(type -> type.isInstance(exampleRequest))) {
399                 continue;
400             }
401 
402             final String result = plugin.serializeExampleRequest(serviceName, methodName, exampleRequest);
403             if (result != null) {
404                 return result;
405             }
406         }
407 
408         throw new IllegalArgumentException("could not find a plugin that can serialize: " + exampleRequest);
409     }

411     /**
412      * Returns a tuple of a service name, a method name and a serialized example request.
413      */
414     private static String[] guessAndSerializeExampleRequest(Object exampleRequest) {
415         checkArgument(!(exampleRequest instanceof CharSequence),
416                       "can't guess service or method name from a string: ", exampleRequest);
417 
418         boolean guessed = false;
419         for (DocServicePlugin plugin : DocService.plugins) {
420             // Skip if the plugin does not support it.
421             if (plugin.supportedExampleRequestTypes().stream()
422                       .noneMatch(type -> type.isInstance(exampleRequest))) {
423                 continue;
424             }
425 
426             final String serviceName = plugin.guessServiceName(exampleRequest);
427             final String methodName = plugin.guessServiceMethodName(exampleRequest);
428 
429             // Skip if the plugin cannot guess the service and method name.
430             if (serviceName == null || methodName == null) {
431                 continue;
432             }
433 
434             guessed = true;
435             final String serialized = plugin.serializeExampleRequest(serviceName, methodName, exampleRequest);
436             if (serialized != null) {
437                 return new String[] { serviceName, methodName, serialized };
438             }
439         }
440 
441         if (guessed) {
442             throw new IllegalArgumentException(
443                     "could not find a plugin that can serialize: " + exampleRequest);
444         } else {
445             throw new IllegalArgumentException(
446                     "could not find a plugin that can guess the service and method name from: " +
447                     exampleRequest);
448         }
449     }

451     /**
452      * Returns a newly-created {@link DocService} based on the properties of this builder.
453      */
454     public DocService build() {
455         return new DocService(exampleHttpHeaders, exampleRequests, injectedScriptSuppliers,
456                               unifyFilter(includeFilter, excludeFilter));
