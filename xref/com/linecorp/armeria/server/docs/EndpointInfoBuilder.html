17  package com.linecorp.armeria.server.docs;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  import static com.google.common.base.Preconditions.checkState;
21  import static com.google.common.collect.ImmutableSet.toImmutableSet;
22  import static java.util.Objects.requireNonNull;
23  
24  import java.util.Set;
25  
26  import javax.annotation.Nullable;
27  
28  import com.google.common.base.MoreObjects;
29  import com.google.common.collect.ImmutableSet;
30  import com.google.common.collect.Iterables;
31  import com.google.common.collect.Streams;
32  
33  import com.linecorp.armeria.common.MediaType;
34  import com.linecorp.armeria.common.SerializationFormat;
35  import com.linecorp.armeria.common.util.UnstableApi;
36  
37  /**
38   * Creates a new {@link EndpointInfo} using the builder pattern.
39   */
40  @UnstableApi
41  public final class EndpointInfoBuilder {
42  
43      private final String hostnamePattern;
44      private final String pathMapping;
45  
46      @Nullable
47      private String regexPathPrefix;
48      @Nullable
49      private String fragment;
50      @Nullable
51      private MediaType defaultMimeType;
52      @Nullable
53      private Set<MediaType> availableMimeTypes;
54  
55      /**
56       * Creates a new {@link EndpointInfoBuilder} that builds the {@link EndpointInfo} with the specified
57       * {@code hostnamePattern} and {@code pathMapping}.
58       *
59       * @deprecated Use {@link EndpointInfo#builder(String, String)}.
60       */
61      @Deprecated
62      public EndpointInfoBuilder(String hostnamePattern, String pathMapping) {
63          this.hostnamePattern = requireNonNull(hostnamePattern, "hostnamePattern");
64          this.pathMapping = requireNonNull(pathMapping, "pathMapping");
65      }
66  
67      /**
68       * Sets the prefix of the pathMapping if the {@code pathMapping} is a regular expression string and
69       * the prefix exists.
70       */
71      public EndpointInfoBuilder regexPathPrefix(String regexPathPrefix) {
72          requireNonNull(regexPathPrefix, "regexPathPrefix");
73          checkState(fragment == null, "regexPathPrefix cannot be set with fragment: %s", fragment);
74          this.regexPathPrefix = regexPathPrefix;
75          return this;
76      }
77  
78      /**
79       * Sets the fragment of the {@code pathMapping}.
80       */
81      public EndpointInfoBuilder fragment(String fragment) {
82          requireNonNull(fragment, "fragment");
83          checkState(regexPathPrefix == null, "fragment cannot be set with regexPathPrefix: %s", regexPathPrefix);
84          this.fragment = fragment;
85          return this;
86      }
87  
88      /**
89       * Sets the default {@link SerializationFormat}.
90       */
91      public EndpointInfoBuilder defaultFormat(SerializationFormat defaultFormat) {
92          requireNonNull(defaultFormat, "defaultFormat");
93          return defaultMimeType(defaultFormat.mediaType());
94      }
95  
96      /**
97       * Sets the default {@link MediaType}.
98       */
99      public EndpointInfoBuilder defaultMimeType(MediaType defaultMimeType) {
100         requireNonNull(defaultMimeType, "defaultMimeType");
101         this.defaultMimeType = defaultMimeType;
102 
103         if (availableMimeTypes == null) {
104             availableMimeTypes = ImmutableSet.of(defaultMimeType);
105         } else if (!availableMimeTypes.contains(defaultMimeType)) {
106             availableMimeTypes = addDefaultMimeType(defaultMimeType, availableMimeTypes);
107         }
108         return this;
109     }
110 
111     private static Set<MediaType> addDefaultMimeType(MediaType defaultMimeType,
112                                                      Iterable<MediaType> availableMimeTypes) {
113         return ImmutableSet.<MediaType>builder().add(defaultMimeType).addAll(availableMimeTypes).build();
114     }
115 
116     /**
117      * Sets the available {@link SerializationFormat}s.
118      */
119     public EndpointInfoBuilder availableFormats(SerializationFormat... availableFormats) {
120         requireNonNull(availableFormats, "availableFormats");
121         return availableFormats(ImmutableSet.copyOf(availableFormats));
122     }
123 
124     /**
125      * Sets the available {@link SerializationFormat}s.
126      */
127     public EndpointInfoBuilder availableFormats(Iterable<SerializationFormat> availableFormats) {
128         requireNonNull(availableFormats, "availableFormats");
129         return availableMimeTypes(Streams.stream(availableFormats).map(SerializationFormat::mediaType)
130                                          .collect(toImmutableSet()));
131     }
132 
133     /**
134      * Sets the available {@link MediaType}s.
135      */
136     public EndpointInfoBuilder availableMimeTypes(MediaType... availableMimeTypes) {
137         requireNonNull(availableMimeTypes, "availableMimeTypes");
138         return availableMimeTypes(ImmutableSet.copyOf(availableMimeTypes));
139     }
140 
141     /**
142      * Sets the available {@link MediaType}s.
143      */
144     public EndpointInfoBuilder availableMimeTypes(Iterable<MediaType> availableMimeTypes) {
145         requireNonNull(availableMimeTypes, "availableMimeTypes");
146         checkArgument(!Iterables.isEmpty(availableMimeTypes), "Should at least have an available media type.");
147         if (defaultMimeType != null && !Iterables.contains(availableMimeTypes, defaultMimeType)) {
148             this.availableMimeTypes = addDefaultMimeType(defaultMimeType, availableMimeTypes);
149         } else {
150             this.availableMimeTypes = ImmutableSet.copyOf(availableMimeTypes);
151         }
152         return this;
153     }
154 
155     /**
156      * Returns a newly-created {@link EndpointInfo} based on the properties of this builder.
157      */
158     public EndpointInfo build() {
159         checkState(availableMimeTypes != null && !availableMimeTypes.isEmpty(),
160                    "Should at least have an available media type.");
161         return new EndpointInfo(hostnamePattern, pathMapping, regexPathPrefix,
162                                 fragment, defaultMimeType, availableMimeTypes);
163     }
164 
165     @Override
166     public String toString() {
167         return MoreObjects.toStringHelper(this).omitNullValues()
168                           .add("hostnamePattern", hostnamePattern)
169                           .add("pathMapping", pathMapping)
170                           .add("regexPathPrefix", regexPathPrefix)
171                           .add("fragment", fragment)
172                           .add("defaultMimeType", defaultMimeType)
173                           .add("availableMimeTypes", availableMimeTypes)
174                           .toString();
175     }
176 }
