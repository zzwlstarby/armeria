58  /**
59   * A decorator which initiates an authentication request to the remote identity provider if the request is
60   * not authenticated.
61   */
62  final class SamlDecorator extends SimpleDecoratingHttpService {
63      private static final Logger logger = LoggerFactory.getLogger(SamlDecorator.class);
64  
65      private final SamlServiceProvider sp;
66      private final SamlPortConfigAutoFiller portConfigHolder;
67  
68      private final String myEntityId;
69      private final Credential signingCredential;
70      private final Authorizer<HttpRequest> authorizer;
71  
72      private final SamlRequestIdManager requestIdManager;
73      private final SamlSingleSignOnHandler ssoHandler;
74  
75      @Nullable
76      private Server server;
77  
78      SamlDecorator(SamlServiceProvider sp, HttpService delegate) {
79          super(delegate);
80          this.sp = sp;
81          portConfigHolder = sp.portConfigAutoFiller();
82  
83          myEntityId = sp.entityId();
84          signingCredential = sp.signingCredential();
85          authorizer = sp.authorizer();
86          ssoHandler = sp.ssoHandler();
87          requestIdManager = sp.requestIdManager();
88      }
89  
90      @Override
91      public void serviceAdded(ServiceConfig cfg) throws Exception {
92          super.serviceAdded(cfg);
93  
94          if (server != null) {
95              if (server != cfg.server()) {
96                  throw new IllegalStateException("cannot be added to more than one server");
97              } else {
98                  return;
99              }
100         }
101 
102         server = cfg.server();
103 
104         // Auto-detect the primary port number and its session protocol after the server started.
105         server.addListener(portConfigHolder);
106     }
107 
108     @Override
109     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
110         return HttpResponse.from(authorizer.authorize(ctx, req).handle((result, cause) -> {
111             if (cause == null && result) {
112                 // Already authenticated.
113                 try {
114                     return delegate().serve(ctx, req);
115                 } catch (Exception e) {
116                     return Exceptions.throwUnsafely(e);
117                 }
118             }
119 
120             final CompletionStage<SamlIdentityProviderConfig> f;
121             if (portConfigHolder.isDone()) {
122                 f = sp.idpConfigSelector().select(sp, ctx, req);
123             } else {
124                 f = portConfigHolder.future().thenCompose(
125                         unused -> sp.idpConfigSelector().select(sp, ctx, req));
126             }
127             // Find an identity provider first where the request is to be sent to.
128             return HttpResponse.from(f.thenApply(idp -> {
129                 if (idp == null) {
130                     throw new RuntimeException("cannot find a suitable identity provider from configurations");
131                 }
132                 final String defaultHostname = firstNonNull(sp.hostname(), ctx.virtualHost().defaultHostname());
133                 final AuthnRequest request = createAuthRequest(idp, defaultHostname);
134                 final MessageContext<AuthnRequest> messageContext = new MessageContext<>();
135                 messageContext.setMessage(request);
136                 return new MessageContextAndIdpConfig(messageContext, idp);
137             }).thenCompose(arg -> {
138                 return ssoHandler.beforeInitiatingSso(ctx, req, arg.messageContext, arg.idpConfig)
139                                  .thenApply(unused -> arg);
140             }).thenApply(arg -> {
141                 final SAMLBindingContext bindingContext =
142                         arg.messageContext.getSubcontext(SAMLBindingContext.class);
143                 final String relayState = bindingContext != null ? bindingContext.getRelayState() : null;
144 
145                 // Support HTTP Redirect and HTTP POST binding protocols when sending
146                 // an authentication request.
147                 final SamlEndpoint endpoint = arg.idpConfig.ssoEndpoint();
148                 try {
149                     if (endpoint.bindingProtocol() == SamlBindingProtocol.HTTP_REDIRECT) {
150                         return responseWithLocation(toRedirectionUrl(
151                                 arg.messageContext.getMessage(),
152                                 endpoint.toUriString(), SAML_REQUEST,
153                                 signingCredential, sp.signatureAlgorithm(),
154                                 relayState));
155                     } else {
156                         final String value = toSignedBase64(
157                                 arg.messageContext.getMessage(),
158                                 signingCredential,
159                                 sp.signatureAlgorithm());
160                         final HttpData body = getSsoForm(endpoint.toUriString(),
161                                                          SAML_REQUEST, value,
162                                                          relayState);
163                         return HttpResponse.of(HttpStatus.OK, MediaType.HTML_UTF_8,
164                                                body);
165                     }
166                 } catch (SamlException e) {
167                     return fail(ctx, e);
168                 }
169             }).exceptionally(e -> fail(ctx, e)));
170         }));
171     }
172 
173     /**
174      * Returns an {@link HttpResponse} for SAML authentication failure.
175      */
176     private static HttpResponse fail(ServiceRequestContext ctx, Throwable cause) {
177         logger.trace("{} Cannot initiate SAML authentication", ctx, cause);
178         return HttpResponse.of(HttpStatus.UNAUTHORIZED);
179     }
180 
181     /**
182      * Returns an {@link AuthnRequest} which is mapped to the specified identity provider.
183      */
184     private AuthnRequest createAuthRequest(SamlIdentityProviderConfig idp, String defaultHostname) {
185         requireNonNull(idp, "idp");
186 
187         final AuthnRequest authnRequest = build(AuthnRequest.DEFAULT_ELEMENT_NAME);
188 
189         final Issuer issuer = build(Issuer.DEFAULT_ELEMENT_NAME);
190         issuer.setValue(myEntityId);
191         authnRequest.setIssuer(issuer);
192 
193         authnRequest.setIssueInstant(DateTime.now());
194         authnRequest.setDestination(idp.ssoEndpoint().toUriString());
195         authnRequest.setID(requestIdManager.newId());
196 
197         // The ProtocolBinding attribute is mutually exclusive with the AssertionConsumerServiceIndex attribute
198         // and is typically accompanied by the AssertionConsumerServiceURL attribute.
199         final SamlPortConfig portConfig = portConfigHolder.config();
200         final SamlEndpoint acsEndpoint = idp.acsEndpoint() != null ? idp.acsEndpoint()
201                                                                    : sp.defaultAcsConfig().endpoint();
202         authnRequest.setAssertionConsumerServiceURL(acsEndpoint.toUriString(portConfig.scheme().uriText(),
203                                                                             defaultHostname,
204                                                                             portConfig.port()));
205         authnRequest.setProtocolBinding(acsEndpoint.bindingProtocol().urn());
206 
207         final SamlNameIdPolicy policy = idp.nameIdPolicy();
208         final NameIDPolicy nameIdPolicy = build(NameIDPolicy.DEFAULT_ELEMENT_NAME);
209         nameIdPolicy.setFormat(policy.format().urn());
210         nameIdPolicy.setAllowCreate(policy.isCreatable());
211         authnRequest.setNameIDPolicy(nameIdPolicy);
212 
213         final AuthnContextClassRef passwordAuthnCtxRef = build(AuthnContextClassRef.DEFAULT_ELEMENT_NAME);
214         passwordAuthnCtxRef.setAuthnContextClassRef(AuthnContext.PASSWORD_AUTHN_CTX);
215 
216         final RequestedAuthnContext requestedAuthnContext = build(RequestedAuthnContext.DEFAULT_ELEMENT_NAME);
217         requestedAuthnContext.setComparison(AuthnContextComparisonTypeEnumeration.EXACT);
218         requestedAuthnContext.getAuthnContextClassRefs().add(passwordAuthnCtxRef);
219 
220         authnRequest.setRequestedAuthnContext(requestedAuthnContext);
221 
222         return authnRequest;
223     }

225     /**
226      * An immutable object holder for {@link MessageContext} and {@link SamlIdentityProviderConfig}.
227      */
228     private static final class MessageContextAndIdpConfig {
229         private final MessageContext<AuthnRequest> messageContext;
230         private final SamlIdentityProviderConfig idpConfig;
231 
232         private MessageContextAndIdpConfig(MessageContext<AuthnRequest> messageContext,
233                                            SamlIdentityProviderConfig idpConfig) {
234             this.messageContext = messageContext;
235             this.idpConfig = idpConfig;
236         }
237     }
