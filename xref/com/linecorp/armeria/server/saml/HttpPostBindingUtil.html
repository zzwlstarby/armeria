44  /**
45   * A utility class which supports HTTP POST binding protocol.
46   */
47  final class HttpPostBindingUtil {
48  
49      private static final ImmutableList<String> XHTML = ImmutableList.of(
50              // 0
51              "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
52              "<!DOCTYPE html PUBLIC \"-//W3C//DTD XHTML 1.0 Strict//EN\" \"DTD/xhtml1-strict.dtd\">" +
53              "<html xmlns=\"http://www.w3.org/1999/xhtml\" xml:lang=\"en\" lang=\"en\">" +
54              "<head><meta http-equiv=\"content-type\" content=\"text/html;charset=utf-8\" /></head>" +
55              "<body onload=\"document.forms[0].submit()\">",
56              // 1, 2
57              "<form method=\"post\" action=\"", "\">",
58              // 3, 4, 5
59              "<input type=\"hidden\" name=\"", "\" value=\"", "\" />",
60              // 6
61              "</form></body></html>");
62  
63      private static final Escaper HTML_ESCAPER = HtmlEscapers.htmlEscaper();
64  
65      /**
66       * Returns an {@link HttpData} which holds a SSO form.
67       */
68      static HttpData getSsoForm(String remoteEndpointUrl,
69                                 String paramName, String paramValue,
70                                 @Nullable String relayState) {
71          requireNonNull(remoteEndpointUrl, "remoteEndpointUrl");
72          requireNonNull(paramName, "paramName");
73          requireNonNull(paramValue, "paramValue");
74  
75          final StringBuilder html = new StringBuilder();
76          html.append(XHTML.get(0))
77              .append(XHTML.get(1)).append(HTML_ESCAPER.escape(remoteEndpointUrl)).append(XHTML.get(2))
78              .append(XHTML.get(3)).append(HTML_ESCAPER.escape(paramName))
79              .append(XHTML.get(4)).append(HTML_ESCAPER.escape(paramValue)).append(XHTML.get(5));
80  
81          if (relayState != null) {
82              html.append(XHTML.get(3)).append(RELAY_STATE)
83                  .append(XHTML.get(4)).append(HTML_ESCAPER.escape(relayState)).append(XHTML.get(5));
84          }
85          html.append(XHTML.get(6));
86  
87          return HttpData.ofUtf8(html.toString());
88      }
89  
90      /**
91       * Signs the specified {@link SignableSAMLObject} with the specified {@link Credential} and
92       * {@code signatureAlgorithm}, and then encodes the object into a base64 string.
93       */
94      static String toSignedBase64(SignableSAMLObject signableObj,
95                                   Credential signingCredential,
96                                   String signatureAlgorithm) {
97          sign(signableObj, signingCredential, signatureAlgorithm);
98          final String messageStr = nodeToString(serialize(signableObj));
99          return Base64.getEncoder().encodeToString(messageStr.getBytes(StandardCharsets.UTF_8));
100     }
101 
102     /**
103      * Converts an {@link AggregatedHttpRequest} which is received from the remote entity to
104      * a {@link SAMLObject}.
105      */
106     static <T extends SAMLObject> MessageContext<T> toSamlObject(AggregatedHttpRequest req, String name) {
107         final SamlParameters parameters = new SamlParameters(req);
108         final byte[] decoded;
109         try {
110             decoded = Base64.getMimeDecoder().decode(parameters.getFirstValue(name));
111         } catch (IllegalArgumentException e) {
112             throw new InvalidSamlRequestException(
113                     "failed to decode a base64 string of the parameter: " + name, e);
114         }
115 
116         @SuppressWarnings("unchecked")
117         final T message = (T) deserialize(decoded);
118 
119         final MessageContext<T> messageContext = new MessageContext<>();
120         messageContext.setMessage(message);
121 
122         final String relayState = parameters.getFirstValueOrNull(RELAY_STATE);
123         if (relayState != null) {
124             final SAMLBindingContext context = messageContext.getSubcontext(SAMLBindingContext.class, true);
125             assert context != null;
126             context.setRelayState(relayState);
127         }
128 
129         return messageContext;
130     }
131 
132     private HttpPostBindingUtil() {}
