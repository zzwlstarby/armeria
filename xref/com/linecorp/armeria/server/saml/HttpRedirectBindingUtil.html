67      private static final String DEFAULT_CACHE_CONTROL =
68              String.join(",", HttpHeaderValues.NO_CACHE, HttpHeaderValues.NO_STORE);
69      private static final String DEFAULT_PRAGMA = HttpHeaderValues.NO_CACHE.toString();

71      /**
72       * Returns a {@link ResponseHeaders} with the specified {@code location}, the default {@code cache-control}
73       * and the default {@code pragma} headers.
74       */
75      static ResponseHeaders headersWithLocation(String location) {
76          return ResponseHeaders.of(HttpStatus.FOUND,
77                                    HttpHeaderNames.LOCATION, location,
78                                    HttpHeaderNames.CACHE_CONTROL, DEFAULT_CACHE_CONTROL,
79                                    HttpHeaderNames.PRAGMA, DEFAULT_PRAGMA);
80      }

82      /**
83       * Returns an {@link HttpResponse} with the specified {@code location}, the default {@code cache-control}
84       * and the default {@code pragma} headers.
85       */
86      static HttpResponse responseWithLocation(String location) {
87          return HttpResponse.of(headersWithLocation(location));
88      }

90      /**
91       * Returns a redirected URL which includes a deflated base64 string that is converted from the specified
92       * {@link SAMLObject}. The URL must contain a signature of the generated query string.
93       */
94      static String toRedirectionUrl(SAMLObject msg,
95                                     String endpointUrl,
96                                     String messageParamName,
97                                     Credential signingCredential,
98                                     String signatureAlgorithm,
99                                     @Nullable String relayState) {
100         requireNonNull(msg, "msg");
101         requireNonNull(endpointUrl, "endpointUrl");
102         requireNonNull(messageParamName, "messageParamName");
103         requireNonNull(signingCredential, "signingCredential");
104         requireNonNull(signatureAlgorithm, "signatureAlgorithm");
105 
106         final QueryParamsBuilder params = QueryParams.builder();
107         params.add(messageParamName, toDeflatedBase64(msg));
108 
109         if (relayState != null) {
110             // RelayState data MAY be included with a SAML protocol message transmitted with this binding.
111             // The value MUST NOT exceed 80 bytes in length and SHOULD be integrity protected by the entity
112             // creating the message independent of any other protections that may or may not exist
113             // during message transmission.
114             if (relayState.length() > 80) {
115                 throw new IllegalArgumentException("too long relayState string: " + relayState.length());
116             }
117             params.add(RELAY_STATE, relayState);
118         }
119 
120         params.add(SIGNATURE_ALGORITHM, signatureAlgorithm);
121 
122         // Use URL-encoded query string as input.
123         final String input = params.toQueryString();
124         final String output = generateSignature(signingCredential, signatureAlgorithm, input);
125         params.add(SIGNATURE, output);
126 
127         return endpointUrl + '?' + params.toQueryString();
128     }

130     /**
131      * Validates a signature in the specified {@link AggregatedHttpRequest}.
132      */
133     private static void validateSignature(Credential validationCredential,
134                                           SamlParameters parameters,
135                                           String messageParamName) {
136         requireNonNull(validationCredential, "validationCredential");
137         requireNonNull(parameters, "parameters");
138         requireNonNull(messageParamName, "messageParamName");
139 
140         final String signature = parameters.getFirstValue(SIGNATURE);
141         final String sigAlg = parameters.getFirstValue(SIGNATURE_ALGORITHM);
142 
143         // The order is one of the followings:
144         // - SAMLRequest={value}&RelayState={value}=SigAlg={value}
145         // - SAMLResponse={value}&RelayState={value}=SigAlg={value}
146         final QueryParamsBuilder params = QueryParams.builder();
147         params.add(messageParamName, parameters.getFirstValue(messageParamName));
148 
149         final String relayState = parameters.getFirstValueOrNull(RELAY_STATE);
150         if (relayState != null) {
151             params.add(RELAY_STATE, relayState);
152         }
153         params.add(SIGNATURE_ALGORITHM, sigAlg);
154 
155         final byte[] input = params.toQueryString().getBytes(StandardCharsets.UTF_8);
156 
157         try {
158             final byte[] decodedSignature = Base64.getMimeDecoder().decode(signature);
159             if (!XMLSigningUtil.verifyWithURI(validationCredential, sigAlg, decodedSignature, input)) {
160                 throw new InvalidSamlRequestException("failed to validate a signature");
161             }
162         } catch (IllegalArgumentException e) {
163             throw new InvalidSamlRequestException("failed to decode a base64 signature string", e);
164         } catch (SecurityException e) {
165             throw new InvalidSamlRequestException("failed to validate a signature", e);
166         }
167     }
168 
169     /**
170      * Generates a signature of the specified {@code input}.
171      */
172     @VisibleForTesting
173     static String generateSignature(Credential signingCredential, String algorithmURI, String input) {
174         try {
175             final byte[] signature =
176                     XMLSigningUtil.signWithURI(signingCredential, algorithmURI,
177                                                input.getBytes(StandardCharsets.UTF_8));
178             return Base64.getEncoder().encodeToString(signature);
179         } catch (SecurityException e) {
180             throw new SamlException("failed to generate a signature", e);
181         }
182     }
183 
184     /**
185      * Encodes the specified {@code message} into a deflated base64 string.
186      */
187     static String toDeflatedBase64(SAMLObject message) {
188         requireNonNull(message, "message");
189 
190         final String messageStr;
191         try {
192             messageStr = nodeToString(XMLObjectSupport.marshall(message));
193         } catch (MarshallingException e) {
194             throw new SamlException("failed to serialize a SAML message", e);
195         }
196 
197         final ByteArrayOutputStream bytesOut = new ByteArrayOutputStream();
198         try (DeflaterOutputStream deflaterStream =
199                      new DeflaterOutputStream(Base64.getEncoder().wrap(bytesOut),
200                                               new Deflater(Deflater.DEFLATED, true))) {
201             deflaterStream.write(messageStr.getBytes(StandardCharsets.UTF_8));
202         } catch (IOException e) {
203             throw new SamlException("failed to deflate a SAML message", e);
204         }
205         return bytesOut.toString();
206     }
207 
208     /**
209      * Decodes, inflates and deserializes the specified base64-encoded message to an {@link XMLObject}.
210      */
211     static XMLObject fromDeflatedBase64(String base64Encoded) {
212         requireNonNull(base64Encoded, "base64Encoded");
213 
214         final byte[] base64decoded;
215         try {
216             base64decoded = Base64.getMimeDecoder().decode(base64Encoded);
217         } catch (IllegalArgumentException e) {
218             throw new InvalidSamlRequestException("failed to decode a deflated base64 string", e);
219         }
220 
221         final ByteArrayOutputStream bytesOut = new ByteArrayOutputStream();
222         try (InflaterOutputStream inflaterOutputStream =
223                      new InflaterOutputStream(bytesOut, new Inflater(true))) {
224             inflaterOutputStream.write(base64decoded);
225         } catch (IOException e) {
226             throw new InvalidSamlRequestException("failed to inflate a SAML message", e);
227         }
228 
229         return SamlMessageUtil.deserialize(bytesOut.toByteArray());
230     }

232     /**
233      * Converts an {@link AggregatedHttpRequest} which is received from the remote entity to
234      * a {@link SAMLObject}.
235      */
236     @SuppressWarnings("unchecked")
237     static <T extends SAMLObject> MessageContext<T> toSamlObject(
238             AggregatedHttpRequest req, String name,
239             Map<String, SamlIdentityProviderConfig> idpConfigs,
240             @Nullable SamlIdentityProviderConfig defaultIdpConfig) {
241         requireNonNull(req, "req");
242         requireNonNull(name, "name");
243         requireNonNull(idpConfigs, "idpConfigs");
244 
245         final SamlParameters parameters = new SamlParameters(req);
246         final T message = (T) fromDeflatedBase64(parameters.getFirstValue(name));
247 
248         final MessageContext<T> messageContext = new MessageContext<>();
249         messageContext.setMessage(message);
250 
251         final Issuer issuer;
252         if (message instanceof RequestAbstractType) {
253             issuer = ((RequestAbstractType) message).getIssuer();
254         } else if (message instanceof StatusResponseType) {
255             issuer = ((StatusResponseType) message).getIssuer();
256         } else {
257             throw new InvalidSamlRequestException(
258                     "invalid message type: " + message.getClass().getSimpleName());
259         }
260 
261         // Use the default identity provider config if there's no issuer.
262         final SamlIdentityProviderConfig config;
263         if (issuer != null) {
264             final String idpEntityId = issuer.getValue();
265             config = idpConfigs.get(idpEntityId);
266             if (config == null) {
267                 throw new InvalidSamlRequestException(
268                         "a message from unknown identity provider: " + idpEntityId);
269             }
270         } else {
271             if (defaultIdpConfig == null) {
272                 throw new InvalidSamlRequestException("failed to get an Issuer element");
273             }
274             config = defaultIdpConfig;
275         }
276 
277         // If this message is sent via HTTP-redirect binding protocol, its signature parameter should
278         // be validated.
279         validateSignature(config.signingCredential(), parameters, name);
280 
281         final String relayState = parameters.getFirstValueOrNull(RELAY_STATE);
282         if (relayState != null) {
283             final SAMLBindingContext context = messageContext.getSubcontext(SAMLBindingContext.class, true);
284             assert context != null;
285             context.setRelayState(relayState);
286         }
287 
288         return messageContext;
289     }
