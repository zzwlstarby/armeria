16  package com.linecorp.armeria.server.saml;
17  
18  import static com.google.common.base.MoreObjects.firstNonNull;
19  import static com.linecorp.armeria.server.saml.SamlPortConfig.validatePort;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.net.URI;
23  import java.net.URISyntaxException;
24  
25  import javax.annotation.Nullable;
26  
27  import com.google.common.base.MoreObjects;
28  
29  import com.linecorp.armeria.common.util.Exceptions;
30  
31  /**
32   * A SAML service URL and its binding protocol.
33   */
34  public final class SamlEndpoint {
35      /**
36       * Creates a {@link SamlEndpoint} of the specified {@code uri} and the HTTP Redirect binding protocol.
37       */
38      public static SamlEndpoint ofHttpRedirect(String uri) {
39          requireNonNull(uri, "uri");
40          try {
41              return ofHttpRedirect(new URI(uri));
42          } catch (URISyntaxException e) {
43              return Exceptions.throwUnsafely(e);
44          }
45      }
46  
47      /**
48       * Creates a {@link SamlEndpoint} of the specified {@code uri} and the HTTP Redirect binding protocol.
49       */
50      public static SamlEndpoint ofHttpRedirect(URI uri) {
51          requireNonNull(uri, "uri");
52          return new SamlEndpoint(uri, SamlBindingProtocol.HTTP_REDIRECT);
53      }
54  
55      /**
56       * Creates a {@link SamlEndpoint} of the specified {@code uri} and the HTTP POST binding protocol.
57       */
58      public static SamlEndpoint ofHttpPost(String uri) {
59          requireNonNull(uri, "uri");
60          try {
61              return ofHttpPost(new URI(uri));
62          } catch (URISyntaxException e) {
63              return Exceptions.throwUnsafely(e);
64          }
65      }
66  
67      /**
68       * Creates a {@link SamlEndpoint} of the specified {@code uri} and the HTTP POST binding protocol.
69       */
70      public static SamlEndpoint ofHttpPost(URI uri) {
71          requireNonNull(uri, "uri");
72          return new SamlEndpoint(uri, SamlBindingProtocol.HTTP_POST);
73      }
74  
75      private final URI uri;
76      private final SamlBindingProtocol bindingProtocol;
77      private final String uriAsString;
78  
79      private SamlEndpoint(URI uri, SamlBindingProtocol bindingProtocol) {
80          this.uri = uri;
81          this.bindingProtocol = bindingProtocol;
82          uriAsString = uri.toString();
83      }
84  
85      /**
86       * Returns a {@link URI} of this endpoint.
87       */
88      public URI uri() {
89          return uri;
90      }
91  
92      /**
93       * Returns a {@link URI} of this endpoint as a string.
94       */
95      public String toUriString() {
96          return uriAsString;
97      }
98  
99      /**
100      * Returns a {@link URI} of this endpoint as a string. The omitted values in the {@link URI} will be
101      * replaced with the specified default values, such as {@code defaultScheme}, {@code defaultHostname}
102      * and {@code defaultPort}.
103      */
104     String toUriString(String defaultScheme, String defaultHostname, int defaultPort) {
105         requireNonNull(defaultScheme, "defaultScheme");
106         requireNonNull(defaultHostname, "defaultHostname");
107         validatePort(defaultPort);
108 
109         final StringBuilder sb = new StringBuilder();
110         sb.append(firstNonNull(uri.getScheme(), defaultScheme)).append("://")
111           .append(firstNonNull(uri.getHost(), defaultHostname)).append(':')
112           .append(uri.getPort() > 0 ? uri.getPort() : defaultPort)
113           .append(uri.getPath());
114         return sb.toString();
115     }
116 
117     /**
118      * Returns a {@link SamlBindingProtocol} of this endpoint.
119      */
120     public SamlBindingProtocol bindingProtocol() {
121         return bindingProtocol;
122     }
123 
124     @Override
125     public boolean equals(@Nullable Object o) {
126         if (this == o) {
127             return true;
128         }
129         if (!(o instanceof SamlEndpoint)) {
130             return false;
131         }
132         final SamlEndpoint that = (SamlEndpoint) o;
133         return uri().equals(that.uri()) && bindingProtocol() == that.bindingProtocol();
134     }
135 
136     @Override
137     public int hashCode() {
138         return uri().hashCode() * 31 + bindingProtocol().hashCode();
139     }
140 
141     @Override
142     public String toString() {
143         return MoreObjects.toStringHelper(this)
144                           .add("uri", uri)
145                           .add("bindingProtocol", bindingProtocol)
146                           .toString();
147     }
148 }
