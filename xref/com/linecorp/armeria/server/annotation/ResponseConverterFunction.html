17  package com.linecorp.armeria.server.annotation;
18  
19  import javax.annotation.Nullable;
20  
21  import com.linecorp.armeria.common.HttpHeaders;
22  import com.linecorp.armeria.common.HttpResponse;
23  import com.linecorp.armeria.common.HttpStatus;
24  import com.linecorp.armeria.common.MediaType;
25  import com.linecorp.armeria.common.ResponseHeaders;
26  import com.linecorp.armeria.server.ServiceRequestContext;
27  
28  /**
29   * Converts a {@code result} object to {@link HttpResponse}. The class implementing this interface would
30   * be specified as {@link ResponseConverter} annotation.
31   *
32   * @see ResponseConverter
33   */
34  @FunctionalInterface
35  public interface ResponseConverterFunction {
36  
37      /**
38       * Returns {@link HttpResponse} instance corresponds to the given {@code result}.
39       * Calls {@link ResponseConverterFunction#fallthrough()} or throws a {@link FallthroughException} if
40       * this converter cannot convert the {@code result} to the {@link HttpResponse}.
41       *
42       * @param headers The HTTP headers that you might want to use to create the {@link HttpResponse}.
43       *                The status of headers is {@link HttpStatus#OK} by default or
44       *                {@link HttpStatus#NO_CONTENT} if the annotated method returns {@code void},
45       *                unless you specify it with {@link StatusCode} on the method.
46       *                The headers also will include a {@link MediaType} if
47       *                {@link ServiceRequestContext#negotiatedResponseMediaType()} returns it.
48       *                If the method returns {@link HttpResult}, this headers is the same headers from
49       *                {@link HttpResult#headers()}
50       *                Please note that the additional headers set by
51       *                {@link ServiceRequestContext#addAdditionalResponseHeader(CharSequence, Object)}
52       *                and {@link AdditionalHeader} are not included in this headers.
53       * @param result The result of the service method.
54       * @param trailers The HTTP trailers that you might want to use to create the {@link HttpResponse}.
55       *                 If the annotated method returns {@link HttpResult}, this trailers is the same
56       *                 trailers from {@link HttpResult#trailers()}.
57       *                 Please note that the additional trailers set by
58       *                 {@link ServiceRequestContext#addAdditionalResponseTrailer(CharSequence, Object)}
59       *                 and {@link AdditionalTrailer} are not included in this trailers.
60       */
61      HttpResponse convertResponse(ServiceRequestContext ctx,
62                                   ResponseHeaders headers,
63                                   @Nullable Object result,
64                                   HttpHeaders trailers) throws Exception;
65  
66      /**
67       * Throws a {@link FallthroughException} in order to try to convert {@code result} to
68       * {@link HttpResponse} by the next converter.
69       */
70      static <T> T fallthrough() {
71          // Always throw the exception quietly.
72          throw FallthroughException.get();
73      }
74  }
