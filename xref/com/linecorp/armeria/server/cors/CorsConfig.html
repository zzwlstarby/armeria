17  package com.linecorp.armeria.server.cors;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.List;
22  import java.util.function.Supplier;
23  
24  import javax.annotation.Nullable;
25  
26  import com.google.common.base.Ascii;
27  import com.google.common.base.MoreObjects;
28  import com.google.common.collect.ImmutableList.Builder;
29  import com.google.common.collect.Iterables;
30  
31  import com.linecorp.armeria.server.Route;
32  import com.linecorp.armeria.server.RoutingContext;
33  
34  /**
35   * <a href="https://en.wikipedia.org/wiki/Cross-origin_resource_sharing">Cross-Origin Resource Sharing
36   * (CORS)</a> configuration.
37   *
38   * @see CorsServiceBuilder
39   * @see CorsService#config()
40   */
41  public final class CorsConfig {
42  
43      private final boolean anyOriginSupported;
44      private final boolean shortCircuit;
45      private final List<CorsPolicy> policies;
46  
47      CorsConfig(CorsServiceBuilder builder) {
48          anyOriginSupported = builder.anyOriginSupported;
49          shortCircuit = builder.shortCircuit;
50          policies = new Builder<CorsPolicy>()
51                  .add(builder.firstPolicyBuilder.build())
52                  .addAll(builder.policies)
53                  .build();
54      }
55  
56      /**
57       * Determines whether a wildcard origin, '*', is supported.
58       *
59       * @return {@code true} if any origin is allowed.
60       */
61      public boolean isAnyOriginSupported() {
62          return anyOriginSupported;
63      }
64  
65      /**
66       * Determines whether a CORS request should be rejected if it's invalid before being
67       * further processing.
68       *
69       * <p>CORS headers are set after a request is processed. This may not always be desired
70       * and this setting will check that the Origin is valid and if it is not valid no
71       * further processing will take place, and a error will be returned to the calling client.
72       *
73       * @return {@code true} if a CORS request should short-circuit upon receiving an invalid Origin header.
74       */
75      public boolean isShortCircuit() {
76          return shortCircuit;
77      }
78  
79      /**
80       * Returns the policies.
81       */
82      public List<CorsPolicy> policies() {
83          return policies;
84      }
85  
86      /**
87       * Returns the policy for the specified {@code origin}.
88       *
89       * @return {@link CorsPolicy} which allows the {@code origin},
90       *         {@code null} if the {@code origin} is {@code null} or not allowed in any policy.
91       */
92      @Nullable
93      public CorsPolicy getPolicy(@Nullable String origin, RoutingContext routingContext) {
94          requireNonNull(routingContext, "routingContext");
95          if (origin == null) {
96              return null;
97          }
98  
99          if (isAnyOriginSupported()) {
100             return Iterables.getFirst(policies, null);
101         }
102 
103         final String lowerCaseOrigin = Ascii.toLowerCase(origin);
104         final boolean isNullOrigin = CorsService.NULL_ORIGIN.equals(lowerCaseOrigin);
105         for (final CorsPolicy policy : policies) {
106             if (isNullOrigin && policy.isNullOriginAllowed() &&
107                 isPathMatched(policy, routingContext)) {
108                 return policy;
109             } else if (!isNullOrigin && policy.origins().contains(lowerCaseOrigin) &&
110                        isPathMatched(policy, routingContext)) {
111                 return policy;
112             }
113         }
114         return null;
115     }
116 
117     private static boolean isPathMatched(CorsPolicy policy, RoutingContext routingContext) {
118         final List<Route> routes = policy.routes();
119         // We do not consider the score of the routing result for simplicity. It'd be enough to find
120         // whether the path is matched or not.
121         return routes.isEmpty() ||
122                routes.stream().anyMatch(route -> route.apply(routingContext).isPresent());
123     }
124 
125     @Override
126     public String toString() {
127         return toString(this, anyOriginSupported, shortCircuit, policies);
128     }
129 
130     static String toString(Object obj, boolean anyOriginSupported, boolean shortCircuit,
131                            List<CorsPolicy> policies) {
132         return MoreObjects.toStringHelper(obj)
133                           .add("policies", policies)
134                           .add("shortCircuit", shortCircuit)
135                           .add("anyOriginSupported", anyOriginSupported)
136                           .toString();
137     }
138 
139     /**
140      * This class is used for preflight HTTP response values that do not need to be
141      * generated, but instead the value is "static" in that the same value will be returned
142      * for each call.
143      */
144     static final class ConstantValueSupplier implements Supplier<Object> {
145 
146         static final ConstantValueSupplier ZERO = new ConstantValueSupplier("0");
147 
148         private final Object value;
149 
150         ConstantValueSupplier(Object value) {
151             this.value = value;
152         }
153 
154         @Override
155         public Object get() {
156             return value;
157         }
158 
159         @Override
160         public String toString() {
161             return String.valueOf(value);
162         }
163     }
164 }
