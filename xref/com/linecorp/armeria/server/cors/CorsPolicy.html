49  /**
50   * Contains information of the CORS policy with the specified origins.
51   */
52  public final class CorsPolicy {
53  
54      private static final String DELIMITER = ",";
55      private static final Joiner HEADER_JOINER = Joiner.on(DELIMITER);
56  
57      /**
58       * Returns a new {@link CorsPolicyBuilder}.
59       */
60      public static CorsPolicyBuilder builder() {
61          return new CorsPolicyBuilder();
62      }
63  
64      /**
65       * Returns a new {@link CorsPolicyBuilder} with the specified {@code origins}.
66       */
67      public static CorsPolicyBuilder builder(String... origins) {
68          return new CorsPolicyBuilder(origins);
69      }
70  
71      /**
72       * Returns a new {@link CorsPolicyBuilder} with the specified {@code origins}.
73       */
74      public static CorsPolicyBuilder builder(Iterable<String> origins) {
75          return new CorsPolicyBuilder(origins);
76      }
77  
78      private final Set<String> origins;
79      private final List<Route> routes;
80      private final boolean credentialsAllowed;
81      private final boolean nullOriginAllowed;
82      private final long maxAge;
83      private final Set<AsciiString> exposedHeaders;
84      private final Set<HttpMethod> allowedRequestMethods;
85      private final Set<AsciiString> allowedRequestHeaders;
86      private final String joinedExposedHeaders;
87      private final String joinedAllowedRequestHeaders;
88      private final String joinedAllowedRequestMethods;
89      private final Map<AsciiString, Supplier<?>> preflightResponseHeaders;
90  
91      CorsPolicy(Set<String> origins, List<Route> routes, boolean credentialsAllowed, long maxAge,
92                 boolean nullOriginAllowed, Set<AsciiString> exposedHeaders,
93                 Set<AsciiString> allowedRequestHeaders, EnumSet<HttpMethod> allowedRequestMethods,
94                 boolean preflightResponseHeadersDisabled,
95                 Map<AsciiString, Supplier<?>> preflightResponseHeaders) {
96          this.origins = ImmutableSet.copyOf(origins);
97          this.routes = ImmutableList.copyOf(routes);
98          this.credentialsAllowed = credentialsAllowed;
99          this.maxAge = maxAge;
100         this.nullOriginAllowed = nullOriginAllowed;
101         this.exposedHeaders = ImmutableSet.copyOf(exposedHeaders);
102         this.allowedRequestMethods = ImmutableSet.copyOf(allowedRequestMethods);
103         this.allowedRequestHeaders = ImmutableSet.copyOf(allowedRequestHeaders);
104         joinedExposedHeaders = HEADER_JOINER.join(this.exposedHeaders);
105         joinedAllowedRequestMethods = this.allowedRequestMethods
106                 .stream().map(HttpMethod::name).collect(Collectors.joining(DELIMITER));
107         joinedAllowedRequestHeaders = HEADER_JOINER.join(this.allowedRequestHeaders);
108         if (preflightResponseHeadersDisabled) {
109             this.preflightResponseHeaders = Collections.emptyMap();
110         } else if (preflightResponseHeaders.isEmpty()) {
111             this.preflightResponseHeaders = ImmutableMap.of(
112                     HttpHeaderNames.DATE, HttpTimestampSupplier::currentTime,
113                     HttpHeaderNames.CONTENT_LENGTH, ConstantValueSupplier.ZERO);
114         } else {
115             this.preflightResponseHeaders = ImmutableMap.copyOf(preflightResponseHeaders);
116         }
117     }
118 
119     /**
120      * Returns the allowed origin. This can either be a wildcard or an origin value.
121      * This method returns the first specified origin if this policy has more than one origin.
122      *
123      * @return the value that will be used for the CORS response header {@code "Access-Control-Allow-Origin"}
124      */
125     public String origin() {
126         return Iterables.getFirst(origins, ANY_ORIGIN);
127     }
128 
129     /**
130      * Returns the set of allowed origins.
131      */
132     public Set<String> origins() {
133         return origins;
134     }
135 
136     /**
137      * Returns the list of {@link Route}s that this policy is supposed to be applied to.
138      */
139     public List<Route> routes() {
140         return routes;
141     }
142 
143     /**
144      * Determines if cookies are supported for CORS requests.
145      *
146      * <p>By default cookies are not included in CORS requests but if {@code isCredentialsAllowed} returns
147      * {@code true} cookies will be added to CORS requests. Setting this value to {@code true} will set the
148      * CORS {@code "Access-Control-Allow-Credentials"} response header to {@code true}.
149      *
150      * <p>Please note that cookie support needs to be enabled on the client side as well.
151      * The client needs to opt-in to send cookies by calling:
152      * <pre>{@code
153      * xhr.withCredentials = true;
154      * }</pre>
155      *
156      * <p>The default value for {@code 'withCredentials'} is {@code false} in which case no cookies are sent.
157      * Setting {@code this} to {@code true} will include cookies in cross origin requests.
158      *
159      * @return {@code true} if cookies are supported.
160      */
161     public boolean isCredentialsAllowed() {
162         return credentialsAllowed;
163     }
164 
165     /**
166      * Gets the {@code maxAge} setting.
167      *
168      * <p>When making a preflight request the client has to perform two requests which can be inefficient.
169      * This setting will set the CORS {@code "Access-Control-Max-Age"} response header and enable the
170      * caching of the preflight response for the specified time. During this time no preflight
171      * request will be made.
172      *
173      * @return the time in seconds that a preflight request may be cached.
174      */
175     public long maxAge() {
176         return maxAge;
177     }
178 
179     /**
180      * Returns a set of headers to be exposed to calling clients.
181      *
182      * <p>During a simple CORS request only certain response headers are made available by the
183      * browser, for example using:
184      * <pre>{@code
185      * xhr.getResponseHeader("Content-Type");
186      * }</pre>
187      * The headers that are available by default are:
188      * <ul>
189      *   <li>{@code Cache-Control}</li>
190      *   <li>{@code Content-Language}</li>
191      *   <li>{@code Content-Type}</li>
192      *   <li>{@code Expires}</li>
193      *   <li>{@code Last-Modified}</li>
194      *   <li>{@code Pragma}</li>
195      * </ul>
196      *
197      * <p>To expose other headers they need to be specified, which is what this method enables by
198      * adding the headers names to the CORS {@code "Access-Control-Expose-Headers"} response header.
199      *
200      * @return the list of the headers to expose.
201      */
202     public Set<AsciiString> exposedHeaders() {
203         return exposedHeaders;
204     }
205 
206     /**
207      * Returns the allowed set of request methods. The Http methods that should be returned in the
208      * CORS {@code "Access-Control-Request-Method"} response header.
209      *
210      * @return the {@link HttpMethod}s that represent the allowed request methods.
211      */
212     public Set<HttpMethod> allowedRequestMethods() {
213         return allowedRequestMethods;
214     }
215 
216     /**
217      * Returns the allowed set of request headers.
218      *
219      * <p>The header names returned from this method will be used to set the CORS
220      * {@code "Access-Control-Allow-Headers"} response header.
221      */
222     public Set<AsciiString> allowedRequestHeaders() {
223         return allowedRequestHeaders;
224     }
225 
226     /**
227      * Determines if the policy allows a {@code "null"} origin.
228      */
229     public boolean isNullOriginAllowed() {
230         return nullOriginAllowed;
231     }
232 
233     /**
234      * Generates immutable HTTP response headers that should be added to a CORS preflight response.
235      *
236      * @return {@link HttpHeaders} the HTTP response headers to be added.
237      */
238     public HttpHeaders generatePreflightResponseHeaders() {
239         final HttpHeadersBuilder headers = HttpHeaders.builder();
240         preflightResponseHeaders.forEach((key, value) -> {
241             final Object val = getValue(value);
242             if (val instanceof Iterable) {
243                 headers.addObject(key, (Iterable<?>) val);
244             } else {
245                 headers.addObject(key, val);
246             }
247         });
248         return headers.build();
249     }
250 
251     /**
252      * This is a non CORS specification feature which enables the setting of preflight
253      * response headers that might be required by intermediaries.
254      *
255      * @param headers the {@link ResponseHeadersBuilder} to which the preflight headers should be added.
256      */
257     void setCorsPreflightResponseHeaders(ResponseHeadersBuilder headers) {
258         headers.add(generatePreflightResponseHeaders());
259     }
260 
261     void setCorsAllowCredentials(ResponseHeadersBuilder headers) {
262         // The string "*" cannot be used for a resource that supports credentials.
263         // https://www.w3.org/TR/cors/#resource-requests
264         if (credentialsAllowed &&
265             !ANY_ORIGIN.equals(headers.get(HttpHeaderNames.ACCESS_CONTROL_ALLOW_ORIGIN))) {
266             headers.set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_CREDENTIALS, "true");
267         }
268     }
269 
270     void setCorsExposeHeaders(ResponseHeadersBuilder headers) {
271         if (exposedHeaders.isEmpty()) {
272             return;
273         }
274 
275         headers.set(HttpHeaderNames.ACCESS_CONTROL_EXPOSE_HEADERS, joinedExposedHeaders);
276     }
277 
278     void setCorsAllowMethods(ResponseHeadersBuilder headers) {
279         headers.set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_METHODS, joinedAllowedRequestMethods);
280     }
281 
282     void setCorsAllowHeaders(ResponseHeadersBuilder headers) {
283         if (allowedRequestHeaders.isEmpty()) {
284             return;
285         }
286 
287         headers.set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_HEADERS, joinedAllowedRequestHeaders);
288     }
289 
290     void setCorsMaxAge(ResponseHeadersBuilder headers) {
291         headers.setLong(HttpHeaderNames.ACCESS_CONTROL_MAX_AGE, maxAge);
292     }
293 
294     @Override
295     public String toString() {
296         return toString(this, origins, routes, nullOriginAllowed, credentialsAllowed, maxAge,
297                         exposedHeaders, allowedRequestMethods, allowedRequestHeaders, preflightResponseHeaders);
298     }
299 
300     static String toString(Object obj, @Nullable Set<String> origins, List<Route> routes,
301                            boolean nullOriginAllowed, boolean credentialsAllowed,
302                            long maxAge, @Nullable Set<AsciiString> exposedHeaders,
303                            @Nullable Set<HttpMethod> allowedRequestMethods,
304                            @Nullable Set<AsciiString> allowedRequestHeaders,
305                            @Nullable Map<AsciiString, Supplier<?>> preflightResponseHeaders) {
306         return MoreObjects.toStringHelper(obj)
307                           .omitNullValues()
308                           .add("origins", origins)
309                           .add("routes", routes)
310                           .add("nullOriginAllowed", nullOriginAllowed)
311                           .add("credentialsAllowed", credentialsAllowed)
312                           .add("maxAge", maxAge)
313                           .add("exposedHeaders", exposedHeaders)
314                           .add("allowedRequestMethods", allowedRequestMethods)
315                           .add("allowedRequestHeaders", allowedRequestHeaders)
316                           .add("preflightResponseHeaders", preflightResponseHeaders).toString();
317     }
318 
319     private static <T> T getValue(Supplier<T> callable) {
320         try {
321             return callable.get();
322         } catch (final Exception e) {
323             throw new IllegalStateException("could not generate value for supplier: " + callable, e);
324         }
325     }
