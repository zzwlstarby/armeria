153     JettyService(@Nullable String hostname,
154                  Function<ScheduledExecutorService, Server> serverFactory,
155                  Consumer<Server> postStopTask) {

158         this.serverFactory = serverFactory;

163     @Override
164     public void serviceAdded(ServiceConfig cfg) {
165         if (armeriaServer != null) {
166             if (armeriaServer != cfg.server()) {
167                 throw new IllegalStateException("cannot be added to more than one server");
168             } else {
169                 return;
170             }
171         }
172 
173         armeriaServer = cfg.server();
174         armeriaServer.addListener(configurator);
175         if (hostname == null) {
176             hostname = armeriaServer.defaultHostname();
177         }
178     }
179 
180     void start() throws Exception {
181         boolean success = false;
182         try {
183             assert armeriaServer != null;
184             server = serverFactory.apply(armeriaServer.config().blockingTaskExecutor());
185             connector = new ArmeriaConnector(server, armeriaServer);
186             server.addConnector(connector);
187 
188             if (!server.isRunning()) {
189                 logger.info("Starting an embedded Jetty: {}", server);
190                 server.start();
191                 startedServer = true;
192             } else {
193                 startedServer = false;
194             }
195             success = true;
196         } finally {
197             if (!success) {
198                 server = null;
199                 connector = null;
200             }
201         }
202     }
203 
204     void stop() {
205         final Server server = this.server;
206         this.server = null;
207         connector = null;
208 
209         if (server == null || !startedServer) {
210             return;
211         }
212 
213         try {
214             logger.info("Stopping an embedded Jetty: {}", server);
215             server.stop();
216         } catch (Exception e) {
217             logger.warn("Failed to stop an embedded Jetty: {}", server, e);
218         }
219 
220         postStopTask.accept(server);
221     }

223     @Override
224     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) {
225         final ArmeriaConnector connector = this.connector;
226         assert connector != null;
227 
228         final HttpResponseWriter res = HttpResponse.streaming();
229 
230         req.aggregate().handle((aReq, cause) -> {
231             if (cause != null) {
232                 logger.warn("{} Failed to aggregate a request:", ctx, cause);
233                 if (res.tryWrite(ResponseHeaders.of(HttpStatus.INTERNAL_SERVER_ERROR))) {
234                     res.close();
235                 }
236                 return null;
237             }
238 
239             boolean success = false;
240             try {
241                 final ArmeriaHttpTransport transport = new ArmeriaHttpTransport(req.method());
242                 final HttpChannel httpChannel = new HttpChannel(
243                         connector,
244                         connector.getHttpConfiguration(),
245                         new ArmeriaEndPoint(hostname, connector.getScheduler(),
246                                             ctx.localAddress(), ctx.remoteAddress()),
247                         transport);
248 
249                 fillRequest(ctx, aReq, httpChannel.getRequest());
250 
251                 ctx.blockingTaskExecutor().execute(() -> invoke(ctx, res, transport, httpChannel));
252                 success = true;
253                 return null;
254             } finally {
255                 if (!success) {
256                     res.close();
257                 }
258             }
259         }).exceptionally(CompletionActions::log);
260 
261         return res;
262     }
263 
264     private void invoke(ServiceRequestContext ctx, HttpResponseWriter res,
265                         ArmeriaHttpTransport transport, HttpChannel httpChannel) {
266 
267         final Queue<HttpData> out = transport.out;
268         try {
269             server.handle(httpChannel);
270             httpChannel.getResponse().getHttpOutput().flush();
271 
272             final Throwable cause = transport.cause;
273             if (cause != null) {
274                 throw cause;
275             }
276 
277             final HttpHeaders headers = toResponseHeaders(transport);
278             if (res.tryWrite(headers)) {
279                 for (;;) {
280                     final HttpData data = out.poll();
281                     if (data == null || !res.tryWrite(data)) {
282                         break;
283                     }
284                 }
285             }
286         } catch (Throwable t) {
287             logger.warn("{} Failed to produce a response:", ctx, t);
288         } finally {
289             res.close();
290         }
291     }

293     private static void fillRequest(
294             ServiceRequestContext ctx, AggregatedHttpRequest aReq, Request jReq) {
295 
296         jReq.setDispatcherType(DispatcherType.REQUEST);
297         jReq.setAsyncSupported(false, "armeria");
298         jReq.setSecure(ctx.sessionProtocol().isTls());
299         jReq.setMetaData(toRequestMetadata(ctx, aReq));
300 
301         final HttpData content = aReq.content();
302         if (!content.isEmpty()) {
303             jReq.getHttpInput().addContent(new Content(ByteBuffer.wrap(content.array())));
304         }
305         jReq.getHttpInput().eof();
306     }
307 
308     private static MetaData.Request toRequestMetadata(ServiceRequestContext ctx, AggregatedHttpRequest aReq) {
309         // Construct the HttpURI
310         final StringBuilder uriBuf = new StringBuilder();
311         final RequestHeaders aHeaders = aReq.headers();
312 
313         uriBuf.append(ctx.sessionProtocol().isTls() ? "https" : "http");
314         uriBuf.append("://");
315         uriBuf.append(aHeaders.authority());
316         uriBuf.append(aHeaders.path());
317 
318         final HttpURI uri = new HttpURI(uriBuf.toString());
319         uri.setPath(ctx.mappedPath());
320 
321         // Convert HttpHeaders to HttpFields
322         final HttpFields jHeaders = new HttpFields(aHeaders.size());
323         aHeaders.forEach(e -> {
324             final AsciiString key = e.getKey();
325             if (!key.isEmpty() && key.byteAt(0) != ':') {
326                 jHeaders.add(key.toString(), e.getValue());
327             }
328         });
329 
330         return new MetaData.Request(aHeaders.get(HttpHeaderNames.METHOD), uri,
331                                     HttpVersion.HTTP_1_1, jHeaders, aReq.content().length());
332     }

334     private static ResponseHeaders toResponseHeaders(ArmeriaHttpTransport transport) {
335         final MetaData.Response info = transport.info;
336         if (info == null) {
337             throw new IllegalStateException("response metadata unavailable");
338         }
339 
340         final ResponseHeadersBuilder headers = ResponseHeaders.builder();
341         headers.status(info.getStatus());
342         info.getFields().forEach(e -> headers.add(HttpHeaderNames.of(e.getName()), e.getValue()));
343 
344         if (transport.method != HttpMethod.HEAD) {
345             headers.setLong(HttpHeaderNames.CONTENT_LENGTH, transport.contentLength);
346         }
347 
348         return headers.build();
349     }

351     private static final class ArmeriaHttpTransport implements HttpTransport {
352 
353         final HttpMethod method;
354         final Queue<HttpData> out = new ArrayDeque<>();
355         long contentLength;
356         @Nullable
357         MetaData.Response info;
358         @Nullable
359         Throwable cause;
360 
361         ArmeriaHttpTransport(HttpMethod method) {
362             this.method = method;
363         }
364 
365         @Override
366         public void send(@Nullable MetaData.Response info, boolean head,
367                          ByteBuffer content, boolean lastContent, Callback callback) {
368 
369             if (info != null) {
370                 this.info = info;
371             }
372 
373             final int length = content.remaining();
374             if (length == 0) {
375                 callback.succeeded();
376                 return;
377             }
378 
379             if (content.hasArray()) {
380                 final int from = content.arrayOffset() + content.position();
381                 out.add(HttpData.wrap(Arrays.copyOfRange(content.array(), from, from + length)));
382                 content.position(content.position() + length);
383             } else {
384                 final byte[] data = new byte[length];
385                 content.get(data);
386                 out.add(HttpData.wrap(data));
387             }
388 
389             contentLength += length;
390             callback.succeeded();
391         }
392 
393         @Override
394         public boolean isPushSupported() {
395             return false;
396         }
397 
398         @Override
399         public void push(MetaData.Request request) {}
400 
401         @Override
402         public void onCompleted() {}
403 
404         @Override
405         public void abort(Throwable failure) {
406             cause = failure;
407         }
408 
409         @Override
410         public boolean isOptimizedForDirectBuffers() {
411             return false;
412         }

415     private static final class ArmeriaEndPoint extends AbstractEndPoint {
416 
417         private final InetSocketAddress localAddress;
418         private final InetSocketAddress remoteAddress;
419 
420         ArmeriaEndPoint(@Nullable String hostname, Scheduler scheduler,
421                         SocketAddress local, SocketAddress remote) {
422             super(scheduler);
423 
424             localAddress = addHostname((InetSocketAddress) local, hostname);
425             remoteAddress = (InetSocketAddress) remote;
426 
427             setIdleTimeout(getIdleTimeout());
428         }
429 
430         @Override
431         public InetSocketAddress getLocalAddress() {
432             return localAddress;
433         }
434 
435         @Override
436         public InetSocketAddress getRemoteAddress() {
437             return remoteAddress;
438         }
439 
440         /**
441          * Adds the hostname string to the specified {@link InetSocketAddress} so that
442          * Jetty's {@code ServletRequest.getLocalName()} implementation returns the configured hostname.
443          */
444         private static InetSocketAddress addHostname(InetSocketAddress address, @Nullable String hostname) {
445             try {
446                 return new InetSocketAddress(InetAddress.getByAddress(
447                         hostname, address.getAddress().getAddress()), address.getPort());
448             } catch (UnknownHostException e) {
449                 throw new Error(e); // Should never happen
450             }
451         }
