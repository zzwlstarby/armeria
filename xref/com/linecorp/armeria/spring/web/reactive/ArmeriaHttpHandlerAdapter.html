16  package com.linecorp.armeria.spring.web.reactive;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.util.concurrent.CompletableFuture;
21  
22  import javax.annotation.Nullable;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  import org.springframework.http.server.reactive.HttpHandler;
27  
28  import com.linecorp.armeria.common.HttpRequest;
29  import com.linecorp.armeria.common.HttpResponse;
30  import com.linecorp.armeria.common.HttpStatus;
31  import com.linecorp.armeria.common.MediaType;
32  import com.linecorp.armeria.server.ServiceRequestContext;
33  
34  import reactor.core.publisher.Mono;
35  
36  /**
37   * An adapter which adapts the {@link HttpHandler} to the Armeria server.
38   */
39  final class ArmeriaHttpHandlerAdapter {
40      private static final Logger logger = LoggerFactory.getLogger(ArmeriaHttpHandlerAdapter.class);
41  
42      private final HttpHandler httpHandler;
43      private final DataBufferFactoryWrapper<?> factoryWrapper;
44  
45      ArmeriaHttpHandlerAdapter(HttpHandler httpHandler, DataBufferFactoryWrapper<?> factoryWrapper) {
46          this.httpHandler = requireNonNull(httpHandler, "httpHandler");
47          this.factoryWrapper = requireNonNull(factoryWrapper, "factoryWrapper");
48      }
49  
50      Mono<Void> handle(ServiceRequestContext ctx, HttpRequest req, CompletableFuture<HttpResponse> future,
51                        @Nullable String serverHeader) {
52          final ArmeriaServerHttpRequest convertedRequest;
53          try {
54              convertedRequest = new ArmeriaServerHttpRequest(ctx, req, factoryWrapper);
55          } catch (Exception e) {
56              final String path = req.path();
57              logger.warn("{} Invalid request path: {}", ctx, path, e);
58              future.complete(HttpResponse.of(HttpStatus.BAD_REQUEST,
59                                              MediaType.PLAIN_TEXT_UTF_8,
60                                              HttpStatus.BAD_REQUEST + "\nInvalid request path: " + path));
61              return Mono.empty();
62          }
63  
64          final ArmeriaServerHttpResponse convertedResponse =
65                  new ArmeriaServerHttpResponse(ctx, future, factoryWrapper, serverHeader);
66          return httpHandler.handle(convertedRequest, convertedResponse)
67                            .doOnSuccess(unused -> {
68                                convertedResponse.setComplete().subscribe();
69                            })
70                            .doOnError(cause -> {
71                                logger.debug("{} Failed to handle a request", ctx, cause);
72                                convertedResponse.setComplete(cause).subscribe();
73                            });
74      }
75  }
