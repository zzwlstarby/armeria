98      private static com.linecorp.armeria.spring.Ssl toArmeriaSslConfiguration(Ssl ssl) {
99          if (!ssl.isEnabled()) {
100             return new com.linecorp.armeria.spring.Ssl();
101         }
102 
103         ClientAuth clientAuth = null;
104         if (ssl.getClientAuth() != null) {
105             switch (ssl.getClientAuth()) {
106                 case NEED:
107                     clientAuth = ClientAuth.REQUIRE;
108                     break;
109                 case WANT:
110                     clientAuth = ClientAuth.OPTIONAL;
111                     break;
112             }
113         }
114         return new com.linecorp.armeria.spring.Ssl()
115                 .setEnabled(ssl.isEnabled())
116                 .setClientAuth(clientAuth)
117                 .setCiphers(ssl.getCiphers() != null ? ImmutableList.copyOf(ssl.getCiphers()) : null)
118                 .setEnabledProtocols(ssl.getEnabledProtocols() != null ? ImmutableList.copyOf(
119                         ssl.getEnabledProtocols()) : null)
120                 .setKeyAlias(ssl.getKeyAlias())
121                 .setKeyPassword(ssl.getKeyPassword())
122                 .setKeyStore(ssl.getKeyStore())
123                 .setKeyStorePassword(ssl.getKeyStorePassword())
124                 .setKeyStoreType(ssl.getKeyStoreType())
125                 .setKeyStoreProvider(ssl.getKeyStoreProvider())
126                 .setTrustStore(ssl.getTrustStore())
127                 .setTrustStorePassword(ssl.getTrustStorePassword())
128                 .setTrustStoreType(ssl.getTrustStoreType())
129                 .setTrustStoreProvider(ssl.getTrustStoreProvider());
130     }

132     @Override
133     public WebServer getWebServer(HttpHandler httpHandler) {
134         final ServerBuilder sb = Server.builder();
135         sb.disableServerHeader();
136         sb.disableDateHeader();
137 
138         final SessionProtocol protocol;
139         final Ssl ssl = getSsl();
140         if (ssl != null) {
141             if (ssl.isEnabled()) {
142                 final SslStoreProvider provider = getSslStoreProvider();
143                 final Supplier<KeyStore> keyStoreSupplier;
144                 final Supplier<KeyStore> trustStoreSupplier;
145                 if (provider != null) {
146                     keyStoreSupplier = () -> {
147                         try {
148                             return provider.getKeyStore();
149                         } catch (Exception e) {
150                             throw new IllegalStateException(e);
151                         }
152                     };
153                     trustStoreSupplier = () -> {
154                         try {
155                             return provider.getTrustStore();
156                         } catch (Exception e) {
157                             throw new IllegalStateException(e);
158                         }
159                     };
160                 } else {
161                     keyStoreSupplier = null;
162                     trustStoreSupplier = null;
163                 }
164                 configureTls(sb, toArmeriaSslConfiguration(ssl), keyStoreSupplier, trustStoreSupplier);
165                 protocol = SessionProtocol.HTTPS;
166             } else {
167                 logger.warn("TLS configuration exists but it is disabled by 'enabled' property.");
168                 protocol = SessionProtocol.HTTP;
169             }
170         } else {
171             protocol = SessionProtocol.HTTP;
172         }
173 
174         final Http2 http2 = getHttp2();
175         if (http2 != null && !http2.isEnabled()) {
176             logger.warn(
177                     "Cannot disable HTTP/2 protocol for Armeria server. It will be enabled automatically.");
178         }
179 
180         final InetAddress address = getAddress();
181         final int port = ensureValidPort(getPort());
182         if (address != null) {
183             sb.port(new InetSocketAddress(address, port), protocol);
184         } else {
185             sb.port(port, protocol);
186         }
187 
188         final Compression compression = getCompression();
189         if (compression != null && compression.getEnabled()) {
190             final long minResponseSize = compression.getMinResponseSize().toBytes();
191             sb.decorator(contentEncodingDecorator(compression.getMimeTypes(),
192                                                   compression.getExcludedUserAgents(),
193                                                   Ints.saturatedCast(minResponseSize)));
194         }
195 
196         final ArmeriaSettings settings = findBean(ArmeriaSettings.class);
197         if (settings != null) {
198             configureArmeriaService(sb, settings);
199         }
200         findBeans(ArmeriaServerConfigurator.class).forEach(configurator -> configurator.configure(sb));
201 
202         final DataBufferFactoryWrapper<?> factoryWrapper =
203                 firstNonNull(findBean(DataBufferFactoryWrapper.class), DataBufferFactoryWrapper.DEFAULT);
204 
205         final Server server = configureService(sb, httpHandler, factoryWrapper, getServerHeader()).build();
206         return new ArmeriaWebServer(server, protocol, address, port);
207     }
208 
209     private static ServerBuilder configureService(ServerBuilder sb, HttpHandler httpHandler,
210                                                   DataBufferFactoryWrapper<?> factoryWrapper,
211                                                   @Nullable String serverHeader) {
212         final ArmeriaHttpHandlerAdapter handler =
213                 new ArmeriaHttpHandlerAdapter(httpHandler, factoryWrapper);
214         return sb.service(Route.ofCatchAll(), (ctx, req) -> {
215             final CompletableFuture<HttpResponse> future = new CompletableFuture<>();
216             final HttpResponse response = HttpResponse.from(future);
217             final Disposable disposable = handler.handle(ctx, req, future, serverHeader).subscribe();
218             response.whenComplete().handle((unused, cause) -> {
219                 if (cause != null) {
220                     if (ctx.method() != HttpMethod.HEAD) {
221                         logger.debug("{} Response stream has been cancelled.", ctx, cause);
222                     }
223                     disposable.dispose();
224                 }
225                 return null;
226             });
227             return response;
228         });
229     }

231     private void configureArmeriaService(ServerBuilder sb, ArmeriaSettings settings) {
232         final MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory =
233                 settings.isEnableMetrics() ? firstNonNull(findBean(MeterIdPrefixFunctionFactory.class),
234                                                           MeterIdPrefixFunctionFactory.ofDefault())
235                                            : null;
236 
237         configurePorts(sb, settings.getPorts());
238         final DocServiceBuilder docServiceBuilder = DocService.builder();
239         configureThriftServices(sb,
240                                 docServiceBuilder,
241                                 findBeans(ThriftServiceRegistrationBean.class),
242                                 meterIdPrefixFunctionFactory,
243                                 settings.getDocsPath());
244         configureGrpcServices(sb,
245                               docServiceBuilder,
246                               findBeans(GrpcServiceRegistrationBean.class),
247                               meterIdPrefixFunctionFactory,
248                               settings.getDocsPath());
249         configureHttpServices(sb,
250                               findBeans(HttpServiceRegistrationBean.class),
251                               meterIdPrefixFunctionFactory);
252         configureAnnotatedServices(sb,
253                                    docServiceBuilder,
254                                    findBeans(AnnotatedServiceRegistrationBean.class),
255                                    meterIdPrefixFunctionFactory,
256                                    settings.getDocsPath());
257         configureServerWithArmeriaSettings(sb, settings,
258                                            firstNonNull(findBean(MeterRegistry.class), Metrics.globalRegistry),
259                                            findBeans(HealthChecker.class));
260         if (settings.getSsl() != null) {
261             configureTls(sb, settings.getSsl());
262         }
263 
264         final ArmeriaSettings.Compression compression = settings.getCompression();
265         if (compression != null && compression.isEnabled()) {
266             final long parsed = parseDataSize(compression.getMinResponseSize());
267             sb.decorator(contentEncodingDecorator(compression.getMimeTypes(),
268                                                   compression.getExcludedUserAgents(),
269                                                   Ints.saturatedCast(parsed)));
270         }
271 
272         if (!Strings.isNullOrEmpty(settings.getDocsPath())) {
273             sb.serviceUnder(settings.getDocsPath(), docServiceBuilder.build());
274         }
275     }

277     @Nullable
278     private <T> T findBean(Class<T> clazz) {
279         try {
280             return beanFactory.getBean(clazz);
281         } catch (NoUniqueBeanDefinitionException e) {
282             throw new IllegalStateException("Too many " + clazz.getSimpleName() + " beans: (expected: 1)", e);
283         } catch (NoSuchBeanDefinitionException e) {
284             return null;
285         }
286     }
287 
288     private <T> List<T> findBeans(Class<T> clazz) {
289         final String[] names = beanFactory.getBeanNamesForType(clazz);
290         if (names.length == 0) {
291             return ImmutableList.of();
292         }
293         return Arrays.stream(names)
294                      .map(name -> beanFactory.getBean(name, clazz))
295                      .collect(toImmutableList());
296     }
297 
298     private static int ensureValidPort(int port) {
299         checkArgument(port >= 0 && port <= 65535,
300                       "port: %s (expected: 0[arbitrary port] or 1-65535)", port);
301         return port;
302     }
