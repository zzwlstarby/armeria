70  /**
71   * {@link HttpService} to handle a {@link WebOperation}. Mostly inspired by reactive implementation in
72   * {@link AbstractWebFluxEndpointHandlerMapping}.
73   */
74  final class WebOperationService implements HttpService {
75  
76      private static final Pattern FILENAME_BAD_CHARS = Pattern.compile("['/\\\\?%*:|\"<> ]");
77  
78      private static final Logger logger = LoggerFactory.getLogger(WebOperationService.class);
79  
80      private static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();
81      private static final TypeReference<Map<String, Object>> JSON_MAP =
82              new TypeReference<Map<String, Object>>() {};
83  
84      @Nullable
85      private static final Class<?> healthComponentClass;
86      @Nullable
87      private static final MethodHandle getStatusMethodHandle;
88  
89      static {
90          final String healthComponentClassName = "org.springframework.boot.actuate.health.HealthComponent";
91          final String getStatusMethodName = "getStatus";
92          Class<?> healthComponentC = null;
93          MethodHandle getStatusMH = null;
94          try {
95              healthComponentC = Class.forName(
96                      healthComponentClassName, false, Health.class.getClassLoader());
97              getStatusMH = MethodHandles.lookup().findVirtual(
98                      healthComponentC, getStatusMethodName, MethodType.methodType(Status.class));
99          } catch (Throwable cause) {
100             logger.debug("Failed to find {}#{}() - not Spring Boot 2.2+?",
101                          healthComponentClassName, getStatusMethodName, cause);
102         }
103 
104         if (getStatusMH != null) {
105             healthComponentClass = healthComponentC;
106             getStatusMethodHandle = getStatusMH;
107         } else {
108             healthComponentClass = null;
109             getStatusMethodHandle = null;
110         }
111     }

113     private final WebOperation operation;
114     private final HealthStatusHttpMapper healthMapper;
115 
116     WebOperationService(WebOperation operation,
117                         HealthStatusHttpMapper healthMapper) {
118         this.operation = operation;
119         this.healthMapper = healthMapper;
120     }
121 
122     @Override
123     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) {
124         final CompletableFuture<HttpResponse> resFuture = new CompletableFuture<>();
125         req.aggregate().handle((aggregatedReq, t) -> {
126             if (t != null) {
127                 resFuture.completeExceptionally(t);
128                 return null;
129             }
130             if (operation.isBlocking()) {
131                 ctx.blockingTaskExecutor().execute(() -> invoke(ctx, aggregatedReq, resFuture));
132             } else {
133                 invoke(ctx, aggregatedReq, resFuture);
134             }
135             return null;
136         });
137         return HttpResponse.from(resFuture);
138     }
139 
140     private void invoke(ServiceRequestContext ctx,
141                         AggregatedHttpRequest req,
142                         CompletableFuture<HttpResponse> resFuture) {
143         final Map<String, Object> arguments = getArguments(ctx, req);
144         final Object result = operation.invoke(new InvocationContext(SecurityContext.NONE, arguments));
145 
146         try {
147             final HttpResponse res = handleResult(ctx, result, req.method());
148             resFuture.complete(res);
149         } catch (Throwable cause) {
150             resFuture.completeExceptionally(cause);
151         }
152     }
153 
154     private static Map<String, Object> getArguments(ServiceRequestContext ctx, AggregatedHttpRequest req) {
155         final Map<String, Object> arguments = new LinkedHashMap<>(ctx.pathParams());
156         if (!req.content().isEmpty()) {
157             final Map<String, Object> bodyParams;
158             try {
159                 bodyParams = OBJECT_MAPPER.readValue(req.content().array(), JSON_MAP);
160             } catch (IOException e) {
161                 throw new IllegalArgumentException("Invalid JSON in request.");
162             }
163             arguments.putAll(bodyParams);
164         }
165 
166         final QueryParams params = QueryParams.fromQueryString(ctx.query());
167         for (String name : params.names()) {
168             final List<String> values = params.getAll(name);
169             arguments.put(name, values.size() != 1 ? values : values.get(0));
170         }
171 
172         return ImmutableMap.copyOf(arguments);
173     }
174 
175     private HttpResponse handleResult(ServiceRequestContext ctx,
176                                       @Nullable Object result, HttpMethod method) throws Throwable {
177         if (result == null) {
178             return HttpResponse.of(method != HttpMethod.GET ? HttpStatus.NO_CONTENT : HttpStatus.NOT_FOUND);
179         }
180 
181         final HttpStatus status;
182         final Object body;
183         if (result instanceof WebEndpointResponse) {
184             final WebEndpointResponse<?> webResult = (WebEndpointResponse<?>) result;
185             status = HttpStatus.valueOf(webResult.getStatus());
186             body = webResult.getBody();
187         } else {
188             if (result instanceof Health) {
189                 status = HttpStatus.valueOf(healthMapper.mapStatus(((Health) result).getStatus()));
190             } else if (healthComponentClass != null && healthComponentClass.isInstance(result)) {
191                 assert getStatusMethodHandle != null; // Always non-null if healthComponentClass is not null.
192                 final Status actuatorStatus = (Status) getStatusMethodHandle.invoke(result);
193                 status = HttpStatus.valueOf(healthMapper.mapStatus(actuatorStatus));
194             } else {
195                 status = HttpStatus.OK;
196             }
197             body = result;
198         }
199 
200         final MediaType contentType = firstNonNull(ctx.negotiatedResponseMediaType(), MediaType.JSON_UTF_8);
201         final String contentSubType = contentType.subtype();
202 
203         if ("json".equals(contentSubType) || contentSubType.endsWith("+json")) {
204             return HttpResponse.of(status, contentType, OBJECT_MAPPER.writeValueAsBytes(body));
205         }
206 
207         if (body instanceof CharSequence) {
208             return HttpResponse.of(status, contentType, (CharSequence) body);
209         }
210 
211         if (body instanceof Resource) {
212             final Resource resource = (Resource) body;
213             final String filename = resource.getFilename();
214             final HttpResponseWriter res = HttpResponse.streaming();
215             final long length = resource.contentLength();
216             final ResponseHeadersBuilder headers = ResponseHeaders.builder(status);
217             headers.contentType(contentType);
218             headers.setLong(HttpHeaderNames.CONTENT_LENGTH, length);
219             headers.setTimeMillis(HttpHeaderNames.LAST_MODIFIED, resource.lastModified());
220             if (filename != null) {
221                 headers.set(HttpHeaderNames.CONTENT_DISPOSITION,
222                             "attachment;filename=" + FILENAME_BAD_CHARS.matcher(filename).replaceAll("_"));
223             }
224 
225             res.write(headers.build());
226 
227             boolean success = false;
228             ReadableByteChannel in = null;
229             try {
230                 in = resource.readableChannel();
231                 final ReadableByteChannel finalIn = in;
232                 ctx.blockingTaskExecutor().execute(() -> streamResource(ctx, res, finalIn, length));
233                 success = true;
234                 return res;
235             } finally {
236                 if (!success && in != null) {
237                     try {
238                         in.close();
239                     } catch (IOException e) {
240                         logger.warn("{} Failed to close an actuator resource: {}", ctx, resource, e);
241                     }
242                 }
243             }
244         }
245 
246         logger.warn("{} Cannot convert an actuator response: {}", ctx, body);
247         return HttpResponse.of(status, contentType, body.toString());
248     }

250     // TODO(trustin): A lot of duplication with StreamingHttpFile. Need to add some utility classes for
251     //                streaming a ReadableByteChannel and an InputStream.
252     private static void streamResource(ServiceRequestContext ctx, HttpResponseWriter res,
253                                        ReadableByteChannel in, long remainingBytes) {
254 
255         final int chunkSize = (int) Math.min(8192, remainingBytes);
256         final ByteBuf buf = ctx.alloc().buffer(chunkSize);
257         final int readBytes;
258         boolean success = false;
259         try {
260             readBytes = read(in, buf);
261             if (readBytes < 0) {
262                 // Should not reach here because we only read up to the end of the stream.
263                 // If reached, it may mean the stream has been truncated.
264                 throw new EOFException();
265             }
266             success = true;
267         } catch (Exception e) {
268             close(res, in, e);
269             return;
270         } finally {
271             if (!success) {
272                 buf.release();
273             }
274         }
275 
276         final long nextRemainingBytes = remainingBytes - readBytes;
277         final boolean endOfStream = nextRemainingBytes == 0;
278         if (readBytes > 0) {
279             if (!res.tryWrite(new ByteBufHttpData(buf, endOfStream))) {
280                 close(in);
281                 return;
282             }
283         } else {
284             buf.release();
285         }
286 
287         if (endOfStream) {
288             close(res, in);
289             return;
290         }
291 
292         res.whenConsumed().thenRun(() -> {
293             try {
294                 ctx.blockingTaskExecutor().execute(() -> streamResource(ctx, res, in, nextRemainingBytes));
295             } catch (Exception e) {
296                 close(res, in, e);
297             }
298         });
299     }
300 
301     private static int read(ReadableByteChannel src, ByteBuf dst) throws IOException {
302         if (src instanceof ScatteringByteChannel) {
303             return dst.writeBytes((ScatteringByteChannel) src, dst.writableBytes());
304         }
305 
306         final int readBytes = src.read(dst.nioBuffer(dst.writerIndex(), dst.writableBytes()));
307         if (readBytes > 0) {
308             dst.writerIndex(dst.writerIndex() + readBytes);
309         }
310         return readBytes;
311     }
312 
313     private static void close(HttpResponseWriter res, Closeable in) {
314         close(in);
315         res.close();
316     }
317 
318     private static void close(HttpResponseWriter res, Closeable in, Exception cause) {
319         close(in);
320         res.close(cause);
321     }
322 
323     private static void close(Closeable in) {
324         try {
325             in.close();
326         } catch (Exception e) {
327             logger.warn("Failed to close a stream for: {}", in, e);
328         }
329     }
