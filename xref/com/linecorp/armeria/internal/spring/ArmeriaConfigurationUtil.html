111     /**
112      * Sets graceful shutdown timeout, health check services and {@link MeterRegistry} for the specified
113      * {@link ServerBuilder}.
114      */
115     public static void configureServerWithArmeriaSettings(ServerBuilder server, ArmeriaSettings settings,
116                                                           MeterRegistry meterRegistry,
117                                                           List<HealthChecker> healthCheckers) {
118         requireNonNull(server, "server");
119         requireNonNull(settings, "settings");
120         requireNonNull(meterRegistry, "meterRegistry");
121         requireNonNull(healthCheckers, "healthCheckers");
122 
123         if (settings.getGracefulShutdownQuietPeriodMillis() >= 0 &&
124             settings.getGracefulShutdownTimeoutMillis() >= 0) {
125             server.gracefulShutdownTimeoutMillis(settings.getGracefulShutdownQuietPeriodMillis(),
126                                                  settings.getGracefulShutdownTimeoutMillis());
127             logger.debug("Set graceful shutdown timeout: quiet period {} ms, timeout {} ms",
128                          settings.getGracefulShutdownQuietPeriodMillis(),
129                          settings.getGracefulShutdownTimeoutMillis());
130         }
131 
132         final String healthCheckPath = settings.getHealthCheckPath();
133         if (!Strings.isNullOrEmpty(healthCheckPath)) {
134             server.service(healthCheckPath, HealthCheckService.of(healthCheckers));
135         }
136 
137         server.meterRegistry(meterRegistry);
138 
139         if (settings.isEnableMetrics() && !Strings.isNullOrEmpty(settings.getMetricsPath())) {
140             final boolean hasPrometheus = hasAllClasses(
141                     "io.micrometer.prometheus.PrometheusMeterRegistry",
142                     "io.prometheus.client.CollectorRegistry");
143 
144             final boolean addedPrometheusExposition;
145             if (hasPrometheus) {
146                 addedPrometheusExposition = PrometheusSupport.addExposition(settings, server, meterRegistry);
147             } else {
148                 addedPrometheusExposition = false;
149             }
150 
151             if (!addedPrometheusExposition) {
152                 final boolean hasDropwizard = hasAllClasses(
153                         "io.micrometer.core.instrument.dropwizard.DropwizardMeterRegistry",
154                         "com.codahale.metrics.MetricRegistry",
155                         "com.codahale.metrics.json.MetricsModule");
156                 if (hasDropwizard) {
157                     DropwizardSupport.addExposition(settings, server, meterRegistry);
158                 }
159             }
160         }
161 
162         if (settings.getSsl() != null) {
163             configureTls(server, settings.getSsl());
164         }
165 
166         final ArmeriaSettings.Compression compression = settings.getCompression();
167         if (compression != null && compression.isEnabled()) {
168             final int minBytesToForceChunkedAndEncoding =
169                     Ints.saturatedCast(parseDataSize(compression.getMinResponseSize()));
170             server.decorator(contentEncodingDecorator(compression.getMimeTypes(),
171                                                       compression.getExcludedUserAgents(),
172                                                       minBytesToForceChunkedAndEncoding));
173         }
174     }

176     private static boolean hasAllClasses(String... classNames) {
177         for (String className : classNames) {
178             try {
179                 Class.forName(className, false, ArmeriaConfigurationUtil.class.getClassLoader());
180             } catch (Throwable t) {
181                 return false;
182             }
183         }
184         return true;
185     }

187     /**
188      * Adds {@link Port}s to the specified {@link ServerBuilder}.
189      */
190     public static void configurePorts(ServerBuilder server, List<Port> ports) {
191         requireNonNull(server, "server");
192         requireNonNull(ports, "ports");
193         ports.forEach(p -> {
194             final String ip = p.getIp();
195             final String iface = p.getIface();
196             final int port = p.getPort();
197             final List<SessionProtocol> protocols = firstNonNull(p.getProtocols(),
198                                                                  ImmutableList.of(SessionProtocol.HTTP));
199 
200             if (ip == null) {
201                 if (iface == null) {
202                     server.port(new ServerPort(port, protocols));
203                 } else {
204                     try {
205                         final Enumeration<InetAddress> e = NetworkInterface.getByName(iface).getInetAddresses();
206                         while (e.hasMoreElements()) {
207                             server.port(new ServerPort(new InetSocketAddress(e.nextElement(), port),
208                                                        protocols));
209                         }
210                     } catch (SocketException e) {
211                         throw new IllegalStateException("Failed to find an iface: " + iface, e);
212                     }
213                 }
214             } else if (iface == null) {
215                 if (NetUtil.isValidIpV4Address(ip) || NetUtil.isValidIpV6Address(ip)) {
216                     final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(ip);
217                     try {
218                         server.port(new ServerPort(new InetSocketAddress(
219                                 InetAddress.getByAddress(bytes), port), protocols));
220                     } catch (UnknownHostException e) {
221                         // Should never happen.
222                         throw new Error(e);
223                     }
224                 } else {
225                     throw new IllegalStateException("invalid IP address: " + ip);
226                 }
227             } else {
228                 throw new IllegalStateException("A port cannot have both IP and iface: " + p);
229             }
230         });
231     }

233     /**
234      * Adds Thrift services to the specified {@link ServerBuilder}.
235      */
236     public static void configureThriftServices(
237             ServerBuilder server, DocServiceBuilder docServiceBuilder,
238             List<ThriftServiceRegistrationBean> beans,
239             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory,
240             @Nullable String docsPath) {
241         requireNonNull(server, "server");
242         requireNonNull(docServiceBuilder, "docServiceBuilder");
243         requireNonNull(beans, "beans");
244 
245         final List<TBase<?, ?>> docServiceRequests = new ArrayList<>();
246         final Map<String, Collection<? extends ExampleHeaders>> docServiceHeaders = new HashMap<>();
247         beans.forEach(bean -> {
248             HttpService service = bean.getService();
249             for (Function<? super HttpService, ? extends HttpService> decorator : bean.getDecorators()) {
250                 service = service.decorate(decorator);
251             }
252             service = setupMetricCollectingService(service, bean, meterIdPrefixFunctionFactory);
253             server.service(bean.getPath(), service);
254             docServiceRequests.addAll(bean.getExampleRequests());
255             ThriftServiceUtils.serviceNames(bean.getService())
256                               .forEach(serviceName ->
257                                                docServiceHeaders.put(serviceName, bean.getExampleHeaders()));
258         });
259 
260         if (Strings.isNullOrEmpty(docsPath)) {
261             return;
262         }
263 
264         docServiceBuilder.exampleRequest(docServiceRequests);
265         for (Entry<String, Collection<? extends ExampleHeaders>> entry : docServiceHeaders.entrySet()) {
266             for (ExampleHeaders exampleHeaders : entry.getValue()) {
267                 configureExampleHeaders(docServiceBuilder, entry.getKey(), exampleHeaders.getMethodName(),
268                                         exampleHeaders.getHeaders());
269             }
270         }
271     }

292     /**
293      * Adds gRPC services to the specified {@link ServerBuilder}.
294      */
295     public static void configureGrpcServices(
296             ServerBuilder server, DocServiceBuilder docServiceBuilder,
297             List<GrpcServiceRegistrationBean> beans,
298             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory,
299             @Nullable String docsPath) {
300         requireNonNull(server, "server");
301         requireNonNull(docServiceBuilder, "docServiceBuilder");
302         requireNonNull(beans, "beans");
303 
304         final List<GrpcExampleRequest> docServiceRequests = new ArrayList<>();
305         final List<GrpcExampleHeaders> docServiceHeaders = new ArrayList<>();
306         beans.forEach(bean -> {
307             final HttpServiceWithRoutes serviceWithRoutes = bean.getService();
308             docServiceRequests.addAll(bean.getExampleRequests());
309             docServiceHeaders.addAll(bean.getExampleHeaders());
310             serviceWithRoutes.routes().forEach(
311                     route -> {
312                         HttpService service = bean.getService();
313                         for (Function<? super HttpService, ? extends HttpService> decorator
314                                 : bean.getDecorators()) {
315                             service = service.decorate(decorator);
316                         }
317                         server.service(route,
318                                        setupMetricCollectingService(service, bean,
319                                                                     meterIdPrefixFunctionFactory));
320                     }
321             );
322         });
323 
324         if (Strings.isNullOrEmpty(docsPath)) {
325             return;
326         }
327 
328         docServiceRequests.forEach(
329                 exampleReq -> docServiceBuilder.exampleRequestForMethod(exampleReq.getServiceType(),
330                                                                         exampleReq.getMethodName(),
331                                                                         exampleReq.getExampleRequest()));
332         docServiceHeaders.forEach(exampleHeader -> configureExampleHeaders(docServiceBuilder,
333                                                                            exampleHeader.getServiceType(),
334                                                                            exampleHeader.getMethodName(),
335                                                                            exampleHeader.getHeaders()));
336     }

338     /**
339      * Adds annotated HTTP services to the specified {@link ServerBuilder}.
340      */
341     public static void configureAnnotatedServices(
342             ServerBuilder server, DocServiceBuilder docServiceBuilder,
343             List<AnnotatedServiceRegistrationBean> beans,
344             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory,
345             @Nullable String docsPath) {
346         requireNonNull(server, "server");
347         requireNonNull(docServiceBuilder, "docServiceBuilder");
348         requireNonNull(beans, "beans");
349 
350         final Map<String, Collection<? extends AnnotatedExampleRequest>> docServiceRequests = new HashMap<>();
351         final Map<String, Collection<? extends ExampleHeaders>> docServiceHeaders = new HashMap<>();
352         beans.forEach(bean -> {
353             Function<? super HttpService, ? extends HttpService> decorator = Function.identity();
354             for (Function<? super HttpService, ? extends HttpService> d : bean.getDecorators()) {
355                 decorator = decorator.andThen(d);
356             }
357             if (meterIdPrefixFunctionFactory != null) {
358                 decorator = decorator.andThen(
359                         metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));
360             }
361             final ImmutableList<Object> exceptionHandlersAndConverters =
362                     ImmutableList.builder()
363                                  .addAll(bean.getExceptionHandlers())
364                                  .addAll(bean.getRequestConverters())
365                                  .addAll(bean.getResponseConverters())
366                                  .build();
367             final String serviceName = bean.getService().getClass().getName();
368             docServiceRequests.put(serviceName, bean.getExampleRequests());
369             docServiceHeaders.put(serviceName, bean.getExampleHeaders());
370             server.annotatedService(bean.getPathPrefix(), bean.getService(), decorator,
371                                     exceptionHandlersAndConverters);
372         });
373 
374         if (Strings.isNullOrEmpty(docsPath)) {
375             return;
376         }
377 
378         for (Entry<String, Collection<? extends AnnotatedExampleRequest>> entry
379                 : docServiceRequests.entrySet()) {
380             for (AnnotatedExampleRequest exampleRequest : entry.getValue()) {
381                 docServiceBuilder.exampleRequestForMethod(entry.getKey(),
382                                                           exampleRequest.getMethodName(),
383                                                           exampleRequest.getExampleRequest());
384             }
385         }
386 
387         for (Entry<String, Collection<? extends ExampleHeaders>> entry : docServiceHeaders.entrySet()) {
388             for (ExampleHeaders exampleHeaders : entry.getValue()) {
389                 configureExampleHeaders(docServiceBuilder, entry.getKey(), exampleHeaders.getMethodName(),
390                                         exampleHeaders.getHeaders());
391             }
392         }
393     }

395     private static HttpService setupMetricCollectingService(
396             HttpService service, AbstractServiceRegistrationBean<?, ?, ?, ?> bean,
397             @Nullable MeterIdPrefixFunctionFactory meterIdPrefixFunctionFactory) {
398         requireNonNull(service, "service");
399         requireNonNull(bean, "bean");

403         }
404         return service.decorate(metricCollectingServiceDecorator(bean, meterIdPrefixFunctionFactory));

438     /**
439      * Adds SSL/TLS context to the specified {@link ServerBuilder}.
440      */
441     public static void configureTls(ServerBuilder sb, Ssl ssl,
442                                     @Nullable Supplier<KeyStore> keyStoreSupplier,
443                                     @Nullable Supplier<KeyStore> trustStoreSupplier) {
444         if (!ssl.isEnabled()) {
445             return;
446         }
447         try {
448             if (keyStoreSupplier == null && trustStoreSupplier == null &&
449                 ssl.getKeyStore() == null && ssl.getTrustStore() == null) {
450                 logger.warn("Configuring TLS with a self-signed certificate " +
451                             "because no key or trust store was specified");
452                 sb.tlsSelfSigned();
453                 return;
454             }
455 
456             final KeyManagerFactory keyManagerFactory = getKeyManagerFactory(ssl, keyStoreSupplier);
457             final TrustManagerFactory trustManagerFactory = getTrustManagerFactory(ssl, trustStoreSupplier);
458 
459             sb.tls(keyManagerFactory);
460             sb.tlsCustomizer(sslContextBuilder -> {
461                 sslContextBuilder.trustManager(trustManagerFactory);
462 
463                 final SslProvider sslProvider = ssl.getProvider();
464                 if (sslProvider != null) {
465                     sslContextBuilder.sslProvider(sslProvider);
466                 }
467                 final List<String> enabledProtocols = ssl.getEnabledProtocols();
468                 if (enabledProtocols != null) {
469                     sslContextBuilder.protocols(enabledProtocols.toArray(EMPTY_PROTOCOL_NAMES));
470                 }
471                 final List<String> ciphers = ssl.getCiphers();
472                 if (ciphers != null) {
473                     sslContextBuilder.ciphers(ImmutableList.copyOf(ciphers),
474                                               SupportedCipherSuiteFilter.INSTANCE);
475                 }
476                 final ClientAuth clientAuth = ssl.getClientAuth();
477                 if (clientAuth != null) {
478                     sslContextBuilder.clientAuth(clientAuth);
479                 }
480             });
481         } catch (Exception e) {
482             throw new IllegalStateException("Failed to configure TLS: " + e, e);
483         }
484     }

486     private static KeyManagerFactory getKeyManagerFactory(
487             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
488         final KeyStore store;
489         if (sslStoreProvider != null) {
490             store = sslStoreProvider.get();
491         } else {
492             store = loadKeyStore(ssl.getKeyStoreType(), ssl.getKeyStore(), ssl.getKeyStorePassword());
493         }
494 
495         KeyManagerFactory keyManagerFactory =
496                 KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
497         if (ssl.getKeyAlias() != null) {
498             keyManagerFactory = new CustomAliasKeyManagerFactory(keyManagerFactory, ssl.getKeyAlias());
499         }
500 
501         String keyPassword = ssl.getKeyPassword();
502         if (keyPassword == null) {
503             keyPassword = ssl.getKeyStorePassword();
504         }
505 
506         keyManagerFactory.init(store, keyPassword != null ? keyPassword.toCharArray()
507                                                           : null);
508         return keyManagerFactory;
509     }
510 
511     private static TrustManagerFactory getTrustManagerFactory(
512             Ssl ssl, @Nullable Supplier<KeyStore> sslStoreProvider) throws Exception {
513         final KeyStore store;
514         if (sslStoreProvider != null) {
515             store = sslStoreProvider.get();
516         } else {
517             store = loadKeyStore(ssl.getTrustStoreType(), ssl.getTrustStore(), ssl.getTrustStorePassword());
518         }
519 
520         final TrustManagerFactory trustManagerFactory =
521                 TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
522         trustManagerFactory.init(store);
523         return trustManagerFactory;
524     }

526     @Nullable
527     private static KeyStore loadKeyStore(
528             @Nullable String type,
529             @Nullable String resource,
530             @Nullable String password) throws IOException, GeneralSecurityException {
531         if (resource == null) {
532             return null;
533         }
534         final KeyStore store = KeyStore.getInstance(firstNonNull(type, "JKS"));
535         final URL url = ResourceUtils.getURL(resource);
536         store.load(url.openStream(), password != null ? password.toCharArray()
537                                                       : null);
538         return store;
539     }

541     /**
542      * Configures a decorator for encoding the content of the HTTP responses sent from the server.
543      */
544     public static Function<? super HttpService, EncodingService> contentEncodingDecorator(
545             @Nullable String[] mimeTypes, @Nullable String[] excludedUserAgents,
546             int minBytesToForceChunkedAndEncoding) {
547         final Predicate<MediaType> encodableContentTypePredicate;
548         if (mimeTypes == null || mimeTypes.length == 0) {
549             encodableContentTypePredicate = contentType -> true;
550         } else {
551             final List<MediaType> encodableContentTypes =
552                     Arrays.stream(mimeTypes).map(MediaType::parse).collect(toImmutableList());
553             encodableContentTypePredicate = contentType ->
554                     encodableContentTypes.stream().anyMatch(contentType::is);
555         }
556 
557         final Predicate<? super RequestHeaders> encodableRequestHeadersPredicate;
558         if (excludedUserAgents == null || excludedUserAgents.length == 0) {
559             encodableRequestHeadersPredicate = headers -> true;
560         } else {
561             final List<Pattern> patterns =
562                     Arrays.stream(excludedUserAgents).map(Pattern::compile).collect(toImmutableList());
563             encodableRequestHeadersPredicate = headers -> {
564                 // No User-Agent header will be converted to an empty string.
565                 final String userAgent = headers.get(HttpHeaderNames.USER_AGENT, "");
566                 return patterns.stream().noneMatch(pattern -> pattern.matcher(userAgent).matches());
567             };
568         }
569 
570         return EncodingService.builder()
571                               .encodableContentTypePredicate(encodableContentTypePredicate)
572                               .encodableRequestHeadersPredicate(encodableRequestHeadersPredicate)
573                               .minBytesToForceChunkedEncoding(minBytesToForceChunkedAndEncoding)
574                               .newDecorator();
575     }

577     /**
578      * Parses the data size text as a decimal {@code long}.
579      *
580      * @param dataSizeText the data size text, i.e. {@code 1}, {@code 1B}, {@code 1KB}, {@code 1MB},
581      *                     {@code 1GB} or {@code 1TB}
582      */
583     public static long parseDataSize(String dataSizeText) {
584         requireNonNull(dataSizeText, "text");
585         final Matcher matcher = DATA_SIZE_PATTERN.matcher(dataSizeText);
586         checkArgument(matcher.matches(),
587                       "Invalid data size text: %s (expected: %s)",
588                       dataSizeText, DATA_SIZE_PATTERN);
589 
590         final long unit;
591         final String unitText = matcher.group(2);
592         if (Strings.isNullOrEmpty(unitText)) {
593             unit = 1L;
594         } else {
595             switch (Ascii.toLowerCase(unitText)) {
596                 case "b":
597                     unit = 1L;
598                     break;
599                 case "kb":
600                     unit = 1024L;
601                     break;
602                 case "mb":
603                     unit = 1024L * 1024L;
604                     break;
605                 case "gb":
606                     unit = 1024L * 1024L * 1024L;
607                     break;
608                 case "tb":
609                     unit = 1024L * 1024L * 1024L * 1024L;
610                     break;
611                 default:
612                     throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
613                                                        " (expected: " + DATA_SIZE_PATTERN + ')');
614             }
615         }
616         try {
617             final long amount = Long.parseLong(matcher.group(1));
618             return LongMath.checkedMultiply(amount, unit);
619         } catch (Exception e) {
620             throw new IllegalArgumentException("Invalid data size text: " + dataSizeText +
621                                                " (expected: " + DATA_SIZE_PATTERN + ')', e);
622         }
623     }
