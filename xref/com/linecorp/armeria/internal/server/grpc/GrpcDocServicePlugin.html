128     @Override
129     public ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs,
130                                                       DocServiceFilter filter) {
131         requireNonNull(serviceConfigs, "serviceConfigs");
132         requireNonNull(filter, "filter");
133 
134         final Map<String, ServiceEntryBuilder> map = new LinkedHashMap<>();
135         for (ServiceConfig serviceConfig : serviceConfigs) {
136             final GrpcService grpcService = serviceConfig.service().as(GrpcService.class);
137             assert grpcService != null;
138 
139             final ImmutableSet.Builder<MediaType> supportedMediaTypesBuilder = ImmutableSet.builder();
140             supportedMediaTypesBuilder.addAll(grpcService.supportedSerializationFormats()
141                                                          .stream()
142                                                          .map(SerializationFormat::mediaType)::iterator);
143 
144             if (!grpcService.isFramed()) {
145                 if (grpcService.supportedSerializationFormats().contains(GrpcSerializationFormats.PROTO)) {
146                     // Normal clients of a GrpcService are not required to set a protocol when using unframed
147                     // requests but we set it here for clarity in DocService, where there may be multiple
148                     // services with similar mime types but different protocols.
149                     supportedMediaTypesBuilder.add(MediaType.PROTOBUF.withParameter("protocol", "gRPC"));
150                 }
151                 if (grpcService.supportedSerializationFormats().contains(GrpcSerializationFormats.JSON)) {
152                     supportedMediaTypesBuilder.add(MediaType.JSON_UTF_8.withParameter("protocol", "gRPC"));
153                 }
154             }
155             final Set<MediaType> supportedMediaTypes = supportedMediaTypesBuilder.build();
156 
157             // Find the ServiceDescriptors of all services and put them all into the 'map'
158             // after wrapping with ServiceEntryBuilder.
159             grpcService.services().stream()
160                        .map(ServerServiceDefinition::getServiceDescriptor)
161                        .filter(Objects::nonNull)
162                        .filter(desc -> desc.getSchemaDescriptor() instanceof ProtoFileDescriptorSupplier)
163                        .forEach(desc -> {
164                            final String serviceName = desc.getName();
165                            map.computeIfAbsent(serviceName, s -> {
166                                final ProtoFileDescriptorSupplier fileDescSupplier =
167                                        (ProtoFileDescriptorSupplier) desc.getSchemaDescriptor();
168                                final FileDescriptor fileDesc = fileDescSupplier.getFileDescriptor();
169                                final ServiceDescriptor serviceDesc =
170                                        fileDesc.getServices().stream()
171                                                .filter(sd -> sd.getFullName().equals(serviceName))
172                                                .findFirst()
173                                                .orElseThrow(IllegalStateException::new);
174                                return new ServiceEntryBuilder(serviceDesc);
175                            });
176                        });
177 
178             final String pathPrefix;
179             final Route route = serviceConfig.route();
180             if (route.pathType() == RoutePathType.PREFIX) {
181                 pathPrefix = route.paths().get(0);
182             } else {
183                 pathPrefix = "/";
184             }
185 
186             for (ServerServiceDefinition service : grpcService.services()) {
187                 final String serviceName = service.getServiceDescriptor().getName();
188                 map.get(serviceName).endpoint(
189                         EndpointInfo.builder(serviceConfig.virtualHost().hostnamePattern(),
190                                              // Only the URL prefix, each method is served
191                                              // at a different path.
192                                              pathPrefix + serviceName + '/')
193                                     .availableMimeTypes(supportedMediaTypes)
194                                     .build());
195             }
196         }
197         return generate(map.values().stream().map(ServiceEntryBuilder::build).collect(toImmutableList()),
198                         filter);
199     }

201     @Override
202     public Map<String, String> loadDocStrings(Set<ServiceConfig> serviceConfigs) {
203         return serviceConfigs.stream()
204                              .flatMap(c -> {
205                                  final GrpcService grpcService = c.service().as(GrpcService.class);
206                                  assert grpcService != null;
207                                  return grpcService.services().stream();
208                              })
209                              .flatMap(s -> docstringExtractor.getAllDocStrings(s.getClass().getClassLoader())
210                                                              .entrySet().stream())
211                              .collect(toImmutableMap(Map.Entry::getKey, Map.Entry::getValue, (a, b) -> a));
212     }

219     @Override
220     public String serializeExampleRequest(String serviceName, String methodName,
221                                           Object exampleRequest) {
222         try {
223             return defaultExamplePrinter.print((MessageOrBuilder) exampleRequest);
224         } catch (InvalidProtocolBufferException e) {
225             throw new UncheckedIOException(
226                     "Invalid example request protobuf. Is it missing required fields?", e);
227         }
228     }

230     @VisibleForTesting
231     ServiceSpecification generate(List<ServiceEntry> entries, DocServiceFilter filter) {
232         final List<ServiceInfo> services = entries.stream()
233                                                   .map(entry -> newServiceInfo(entry, filter))
234                                                   .filter(Objects::nonNull)
235                                                   .collect(toImmutableList());
236 
237         return ServiceSpecification.generate(services, this::newNamedTypeInfo);
238     }

240     private NamedTypeInfo newNamedTypeInfo(TypeSignature typeSignature) {
241         final Object descriptor = typeSignature.namedTypeDescriptor();
242         if (descriptor instanceof Descriptor) {
243             return newStructInfo((Descriptor) descriptor);
244         }
245 
246         assert descriptor instanceof EnumDescriptor;
247         return newEnumInfo((EnumDescriptor) descriptor);
248     }

250     @Nullable
251     ServiceInfo newServiceInfo(ServiceEntry entry, DocServiceFilter filter) {
252         final List<MethodInfo> methodInfos =
253                 entry.methods().stream()
254                      .filter(m -> filter.test(name(), entry.name(), m.getName()))
255                      .map(m -> newMethodInfo(m, entry))
256                      .collect(toImmutableList());
257         if (methodInfos.isEmpty()) {
258             return null;
259         }
260         return new ServiceInfo(entry.name(), methodInfos);
261     }

263     @VisibleForTesting
264     static MethodInfo newMethodInfo(MethodDescriptor method, ServiceEntry service) {
265         final Set<EndpointInfo> methodEndpoints =
266                 service.endpointInfos.stream()
267                                      .map(e -> {
268                                          final EndpointInfoBuilder builder = EndpointInfo.builder(
269                                                  e.hostnamePattern(), e.pathMapping() + method.getName());
270                                          if (e.fragment() != null) {
271                                              builder.fragment(e.fragment());
272                                          }
273                                          if (e.defaultMimeType() != null) {
274                                              builder.defaultMimeType(e.defaultMimeType());
275                                          }
276                                          return builder.availableMimeTypes(e.availableMimeTypes()).build();
277                                      })
278                                      .collect(toImmutableSet());
279         return new MethodInfo(
280                 method.getName(),
281                 namedMessageSignature(method.getOutputType()),
282                 // gRPC methods always take a single request parameter of message type.
283                 ImmutableList.of(FieldInfo.builder("request", namedMessageSignature(method.getInputType()))
284                                           .requirement(FieldRequirement.REQUIRED).build()),
285                 ImmutableList.of(),
286                 methodEndpoints,
287                 ImmutableList.of(),
288                 defaultExamples(method),
289                 HttpMethod.POST,
290                 null);
291     }

293     private static List<String> defaultExamples(MethodDescriptor method) {
294         try {
295             final DynamicMessage defaultInput = DynamicMessage.getDefaultInstance(method.getInputType());
296             final String serialized = defaultExamplePrinter.print(defaultInput).trim();
297             if ("{\n}".equals(serialized) || "{}".equals(serialized)) {
298                 // Ignore an empty object.
299                 return ImmutableList.of();
300             }
301             return ImmutableList.of(serialized);
302         } catch (InvalidProtocolBufferException e) {
303             return ImmutableList.of();
304         }
305     }

316     private static FieldInfo newFieldInfo(FieldDescriptor fieldDescriptor) {
317         return FieldInfo.builder(fieldDescriptor.getName(), newFieldTypeInfo(fieldDescriptor))
318                         .requirement(fieldDescriptor.isRequired() ? FieldRequirement.REQUIRED
319                                                                   : FieldRequirement.OPTIONAL)
320                         .build();
321     }

323     @VisibleForTesting
324     static TypeSignature newFieldTypeInfo(FieldDescriptor fieldDescriptor) {
325         if (fieldDescriptor.isMapField()) {
326             return TypeSignature.ofMap(
327                     newFieldTypeInfo(fieldDescriptor.getMessageType().findFieldByNumber(1)),
328                     newFieldTypeInfo(fieldDescriptor.getMessageType().findFieldByNumber(2)));
329         }
330         final TypeSignature fieldType;
331         switch (fieldDescriptor.getType()) {
332             case BOOL:
333                 fieldType = BOOL;
334                 break;
335             case BYTES:
336                 fieldType = BYTES;
337                 break;
338             case DOUBLE:
339                 fieldType = DOUBLE;
340                 break;
341             case FIXED32:
342                 fieldType = FIXED32;
343                 break;
344             case FIXED64:
345                 fieldType = FIXED64;
346                 break;
347             case FLOAT:
348                 fieldType = FLOAT;
349                 break;
350             case INT32:
351                 fieldType = INT32;
352                 break;
353             case INT64:
354                 fieldType = INT64;
355                 break;
356             case SFIXED32:
357                 fieldType = SFIXED32;
358                 break;
359             case SFIXED64:
360                 fieldType = SFIXED64;
361                 break;
362             case SINT32:
363                 fieldType = SINT32;
364                 break;
365             case SINT64:
366                 fieldType = SINT64;
367                 break;
368             case STRING:
369                 fieldType = STRING;
370                 break;
371             case UINT32:
372                 fieldType = UINT32;
373                 break;
374             case UINT64:
375                 fieldType = UINT64;
376                 break;
377             case MESSAGE:
378                 fieldType = namedMessageSignature(fieldDescriptor.getMessageType());
379                 break;
380             case GROUP:
381                 // This type has been deprecated since the launch of protocol buffers to open source.
382                 // There is no real metadata for this in the descriptor so we just treat as UNKNOWN
383                 // since it shouldn't happen in practice anyways.
384                 fieldType = UNKNOWN;
385                 break;
386             case ENUM:
387                 fieldType = TypeSignature.ofNamed(
388                         fieldDescriptor.getEnumType().getFullName(), fieldDescriptor.getEnumType());
389                 break;
390             default:
391                 fieldType = UNKNOWN;
392                 break;
393         }
394         return fieldDescriptor.isRepeated() ? TypeSignature.ofContainer("repeated", fieldType) : fieldType;
395     }

406     private static TypeSignature namedMessageSignature(Descriptor descriptor) {
407         return TypeSignature.ofNamed(descriptor.getFullName(), descriptor);

410     @VisibleForTesting
411     static final class ServiceEntry {
412         final ServiceDescriptor service;
413         final List<EndpointInfo> endpointInfos;
414 
415         ServiceEntry(ServiceDescriptor service,
416                      List<EndpointInfo> endpointInfos) {
417             this.service = service;
418             this.endpointInfos = ImmutableList.copyOf(endpointInfos);
419         }
420 
421         String name() {
422             return service.getFullName();
423         }
424 
425         List<MethodDescriptor> methods() {
426             return ImmutableList.copyOf(service.getMethods());
427         }
428     }

430     @VisibleForTesting
431     static final class ServiceEntryBuilder {
432         private final ServiceDescriptor service;
433         private final List<EndpointInfo> endpointInfos = new ArrayList<>();
434 
435         ServiceEntryBuilder(ServiceDescriptor service) {
436             this.service = service;
437         }
438 
439         ServiceEntryBuilder endpoint(EndpointInfo endpointInfo) {
440             endpointInfos.add(requireNonNull(endpointInfo, "endpointInfo"));
441             return this;
442         }
443 
444         ServiceEntry build() {
445             return new ServiceEntry(service, endpointInfos);
446         }
