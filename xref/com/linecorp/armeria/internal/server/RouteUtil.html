17  package com.linecorp.armeria.internal.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import javax.annotation.Nullable;
22  
23  import com.linecorp.armeria.server.Route;
24  
25  /**
26   * A utility class for {@link Route}.
27   */
28  public final class RouteUtil {
29  
30      /**
31       * The prefix which represents an exact path.
32       */
33      public static final String EXACT = "exact:";
34  
35      /**
36       * The prefix which represents a prefix path.
37       */
38      public static final String PREFIX = "prefix:";
39  
40      /**
41       * The prefix which represents a glob path.
42       */
43      public static final String GLOB = "glob:";
44  
45      /**
46       * The prefix which represents a regex path.
47       */
48      public static final String REGEX = "regex:";
49  
50      public static final String UNKNOWN_LOGGER_NAME = "__UNKNOWN__";
51  
52      public static final String ROOT_LOGGER_NAME = "__ROOT__";
53  
54      /**
55       * Ensures that the specified {@code path} is an absolute path that starts with {@code "/"}.
56       *
57       * @return {@code path}
58       *
59       * @throws NullPointerException if {@code path} is {@code null}
60       * @throws IllegalArgumentException if {@code path} is not an absolute path
61       */
62      public static String ensureAbsolutePath(String path, String paramName) {
63          requireNonNull(path, paramName);
64          if (path.isEmpty() || path.charAt(0) != '/') {
65              throw new IllegalArgumentException(paramName + ": " + path +
66                                                 " (expected: an absolute path starting with '/')");
67          }
68          return path;
69      }
70  
71      /**
72       * Returns the logger name from the specified {@code pathish}.
73       */
74      public static String newLoggerName(@Nullable String pathish) {
75          if (pathish == null) {
76              return UNKNOWN_LOGGER_NAME;
77          }
78  
79          String normalized = pathish;
80          if ("/".equals(normalized)) {
81              return ROOT_LOGGER_NAME;
82          }
83  
84          if (normalized.startsWith("/")) {
85              normalized = normalized.substring(1); // Strip the first slash.
86          }
87  
88          final int end;
89          if (normalized.endsWith("/")) {
90              end = normalized.length() - 1;
91          } else {
92              end = normalized.length();
93          }
94  
95          final StringBuilder buf = new StringBuilder(end);
96          boolean start = true;
97          for (int i = 0; i < end; i++) {
98              final char ch = normalized.charAt(i);
99              if (ch != '/') {
100                 if (start) {
101                     start = false;
102                     if (Character.isJavaIdentifierStart(ch)) {
103                         buf.append(ch);
104                     } else {
105                         buf.append('_');
106                         if (Character.isJavaIdentifierPart(ch)) {
107                             buf.append(ch);
108                         }
109                     }
110                 } else {
111                     if (Character.isJavaIdentifierPart(ch)) {
112                         buf.append(ch);
113                     } else {
114                         buf.append('_');
115                     }
116                 }
117             } else {
118                 start = true;
119                 buf.append('.');
120             }
121         }
122 
123         return buf.toString();
124     }
125 
126     private RouteUtil() {}
127 }
