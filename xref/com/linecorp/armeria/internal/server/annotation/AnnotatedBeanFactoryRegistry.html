54  /**
55   * A singleton class which manages factories for creating a bean. {@link #register(Class, Set, List)} should
56   * be called at first to let {@link AnnotatedBeanFactoryRegistry} create a factory for a bean.
57   */
58  final class AnnotatedBeanFactoryRegistry {
59      private static final Logger logger = LoggerFactory.getLogger(AnnotatedBeanFactoryRegistry.class);
60  
61      // FIXME(trustin): Fix class loader leak. Should use Class as a weak key.
62      private static final ConcurrentMap<BeanFactoryId, AnnotatedBeanFactory<?>> factories =
63              new MapMaker().makeMap();
64      private static final AnnotatedBeanFactory<?> unsupportedBeanFactory;
65  
66      static {
67          final BeanFactoryId beanFactoryId = new BeanFactoryId(Object.class, ImmutableSet.of());
68          final Entry<Constructor<Object>, List<AnnotatedValueResolver>> constructor =
69                  new SimpleImmutableEntry<>(null, ImmutableList.of());
70  
71          unsupportedBeanFactory = new AnnotatedBeanFactory<>(beanFactoryId, constructor,
72                                                              ImmutableMap.of(), ImmutableMap.of());
73      }
74  
75      /**
76       * Returns a {@link BeanFactoryId} of the specified {@link Class} and {@code pathParams} for finding its
77       * factory from the factory cache later.
78       */
79      static synchronized BeanFactoryId register(Class<?> clazz, Set<String> pathParams,
80                                                 List<RequestObjectResolver> objectResolvers) {
81          final BeanFactoryId beanFactoryId = new BeanFactoryId(clazz, pathParams);
82          if (!factories.containsKey(beanFactoryId)) {
83              final AnnotatedBeanFactory<Object> factory = createFactory(beanFactoryId, objectResolvers);
84              if (factory != null) {
85                  factories.put(beanFactoryId, factory);
86                  logger.debug("Registered a bean factory: {}", beanFactoryId);
87              } else {
88                  factories.put(beanFactoryId, unsupportedBeanFactory);
89              }
90          }
91          return beanFactoryId;
92      }
93  
94      /**
95       * Returns a factory of the specified {@link BeanFactoryId}.
96       */
97      @Nullable
98      static AnnotatedBeanFactory<?> find(@Nullable BeanFactoryId beanFactoryId) {
99          if (beanFactoryId == null) {
100             return null;
101         }
102         final AnnotatedBeanFactory<?> factory = factories.get(beanFactoryId);
103         return factory != null && factory != unsupportedBeanFactory ? factory : null;
104     }

106     static Set<AnnotatedValueResolver> uniqueResolverSet() {
107         return new TreeSet<>((o1, o2) -> {
108             final String o1Name = o1.httpElementName();
109             final String o2Name = o2.httpElementName();
110             if (o1Name != null && o2Name != null && o1Name.equals(o2Name) &&
111                 o1.annotationType() == o2.annotationType()) {
112                 return 0;
113             }
114             // We are not ordering, but just finding duplicate elements.
115             return -1;
116         });
117     }
118 
119     static void warnRedundantUse(AnnotatedValueResolver resolver, String genericString) {
120         assert resolver.annotationType() != null;
121         logger.warn("Found a redundant use of annotation in {}." +
122                     " httpElementName: {}, annotation: {}", genericString,
123                     resolver.httpElementName(), resolver.annotationType().getSimpleName());
124     }

126     @Nullable
127     private static <T> AnnotatedBeanFactory<T> createFactory(BeanFactoryId beanFactoryId,
128                                                              List<RequestObjectResolver> objectResolvers) {
129         requireNonNull(beanFactoryId, "beanFactoryId");
130         requireNonNull(objectResolvers, "objectResolvers");
131 
132         final int modifiers = beanFactoryId.type.getModifiers();
133         if (Modifier.isAbstract(modifiers) || Modifier.isInterface(modifiers)) {
134             // Only concrete classes can be handled.
135             return null;
136         }
137 
138         // Support request converters which are specified for a bean class. e.g.
139         //
140         // @RequestConverter(BeanConverterA.class)
141         // @RequestConverter(BeanConverterB.class)
142         // class CompositeBean {
143         //     @RequestObject
144         //     BeanA a;
145         //     ...
146         // }
147         final List<RequestObjectResolver> resolvers = addToFirstIfExists(
148                 objectResolvers, AnnotationUtil.findDeclared(beanFactoryId.type, RequestConverter.class));
149 
150         final Entry<Constructor<T>, List<AnnotatedValueResolver>> constructor =
151                 findConstructor(beanFactoryId, resolvers);
152         if (constructor == null) {
153             // There is no constructor, so we cannot create a new instance.
154             return null;
155         }
156 
157         final List<AnnotatedValueResolver> constructorAnnotatedResolvers = constructor.getValue();
158 
159         // Find the methods whose parameters are not annotated with the same annotations in the constructor.
160         // If there're parameters used redundantly, it would warn it.
161         final Map<Method, List<AnnotatedValueResolver>> methods =
162                 findMethods(constructorAnnotatedResolvers, beanFactoryId, resolvers);
163 
164         // Find the fields which are not annotated with the same annotations in the constructor and methods.
165         // If there're parameters used redundantly, it would warn it.
166         final Map<Field, AnnotatedValueResolver> fields = findFields(constructorAnnotatedResolvers, methods,
167                                                                      beanFactoryId, resolvers);
168 
169         if (constructor.getValue().isEmpty() && methods.isEmpty() && fields.isEmpty()) {
170             // A default constructor exists but there is no annotated field or method.
171             return null;
172         }
173 
174         // Suppress Java language access checking.
175         constructor.getKey().setAccessible(true);
176         methods.keySet().forEach(method -> method.setAccessible(true));
177         fields.keySet().forEach(field -> field.setAccessible(true));
178         return new AnnotatedBeanFactory<>(beanFactoryId, constructor, methods, fields);
179     }

181     @Nullable
182     @SuppressWarnings({ "unchecked", "rawtypes" })
183     private static <T> Entry<Constructor<T>, List<AnnotatedValueResolver>> findConstructor(
184             BeanFactoryId beanFactoryId, List<RequestObjectResolver> objectResolvers) {
185 
186         Entry<Constructor<T>, List<AnnotatedValueResolver>> candidate = null;
187 
188         final Set<Constructor> constructors = getConstructors(beanFactoryId.type);
189         for (final Constructor<T> constructor : constructors) {
190             // A default constructor can be a candidate only if there has been no candidate yet.
191             if (constructor.getParameterCount() == 0 && candidate == null) {
192                 candidate = new SimpleImmutableEntry<>(constructor, ImmutableList.of());
193                 continue;
194             }
195 
196             try {
197                 final List<RequestConverter> converters =
198                         AnnotationUtil.findDeclared(constructor, RequestConverter.class);
199                 final List<AnnotatedValueResolver> resolvers =
200                         AnnotatedValueResolver.ofBeanConstructorOrMethod(
201                                 constructor, beanFactoryId.pathParams,
202                                 addToFirstIfExists(objectResolvers, converters));
203                 if (!resolvers.isEmpty()) {
204                     // Can overwrite only if the current candidate is a default constructor.
205                     if (candidate == null || candidate.getValue().isEmpty()) {
206                         candidate = new SimpleImmutableEntry<>(constructor, resolvers);
207                     } else {
208                         throw new IllegalArgumentException(
209                                 "too many annotated constructors in " + beanFactoryId.type.getSimpleName() +
210                                 " (expected: 0 or 1)");
211                     }
212                 }
213             } catch (NoAnnotatedParameterException ignored) {
214                 // There's no annotated parameters in the constructor.
215             }
216         }
217         return candidate;
218     }

220     private static Map<Method, List<AnnotatedValueResolver>> findMethods(
221             List<AnnotatedValueResolver> constructorAnnotatedResolvers,
222             BeanFactoryId beanFactoryId, List<RequestObjectResolver> objectResolvers) {
223         final Set<AnnotatedValueResolver> uniques = uniqueResolverSet();
224         uniques.addAll(constructorAnnotatedResolvers);
225 
226         final Builder<Method, List<AnnotatedValueResolver>> methodsBuilder = ImmutableMap.builder();
227         final Set<Method> methods = getAllMethods(beanFactoryId.type);
228         for (final Method method : methods) {
229             final List<RequestConverter> converters =
230                     AnnotationUtil.findDeclared(method, RequestConverter.class);
231             try {
232                 final List<AnnotatedValueResolver> resolvers =
233                         AnnotatedValueResolver.ofBeanConstructorOrMethod(
234                                 method, beanFactoryId.pathParams,
235                                 addToFirstIfExists(objectResolvers, converters));
236                 if (!resolvers.isEmpty()) {
237                     boolean redundant = false;
238                     for (AnnotatedValueResolver resolver : resolvers) {
239                         if (!uniques.add(resolver)) {
240                             redundant = true;
241                             warnRedundantUse(resolver, method.toGenericString());
242                         }
243                     }
244                     if (redundant && resolvers.size() == 1) {
245                         // Prevent redundant injection only when the size of parameter is 1.
246                         // If the method contains more than 2 parameters and if one of them is used redundantly,
247                         // we'd better to inject the method rather than ignore it.
248                         continue;
249                     }
250                     methodsBuilder.put(method, resolvers);
251                 }
252             } catch (NoAnnotatedParameterException ignored) {
253                 // There's no annotated parameters in the method.
254             }
255         }
256         return methodsBuilder.build();
257     }
258 
259     private static Map<Field, AnnotatedValueResolver> findFields(
260             List<AnnotatedValueResolver> constructorAnnotatedResolvers,
261             Map<Method, List<AnnotatedValueResolver>> methods,
262             BeanFactoryId beanFactoryId, List<RequestObjectResolver> objectResolvers) {
263         final Set<AnnotatedValueResolver> uniques = uniqueResolverSet();
264         uniques.addAll(constructorAnnotatedResolvers);
265         methods.values().forEach(uniques::addAll);
266 
267         final Builder<Field, AnnotatedValueResolver> builder = ImmutableMap.builder();
268         final Set<Field> fields = getAllFields(beanFactoryId.type);
269         for (final Field field : fields) {
270             final List<RequestConverter> converters =
271                     AnnotationUtil.findDeclared(field, RequestConverter.class);
272             final AnnotatedValueResolver resolver =
273                     AnnotatedValueResolver.ofBeanField(field, beanFactoryId.pathParams,
274                                                        addToFirstIfExists(objectResolvers, converters));
275             if (resolver != null) {
276                 if (uniques.add(resolver)) {
277                     builder.put(field, resolver);
278                 } else {
279                     warnRedundantUse(resolver, field.toGenericString());
280                 }
281             }
282         }
283         return builder.build();
284     }

288     /**
289      * An identifier of the registered bean factory.
290      */
291     static final class BeanFactoryId {
292         private final Class<?> type;
293         private final Set<String> pathParams;
294 
295         private BeanFactoryId(Class<?> type, Set<String> pathParams) {
296             this.type = type;
297             this.pathParams = ImmutableSortedSet.copyOf(pathParams);
298         }
299 
300         Class<?> type() {
301             return type;
302         }
303 
304         @Override
305         public boolean equals(@Nullable Object o) {
306             if (this == o) {
307                 return true;
308             }
309             if (o == null || o.getClass() != getClass()) {
310                 return false;
311             }
312             final BeanFactoryId that = (BeanFactoryId) o;
313             return type == that.type &&
314                    pathParams.equals(that.pathParams);
315         }
316 
317         @Override
318         public int hashCode() {
319             return Objects.hash(type, pathParams);
320         }
321 
322         @Override
323         public String toString() {
324             return MoreObjects.toStringHelper(this)
325                               .add("type", type.getName())
326                               .add("pathParams", pathParams)
327                               .toString();
328         }
329     }
