133     /**
134      * Mapping from HTTP method annotation to {@link HttpMethod}, like following.
135      * <ul>
136      *   <li>{@link Options} -> {@link HttpMethod#OPTIONS}
137      *   <li>{@link Get} -> {@link HttpMethod#GET}
138      *   <li>{@link Head} -> {@link HttpMethod#HEAD}
139      *   <li>{@link Post} -> {@link HttpMethod#POST}
140      *   <li>{@link Put} -> {@link HttpMethod#PUT}
141      *   <li>{@link Patch} -> {@link HttpMethod#PATCH}
142      *   <li>{@link Delete} -> {@link HttpMethod#DELETE}
143      *   <li>{@link Trace} -> {@link HttpMethod#TRACE}
144      * </ul>
145      */
146     private static final Map<Class<?>, HttpMethod> HTTP_METHOD_MAP =
147             ImmutableMap.<Class<?>, HttpMethod>builder()
148                     .put(Options.class, HttpMethod.OPTIONS)
149                     .put(Get.class, HttpMethod.GET)
150                     .put(Head.class, HttpMethod.HEAD)
151                     .put(Post.class, HttpMethod.POST)
152                     .put(Put.class, HttpMethod.PUT)
153                     .put(Patch.class, HttpMethod.PATCH)
154                     .put(Delete.class, HttpMethod.DELETE)
155                     .put(Trace.class, HttpMethod.TRACE)
156                     .build();

158     /**
159      * Returns the list of {@link AnnotatedService} defined by {@link Path} and HTTP method annotations
160      * from the specified {@code object}, {@link RequestConverterFunction}s, {@link ResponseConverterFunction}s,
161      * {@link ExceptionHandlerFunction}s and {@link AnnotatedServiceExtensions}.
162      */
163     public static List<AnnotatedServiceElement> find(
164             String pathPrefix, Object object,
165             List<RequestConverterFunction> requestConverterFunctions,
166             List<ResponseConverterFunction> responseConverterFunctions,
167             List<ExceptionHandlerFunction> exceptionHandlerFunctions) {
168         final List<Method> methods = requestMappingMethods(object);
169         return methods.stream()
170                       .flatMap((Method method) ->
171                                        create(pathPrefix, object, method, requestConverterFunctions,
172                                               responseConverterFunctions, exceptionHandlerFunctions).stream())
173                       .collect(toImmutableList());
174     }

176     private static HttpStatus defaultResponseStatus(Method method) {
177         final StatusCode statusCodeAnnotation = AnnotationUtil.findFirst(method, StatusCode.class);
178         if (statusCodeAnnotation == null) {
179             // Set a default HTTP status code for a response depending on the return type of the method.
180             final Class<?> returnType = method.getReturnType();
181             return returnType == Void.class ||
182                    returnType == void.class ? HttpStatus.NO_CONTENT : HttpStatus.OK;
183         }
184 
185         final int statusCode = statusCodeAnnotation.value();
186         checkArgument(statusCode >= 0,
187                       "invalid HTTP status code: %s (expected: >= 0)", statusCode);
188         return HttpStatus.valueOf(statusCode);
189     }

191     private static <T extends Annotation> void setAdditionalHeader(HttpHeadersBuilder headers,
192                                                                    AnnotatedElement element,
193                                                                    String clsAlias,
194                                                                    String elementAlias,
195                                                                    String level,
196                                                                    Class<T> annotation,
197                                                                    Function<T, String> nameGetter,
198                                                                    Function<T, String[]> valueGetter) {
199         requireNonNull(headers, "headers");
200         requireNonNull(element, "element");
201         requireNonNull(level, "level");
202 
203         final Set<String> addedHeaderSets = new HashSet<>();
204         AnnotationUtil.findAll(element, annotation).forEach(header -> {
205             final String name = nameGetter.apply(header);
206             final String[] value = valueGetter.apply(header);
207 
208             if (addedHeaderSets.contains(name)) {
209                 logger.warn("The additional {} named '{}' at '{}' is set at the same {} level already;" +
210                             "ignoring.",
211                             clsAlias, name, elementAlias, level);
212                 return;
213             }
214             headers.set(HttpHeaderNames.of(name), value);
215             addedHeaderSets.add(name);
216         });
217     }

219     /**
220      * Returns a list of {@link AnnotatedService} instances. A single {@link AnnotatedService} is
221      * created per each {@link Route} associated with the {@code method}.
222      */
223     @VisibleForTesting
224     static List<AnnotatedServiceElement> create(String pathPrefix, Object object, Method method,
225                                                 List<RequestConverterFunction> baseRequestConverters,
226                                                 List<ResponseConverterFunction> baseResponseConverters,
227                                                 List<ExceptionHandlerFunction> baseExceptionHandlers) {
228 
229         final Set<Annotation> methodAnnotations = httpMethodAnnotations(method);
230         if (methodAnnotations.isEmpty()) {
231             throw new IllegalArgumentException("HTTP Method specification is missing: " + method.getName());
232         }
233 
234         final Class<?> clazz = object.getClass();
235         final Map<HttpMethod, List<String>> httpMethodPatternsMap = getHttpMethodPatternsMap(method,
236                                                                                              methodAnnotations);
237         final String computedPathPrefix = computePathPrefix(clazz, pathPrefix);
238         final Set<MediaType> consumableMediaTypes = consumableMediaTypes(method, clazz);
239         final Set<MediaType> producibleMediaTypes = producibleMediaTypes(method, clazz);
240 
241         final List<Route> routes = httpMethodPatternsMap.entrySet().stream().flatMap(
242                 pattern -> {
243                     final HttpMethod httpMethod = pattern.getKey();
244                     final List<String> pathMappings = pattern.getValue();
245                     return pathMappings.stream().map(
246                             pathMapping -> Route.builder()
247                                                 .path(computedPathPrefix, pathMapping)
248                                                 .methods(httpMethod)
249                                                 .consumes(consumableMediaTypes)
250                                                 .produces(producibleMediaTypes)
251                                                 .matchesParams(
252                                                         predicates(method, clazz, MatchesParam.class,
253                                                                    MatchesParam::value))
254                                                 .matchesHeaders(
255                                                         predicates(method, clazz, MatchesHeader.class,
256                                                                    MatchesHeader::value))
257                                                 .build());
258                 }).collect(toImmutableList());
259 
260         final List<RequestConverterFunction> req =
261                 getAnnotatedInstances(method, clazz, RequestConverter.class, RequestConverterFunction.class)
262                         .addAll(baseRequestConverters).build();
263         final List<ResponseConverterFunction> res =
264                 getAnnotatedInstances(method, clazz, ResponseConverter.class, ResponseConverterFunction.class)
265                         .addAll(baseResponseConverters).build();
266         final List<ExceptionHandlerFunction> eh =
267                 getAnnotatedInstances(method, clazz, ExceptionHandler.class, ExceptionHandlerFunction.class)
268                         .addAll(baseExceptionHandlers).add(defaultExceptionHandler).build();
269 
270         final ResponseHeadersBuilder defaultHeaders = ResponseHeaders.builder(defaultResponseStatus(method));
271 
272         final HttpHeadersBuilder defaultTrailers = HttpHeaders.builder();
273         final String classAlias = clazz.getName();
274         final String methodAlias = String.format("%s.%s()", classAlias, method.getName());
275         setAdditionalHeader(defaultHeaders, clazz, "header", classAlias, "class", AdditionalHeader.class,
276                             AdditionalHeader::name, AdditionalHeader::value);
277         setAdditionalHeader(defaultHeaders, method, "header", methodAlias, "method", AdditionalHeader.class,
278                             AdditionalHeader::name, AdditionalHeader::value);
279         setAdditionalHeader(defaultTrailers, clazz, "trailer", classAlias, "class",
280                             AdditionalTrailer.class, AdditionalTrailer::name, AdditionalTrailer::value);
281         setAdditionalHeader(defaultTrailers, method, "trailer", methodAlias, "method",
282                             AdditionalTrailer.class, AdditionalTrailer::name, AdditionalTrailer::value);
283 
284         if (defaultHeaders.status().isContentAlwaysEmpty() && !defaultTrailers.isEmpty()) {
285             logger.warn("A response with HTTP status code '{}' cannot have a content. " +
286                         "Trailers defined at '{}' might be ignored.",
287                         defaultHeaders.status().code(), methodAlias);
288         }
289 
290         final ResponseHeaders responseHeaders = defaultHeaders.build();
291         final HttpHeaders responseTrailers = defaultTrailers.build();
292 
293         final boolean useBlockingTaskExecutor = AnnotationUtil.findFirst(method, Blocking.class) != null;
294 
295         return routes.stream().map(route -> {
296             final List<AnnotatedValueResolver> resolvers = getAnnotatedValueResolvers(req, route, method,
297                                                                                       clazz);
298             return new AnnotatedServiceElement(
299                     route,
300                     new AnnotatedService(object, method, resolvers, eh, res, route, responseHeaders,
301                                          responseTrailers, useBlockingTaskExecutor),
302                     decorator(method, clazz));
303         }).collect(toImmutableList());
304     }

306     private static List<AnnotatedValueResolver> getAnnotatedValueResolvers(List<RequestConverterFunction> req,
307                                                                            Route route, Method method,
308                                                                            Class<?> clazz) {
309         final Set<String> expectedParamNames = route.paramNames();
310         List<AnnotatedValueResolver> resolvers;
311         try {
312             resolvers = AnnotatedValueResolver.ofServiceMethod(method, expectedParamNames,
313                                                                AnnotatedValueResolver
314                                                                        .toRequestObjectResolvers(req));
315         } catch (NoParameterException ignored) {
316             // Allow no parameter like below:
317             //
318             // @Get("/")
319             // public String method1() { ... }
320             //
321             resolvers = ImmutableList.of();
322         }
323 
324         final Set<String> requiredParamNames =
325                 resolvers.stream()
326                          .filter(AnnotatedValueResolver::isPathVariable)
327                          .map(AnnotatedValueResolver::httpElementName)
328                          .collect(toImmutableSet());
329 
330         if (!expectedParamNames.containsAll(requiredParamNames)) {
331             final Set<String> missing = Sets.difference(requiredParamNames, expectedParamNames);
332             throw new IllegalArgumentException("cannot find path variables: " + missing);
333         }
334 
335         // Warn unused path variables only if there's no '@RequestObject' annotation.
336         if (resolvers.stream().noneMatch(r -> r.annotationType() == RequestObject.class) &&
337             !requiredParamNames.containsAll(expectedParamNames)) {
338             final Set<String> missing = Sets.difference(expectedParamNames, requiredParamNames);
339             logger.warn("Some path variables of the method '" + method.getName() +
340                         "' of the class '" + clazz.getName() +
341                         "' do not have their corresponding parameters annotated with @Param. " +
342                         "They would not be automatically injected: " + missing);
343         }
344 
345         return resolvers;
346     }

348     /**
349      * Returns the list of {@link Path} annotated methods.
350      */
351     private static List<Method> requestMappingMethods(Object object) {
352         return getAllMethods(object.getClass(), withModifier(Modifier.PUBLIC))
353                 .stream()
354                 // Lookup super classes just in case if the object is a proxy.
355                 .filter(m -> AnnotationUtil.getAnnotations(m, FindOption.LOOKUP_SUPER_CLASSES)
356                                            .stream()
357                                            .map(Annotation::annotationType)
358                                            .anyMatch(a -> a == Path.class ||
359                                        HTTP_METHOD_MAP.containsKey(a)))
360                 .sorted(Comparator.comparingInt(AnnotatedServiceFactory::order))
361                 .collect(toImmutableList());
362     }

393     /**
394      * Returns the set of {@link MediaType}s specified by {@link Consumes} annotation.
395      */
396     private static Set<MediaType> consumableMediaTypes(Method method, Class<?> clazz) {
397         List<Consumes> consumes = AnnotationUtil.findAll(method, Consumes.class);
398         List<ConsumeType> consumeTypes = AnnotationUtil.findAll(method, ConsumeType.class);
399 
400         if (consumes.isEmpty() && consumeTypes.isEmpty()) {
401             consumes = AnnotationUtil.findAll(clazz, Consumes.class);
402             consumeTypes = AnnotationUtil.findAll(clazz, ConsumeType.class);
403         }
404 
405         final List<MediaType> types =
406                 Stream.concat(consumes.stream().map(Consumes::value),
407                               consumeTypes.stream().map(ConsumeType::value))
408                       .map(MediaType::parse)
409                       .collect(toImmutableList());
410         return listToSet(types, Consumes.class);
411     }

413     /**
414      * Returns the list of {@link MediaType}s specified by {@link Produces} annotation.
415      */
416     private static Set<MediaType> producibleMediaTypes(Method method, Class<?> clazz) {
417         List<Produces> produces = AnnotationUtil.findAll(method, Produces.class);
418         List<ProduceType> produceTypes = AnnotationUtil.findAll(method, ProduceType.class);
419 
420         if (produces.isEmpty() && produceTypes.isEmpty()) {
421             produces = AnnotationUtil.findAll(clazz, Produces.class);
422             produceTypes = AnnotationUtil.findAll(clazz, ProduceType.class);
423         }
424 
425         final List<MediaType> types =
426                 Stream.concat(produces.stream().map(Produces::value),
427                               produceTypes.stream().map(ProduceType::value))
428                       .map(MediaType::parse)
429                       .peek(type -> {
430                           if (type.hasWildcard()) {
431                               throw new IllegalArgumentException(
432                                       "Producible media types must not have a wildcard: " + type);
433                           }
434                       })
435                       .collect(toImmutableList());
436         return listToSet(types, Produces.class);
437     }

439     /**
440      * Returns a list of predicates which will be used to evaluate whether a request can be accepted
441      * by a service method.
442      */
443     private static <T extends Annotation> List<String> predicates(Method method, Class<?> clazz,
444                                                                   Class<T> annotationType,
445                                                                   Function<T, String> toStringPredicate) {
446         final List<T> classLevel = AnnotationUtil.findAll(clazz, annotationType);
447         final List<T> methodLevel = AnnotationUtil.findAll(method, annotationType);
448         return Streams.concat(classLevel.stream(), methodLevel.stream())
449                       .map(toStringPredicate).collect(toImmutableList());

452     /**
453      * Converts the list of {@link MediaType}s to a set. It raises an {@link IllegalArgumentException} if the
454      * list has duplicate elements.
455      */
456     private static Set<MediaType> listToSet(List<MediaType> types, Class<?> annotationClass) {
457         final Set<MediaType> set = new LinkedHashSet<>();
458         for (final MediaType type : types) {
459             if (!set.add(type)) {
460                 throw new IllegalArgumentException(
461                         "Duplicated media type for @" + annotationClass.getSimpleName() + ": " + type);
462             }
463         }
464         return ImmutableSet.copyOf(set);
465     }

467     /**
468      * Returns path patterns for each {@link HttpMethod}. The path pattern might be specified by
469      * {@link Path} or HTTP method annotations such as {@link Get} and {@link Post}. Path patterns
470      * may be specified by either HTTP method annotations, or {@link Path} annotations but not both
471      * simultaneously.
472      */
473     private static Map<HttpMethod, List<String>> getHttpMethodPatternsMap(Method method,
474                                                                           Set<Annotation> methodAnnotations) {
475         final List<String> pathPatterns = AnnotationUtil.findAll(method, Path.class).stream().map(Path::value)
476                                                         .collect(toImmutableList());
477         final boolean usePathPatterns = !pathPatterns.isEmpty();
478 
479         final Map<HttpMethod, List<String>> httpMethodAnnotatedPatternMap =
480                 getHttpMethodAnnotatedPatternMap(methodAnnotations);
481         if (httpMethodAnnotatedPatternMap.isEmpty()) {
482             throw new IllegalArgumentException(method.getDeclaringClass().getName() + '#' + method.getName() +
483                                                " must have an HTTP method annotation.");
484         }
485         return httpMethodAnnotatedPatternMap.entrySet().stream().collect(
486                 ImmutableMap.toImmutableMap(
487                         Entry::getKey,
488                         entry -> {
489                             final List<String> httpMethodPaths = entry.getValue();
490                             if (usePathPatterns && !httpMethodPaths.isEmpty()) {
491                                 throw new IllegalArgumentException(
492                                         method.getDeclaringClass().getName() + '#' + method.getName() +
493                                         " cannot specify both an HTTP mapping and a Path mapping.");
494                             }
495                             if (usePathPatterns) {
496                                 httpMethodPaths.addAll(pathPatterns);
497                             }
498                             if (httpMethodPaths.isEmpty()) {
499                                 throw new IllegalArgumentException("A path pattern should be specified by" +
500                                                                    " @Path or HTTP method annotations.");
501                             }
502                             return ImmutableList.copyOf(httpMethodPaths);
503                         }));
504     }
505 
506     private static Map<HttpMethod, List<String>> getHttpMethodAnnotatedPatternMap(
507             Set<Annotation> methodAnnotations) {
508         final Map<HttpMethod, List<String>> httpMethodPatternMap = new EnumMap<>(HttpMethod.class);
509         methodAnnotations.stream()
510                          .filter(annotation -> HTTP_METHOD_MAP.containsKey(annotation.annotationType()))
511                          .forEach(annotation -> {
512                              final HttpMethod httpMethod = HTTP_METHOD_MAP.get(annotation.annotationType());
513                              final String value = (String) invokeValueMethod(annotation);
514                              final List<String> patterns = httpMethodPatternMap
515                                      .computeIfAbsent(httpMethod, ignored -> new ArrayList<>());
516                              if (DefaultValues.isSpecified(value)) {
517                                  patterns.add(value);
518                              }
519                          });
520         return httpMethodPatternMap;
521     }

523     /**
524      * Returns a decorator chain which is specified by {@link Decorator} annotations and user-defined
525      * decorator annotations.
526      */
527     private static Function<? super HttpService, ? extends HttpService> decorator(
528             Method method, Class<?> clazz) {
529 
530         final List<DecoratorAndOrder> decorators = collectDecorators(clazz, method);
531 
532         Function<? super HttpService, ? extends HttpService> decorator = Function.identity();
533         for (int i = decorators.size() - 1; i >= 0; i--) {
534             final DecoratorAndOrder d = decorators.get(i);
535             decorator = decorator.andThen(d.decorator());
536         }
537         return decorator;
538     }

540     /**
541      * Returns a decorator list which is specified by {@link Decorator} annotations and user-defined
542      * decorator annotations.
543      */
544     @VisibleForTesting
545     static List<DecoratorAndOrder> collectDecorators(Class<?> clazz, Method method) {
546         final List<DecoratorAndOrder> decorators = new ArrayList<>();
547 
548         // Class-level decorators are applied before method-level decorators.
549         collectDecorators(decorators, AnnotationUtil.getAllAnnotations(clazz));
550         collectDecorators(decorators, AnnotationUtil.getAllAnnotations(method));
551 
552         // Sort decorators by "order" attribute values.
553         decorators.sort(Comparator.comparing(DecoratorAndOrder::order));
554 
555         return decorators;
556     }

558     /**
559      * Adds decorators to the specified {@code list}. Decorators which are annotated with {@link Decorator}
560      * and user-defined decorators will be collected.
561      */
562     private static void collectDecorators(List<DecoratorAndOrder> list, List<Annotation> annotations) {
563         if (annotations.isEmpty()) {
564             return;
565         }
566 
567         // Respect the order of decorators which is specified by a user. The first one is first applied
568         // for most of the cases. But if @Decorator and user-defined decorators are specified in a mixed order,
569         // the specified order and the applied order can be different. To overcome this problem, we introduce
570         // "order" attribute to @Decorator annotation to sort decorators. If a user-defined decorator
571         // annotation has "order" attribute, it will be also used for sorting.
572         for (final Annotation annotation : annotations) {
573             if (annotation instanceof Decorator) {
574                 final Decorator d = (Decorator) annotation;
575                 list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
576                 continue;
577             }
578 
579             if (annotation instanceof Decorators) {
580                 final Decorator[] decorators = ((Decorators) annotation).value();
581                 for (final Decorator d : decorators) {
582                     list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
583                 }
584                 continue;
585             }
586 
587             DecoratorAndOrder udd = userDefinedDecorator(annotation);
588             if (udd != null) {
589                 list.add(udd);
590                 continue;
591             }
592 
593             // If user-defined decorators are repeatable and they are specified more than once.
594             try {
595                 final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
596                                                                     withName("value")), null);
597                 assert method != null : "No 'value' method is found from " + annotation;
598                 final Annotation[] decorators = (Annotation[]) method.invoke(annotation);
599                 for (final Annotation decorator : decorators) {
600                     udd = userDefinedDecorator(decorator);
601                     if (udd == null) {
602                         break;
603                     }
604                     list.add(udd);
605                 }
606             } catch (Throwable ignore) {
607                 // The annotation may be a container of a decorator or may be not, so we just ignore
608                 // any exception from this clause.
609             }
610         }
611     }
612 
613     /**
614      * Returns a decorator with its order if the specified {@code annotation} is one of the user-defined
615      * decorator annotation.
616      */
617     @Nullable
618     private static DecoratorAndOrder userDefinedDecorator(Annotation annotation) {
619         // User-defined decorator MUST be annotated with @DecoratorFactory annotation.
620         final DecoratorFactory d = AnnotationUtil.findFirstDeclared(annotation.annotationType(),
621                                                                     DecoratorFactory.class);
622         if (d == null) {
623             return null;
624         }
625 
626         // In case of user-defined decorator, we need to create a new decorator from its factory.
627         @SuppressWarnings("unchecked")
628         final DecoratorFactoryFunction<Annotation> factory = getInstance(d, DecoratorFactoryFunction.class);
629 
630         // If the annotation has "order" attribute, we can use it when sorting decorators.
631         int order = 0;
632         try {
633             final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
634                                                                 withName("order")), null);
635             if (method != null) {
636                 final Object value = method.invoke(annotation);
637                 if (value instanceof Integer) {
638                     order = (Integer) value;
639                 }
640             }
641         } catch (Throwable ignore) {
642             // A user-defined decorator may not have an 'order' attribute.
643             // If it does not exist, '0' is used by default.
644         }
645         return new DecoratorAndOrder(annotation, factory.newDecorator(annotation), order);
646     }

656     /**
657      * Returns a {@link Builder} which has the instances specified by the annotations of the
658      * {@code annotationType}. The annotations of the specified {@code method} and {@code clazz} will be
659      * collected respectively.
660      */
661     private static <T extends Annotation, R> Builder<R> getAnnotatedInstances(
662             AnnotatedElement method, AnnotatedElement clazz, Class<T> annotationType, Class<R> resultType) {
663         final Builder<R> builder = new Builder<>();
664         Stream.concat(AnnotationUtil.findAll(method, annotationType).stream(),
665                       AnnotationUtil.findAll(clazz, annotationType).stream())
666               .forEach(annotation -> builder.add(getInstance(annotation, resultType)));
667         return builder;
668     }

670     /**
671      * Returns a cached instance of the specified {@link Class} which is specified in the given
672      * {@link Annotation}.
673      */
674     static <T> T getInstance(Annotation annotation, Class<T> expectedType) {
675         try {
676             @SuppressWarnings("unchecked")
677             final Class<? extends T> clazz = (Class<? extends T>) invokeValueMethod(annotation);
678             return expectedType.cast(instanceCache.computeIfAbsent(clazz, type -> {
679                 try {
680                     return getInstance0(clazz);
681                 } catch (Exception e) {
682                     throw new IllegalStateException(
683                             "A class specified in @" + annotation.getClass().getSimpleName() +
684                             " annotation must have an accessible default constructor: " + clazz.getName(), e);
685                 }
686             }));
687         } catch (ClassCastException e) {
688             throw new IllegalArgumentException(
689                     "A class specified in @" + annotation.getClass().getSimpleName() +
690                     " annotation cannot be cast to " + expectedType, e);
691         }
692     }

694     /**
695      * Returns a cached instance of the specified {@link Class}.
696      */
697     static <T> T getInstance(Class<T> clazz) {
698         @SuppressWarnings("unchecked")
699         final T casted = (T) instanceCache.computeIfAbsent(clazz, type -> {
700             try {
701                 return getInstance0(clazz);
702             } catch (Exception e) {
703                 throw new IllegalStateException("A class must have an accessible default constructor: " +
704                                                 clazz.getName(), e);
705             }
706         });
707         return casted;
708     }
709 
710     private static <T> T getInstance0(Class<? extends T> clazz) throws Exception {
711         @SuppressWarnings("unchecked")
712         final Constructor<? extends T> constructor =
713                 Iterables.getFirst(getConstructors(clazz, withParametersCount(0)), null);
714         assert constructor != null : "No default constructor is found from " + clazz.getName();
715         constructor.setAccessible(true);
716         return constructor.newInstance();
717     }

719     /**
720      * Returns an object which is returned by {@code value()} method of the specified annotation {@code a}.
721      */
722     private static Object invokeValueMethod(Annotation a) {
723         try {
724             final Method method = Iterables.getFirst(getMethods(a.getClass(), withName("value")), null);
725             assert method != null : "No 'value' method is found from " + a;
726             return method.invoke(a);
727         } catch (Exception e) {
728             throw new IllegalStateException("An annotation @" + a.getClass().getSimpleName() +
729                                             " must have a 'value' method", e);
730         }
731     }

733     /**
734      * Returns the description of the specified {@link AnnotatedElement}.
735      */
736     @Nullable
737     static String findDescription(AnnotatedElement annotatedElement) {
738         requireNonNull(annotatedElement, "annotatedElement");
739         final Description description = AnnotationUtil.findFirst(annotatedElement, Description.class);
740         if (description != null) {
741             final String value = description.value();
742             if (DefaultValues.isSpecified(value)) {
743                 checkArgument(!value.isEmpty(), "value is empty");
744                 return value;
745             }
746         }
747         return null;
748     }

750     /**
751      * Returns the path prefix to use. If there is {@link PathPrefix} annotation on the class
752      * then path prefix is computed by concatenating pathPrefix and value from annotation else
753      * returns pathPrefix.
754      */
755     private static String computePathPrefix(Class<?> clazz, String pathPrefix) {
756         ensureAbsolutePath(pathPrefix, "pathPrefix");
757         final PathPrefix pathPrefixAnnotation = AnnotationUtil.findFirst(clazz, PathPrefix.class);
758         if (pathPrefixAnnotation == null) {
759             return pathPrefix;
760         }
761         final String pathPrefixFromAnnotation = pathPrefixAnnotation.value();
762         ensureAbsolutePath(pathPrefixFromAnnotation, "pathPrefixFromAnnotation");
763         return concatPaths(pathPrefix, pathPrefixFromAnnotation);
764     }
