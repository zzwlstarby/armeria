119     @Override
120     public ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs,
121                                                       DocServiceFilter filter) {
122         requireNonNull(serviceConfigs, "serviceConfigs");
123         requireNonNull(filter, "filter");
124 
125         final Map<Class<?>, Set<MethodInfo>> methodInfos = new HashMap<>();
126         final Map<Class<?>, String> serviceDescription = new HashMap<>();
127         serviceConfigs.forEach(sc -> {
128             final AnnotatedService service = sc.service().as(AnnotatedService.class);
129             if (service != null) {
130                 final String className = service.object().getClass().getName();
131                 final String methodName = service.method().getName();
132                 if (!filter.test(name(), className, methodName)) {
133                     return;
134                 }
135                 addMethodInfo(methodInfos, sc.virtualHost().hostnamePattern(), service);
136                 addServiceDescription(serviceDescription, service);
137             }
138         });
139 
140         return generate(serviceDescription, methodInfos);
141     }

143     private static void addServiceDescription(Map<Class<?>, String> serviceDescription,
144                                               AnnotatedService service) {
145         final Class<?> clazz = service.object().getClass();
146         serviceDescription.computeIfAbsent(clazz, AnnotatedServiceFactory::findDescription);
147     }

149     private static void addMethodInfo(Map<Class<?>, Set<MethodInfo>> methodInfos,
150                                       String hostnamePattern, AnnotatedService service) {
151         final Route route = service.route();
152         final EndpointInfo endpoint = endpointInfo(route, hostnamePattern);
153         final Method method = service.method();
154         final String name = method.getName();
155         final TypeSignature returnTypeSignature = toTypeSignature(method.getGenericReturnType());
156         final List<FieldInfo> fieldInfos = fieldInfos(service.annotatedValueResolvers());
157         final Class<?> clazz = service.object().getClass();
158         route.methods().forEach(
159                 httpMethod -> {
160                     final MethodInfo methodInfo = new MethodInfo(
161                             name, returnTypeSignature, fieldInfos, ImmutableList.of(), // Ignore exceptions.
162                             ImmutableList.of(endpoint), httpMethod, AnnotatedServiceFactory
163                                     .findDescription(method));
164                     methodInfos.computeIfAbsent(clazz, unused -> new HashSet<>()).add(methodInfo);
165                 });
166     }

168     @VisibleForTesting
169     static EndpointInfo endpointInfo(Route route, String hostnamePattern) {
170         final EndpointInfoBuilder builder;
171         final RoutePathType pathType = route.pathType();
172         final List<String> paths = route.paths();
173         switch (pathType) {
174             case EXACT:
175                 builder = EndpointInfo.builder(hostnamePattern, RouteUtil.EXACT + paths.get(0));
176                 break;
177             case PREFIX:
178                 builder = EndpointInfo.builder(hostnamePattern, RouteUtil.PREFIX + paths.get(0));
179                 break;
180             case PARAMETERIZED:
181                 builder = EndpointInfo.builder(hostnamePattern, normalizeParameterized(route));
182                 break;
183             case REGEX:
184                 builder = EndpointInfo.builder(hostnamePattern, RouteUtil.REGEX + paths.get(0));
185                 break;
186             case REGEX_WITH_PREFIX:
187                 builder = EndpointInfo.builder(hostnamePattern, RouteUtil.REGEX + paths.get(0));
188                 builder.regexPathPrefix(RouteUtil.PREFIX + paths.get(1));
189                 break;
190             default:
191                 // Should never reach here.
192                 throw new Error();
193         }
194 
195         builder.availableMimeTypes(availableMimeTypes(route));
196         return builder.build();
197     }

199     private static String normalizeParameterized(Route route) {
200         final String path = route.paths().get(0);
201         int beginIndex = 0;
202 
203         final StringBuilder sb = new StringBuilder();
204         for (String paramName : route.paramNames()) {
205             final int colonIndex = path.indexOf(':', beginIndex);
206             assert colonIndex != -1;
207             sb.append(path, beginIndex, colonIndex);
208             sb.append('{');
209             sb.append(paramName);
210             sb.append('}');
211             beginIndex = colonIndex + 1;
212         }
213         if (beginIndex < path.length()) {
214             sb.append(path, beginIndex, path.length());
215         }
216         return sb.toString();
217     }

219     private static Set<MediaType> availableMimeTypes(Route route) {
220         final ImmutableSet.Builder<MediaType> builder = ImmutableSet.builder();
221         final Set<MediaType> consumeTypes = route.consumes();
222         builder.addAll(consumeTypes);
223         if (!consumeTypes.contains(MediaType.JSON_UTF_8)) {
224             builder.add(MediaType.JSON_UTF_8);
225         }
226         return builder.build();
227     }

229     private static List<FieldInfo> fieldInfos(List<AnnotatedValueResolver> resolvers) {
230         final ImmutableList.Builder<FieldInfo> fieldInfosBuilder = ImmutableList.builder();
231         for (AnnotatedValueResolver resolver : resolvers) {
232             final FieldInfo fieldInfo = fieldInfo(resolver);
233             if (fieldInfo != null) {
234                 fieldInfosBuilder.add(fieldInfo);
235             }
236         }
237         return fieldInfosBuilder.build();
238     }

240     @Nullable
241     private static FieldInfo fieldInfo(AnnotatedValueResolver resolver) {
242         final Class<? extends Annotation> annotationType = resolver.annotationType();
243         if (annotationType == RequestObject.class) {
244             final BeanFactoryId beanFactoryId = resolver.beanFactoryId();
245             final AnnotatedBeanFactory<?> factory = AnnotatedBeanFactoryRegistry.find(beanFactoryId);
246             if (factory != null) {
247                 final Builder<AnnotatedValueResolver> builder = ImmutableList.builder();
248                 factory.constructor().getValue().forEach(builder::add);
249                 factory.methods().values().forEach(resolvers -> resolvers.forEach(builder::add));
250                 factory.fields().values().forEach(builder::add);
251                 final List<AnnotatedValueResolver> resolvers = builder.build();
252                 if (!resolvers.isEmpty()) {
253                     // Just use the simple name of the bean class as the field name.
254                     return FieldInfo.builder(beanFactoryId.type().getSimpleName(), BEAN,
255                                              fieldInfos(resolvers)).build();
256                 }
257             }
258             return null;
259         }
260 
261         if (annotationType != Param.class && annotationType != Header.class) {
262             return null;
263         }
264         final TypeSignature signature;
265         if (resolver.hasContainer()) {
266             final Class<?> containerType = resolver.containerType();
267             assert containerType != null;
268             final TypeSignature parameterTypeSignature = toTypeSignature(resolver.elementType());
269             if (List.class.isAssignableFrom(containerType)) {
270                 signature = TypeSignature.ofList(parameterTypeSignature);
271             } else if (Set.class.isAssignableFrom(containerType)) {
272                 signature = TypeSignature.ofSet(parameterTypeSignature);
273             } else {
274                 // Only List and Set are supported for the containerType.
275                 return null;
276             }
277         } else {
278             signature = toTypeSignature(resolver.elementType());
279         }
280         final String name = resolver.httpElementName();
281         assert name != null;
282 
283         final FieldInfoBuilder builder =
284                 FieldInfo.builder(name, signature)
285                          .location(location(resolver))
286                          .requirement(resolver.shouldExist() ? FieldRequirement.REQUIRED
287                                                              : FieldRequirement.OPTIONAL);
288         if (resolver.description() != null) {
289             builder.docString(resolver.description());
290         }
291         return builder.build();
292     }

294     @VisibleForTesting
295     static TypeSignature toTypeSignature(Type type) {
296         requireNonNull(type, "type");
297 
298         // The data types defined by the OpenAPI Specification:
299 
300         if (type == Void.class || type == void.class) {
301             return VOID;
302         }
303         if (type == Boolean.class || type == boolean.class) {
304             return BOOLEAN;
305         }
306         if (type == Byte.class || type == byte.class) {
307             return BYTE;
308         }
309         if (type == Short.class || type == short.class) {
310             return SHORT;
311         }
312         if (type == Integer.class || type == int.class) {
313             return INT;
314         }
315         if (type == Long.class || type == long.class) {
316             return LONG;
317         }
318         if (type == Float.class || type == float.class) {
319             return FLOAT;
320         }
321         if (type == Double.class || type == double.class) {
322             return DOUBLE;
323         }
324         if (type == String.class) {
325             return STRING;
326         }
327         if (type == byte[].class || type == Byte[].class) {
328             return BINARY;
329         }
330         // End of data types defined by the OpenAPI Specification.
331 
332         if (type instanceof ParameterizedType) {
333             final ParameterizedType parameterizedType = (ParameterizedType) type;
334             final Class<?> rawType = (Class<?>) parameterizedType.getRawType();
335             if (List.class.isAssignableFrom(rawType)) {
336                 return TypeSignature.ofList(toTypeSignature(parameterizedType.getActualTypeArguments()[0]));
337             }
338             if (Set.class.isAssignableFrom(rawType)) {
339                 return TypeSignature.ofSet(toTypeSignature(parameterizedType.getActualTypeArguments()[0]));
340             }
341 
342             if (Map.class.isAssignableFrom(rawType)) {
343                 final TypeSignature key = toTypeSignature(parameterizedType.getActualTypeArguments()[0]);
344                 final TypeSignature value = toTypeSignature(parameterizedType.getActualTypeArguments()[1]);
345                 return TypeSignature.ofMap(key, value);
346             }
347 
348             final List<TypeSignature> actualTypes = Stream.of(parameterizedType.getActualTypeArguments())
349                                                           .map(AnnotatedDocServicePlugin::toTypeSignature)
350                                                           .collect(toImmutableList());
351             return TypeSignature.ofContainer(rawType.getSimpleName(), actualTypes);
352         }
353 
354         if (type instanceof WildcardType) {
355             // Create an unresolved type with an empty string so that the type name will be '?'.
356             return TypeSignature.ofUnresolved("");
357         }
358         if (type instanceof TypeVariable) {
359             return TypeSignature.ofBase(type.getTypeName());
360         }
361         if (type instanceof GenericArrayType) {
362             return TypeSignature.ofList(toTypeSignature(((GenericArrayType) type).getGenericComponentType()));
363         }
364 
365         if (!(type instanceof Class)) {
366             return TypeSignature.ofBase(type.getTypeName());
367         }
368 
369         final Class<?> clazz = (Class<?>) type;
370         if (clazz.isArray()) {
371             // If it's an array, return it as a list.
372             return TypeSignature.ofList(toTypeSignature(clazz.getComponentType()));
373         }
374 
375         return TypeSignature.ofBase(clazz.getSimpleName());
376     }

378     private static FieldLocation location(AnnotatedValueResolver resolver) {
379         if (resolver.isPathVariable()) {
380             return PATH;
381         }
382         if (resolver.annotationType() == Param.class) {
383             return QUERY;
384         }
385         if (resolver.annotationType() == Header.class) {
386             return HEADER;
387         }
388         return UNSPECIFIED;
389     }

391     @VisibleForTesting
392     static ServiceSpecification generate(Map<Class<?>, String> serviceDescription,
393                                          Map<Class<?>, Set<MethodInfo>> methodInfos) {
394         final Set<ServiceInfo> serviceInfos = methodInfos
395                 .entrySet().stream()
396                 .map(entry -> {
397                     final Class<?> service = entry.getKey();
398                     return new ServiceInfo(service.getName(), entry.getValue(),
399                                            serviceDescription.get(service));
400                 })
401                 .collect(toImmutableSet());
402 
403         return ServiceSpecification.generate(serviceInfos, AnnotatedDocServicePlugin::newNamedTypeInfo);
404     }

406     private static NamedTypeInfo newNamedTypeInfo(TypeSignature typeSignature) {
407         final Class<?> type = (Class<?>) typeSignature.namedTypeDescriptor();
408         if (type == null) {
409             throw new IllegalArgumentException("cannot create a named type from: " + typeSignature);
410         }
411 
412         if (type.isEnum()) {
413             @SuppressWarnings("unchecked")
414             final Class<? extends Enum<?>> enumType = (Class<? extends Enum<?>>) type;
415             return new EnumInfo(enumType);
416         }
417 
418         return newStructInfo(type);
419     }
420 
421     private static StructInfo newStructInfo(Class<?> structClass) {
422         final String name = structClass.getName();
423 
424         final Field[] declaredFields = structClass.getDeclaredFields();
425         final List<FieldInfo> fields =
426                 Stream.of(declaredFields)
427                       .map(f -> FieldInfo.of(f.getName(), toTypeSignature(f.getGenericType())))
428                       .collect(Collectors.toList());
429         return new StructInfo(name, fields);
430     }

439     public String serializeExampleRequest(String serviceName, String methodName,
440                                           Object exampleRequest) {
441         try {
442             return mapper.writerWithDefaultPrettyPrinter().writeValueAsString(exampleRequest);

444             // Ignore the exception and just return Optional.empty().
445         }
