78  /**
79   * Encapsulates the state of a single client call, writing messages from the client and reading responses
80   * from the server, passing to business logic via {@link ClientCall.Listener}.
81   */
82  final class ArmeriaClientCall<I, O> extends ClientCall<I, O>
83          implements ArmeriaMessageDeframer.Listener, TransportStatusListener {
84  
85      private static final Runnable NO_OP = () -> {
86      };

122     ArmeriaClientCall(
123             DefaultClientRequestContext ctx,
124             EndpointGroup endpointGroup,
125             HttpClient httpClient,
126             HttpRequestWriter req,
127             MethodDescriptor<I, O> method,
128             int maxOutboundMessageSizeBytes,
129             int maxInboundMessageSizeBytes,
130             CallOptions callOptions,
131             CompressorRegistry compressorRegistry,
132             DecompressorRegistry decompressorRegistry,
133             SerializationFormat serializationFormat,
134             @Nullable MessageMarshaller jsonMarshaller,
135             boolean unsafeWrapResponseBuffers,
136             String advertisedEncodingsHeader) {
137         this.ctx = ctx;
138         this.endpointGroup = endpointGroup;
139         this.httpClient = httpClient;
140         this.req = req;
141         this.method = method;
142         this.callOptions = callOptions;
143         this.compressorRegistry = compressorRegistry;
144         this.unsafeWrapResponseBuffers = unsafeWrapResponseBuffers;
145         this.advertisedEncodingsHeader = advertisedEncodingsHeader;
146         this.serializationFormat = serializationFormat;
147         messageFramer = new ArmeriaMessageFramer(ctx.alloc(), maxOutboundMessageSizeBytes);
148         marshaller = new GrpcMessageMarshaller<>(
149                 ctx.alloc(), serializationFormat, method, jsonMarshaller,
150                 unsafeWrapResponseBuffers);
151         responseReader = new HttpStreamReader(
152                 decompressorRegistry,
153                 new ArmeriaMessageDeframer(this, maxInboundMessageSizeBytes, ctx.alloc()),
154                 this);
155         executor = callOptions.getExecutor();
156 
157         req.whenComplete().handle((unused1, unused2) -> {
158             if (!ctx.log().isAvailable(RequestLogProperty.REQUEST_CONTENT)) {
159                 // Can reach here if the request stream was empty.
160                 ctx.logBuilder().requestContent(GrpcLogUtil.rpcRequest(method), null);
161             }
162             return null;
163         });
164     }

166     @Override
167     public void start(Listener<O> responseListener, Metadata metadata) {
168         requireNonNull(responseListener, "responseListener");
169         requireNonNull(metadata, "metadata");
170         final Compressor compressor;
171         if (callOptions.getCompressor() != null) {
172             compressor = compressorRegistry.lookupCompressor(callOptions.getCompressor());
173             if (compressor == null) {
174                 responseListener.onClose(
175                         Status.INTERNAL.withDescription(
176                                 "Unable to find compressor by name " + callOptions.getCompressor()),
177                         new Metadata());
178                 return;
179             }
180         } else {
181             compressor = Identity.NONE;
182         }
183         messageFramer.setCompressor(ForwardingCompressor.forGrpc(compressor));
184         prepareHeaders(compressor, metadata);
185         listener = responseListener;
186 
187         if (callOptions.getDeadline() != null) {
188             final long remainingMillis = callOptions.getDeadline().timeRemaining(TimeUnit.MILLISECONDS);
189             if (remainingMillis <= 0) {
190                 final Status status = Status.DEADLINE_EXCEEDED
191                         .augmentDescription(
192                                 "ClientCall started after deadline exceeded: " +
193                                 callOptions.getDeadline());
194                 close(status, new Metadata());
195             } else {
196                 ctx.setResponseTimeoutAfterMillis(remainingMillis);
197                 ctx.setResponseTimeoutHandler(() -> {
198                     final Status status = Status.DEADLINE_EXCEEDED
199                             .augmentDescription(
200                                     "deadline exceeded after " +
201                                     TimeUnit.MILLISECONDS.toNanos(remainingMillis) + "ns.");
202                     close(status, new Metadata());
203                 });
204             }
205         }
206 
207         final HttpResponse res = initContextAndExecuteWithFallback(
208                 httpClient, ctx, endpointGroup,
209                 (unused, cause) -> HttpResponse.ofFailure(GrpcStatus.fromThrowable(cause)
210                                                                     .withDescription(cause.getMessage())
211                                                                     .asRuntimeException()));
212 
213         res.subscribe(responseReader, ctx.eventLoop(), WITH_POOLED_OBJECTS);
214         res.whenComplete().handleAsync(responseReader, ctx.eventLoop());
215         responseListener.onReady();
216     }

218     @Override
219     public void request(int numMessages) {
220         if (ctx.eventLoop().inEventLoop()) {
221             responseReader.request(numMessages);
222         } else {
223             ctx.eventLoop().submit(() -> responseReader.request(numMessages));
224         }
225     }

227     @Override
228     public void cancel(@Nullable String message, @Nullable Throwable cause) {
229         if (ctx.eventLoop().inEventLoop()) {
230             doCancel(message, cause);
231         } else {
232             ctx.eventLoop().submit(() -> doCancel(message, cause));
233         }
234     }
235 
236     private void doCancel(@Nullable String message, @Nullable Throwable cause) {
237         if (message == null && cause == null) {
238             cause = new CancellationException("Cancelled without a message or cause");
239             logger.warn("Cancelling without a message or cause is suboptimal", cause);
240         }
241         if (cancelCalled) {
242             return;
243         }
244         cancelCalled = true;
245         Status status = Status.CANCELLED;
246         if (message != null) {
247             status = status.withDescription(message);
248         }
249         if (cause != null) {
250             status = status.withCause(cause);
251         }
252         close(status, new Metadata());
253         if (cause == null) {
254             req.abort();
255         } else {
256             req.abort(cause);
257         }
258     }
259 
260     @Override
261     public void halfClose() {
262         if (ctx.eventLoop().inEventLoop()) {
263             req.close();
264         } else {
265             ctx.eventLoop().submit((Runnable) req::close);
266         }
267     }

269     @Override
270     public void sendMessage(I message) {
271         pendingMessagesUpdater.incrementAndGet(this);
272         if (ctx.eventLoop().inEventLoop()) {
273             doSendMessage(message);
274         } else {
275             ctx.eventLoop().submit(() -> doSendMessage(message));
276         }
277     }
278 
279     @Override
280     public boolean isReady() {
281         return pendingMessages == 0;
282     }
283 
284     private void doSendMessage(I message) {
285         final RequestLogAccess log = ctx.log();
286         if (log.isComplete()) {
287             // Completed already; no need to send anymore.
288             return;
289         }
290 
291         try {
292             if (!log.isAvailable(RequestLogProperty.REQUEST_CONTENT)) {
293                 ctx.logBuilder().requestContent(GrpcLogUtil.rpcRequest(method, message), null);
294             }
295             final ByteBuf serialized = marshaller.serializeRequest(message);
296             req.write(messageFramer.writePayload(serialized));
297             req.whenConsumed().thenRun(() -> {
298                 if (pendingMessagesUpdater.decrementAndGet(this) == 0) {
299                     try (SafeCloseable ignored = ctx.push()) {
300                         assert listener != null;
301                         listener.onReady();
302                     } catch (Throwable t) {
303                         close(GrpcStatus.fromThrowable(t), new Metadata());
304                     }
305                 }
306             });
307         } catch (Throwable t) {
308             cancel(null, t);
309         }
310     }

317     @Override
318     public void messageRead(DeframedMessage message) {
319         if (GrpcSerializationFormats.isGrpcWeb(serializationFormat) && message.type() >> 7 == 1) {
320             // grpc-web trailers
321             final ByteBuf buf = message.buf();
322             // trailers never compressed
323             assert buf != null;
324             try {
325                 final HttpHeaders trailers = parseGrpcWebTrailers(buf);
326                 if (trailers == null) {
327                     // Malformed trailers.
328                     close(Status.INTERNAL
329                                   .withDescription("grpc-web trailers malformed: " +
330                                                    buf.toString(StandardCharsets.UTF_8)),
331                           new Metadata());
332                 } else {
333                     GrpcStatus.reportStatus(trailers, responseReader, this);
334                 }
335             } finally {
336                 buf.release();
337             }
338             return;
339         }
340         try {
341             final O msg = marshaller.deserializeResponse(message);
342             if (firstResponse == null) {
343                 firstResponse = msg;
344             }
345 
346             final ByteBuf buf = message.buf();
347             if (unsafeWrapResponseBuffers && buf != null) {
348                 GrpcUnsafeBufferUtil.storeBuffer(buf, msg, ctx);
349             }
350 
351             try (SafeCloseable ignored = ctx.push()) {
352                 assert listener != null;
353                 listener.onMessage(msg);
354             }
355         } catch (Throwable t) {
356             req.close(GrpcStatus.fromThrowable(t).asException());
357             throw t instanceof RuntimeException ? (RuntimeException) t : new RuntimeException(t);
358         }
359 
360         notifyExecutor();
361     }

373     private void prepareHeaders(Compressor compressor, Metadata metadata) {
374         final RequestHeadersBuilder newHeaders = req.headers().toBuilder();
375         if (compressor != Identity.NONE) {
376             newHeaders.set(GrpcHeaderNames.GRPC_ENCODING, compressor.getMessageEncoding());
377         }
378 
379         if (!advertisedEncodingsHeader.isEmpty()) {
380             newHeaders.add(GrpcHeaderNames.GRPC_ACCEPT_ENCODING, advertisedEncodingsHeader);
381         }
382 
383         newHeaders.add(GrpcHeaderNames.GRPC_TIMEOUT,
384                        TimeoutHeaderUtil.toHeaderValue(
385                                TimeUnit.MILLISECONDS.toNanos(ctx.responseTimeoutMillis())));
386 
387         MetadataUtil.fillHeaders(metadata, newHeaders);
388 
389         final HttpRequest newReq = req.withHeaders(newHeaders);
390         ctx.updateRequest(newReq);
391     }
392 
393     private void close(Status status, Metadata metadata) {
394         final Deadline deadline = callOptions.getDeadline();
395         if (status.getCode() == Code.CANCELLED && deadline != null && deadline.isExpired()) {
396             status = Status.DEADLINE_EXCEEDED.augmentDescription(
397                     "ClientCall was cancelled at or after deadline.");
398             // Replace trailers to prevent mixing sources of status and trailers.
399             metadata = new Metadata();
400         }
401         ctx.logBuilder().responseContent(GrpcLogUtil.rpcResponse(status, firstResponse), null);
402         if (status.isOk()) {
403             req.abort();
404         } else {
405             req.abort(status.asRuntimeException(metadata));
406         }
407         responseReader.cancel();
408 
409         try (SafeCloseable ignored = ctx.push()) {
410             assert listener != null;
411             listener.onClose(status, metadata);
412         }
413 
414         notifyExecutor();
415     }

417     /**
418      * Armeria does not support {@link CallOptions} set by the user, however gRPC stubs set an {@link Executor}
419      * within blocking stubs which is used to notify the stub when processing is finished. It's unclear why
420      * the stubs use a loop and {@link Future#isDone()} instead of just blocking on
421      * {@link Future#get}, but we make sure to run the {@link Executor} so the stub can
422      * be notified of completion.
423      */
424     private void notifyExecutor() {
425         if (executor != null) {
426             executor.execute(NO_OP);
427         }
428     }

430     @Nullable
431     private static HttpHeaders parseGrpcWebTrailers(ByteBuf buf) {
432         final HttpHeadersBuilder trailers = HttpHeaders.builder();
433         while (buf.readableBytes() > 0) {
434             int start = buf.forEachByte(ByteProcessor.FIND_NON_LINEAR_WHITESPACE);
435             if (start == -1) {
436                 return null;
437             }
438             int endExclusive;
439             if (buf.getByte(start) == ':') {
440                 // We need to skip the pseudoheader colon when searching for the separator.
441                 buf.skipBytes(1);
442                 endExclusive = buf.forEachByte(FIND_COLON);
443                 buf.readerIndex(start);
444             } else {
445                 endExclusive = buf.forEachByte(FIND_COLON);
446             }
447             if (endExclusive == -1) {
448                 return null;
449             }
450             final CharSequence name = buf.readCharSequence(endExclusive - start, StandardCharsets.UTF_8);
451             buf.readerIndex(endExclusive + 1);
452             start = buf.forEachByte(ByteProcessor.FIND_NON_LINEAR_WHITESPACE);
453             buf.readerIndex(start);
454             endExclusive = buf.forEachByte(ByteProcessor.FIND_CRLF);
455             final CharSequence value = buf.readCharSequence(endExclusive - start, StandardCharsets.UTF_8);
456             trailers.add(name, value.toString());
457             start = buf.forEachByte(ByteProcessor.FIND_NON_CRLF);
458             if (start != -1) {
459                 buf.readerIndex(start);
460             } else {
461                 // Nothing but CRLF remaining, we're done.
462                 buf.skipBytes(buf.readableBytes());
463             }
464         }
465         return trailers.build();
466     }
