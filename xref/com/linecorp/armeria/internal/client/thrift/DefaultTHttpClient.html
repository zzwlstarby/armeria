17  package com.linecorp.armeria.internal.client.thrift;
18  
19  import static com.linecorp.armeria.internal.common.ArmeriaHttpUtil.concatPaths;
20  import static java.util.Objects.requireNonNull;
21  
22  import javax.annotation.Nullable;
23  
24  import com.linecorp.armeria.client.ClientBuilderParams;
25  import com.linecorp.armeria.client.RpcClient;
26  import com.linecorp.armeria.client.UserClient;
27  import com.linecorp.armeria.client.thrift.THttpClient;
28  import com.linecorp.armeria.common.HttpMethod;
29  import com.linecorp.armeria.common.RpcRequest;
30  import com.linecorp.armeria.common.RpcResponse;
31  import com.linecorp.armeria.internal.common.PathAndQuery;
32  
33  import io.micrometer.core.instrument.MeterRegistry;
34  
35  final class DefaultTHttpClient extends UserClient<RpcRequest, RpcResponse> implements THttpClient {
36  
37      DefaultTHttpClient(ClientBuilderParams params, RpcClient delegate, MeterRegistry meterRegistry) {
38          super(params, delegate, meterRegistry);
39      }
40  
41      @Override
42      public RpcResponse execute(String path, Class<?> serviceType, String method, Object... args) {
43          return execute0(path, serviceType, null, method, args);
44      }
45  
46      @Override
47      public RpcResponse executeMultiplexed(
48              String path, Class<?> serviceType, String serviceName, String method, Object... args) {
49          requireNonNull(serviceName, "serviceName");
50          return execute0(path, serviceType, serviceName, method, args);
51      }
52  
53      private RpcResponse execute0(
54              String path, Class<?> serviceType, @Nullable String serviceName, String method, Object[] args) {
55  
56          path = concatPaths(uri().getRawPath(), path);
57          final PathAndQuery pathAndQuery = PathAndQuery.parse(path);
58          if (pathAndQuery == null) {
59              return RpcResponse.ofFailure(new IllegalArgumentException("invalid path: " + path));
60          }
61  
62          // A thrift path is always good to cache as it cannot have non-fixed parameters.
63          pathAndQuery.storeInCache(path);
64  
65          final RpcRequest call = RpcRequest.of(serviceType, method, args);
66          return execute(HttpMethod.POST, pathAndQuery.path(), null, serviceName, call,
67                         (ctx, cause) -> RpcResponse.ofFailure(cause));
68      }
69  }
