56      /**
57       * Sets up request metrics.
58       */
59      public static void setup(RequestContext ctx, AttributeKey<Boolean> requestMetricsSetKey,
60                               MeterIdPrefixFunction meterIdPrefixFunction, boolean server) {
61          final Boolean isRequestMetricsSet = ctx.attr(requestMetricsSetKey);
62  
63          if (Boolean.TRUE.equals(isRequestMetricsSet)) {
64              return;
65          }
66          ctx.setAttr(requestMetricsSetKey, true);
67  
68          ctx.log()
69             .whenAvailable(RequestLogProperty.REQUEST_START_TIME,
70                            RequestLogProperty.REQUEST_HEADERS,
71                            RequestLogProperty.NAME,
72                            RequestLogProperty.SESSION)
73             .thenAccept(log -> onRequest(log, meterIdPrefixFunction, server));
74      }
75  
76      /**
77       * Appends {@link HttpStatus} to {@link Tag}.
78       */
79      public static void appendHttpStatusTag(ImmutableList.Builder<Tag> tagListBuilder, RequestLog log) {
80          requireNonNull(tagListBuilder, "tagListBuilder");
81          requireNonNull(log, "log");
82          // Add the 'httpStatus' tag.
83          final HttpStatus status;
84          if (log.isAvailable(RequestLogProperty.RESPONSE_HEADERS)) {
85              status = log.responseHeaders().status();
86          } else {
87              status = HttpStatus.UNKNOWN;
88          }
89          tagListBuilder.add(Tag.of(Flags.useLegacyMeterNames() ? "httpStatus" : "http.status",
90                                    status.codeAsText()));
91      }

93      private static void onRequest(RequestLog log, MeterIdPrefixFunction meterIdPrefixFunction, boolean server) {
94          final RequestContext ctx = log.context();
95          final MeterRegistry registry = ctx.meterRegistry();
96          final MeterIdPrefix activeRequestsId =
97                  meterIdPrefixFunction.activeRequestPrefix(registry, log)
98                                       .append(Flags.useLegacyMeterNames() ? "activeRequests"
99                                                                           : "active.requests");
100 
101         final ActiveRequestMetrics activeRequestMetrics = MicrometerUtil.register(
102                 registry, activeRequestsId, ActiveRequestMetrics.class,
103                 (reg, prefix) ->
104                         reg.gauge(prefix.name(), prefix.tags(),
105                                   new ActiveRequestMetrics(), ActiveRequestMetrics::doubleValue));
106         activeRequestMetrics.increment();
107         ctx.log().whenComplete().thenAccept(requestLog -> {
108             onResponse(requestLog, meterIdPrefixFunction, server);
109             activeRequestMetrics.decrement();
110         });
111     }
112 
113     private static void onResponse(RequestLog log, MeterIdPrefixFunction meterIdPrefixFunction,
114                                    boolean server) {
115         final RequestContext ctx = log.context();
116         final MeterRegistry registry = ctx.meterRegistry();
117         final MeterIdPrefix idPrefix = meterIdPrefixFunction.completeRequestPrefix(registry, log);
118 
119         if (server) {
120             final ServiceRequestMetrics metrics = MicrometerUtil.register(registry, idPrefix,
121                                                                           ServiceRequestMetrics.class,
122                                                                           DefaultServiceRequestMetrics::new);
123             updateMetrics(log, metrics);
124             if (log.responseCause() instanceof RequestTimeoutException) {
125                 metrics.requestTimeouts().increment();
126             }
127             return;
128         }
129 
130         final ClientRequestMetrics metrics = MicrometerUtil.register(registry, idPrefix,
131                                                                      ClientRequestMetrics.class,
132                                                                      DefaultClientRequestMetrics::new);
133         updateMetrics(log, metrics);
134         final ClientConnectionTimings timings = log.connectionTimings();
135         if (timings != null) {
136             metrics.connectionAcquisitionDuration().record(timings.connectionAcquisitionDurationNanos(),
137                                                            TimeUnit.NANOSECONDS);
138             final long dnsResolutionDurationNanos = timings.dnsResolutionDurationNanos();
139             if (dnsResolutionDurationNanos >= 0) {
140                 metrics.dnsResolutionDuration().record(dnsResolutionDurationNanos, TimeUnit.NANOSECONDS);
141             }
142             final long socketConnectDurationNanos = timings.socketConnectDurationNanos();
143             if (socketConnectDurationNanos >= 0) {
144                 metrics.socketConnectDuration().record(socketConnectDurationNanos, TimeUnit.NANOSECONDS);
145             }
146             final long pendingAcquisitionDurationNanos = timings.pendingAcquisitionDurationNanos();
147             if (pendingAcquisitionDurationNanos >= 0) {
148                 metrics.pendingAcquisitionDuration().record(pendingAcquisitionDurationNanos,
149                                                             TimeUnit.NANOSECONDS);
150             }
151         }
152         if (log.requestCause() != null) {
153             if (log.requestCause() instanceof WriteTimeoutException) {
154                 metrics.writeTimeouts().increment();
155             }
156             return;
157         }
158 
159         if (log.responseCause() instanceof ResponseTimeoutException) {
160             metrics.responseTimeouts().increment();
161         }
162 
163         final int childrenSize = log.children().size();
164         if (childrenSize > 0) {
165             metrics.actualRequests().increment(childrenSize);
166         }
167     }
168 
169     private static void updateMetrics(RequestLog log, RequestMetrics metrics) {
170         if (log.requestCause() != null) {
171             metrics.failure().increment();
172             return;
173         }
174 
175         metrics.requestDuration().record(log.requestDurationNanos(), TimeUnit.NANOSECONDS);
176         metrics.requestLength().record(log.requestLength());
177         metrics.responseDuration().record(log.responseDurationNanos(), TimeUnit.NANOSECONDS);
178         metrics.responseLength().record(log.responseLength());
179         metrics.totalDuration().record(log.totalDurationNanos(), TimeUnit.NANOSECONDS);
180 
181         if (isSuccess(log)) {
182             metrics.success().increment();
183         } else {
184             metrics.failure().increment();
185         }
186     }
187 
188     private static boolean isSuccess(RequestLog log) {
189         if (log.responseCause() != null) {
190             return false;
191         }
192 
193         final int statusCode = log.responseHeaders().status().code();
194         if (statusCode < 100 || statusCode >= 400) {
195             return false;
196         }
197 
198         final Object responseContent = log.responseContent();
199         if (responseContent instanceof RpcResponse) {
200             return !((RpcResponse) responseContent).isCompletedExceptionally();
201         }
202 
203         return true;
204     }

247     private abstract static class AbstractRequestMetrics implements RequestMetrics {
248 
249         private final Counter success;
250         private final Counter failure;
251         private final Timer requestDuration;
252         private final DistributionSummary requestLength;
253         private final Timer responseDuration;
254         private final DistributionSummary responseLength;
255         private final Timer totalDuration;
256 
257         AbstractRequestMetrics(MeterRegistry parent, MeterIdPrefix idPrefix) {
258             final String requests = idPrefix.name("requests");
259             success = parent.counter(requests, idPrefix.tags("result", "success"));
260             failure = parent.counter(requests, idPrefix.tags("result", "failure"));
261 
262             requestDuration = newTimer(
263                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "requestDuration"
264                                                                       : "request.duration"), idPrefix.tags());
265             requestLength = newDistributionSummary(
266                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "requestLength"
267                                                                       : "request.length"), idPrefix.tags());
268             responseDuration = newTimer(
269                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "responseDuration"
270                                                                       : "response.duration"), idPrefix.tags());
271             responseLength = newDistributionSummary(
272                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "responseLength"
273                                                                       : "response.length"), idPrefix.tags());
274             totalDuration = newTimer(
275                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "totalDuration"
276                                                                       : "total.duration"), idPrefix.tags());
277         }
278 
279         @Override
280         public Counter success() {
281             return success;
282         }
283 
284         @Override
285         public Counter failure() {
286             return failure;
287         }
288 
289         @Override
290         public Timer requestDuration() {
291             return requestDuration;
292         }
293 
294         @Override
295         public DistributionSummary requestLength() {
296             return requestLength;
297         }
298 
299         @Override
300         public Timer responseDuration() {
301             return responseDuration;
302         }
303 
304         @Override
305         public DistributionSummary responseLength() {
306             return responseLength;
307         }
308 
309         @Override
310         public Timer totalDuration() {
311             return totalDuration;
312         }
313     }

315     private static class DefaultClientRequestMetrics
316             extends AbstractRequestMetrics implements ClientRequestMetrics {
317 
318         private static final AtomicReferenceFieldUpdater<DefaultClientRequestMetrics, Counter>
319                 actualRequestsUpdater = AtomicReferenceFieldUpdater.newUpdater(
320                 DefaultClientRequestMetrics.class, Counter.class, "actualRequests");
321 
322         private final MeterRegistry parent;
323         private final MeterIdPrefix idPrefix;
324 
325         private final Timer connectionAcquisitionDuration;
326         private final Timer dnsResolutionDuration;
327         private final Timer socketConnectDuration;
328         private final Timer pendingAcquisitionDuration;
329 
330         private final Counter writeTimeouts;
331         private final Counter responseTimeouts;
332 
333         @Nullable
334         private volatile Counter actualRequests;
335 
336         DefaultClientRequestMetrics(MeterRegistry parent, MeterIdPrefix idPrefix) {
337             super(parent, idPrefix);
338             this.parent = parent;
339             this.idPrefix = idPrefix;
340 
341             connectionAcquisitionDuration = newTimer(
342                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "connectionAcquisitionDuration"
343                                                                       : "connection.acquisition.duration"),
344                     idPrefix.tags());
345             dnsResolutionDuration = newTimer(
346                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "dnsResolutionDuration"
347                                                                       : "dns.resolution.duration"),
348                     idPrefix.tags());
349             socketConnectDuration = newTimer(
350                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "socketConnectDuration"
351                                                                       : "socket.connect.duration"),
352                     idPrefix.tags());
353             pendingAcquisitionDuration = newTimer(
354                     parent, idPrefix.name(Flags.useLegacyMeterNames() ? "pendingAcquisitionDuration"
355                                                                       : "pending.acquisition.duration"),
356                     idPrefix.tags());
357 
358             final String timeouts = idPrefix.name("timeouts");
359             writeTimeouts = parent.counter(timeouts, idPrefix.tags("cause", "WriteTimeoutException"));
360             responseTimeouts = parent.counter(timeouts, idPrefix.tags("cause", "ResponseTimeoutException"));
361         }
362 
363         @Override
364         public Counter actualRequests() {
365             final Counter actualRequests = this.actualRequests;
366             if (actualRequests != null) {
367                 return actualRequests;
368             }
369 
370             final Counter counter =
371                     parent.counter(idPrefix.name(Flags.useLegacyMeterNames() ? "actualRequests"
372                                                                              : "actual.requests"),
373                                    idPrefix.tags());
374             if (actualRequestsUpdater.compareAndSet(this, null, counter)) {
375                 return counter;
376             }
377             return this.actualRequests;
378         }
379 
380         @Override
381         public Timer connectionAcquisitionDuration() {
382             return connectionAcquisitionDuration;
383         }
384 
385         @Override
386         public Timer dnsResolutionDuration() {
387             return dnsResolutionDuration;
388         }
389 
390         @Override
391         public Timer socketConnectDuration() {
392             return socketConnectDuration;
393         }
394 
395         @Override
396         public Timer pendingAcquisitionDuration() {
397             return pendingAcquisitionDuration;
398         }
399 
400         @Override
401         public Counter writeTimeouts() {
402             return writeTimeouts;
403         }
404 
405         @Override
406         public Counter responseTimeouts() {
407             return responseTimeouts;
408         }
409     }
