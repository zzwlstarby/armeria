49  /**
50   * Registers the stats of Caffeine {@link Cache}.
51   */
52  public final class CaffeineMetricSupport {
53  
54      @VisibleForTesting
55      static final long UPDATE_INTERVAL_NANOS = TimeUnit.SECONDS.toNanos(3);
56  
57      public static void setup(MeterRegistry registry, MeterIdPrefix idPrefix, Cache<?, ?> cache) {
58          setup(registry, idPrefix, cache, Ticker.systemTicker());
59      }
60  
61      public static void setup(MeterRegistry registry, MeterIdPrefix idPrefix, Cache<?, ?> cache, Ticker ticker) {
62          requireNonNull(cache, "cache");
63          if (!cache.policy().isRecordingStats()) {
64              return;
65          }
66          final CaffeineMetrics metrics = MicrometerUtil.register(
67                  registry, idPrefix, CaffeineMetrics.class, CaffeineMetrics::new);
68          metrics.add(cache, ticker);
69      }
70  
71      private CaffeineMetricSupport() {}
72  
73      enum Type {
74          HIT_COUNT,
75          MISS_COUNT,
76          EVICTION_COUNT,
77          EVICTION_WEIGHT,
78          LOAD_SUCCESS_COUNT,
79          LOAD_FAILURE_COUNT,
80          TOTAL_LOAD_TIME;
81  
82          static final int count = Type.values().length;
83      }
84  
85      private static final class CaffeineMetrics {
86  
87          private final MeterRegistry parent;
88          private final MeterIdPrefix idPrefix;
89          private final List<CacheReference> cacheRefs = new ArrayList<>(2);
90          private final AtomicBoolean hasLoadingCache = new AtomicBoolean();
91  
92          /**
93           * An array whose each element is the sum of the garbage-collected {@link Cache} stats.
94           * {@link Type#ordinal()} signifies the index of this array.
95           */
96          private final double[] statsForGarbageCollected = new double[Type.count];
97  
98          CaffeineMetrics(MeterRegistry parent, MeterIdPrefix idPrefix) {
99              this.parent = requireNonNull(parent, "parent");
100             this.idPrefix = requireNonNull(idPrefix, "idPrefix");
101 
102             final String requests = idPrefix.name("requests");
103             parent.more().counter(requests, idPrefix.tags("result", "hit"), this,
104                                   func(HIT_COUNT, ref -> ref.cacheStats.hitCount()));
105             parent.more().counter(requests, idPrefix.tags("result", "miss"), this,
106                                   func(MISS_COUNT, ref -> ref.cacheStats.missCount()));
107             parent.more().counter(idPrefix.name("evictions"), idPrefix.tags(), this,
108                                   func(EVICTION_COUNT, ref -> ref.cacheStats.evictionCount()));
109             parent.more().counter(idPrefix.name(Flags.useLegacyMeterNames() ? "evictionWeight"
110                                                                             : "eviction.weight"),
111                                   idPrefix.tags(), this,
112                                   func(EVICTION_WEIGHT, ref -> ref.cacheStats.evictionWeight()));
113             parent.gauge(idPrefix.name(Flags.useLegacyMeterNames() ? "estimatedSize"
114                                                                    : "estimated.size"),
115                          idPrefix.tags(), this,
116                          func(null, ref -> ref.estimatedSize));
117         }
118 
119         void add(Cache<?, ?> cache, Ticker ticker) {
120             synchronized (cacheRefs) {
121                 for (CacheReference ref : cacheRefs) {
122                     if (ref.get() == cache) {
123                         // Do not aggregate more than once for the same instance.
124                         return;
125                     }
126                 }
127 
128                 cacheRefs.add(new CacheReference(cache, ticker));
129             }
130 
131             if (cache instanceof LoadingCache && hasLoadingCache.compareAndSet(false, true)) {
132                 // Add the following meters only for LoadingCache and only once.
133                 final String loads = idPrefix.name("loads");
134 
135                 parent.more().counter(loads, idPrefix.tags("result", "success"), this,
136                                       func(LOAD_SUCCESS_COUNT, ref -> ref.cacheStats.loadSuccessCount()));
137                 parent.more().counter(loads, idPrefix.tags("result", "failure"), this,
138                                       func(LOAD_FAILURE_COUNT, ref -> ref.cacheStats.loadFailureCount()));
139                 parent.more().counter(idPrefix.name(Flags.useLegacyMeterNames() ? "loadDuration"
140                                                                                 : "load.duration"),
141                                       idPrefix.tags(), this,
142                                       func(TOTAL_LOAD_TIME, ref -> ref.cacheStats.totalLoadTime()));
143             }
144         }
145 
146         private ToDoubleFunction<CaffeineMetrics> func(@Nullable Type type,
147                                                        ToDoubleFunction<CacheReference> valueFunction) {
148             return value -> {
149                 double sum = 0;
150                 synchronized (cacheRefs) {
151                     for (final Iterator<CacheReference> i = cacheRefs.iterator(); i.hasNext();) {
152                         final CacheReference ref = i.next();
153                         final boolean garbageCollected = ref.updateCacheStats();
154                         if (!garbageCollected) {
155                             sum += valueFunction.applyAsDouble(ref);
156                         } else {
157                             // Remove the garbage-collected reference from the list to prevent it from
158                             // growing infinitely.
159                             i.remove();
160 
161                             // Accumulate the stats of the removed reference so the counters do not decrease.
162                             // NB: We do not accumulate 'estimatedSize' because it's not a counter but a gauge.
163                             final CacheStats stats = ref.cacheStats;
164                             statsForGarbageCollected[HIT_COUNT.ordinal()] += stats.hitCount();
165                             statsForGarbageCollected[MISS_COUNT.ordinal()] += stats.missCount();
166                             statsForGarbageCollected[EVICTION_COUNT.ordinal()] += stats.evictionCount();
167                             statsForGarbageCollected[EVICTION_WEIGHT.ordinal()] += stats.evictionWeight();
168                             statsForGarbageCollected[LOAD_SUCCESS_COUNT.ordinal()] += stats.loadSuccessCount();
169                             statsForGarbageCollected[LOAD_FAILURE_COUNT.ordinal()] += stats.loadFailureCount();
170                             statsForGarbageCollected[TOTAL_LOAD_TIME.ordinal()] += stats.totalLoadTime();
171                         }
172                     }
173 
174                     if (type != null) {
175                         // Add the value of the garbage-collected caches.
176                         sum += statsForGarbageCollected[type.ordinal()];
177                     }
178                 }
179 
180                 return sum;
181             };
182         }
183     }
184 
185     private static final class CacheReference extends WeakReference<Cache<?, ?>> {
186 
187         private final Ticker ticker;
188         private volatile long lastStatsUpdateTime;
189         private CacheStats cacheStats = CacheStats.empty();
190         private long estimatedSize;
191 
192         CacheReference(Cache<?, ?> cache, Ticker ticker) {
193             super(requireNonNull(cache, "cache"));
194             this.ticker = requireNonNull(ticker, "ticker");
195             updateCacheStats(true);
196         }
197 
198         boolean updateCacheStats() {
199             return updateCacheStats(false);
200         }
201 
202         private boolean updateCacheStats(boolean force) {
203             final Cache<?, ?> cache = get();
204             if (cache == null) {
205                 return true; // GC'd
206             }
207 
208             final long currentTimeNanos = ticker.read();
209             if (!force) {
210                 if (currentTimeNanos - lastStatsUpdateTime < UPDATE_INTERVAL_NANOS) {
211                     return false; // Not GC'd
212                 }
213             }
214 
215             cacheStats = cache.stats();
216             estimatedSize = cache.estimatedSize();
217 
218             // Write the volatile field last so that cacheStats and estimatedSize are visible
219             // after reading the volatile field.
220             lastStatsUpdateTime = currentTimeNanos;
221             return false; // Not GC'd
222         }
223     }
