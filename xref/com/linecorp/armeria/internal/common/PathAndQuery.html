47      private static final PathAndQuery ROOT_PATH_QUERY = new PathAndQuery("/", null);

49      private static final BitSet ALLOWED_PATH_CHARS = new BitSet();
50      private static final BitSet ALLOWED_QUERY_CHARS = new BitSet();
51  
52      private static final int PERCENT_ENCODING_MARKER = 0xFF;
53  
54      private static final byte[] RAW_CHAR_TO_MARKER = new byte[256];
55      private static final String[] MARKER_TO_PERCENT_ENCODED_CHAR = new String[256];

57      static {
58          final String allowedPathChars =
59                  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-._~:/?#[]@!$&'()*+,;=";
60          for (int i = 0; i < allowedPathChars.length(); i++) {
61              ALLOWED_PATH_CHARS.set(allowedPathChars.charAt(i));
62          }
63  
64          final String allowedQueryChars =
65                  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-._~:/?#[]@!$&'()*,;=";
66          for (int i = 0; i < allowedQueryChars.length(); i++) {
67              ALLOWED_QUERY_CHARS.set(allowedQueryChars.charAt(i));
68          }
69  
70          for (final ReservedChar reservedChar : ReservedChar.values()) {
71              RAW_CHAR_TO_MARKER[reservedChar.rawChar] = reservedChar.marker;
72              MARKER_TO_PERCENT_ENCODED_CHAR[reservedChar.marker] = reservedChar.percentEncodedChar;
73          }
74      }

111     /**
112      * Validates the {@link String} that contains an absolute path and a query, and splits them into
113      * the path part and the query part. If the path is usable (e.g., can be served a successful response from
114      * the server and doesn't have variable path parameters), {@link PathAndQuery#storeInCache(String)} should
115      * be called to cache the parsing result for faster future invocations.
116      *
117      * @return a {@link PathAndQuery} with the absolute path and query, or {@code null} if the specified
118      *         {@link String} is not an absolute path or invalid.
119      */
120     @Nullable
121     public static PathAndQuery parse(@Nullable String rawPath) {
122         if (CACHE != null && rawPath != null) {
123             final PathAndQuery parsed = CACHE.getIfPresent(rawPath);
124             if (parsed != null) {
125                 return parsed;
126             }
127         }
128         return splitPathAndQuery(rawPath);
129     }

131     /**
132      * Stores this {@link PathAndQuery} into cache for the given raw path. This should be used by callers when
133      * the parsed result was valid (e.g., when a server is able to successfully handle the parsed path).
134      */
135     public void storeInCache(@Nullable String rawPath) {
136         if (CACHE != null && rawPath != null) {
137             CACHE.put(rawPath, this);
138         }
139     }

187     @Nullable
188     private static PathAndQuery splitPathAndQuery(@Nullable final String pathAndQuery) {
189         final Bytes path;
190         final Bytes query;
191 
192         if (pathAndQuery == null) {
193             return ROOT_PATH_QUERY;
194         }
195 
196         // Split by the first '?'.
197         final int queryPos = pathAndQuery.indexOf('?');
198         if (queryPos >= 0) {
199             if ((path = decodePercentsAndEncodeToUtf8(
200                     pathAndQuery, 0, queryPos, true)) == null) {
201                 return null;
202             }
203             if ((query = decodePercentsAndEncodeToUtf8(
204                     pathAndQuery, queryPos + 1, pathAndQuery.length(), false)) == null) {
205                 return null;
206             }
207         } else {
208             if ((path = decodePercentsAndEncodeToUtf8(
209                     pathAndQuery, 0, pathAndQuery.length(), true)) == null) {
210                 return null;
211             }
212             query = null;
213         }
214 
215         if (path.data[0] != '/') {
216             // Do not accept a relative path.
217             return null;
218         }
219 
220         // Reject the prohibited patterns.
221         if (firstPathComponentContainsColon(path) || pathContainsDoubleDots(path)) {
222             return null;
223         }
224 
225         return new PathAndQuery(encodeToPercents(path, true),
226                                 query != null ? encodeToPercents(query, false) : null);
227     }

229     @Nullable
230     private static Bytes decodePercentsAndEncodeToUtf8(String value, int start, int end, boolean isPath) {
231         final int length = end - start;
232         if (length == 0) {
233             return isPath ? ROOT_PATH : EMPTY_QUERY;
234         }
235 
236         final Bytes buf = new Bytes(Math.max(length * 3 / 2, 4));
237         boolean wasSlash = false;
238         for (final CodePointIterator i = new CodePointIterator(value, start, end);
239              i.hasNextCodePoint();/* noop */) {
240             final int pos = i.position();
241             final int cp = i.nextCodePoint();
242 
243             if (cp == '%') {
244                 final int hexEnd = pos + 3;
245                 if (hexEnd > end) {
246                     // '%' or '%x' (must be followed by two hexadigits)
247                     return null;
248                 }
249 
250                 final int digit1 = decodeHexNibble(value.charAt(pos + 1));
251                 final int digit2 = decodeHexNibble(value.charAt(pos + 2));
252                 if (digit1 < 0 || digit2 < 0) {
253                     // The first or second digit is not hexadecimal.
254                     return null;
255                 }
256 
257                 final int decoded = (digit1 << 4) | digit2;
258                 if (isPath) {
259                     if (appendOneByte(buf, decoded, wasSlash, isPath)) {
260                         wasSlash = decoded == '/';
261                     } else {
262                         return null;
263                     }
264                 } else {
265                     // If query:
266                     final byte marker = RAW_CHAR_TO_MARKER[decoded];
267                     if (marker != 0) {
268                         // Insert a special mark so we can distinguish a raw character and percent-encoded
269                         // character in a query string, such as '&' and '%26'.
270                         // We will encode this mark back into a percent-encoded character later.
271                         buf.ensure(2);
272                         buf.add((byte) PERCENT_ENCODING_MARKER);
273                         buf.add(marker);
274                         wasSlash = false;
275                     } else if (appendOneByte(buf, decoded, wasSlash, isPath)) {
276                         wasSlash = decoded == '/';
277                     } else {
278                         return null;
279                     }
280                 }
281 
282                 i.position(hexEnd);
283                 continue;
284             }
285 
286             if (cp == '+' && !isPath) {
287                 buf.ensure(1);
288                 buf.add((byte) ' ');
289                 wasSlash = false;
290                 continue;
291             }
292 
293             if (cp <= 0x7F) {
294                 if (!appendOneByte(buf, cp, wasSlash, isPath)) {
295                     return null;
296                 }
297                 wasSlash = cp == '/';
298                 continue;
299             }
300 
301             if (cp <= 0x7ff) {
302                 buf.ensure(2);
303                 buf.add((byte) ((cp >>> 6) | 0b110_00000));
304                 buf.add((byte) (cp & 0b111111 | 0b10_000000));
305             } else if (cp <= 0xffff) {
306                 buf.ensure(3);
307                 buf.add((byte) ((cp >>> 12) | 0b1110_0000));
308                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
309                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
310             } else if (cp <= 0x1fffff) {
311                 buf.ensure(4);
312                 buf.add((byte) ((cp >>> 18) | 0b11110_000));
313                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
314                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
315                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
316             } else if (cp <= 0x3ffffff) {
317                 // A valid unicode character will never reach here, but for completeness.
318                 // http://unicode.org/mail-arch/unicode-ml/Archives-Old/UML018/0330.html
319                 buf.ensure(5);
320                 buf.add((byte) ((cp >>> 24) | 0b111110_00));
321                 buf.add((byte) (((cp >>> 18) & 0b111111) | 0b10_000000));
322                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
323                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
324                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
325             } else {
326                 // A valid unicode character will never reach here, but for completeness.
327                 // http://unicode.org/mail-arch/unicode-ml/Archives-Old/UML018/0330.html
328                 buf.ensure(6);
329                 buf.add((byte) ((cp >>> 30) | 0b1111110_0));
330                 buf.add((byte) (((cp >>> 24) & 0b111111) | 0b10_000000));
331                 buf.add((byte) (((cp >>> 18) & 0b111111) | 0b10_000000));
332                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
333                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
334                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
335             }
336 
337             wasSlash = false;
338         }
339 
340         return buf;
341     }

343     private static boolean appendOneByte(Bytes buf, int cp, boolean wasSlash, boolean isPath) {
344         if (cp == 0x7F) {
345             // Reject the control character: 0x7F
346             return false;
347         }
348 
349         if (cp >>> 5 == 0) {
350             // Reject the control characters: 0x00..0x1F
351             if (isPath) {
352                 return false;
353             } else if (cp != 0x0A && cp != 0x0D && cp != 0x09) {
354                 // .. except 0x0A (LF), 0x0D (CR) and 0x09 (TAB) because they are used in a form.
355                 return false;
356             }
357         }
358 
359         if (cp == '/' && isPath) {
360             if (!wasSlash) {
361                 buf.ensure(1);
362                 buf.add((byte) '/');
363             } else {
364                 // Remove the consecutive slashes: '/path//with///consecutive////slashes'.
365             }
366         } else {
367             buf.ensure(1);
368             buf.add((byte) cp);
369         }
370 
371         return true;
372     }

374     /**
375      * According to RFC 3986 section 3.3, path can contain a colon, except the first segment.
376      *
377      * <p>Should allow the asterisk character in the path, query, or fragment components of a URL(RFC2396).
378      * @see <a href="https://tools.ietf.org/html/rfc3986#section-3.3">RFC 3986, section 3.3</a>
379      */
380     private static boolean firstPathComponentContainsColon(Bytes path) {
381         final int length = path.length;
382         for (int i = 1; i < length; i++) {
383             final byte b = path.data[i];
384             if (b == '/') {
385                 break;
386             }
387             if (b == ':') {
388                 return true;
389             }
390         }
391         return false;
392     }
393 
394     private static boolean pathContainsDoubleDots(Bytes path) {
395         final int length = path.length;
396         byte b0 = 0;
397         byte b1 = 0;
398         byte b2 = '/';
399         for (int i = 1; i < length; i++) {
400             final byte b3 = path.data[i];
401             if (b3 == '/' && b2 == '.' && b1 == '.' && b0 == '/') {
402                 return true;
403             }
404             b0 = b1;
405             b1 = b2;
406             b2 = b3;
407         }
408 
409         return b0 == '/' && b1 == '.' && b2 == '.';
410     }

412     private static String encodeToPercents(Bytes value, boolean isPath) {
413         final BitSet allowedChars = isPath ? ALLOWED_PATH_CHARS : ALLOWED_QUERY_CHARS;
414         final int length = value.length;
415         boolean needsEncoding = false;
416         for (int i = 0; i < length; i++) {
417             if (!allowedChars.get(value.data[i] & 0xFF)) {
418                 needsEncoding = true;
419                 break;
420             }
421         }
422 
423         if (!needsEncoding) {
424             // Deprecated, but it fits perfect for our use case.
425             // noinspection deprecation
426             return new String(value.data, 0, 0, length);
427         }
428 
429         final StringBuilder buf = new StringBuilder(length);
430         for (int i = 0; i < length; i++) {
431             final int b = value.data[i] & 0xFF;
432 
433             if (b == PERCENT_ENCODING_MARKER && (i + 1) < length) {
434                 final int marker = value.data[i + 1] & 0xFF;
435                 final String percentEncodedChar = MARKER_TO_PERCENT_ENCODED_CHAR[marker];
436                 if (percentEncodedChar != null) {
437                     buf.append(percentEncodedChar);
438                     i++;
439                     continue;
440                 }
441             }
442 
443             if (allowedChars.get(b)) {
444                 buf.append((char) b);
445             } else if (b == ' ') {
446                 if (isPath) {
447                     buf.append("%20");
448                 } else {
449                     buf.append('+');
450                 }
451             } else {
452                 buf.append('%');
453                 appendHexNibble(buf, b >>> 4);
454                 appendHexNibble(buf, b & 0xF);
455             }
456         }
457 
458         return buf.toString();
459     }
460 
461     private static void appendHexNibble(StringBuilder buf, int nibble) {
462         if (nibble < 10) {
463             buf.append((char) ('0' + nibble));
464         } else {
465             buf.append((char) ('A' + nibble - 10));
466         }
467     }

469     private static final class Bytes {
470         byte[] data;
471         int length;
472 
473         Bytes(int initialCapacity) {
474             data = new byte[initialCapacity];
475         }
476 
477         Bytes(byte[] data) {
478             this.data = data;
479             length = data.length;
480         }
481 
482         void add(byte b) {
483             data[length++] = b;
484         }
485 
486         void ensure(int numBytes) {
487             int newCapacity = length + numBytes;
488             if (newCapacity <= data.length) {
489                 return;
490             }
491 
492             newCapacity =
493                     (int) Math.max(Math.min((long) data.length + (data.length >> 1), Arrays.MAX_ARRAY_SIZE),
494                                    newCapacity);
495 
496             data = ByteArrays.forceCapacity(data, newCapacity, length);
497         }
498     }

500     private static final class CodePointIterator {
501         private final CharSequence str;
502         private final int end;
503         private int pos;
504 
505         CodePointIterator(CharSequence str, int start, int end) {
506             this.str = str;
507             this.end = end;
508             pos = start;
509         }
510 
511         int position() {
512             return pos;
513         }
514 
515         void position(int pos) {
516             this.pos = pos;
517         }
518 
519         boolean hasNextCodePoint() {
520             return pos < end;
521         }
522 
523         int nextCodePoint() {
524             assert pos < end;
525 
526             final char c1 = str.charAt(pos++);
527             if (Character.isHighSurrogate(c1) && pos < end) {
528                 final char c2 = str.charAt(pos);
529                 if (Character.isLowSurrogate(c2)) {
530                     pos++;
531                     return Character.toCodePoint(c1, c2);
532                 }
533             }
534 
535             return c1;
536         }
537     }

539     /**
540      * Reserved characters which require percent-encoding. These values are only used for constructing
541      * {@link #RAW_CHAR_TO_MARKER} and {@link #MARKER_TO_PERCENT_ENCODED_CHAR} mapping tables.
542      *
543      * @see <a href="https://tools.ietf.org/html/rfc3986#section-2.2">RFC 3986, section 2.2</a>
544      */
545     private enum ReservedChar {
546         GEN_DELIM_01(':', "%3A", (byte) 0x01),
547         GEN_DELIM_02('/', "%2F", (byte) 0x02),
548         GEN_DELIM_03('?', "%3F", (byte) 0x03),
549         GEN_DELIM_04('#', "%23", (byte) 0x04),
550         GEN_DELIM_05('[', "%5B", (byte) 0x05),
551         GEN_DELIM_06(']', "%5D", (byte) 0x06),
552         GEN_DELIM_07('@', "%40", (byte) 0x07),
553 
554         SUB_DELIM_01('!', "%21", (byte) 0x11),
555         SUB_DELIM_02('$', "%24", (byte) 0x12),
556         SUB_DELIM_03('&', "%26", (byte) 0x13),
557         SUB_DELIM_04('\'', "%27", (byte) 0x14),
558         SUB_DELIM_05('(', "%28", (byte) 0x15),
559         SUB_DELIM_06(')', "%29", (byte) 0x16),
560         SUB_DELIM_07('*', "%2A", (byte) 0x17),
561         SUB_DELIM_08('+', "%2B", (byte) 0x18),
562         SUB_DELIM_09(',', "%2C", (byte) 0x19),
563         SUB_DELIM_10(';', "%3B", (byte) 0x1A),
564         SUB_DELIM_11('=', "%3D", (byte) 0x1B);
565 
566         private final int rawChar;
567         private final String percentEncodedChar;
568         private final byte marker;
569 
570         ReservedChar(int rawChar, String percentEncodedChar, byte marker) {
571             this.rawChar = rawChar;
572             this.percentEncodedChar = percentEncodedChar;
573             this.marker = marker;
574         }
575     }
