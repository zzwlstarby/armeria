66      /**
67       * The maximum allowed length of an HTTP chunk when TLS is enabled.
68       * <ul>
69       *   <li>16384 - The maximum length of a cleartext TLS record.</li>
70       *   <li>6 - The maximum header length of an HTTP chunk. i.e. "4000\r\n".length()</li>
71       * </ul>
72       *
73       * <p>To be precise, we have a chance of wasting 6 bytes because we may not use chunked encoding,
74       * but it is not worth adding complexity to be that precise.
75       */
76      private static final int MAX_TLS_DATA_LENGTH = 16384 - 6;

118     @Override
119     protected ChannelFuture doWriteHeaders(int id, int streamId, HttpHeaders headers, boolean endStream) {
120         if (!isWritable(id)) {
121             return newClosedSessionFuture();
122         }
123 
124         try {
125             return server ? writeServerHeaders(id, streamId, headers, endStream)
126                           : writeClientHeaders(id, streamId, headers, endStream);
127         } catch (Throwable t) {
128             return newFailedFuture(t);
129         }
130     }
131 
132     private ChannelFuture writeServerHeaders(
133             int id, int streamId, HttpHeaders headers, boolean endStream) throws Http2Exception {
134 
135         final HttpObject converted = convertServerHeaders(streamId, headers, endStream);
136         final String status = headers.get(HttpHeaderNames.STATUS);
137         if (status == null) {
138             // Trailers
139             final ChannelFuture f = write(id, converted, endStream);
140             ch.flush();
141             return f;
142         }
143 
144         if (!status.isEmpty() && status.charAt(0) == '1') {
145             // Informational status headers.
146             final ChannelFuture f = write(id, converted, false);
147             if (endStream) {
148                 // Can't end a stream with informational status in HTTP/1.
149                 f.addListener(ChannelFutureListener.CLOSE);
150             }
151             ch.flush();
152             return f;
153         }
154 
155         // Non-informational status headers.
156         return writeNonInformationalHeaders(id, converted, endStream);
157     }
158 
159     private ChannelFuture writeClientHeaders(
160             int id, int streamId, HttpHeaders headers, boolean endStream) throws Http2Exception {
161 
162         return writeNonInformationalHeaders(id, convertClientHeaders(streamId, headers, endStream), endStream);
163     }
164 
165     private ChannelFuture writeNonInformationalHeaders(int id, HttpObject converted, boolean endStream) {
166 
167         ChannelFuture f;
168         if (converted instanceof LastHttpContent) {
169             assert endStream;
170             f = write(id, converted, true);
171         } else {
172             f = write(id, converted, false);
173             if (endStream) {
174                 f = write(id, LastHttpContent.EMPTY_LAST_CONTENT, true);
175             }
176         }
177 
178         ch.flush();
179         return f;
180     }

182     private HttpObject convertServerHeaders(
183             int streamId, HttpHeaders headers, boolean endStream) throws Http2Exception {
184 
185         // Leading headers will always have :status, trailers will never have it.
186         final String status = headers.get(HttpHeaderNames.STATUS);
187         if (status == null) {
188             return convertTrailingHeaders(streamId, headers);
189         }
190 
191         // Convert leading headers.
192         final HttpResponse res;
193         final int statusCode = Integer.parseInt(status);
194         final boolean informational = HttpStatusClass.INFORMATIONAL.contains(statusCode);
195         final HttpResponseStatus nettyStatus = HttpResponseStatus.valueOf(statusCode);
196 
197         if (endStream || informational) {
198 
199             res = new DefaultFullHttpResponse(
200                     HttpVersion.HTTP_1_1, nettyStatus,
201                     Unpooled.EMPTY_BUFFER, false);
202 
203             final io.netty.handler.codec.http.HttpHeaders outHeaders = res.headers();
204             convert(streamId, headers, outHeaders, false, false);
205 
206             if (HttpStatus.isContentAlwaysEmpty(statusCode)) {
207                 outHeaders.remove(HttpHeaderNames.CONTENT_LENGTH);
208             } else if (!headers.contains(HttpHeaderNames.CONTENT_LENGTH)) {
209                 // NB: Set the 'content-length' only when not set rather than always setting to 0.
210                 //     It's because a response to a HEAD request can have empty content while having
211                 //     non-zero 'content-length' header.
212                 //     However, this also opens the possibility of sending a non-zero 'content-length'
213                 //     header even when it really has to be zero. e.g. a response to a non-HEAD request
214                 outHeaders.setInt(HttpHeaderNames.CONTENT_LENGTH, 0);
215             }
216         } else {
217             res = new DefaultHttpResponse(HttpVersion.HTTP_1_1, nettyStatus, false);
218             // Perform conversion.
219             convert(streamId, headers, res.headers(), false, false);
220             setTransferEncoding(res);
221         }
222 
223         return res;
224     }

226     private HttpObject convertClientHeaders(int streamId, HttpHeaders headers, boolean endStream)
227             throws Http2Exception {
228 
229         // Leading headers will always have :method, trailers will never have it.
230         final String method = headers.get(HttpHeaderNames.METHOD);
231         if (method == null) {
232             return convertTrailingHeaders(streamId, headers);
233         }
234 
235         // Convert leading headers.
236         final String path = headers.get(HttpHeaderNames.PATH);
237         assert path != null;
238         final HttpRequest req = new DefaultHttpRequest(
239                 HttpVersion.HTTP_1_1,
240                 HttpMethod.valueOf(method),
241                 path, false);
242 
243         convert(streamId, headers, req.headers(), false, true);
244 
245         if (endStream) {
246             req.headers().remove(HttpHeaderNames.TRANSFER_ENCODING);
247 
248             // Set or remove the 'content-length' header depending on request method.
249             // See: https://tools.ietf.org/html/rfc7230#section-3.3.2
250             //
251             // > A user agent SHOULD send a Content-Length in a request message when
252             // > no Transfer-Encoding is sent and the request method defines a meaning
253             // > for an enclosed payload body.  For example, a Content-Length header
254             // > field is normally sent in a POST request even when the value is 0
255             // > (indicating an empty payload body).  A user agent SHOULD NOT send a
256             // > Content-Length header field when the request message does not contain
257             // > a payload body and the method semantics do not anticipate such a
258             // > body.
259             switch (method) {
260                 case "POST":
261                 case "PUT":
262                 case "PATCH":
263                     req.headers().set(HttpHeaderNames.CONTENT_LENGTH, "0");
264                     break;
265                 default:
266                     req.headers().remove(HttpHeaderNames.CONTENT_LENGTH);
267             }
268         } else if (HttpUtil.getContentLength(req, -1L) >= 0) {
269             // Avoid the case where both 'content-length' and 'transfer-encoding' are set.
270             req.headers().remove(HttpHeaderNames.TRANSFER_ENCODING);
271         } else {
272             req.headers().set(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
273         }
274 
275         return req;
276     }

278     private void convert(int streamId, HttpHeaders inHeaders,
279                          io.netty.handler.codec.http.HttpHeaders outHeaders, boolean trailer,
280                          boolean isRequest) throws Http2Exception {
281 
282         ArmeriaHttpUtil.toNettyHttp1(
283                 streamId, inHeaders, outHeaders, HttpVersion.HTTP_1_1, trailer, isRequest);
284 
285         outHeaders.remove(ExtensionHeaderNames.STREAM_ID.text());
286         if (server) {
287             outHeaders.remove(ExtensionHeaderNames.SCHEME.text());
288         } else {
289             outHeaders.remove(ExtensionHeaderNames.PATH.text());
290         }
291     }

293     private LastHttpContent convertTrailingHeaders(int streamId, HttpHeaders headers) throws Http2Exception {
294         final LastHttpContent lastContent;
295         if (headers.isEmpty()) {
296             lastContent = LastHttpContent.EMPTY_LAST_CONTENT;
297         } else {
298             lastContent = new DefaultLastHttpContent(Unpooled.EMPTY_BUFFER, false);
299             convert(streamId, headers, lastContent.trailingHeaders(), true, false);
300         }
301         return lastContent;
302     }
303 
304     private static void setTransferEncoding(HttpMessage out) {
305         final io.netty.handler.codec.http.HttpHeaders outHeaders = out.headers();
306         final long contentLength = HttpUtil.getContentLength(out, -1L);
307         if (contentLength < 0) {
308             // Use chunked encoding.
309             outHeaders.set(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
310             outHeaders.remove(HttpHeaderNames.CONTENT_LENGTH);
311         }
312     }

314     @Override
315     protected ChannelFuture doWriteData(int id, int streamId, HttpData data, boolean endStream) {
316         if (!isWritable(id)) {
317             ReferenceCountUtil.safeRelease(data);
318             return newClosedSessionFuture();
319         }
320 
321         final int length = data.length();
322         if (length == 0) {
323             ReferenceCountUtil.safeRelease(data);
324             final HttpContent content = endStream ? LastHttpContent.EMPTY_LAST_CONTENT : EMPTY_CONTENT;
325             final ChannelFuture future = write(id, content, endStream);
326             ch.flush();
327             return future;
328         }
329 
330         try {
331             if (!isTls || length <= MAX_TLS_DATA_LENGTH) {
332                 // Cleartext connection or data.length() <= MAX_TLS_DATA_LENGTH
333                 return doWriteUnsplitData(id, data, endStream);
334             } else {
335                 // TLS and data.length() > MAX_TLS_DATA_LENGTH
336                 return doWriteSplitData(id, data, endStream);
337             }
338         } catch (Throwable t) {
339             return newFailedFuture(t);
340         }
341     }
342 
343     private ChannelFuture doWriteUnsplitData(int id, HttpData data, boolean endStream) {
344         final ByteBuf buf = toByteBuf(data);
345         boolean handled = false;
346         try {
347             final HttpContent content;
348             if (endStream) {
349                 content = new DefaultLastHttpContent(buf);
350             } else {
351                 content = new DefaultHttpContent(buf);
352             }
353 
354             final ChannelFuture future = write(id, content, endStream);
355             handled = true;
356             ch.flush();
357             return future;
358         } finally {
359             if (!handled) {
360                 ReferenceCountUtil.safeRelease(buf);
361             }
362         }
363     }
364 
365     private ChannelFuture doWriteSplitData(int id, HttpData data, boolean endStream) {
366         try {
367             int offset = 0;
368             int remaining = data.length();
369             ChannelFuture lastFuture;
370             for (;;) {
371                 // Ensure an HttpContent does not exceed the maximum length of a cleartext TLS record.
372                 final int chunkSize = Math.min(MAX_TLS_DATA_LENGTH, remaining);
373                 lastFuture = write(id, new DefaultHttpContent(dataChunk(data, offset, chunkSize)), false);
374                 remaining -= chunkSize;
375                 if (remaining == 0) {
376                     break;
377                 }
378                 offset += chunkSize;
379             }
380 
381             if (endStream) {
382                 lastFuture = write(id, LastHttpContent.EMPTY_LAST_CONTENT, true);
383             }
384 
385             ch.flush();
386             return lastFuture;
387         } finally {
388             ReferenceCountUtil.safeRelease(data);
389         }
390     }

392     private static ByteBuf dataChunk(HttpData data, int offset, int chunkSize) {
393         if (data instanceof ByteBufHolder) {
394             final ByteBuf buf = ((ByteBufHolder) data).content();
395             return buf.retainedSlice(offset, chunkSize);
396         } else {
397             return Unpooled.wrappedBuffer(data.array(), offset, chunkSize);
398         }
399     }

401     private ChannelFuture write(int id, HttpObject obj, boolean endStream) {
402         if (id < currentId) {
403             // Attempted to write something on a finished request/response; discard.
404             // e.g. the request already timed out.
405             ReferenceCountUtil.safeRelease(obj);
406             return newFailedFuture(ClosedStreamException.get());
407         }
408 
409         final PendingWrites currentPendingWrites = pendingWritesMap.get(id);
410         if (id == currentId) {
411             if (currentPendingWrites != null) {
412                 pendingWritesMap.remove(id);
413                 flushPendingWrites(currentPendingWrites);
414             }
415 
416             final ChannelFuture future = ch.write(obj);
417             if (endStream) {
418                 currentId++;
419 
420                 // The next PendingWrites might be complete already.
421                 for (;;) {
422                     final PendingWrites nextPendingWrites = pendingWritesMap.get(currentId);
423                     if (nextPendingWrites == null) {
424                         break;
425                     }
426 
427                     flushPendingWrites(nextPendingWrites);
428                     if (!nextPendingWrites.isEndOfStream()) {
429                         break;
430                     }
431 
432                     pendingWritesMap.remove(currentId);
433                     currentId++;
434                 }
435             }
436 
437             return future;
438         } else {
439             final ChannelPromise promise = ch.newPromise();
440             final Entry<HttpObject, ChannelPromise> entry = new SimpleImmutableEntry<>(obj, promise);
441             final PendingWrites pendingWrites;
442             if (currentPendingWrites == null) {
443                 pendingWrites = new PendingWrites();
444                 maxIdWithPendingWrites = Math.max(maxIdWithPendingWrites, id);
445                 pendingWritesMap.put(id, pendingWrites);
446             } else {
447                 pendingWrites = currentPendingWrites;
448             }
449 
450             pendingWrites.add(entry);
451 
452             if (endStream) {
453                 pendingWrites.setEndOfStream();
454             }
455 
456             return promise;
457         }
458     }
459 
460     private void flushPendingWrites(PendingWrites pendingWrites) {
461         for (;;) {
462             final Entry<HttpObject, ChannelPromise> e = pendingWrites.poll();
463             if (e == null) {
464                 break;
465             }
466 
467             ch.write(e.getKey(), e.getValue());
468         }
469     }

471     @Override
472     protected ChannelFuture doWriteReset(int id, int streamId, Http2Error error) {
473         // NB: this.minClosedId can be overwritten more than once when 3+ pipelined requests are received
474         //     and they are handled by different threads simultaneously.
475         //     e.g. when the 3rd request triggers a reset and then the 2nd one triggers another.
476         minClosedId = Math.min(minClosedId, id);
477 
478         if (minClosedId <= maxIdWithPendingWrites) {
479             final ClosedSessionException cause =
480                     new ClosedSessionException("An HTTP/1 stream has been reset: " + error);
481             for (int i = minClosedId; i <= maxIdWithPendingWrites; i++) {
482                 final PendingWrites pendingWrites = pendingWritesMap.remove(i);
483                 for (;;) {
484                     final Entry<HttpObject, ChannelPromise> e = pendingWrites.poll();
485                     if (e == null) {
486                         break;
487                     }
488 
489                     e.getValue().tryFailure(cause);
490                 }
491             }
492         }
493 
494         final ChannelFuture f = ch.write(Unpooled.EMPTY_BUFFER);
495         if (!isWritable(currentId)) {
496             f.addListener(ChannelFutureListener.CLOSE);
497         }
498 
499         return f;
500     }

502     @Override
503     public boolean isWritable(int id, int streamId) {
504         return isWritable(id);
505     }
506 
507     private boolean isWritable(int id) {
508         return id < minClosedId;
509     }
510 
511     @Override
512     protected void doClose() {
513         if (pendingWritesMap.isEmpty()) {
514             return;
515         }
516 
517         final ClosedSessionException cause = ClosedSessionException.get();
518         for (Queue<Entry<HttpObject, ChannelPromise>> queue : pendingWritesMap.values()) {
519             for (;;) {
520                 final Entry<HttpObject, ChannelPromise> e = queue.poll();
521                 if (e == null) {
522                     break;
523                 }
524 
525                 e.getValue().tryFailure(cause);
526             }
527         }
528 
529         pendingWritesMap.clear();
530     }

532     private static final class PendingWrites extends ArrayDeque<Entry<HttpObject, ChannelPromise>> {
533 
534         private static final long serialVersionUID = 4241891747461017445L;

536         private boolean endOfStream;

538         PendingWrites() {
539             super(4);
540         }

542         @Override
543         public boolean add(Entry<HttpObject, ChannelPromise> httpObjectChannelPromiseEntry) {
544             return isEndOfStream() ? false : super.add(httpObjectChannelPromiseEntry);
545         }
546 
547         boolean isEndOfStream() {
548             return endOfStream;
549         }
550 
551         void setEndOfStream() {
552             endOfStream = true;
553         }
